37// Thread-safe bounded queue (ring buffer)
38typedef struct frame_queue_s {
39 void **slots;
40 int capacity;
41 int head, tail, count;
42 mutex_t mu;
43 cond_t not_empty;
44 cond_t not_full;
45 const char *name;
46} frame_queue_t;
47
48static frame_queue_t *frame_queue_create(int capacity, const char *name) {
49 frame_queue_t *q = SAFE_CALLOC(1, sizeof(*q), frame_queue_t *);
50 q->slots = SAFE_CALLOC(capacity, sizeof(void *), void *);
51 q->capacity = capacity;
52 q->head = q->tail = q->count = 0;
53 q->name = name;
54 mutex_init(&q->mu, name);
55 cond_init(&q->not_empty, "queue_not_empty");
56 cond_init(&q->not_full, "queue_not_full");
57 return q;
58}
59
60static void frame_queue_destroy(frame_queue_t *q) {
61 if (!q) return;
62 mutex_destroy(&q->mu);
63 cond_destroy(&q->not_empty);
64 cond_destroy(&q->not_full);
65 SAFE_FREE(q->slots);
66 SAFE_FREE(q);
67}

69// Push item, timeout_ns=0 means non-blocking (drops if full)
70static bool frame_queue_push(frame_queue_t *q, void *item, uint64_t timeout_ns) {
71 mutex_lock(&q->mu);
72 while (q->count >= q->capacity) {
73 if (timeout_ns == 0) {
74 // Non-blocking: drop the item
75 mutex_unlock(&q->mu);
76 return false;
77 }
78 if (!cond_timedwait(&q->not_full, &q->mu, timeout_ns)) {
79 mutex_unlock(&q->mu);
80 return false; // timeout
81 }
82 }
83 q->slots[q->tail] = item;
84 q->tail = (q->tail + 1) % q->capacity;
85 q->count++;
86 cond_signal(&q->not_empty);
87 mutex_unlock(&q->mu);
88 return true;
89}
90
91// Pop item, timeout_ns=0 means blocking forever
92static void *frame_queue_pop(frame_queue_t *q, uint64_t timeout_ns) {
93 mutex_lock(&q->mu);
94 while (q->count == 0) {
95 if (timeout_ns == 0) {
96 // Block forever
97 cond_wait(&q->not_empty, &q->mu);
98 } else {
99 if (!cond_timedwait(&q->not_empty, &q->mu, timeout_ns)) {
100 mutex_unlock(&q->mu);
101 return NULL; // timeout
102 }
103 }
104 }
105 void *item = q->slots[q->head];
106 q->head = (q->head + 1) % q->capacity;
107 q->count--;
108 cond_signal(&q->not_full);
109 mutex_unlock(&q->mu);
110 return item;
111}
112
113static int frame_queue_count(frame_queue_t *q) {
114 if (!q) return 0;
115 mutex_lock(&q->mu);
116 int count = q->count;
117 mutex_unlock(&q->mu);
118 return count;
119}
120
121static void frame_queue_flush(frame_queue_t *q, void (*free_fn)(void *)) {
122 if (!q) return;
123 mutex_lock(&q->mu);
124 while (q->count > 0) {
125 void *item = q->slots[q->head];
126 q->head = (q->head + 1) % q->capacity;
127 q->count--;
128 if (item && free_fn) free_fn(item);
129 }
130 mutex_unlock(&q->mu);
131}

137static pipeline_frame_t *make_frame_copy(const image_t *src) {
138 pipeline_frame_t *f = SAFE_CALLOC(1, sizeof(*f), pipeline_frame_t *);
139 f->w = src->w;
140 f->h = src->h;
141 f->captured_ns = time_get_ns();
142 size_t bytes = src->w * src->h * 3;
143 f->pixels = SAFE_MALLOC(bytes, uint8_t *);
144 memcpy(f->pixels, src->pixels, bytes);
145 return f;
146}

149 if (!f) return;

161 if (!f) {

163 return false;

200static void *pipeline_capture_thread(void *arg) {
201 session_pipeline_t *pipeline = (session_pipeline_t *)arg;
202 log_info("[PIPELINE_CAPTURE] Starting capture thread");
203
204 bool snapshot_mode = GET_OPTION(snapshot_mode);
205 if (snapshot_mode) {
206 // Initialize duration estimate BEFORE encoding frames so PTS scaling works from frame 1
207 extern uint64_t g_snapshot_actual_duration_ms;
208 double snapshot_delay = GET_OPTION(snapshot_delay);
209 g_snapshot_actual_duration_ms = (uint64_t)(snapshot_delay * 1000.0);
210 log_info("[PIPELINE_CAPTURE] Snapshot mode: initialized g_snapshot_actual_duration_ms=%llu ms (snapshot_delay=%.2f)",
211 (unsigned long long)g_snapshot_actual_duration_ms, snapshot_delay);
212 }
213
214 while (!atomic_load_bool(&pipeline->stop)) {
215 image_t *img = session_capture_read_frame(pipeline->capture);
216
217 if (!img) {
218 if (session_capture_at_end(pipeline->capture)) {
219 log_info("[PIPELINE_CAPTURE] End of media reached");
220 // Push EOF sentinels (zero-initialized)
221 pipeline_frame_t *sentinel1 = SAFE_CALLOC(1, sizeof(*sentinel1), pipeline_frame_t *);
222 frame_queue_push(pipeline->display_queue, sentinel1, 10 * NS_PER_MS_INT);
223 if (pipeline->has_render_file) {
224 pipeline_frame_t *sentinel2 = SAFE_CALLOC(1, sizeof(*sentinel2), pipeline_frame_t *);
225 frame_queue_push(pipeline->encode_queue, sentinel2, 10 * NS_PER_MS_INT);
226 }
227 break;
228 }
229 platform_sleep_ns(1 * NS_PER_MS_INT);
230 continue;
231 }
232
233 // Check snapshot mode elapsed time (measure from first capture, not first render)
234 if (snapshot_mode) {
235 uint64_t now_ns = time_get_ns();
236
237 // Set first capture timestamp on first frame (only once)
238 if (g_snapshot_first_capture_ns == 0) {
239 g_snapshot_first_capture_ns = now_ns;
240 log_info("[SNAPSHOT_CAPTURE] First frame captured at %llu ns", (unsigned long long)now_ns);
241 }
242
243 // Calculate elapsed time from first capture
244 double elapsed = (double)(now_ns - g_snapshot_first_capture_ns) / NS_PER_SEC_INT;
245 double snapshot_delay = GET_OPTION(snapshot_delay);
246
247 if (elapsed >= snapshot_delay) {
248 log_info("[PIPELINE_CAPTURE] Snapshot video elapsed=%.3f reached delay=%.2f - stopping capture but waiting for encode queue to drain",
249 elapsed, snapshot_delay);
250
251 // Update with actual measured duration so encoder can scale frames accurately
252 extern uint64_t g_snapshot_actual_duration_ms;
253 extern uint64_t g_snapshot_last_capture_elapsed_ns;
254 uint64_t actual_ms = (uint64_t)(elapsed * 1000.0);
255 g_snapshot_actual_duration_ms = actual_ms;
256 uint64_t last_frame_elapsed_ns = now_ns - g_snapshot_first_capture_ns;
257 g_snapshot_last_capture_elapsed_ns = last_frame_elapsed_ns;
258 log_info("[PIPELINE_CAPTURE] g_snapshot_actual_duration_ms=%llu, last_capture_elapsed_ns=%llu",
259 (unsigned long long)actual_ms, (unsigned long long)last_frame_elapsed_ns);
260
261 // Wait for encode queue to drain (all buffered frames processed)
262 // This ensures slow encoders can catch up and encode all captured frames
263 if (pipeline->has_render_file) {
264 while (frame_queue_count(pipeline->encode_queue) > 0) {
265 log_debug("[PIPELINE_CAPTURE] Waiting for encode queue to drain (%d frames pending)",
266 frame_queue_count(pipeline->encode_queue));
267 platform_sleep_ns(10 * NS_PER_MS_INT);
268 }
269 log_info("[PIPELINE_CAPTURE] Encode queue drained, sending EOF sentinel");
270 }
271
272 // Push EOF sentinels (zero-initialized)
273 pipeline_frame_t *sentinel1 = SAFE_CALLOC(1, sizeof(*sentinel1), pipeline_frame_t *);
274 frame_queue_push(pipeline->display_queue, sentinel1, 10 * NS_PER_MS_INT);
275 if (pipeline->has_render_file) {
276 pipeline_frame_t *sentinel2 = SAFE_CALLOC(1, sizeof(*sentinel2), pipeline_frame_t *);
277 frame_queue_push(pipeline->encode_queue, sentinel2, 10 * NS_PER_MS_INT);
278 }
279 break;
280 }
281 }
282
283 // Respect configured FPS
284 session_capture_sleep_for_fps(pipeline->capture);
285
286 // Copy frame
287 pipeline_frame_t *frame = make_frame_copy(img);
288
289 // Record first frame timestamp for snapshot timer
290 uint64_t zero = 0;
291 atomic_cas_u64(&pipeline->first_frame_ns, &zero, frame->captured_ns);
292
293 // Fan out: display queue (fast, can drop), encode queue (slow, keep all)
294 pipeline_frame_t *display_copy = SAFE_MALLOC(sizeof(*display_copy), pipeline_frame_t *);
295 memcpy(display_copy, frame, sizeof(*display_copy));
296 display_copy->pixels = SAFE_MALLOC(frame->w * frame->h * 3, uint8_t *);
297 memcpy(display_copy->pixels, frame->pixels, frame->w * frame->h * 3);
298
299 log_info("[PIPELINE_CAPTURE_PUSH_DISPLAY] Pushing %dx%d frame to display_queue", display_copy->w, display_copy->h);
300 if (!frame_queue_push(pipeline->display_queue, display_copy, 0)) {
301 // Non-blocking: drop if queue full
302 log_warn("[PIPELINE_CAPTURE_DROP] Display queue full, dropping frame");
303 free_frame(display_copy);
304 }
305
306 if (pipeline->has_render_file) {
307 // Encode all captured frames, including frames before display rendering starts
308 // The encoder will use them with correct timestamps regardless of display timing
309 if (!frame_queue_push(pipeline->encode_queue, frame, 500 * NS_PER_MS_INT)) {
310 // Non-blocking: drop if queue full after 500ms
311 log_warn("[PIPELINE_CAPTURE] Encode queue blocked, dropping frame");
312 free_frame(frame);
313 } else {
314 log_debug_every(60 * NS_PER_SEC_INT, "[PIPELINE_CAPTURE] Enqueued frame to encode_queue (%dx%d)", frame->w, frame->h);
315 }
316 } else {
317 log_warn_every(1000 * NS_PER_MS_INT, "[PIPELINE_CAPTURE] has_render_file=false, NOT encoding frames");
318 free_frame(frame);
319 }
320 }
321
322 log_info("[PIPELINE_CAPTURE] Capture thread exiting");
323 return NULL;
324}

330static void *pipeline_encode_thread(void *arg) {
331 session_pipeline_t *pipeline = (session_pipeline_t *)arg;
332 log_info("[PIPELINE_ENCODE] Starting encode thread");
333
334 uint64_t frames_processed = 0;
335
336 while (!atomic_load_bool(&pipeline->stop)) {
337 pipeline_frame_t *frame = (pipeline_frame_t *)frame_queue_pop(pipeline->encode_queue, 100 * NS_PER_MS_INT);
338
339 if (!frame) {
340 log_debug_every(1 * NS_PER_SEC_INT, "[PIPELINE_ENCODE] Waiting for frames (processed=%llu)", (unsigned long long)frames_processed);
341 continue; // timeout
342 }
343
344 if (!frame->pixels) {
345 // EOF sentinel
346 log_info("[PIPELINE_ENCODE] Received EOF sentinel, exiting (processed=%llu frames)", (unsigned long long)frames_processed);
347 free_frame(frame);
348 break;
349 }
350
351 // Validate frame before processing (catch memory corruption)
352 if (!frame_is_valid(frame)) {
353 log_error("Skipping frame with corrupted dimensions in encode thread");
354 free_frame(frame);
355 continue;
356 }
357
358 if (frames_processed == 0 || frames_processed % 30 == 0) {
359 log_info("[PIPELINE_ENCODE] Processing frame %llu: %dx%d", (unsigned long long)frames_processed, frame->w, frame->h);
360 }
361
362 // Encode frame (convert_to_ascii is called internally by encode_frame)
363 // Avoid double-conversion that was causing state desynchronization between display/encode threads
364 image_t raw_image = { .w = frame->w, .h = frame->h, .pixels = (rgb_pixel_t *)frame->pixels };
365 session_display_encode_frame(pipeline->display, &raw_image, frame->captured_ns);
366 frames_processed++;
367
368 free_frame(frame);
369 }
370
371 log_info("[PIPELINE_ENCODE] Encode thread exiting (total frames encoded=%llu)", (unsigned long long)frames_processed);
372 return NULL;
373}

379asciichat_error_t session_pipeline_create(
380 session_capture_ctx_t *capture,
381 session_display_ctx_t *display,
382 session_pipeline_t **out)
383{
384 if (!capture || !display || !out) {
385 return SET_ERRNO(ERROR_INVALID_PARAM, "session_pipeline_create: NULL argument");
386 }
387
388 session_pipeline_t *p = SAFE_CALLOC(1, sizeof(*p), session_pipeline_t *);
389 p->capture = capture;
390 p->display = display;
391 p->display_queue = frame_queue_create(4, "display");
392 p->encode_queue = frame_queue_create(256, "encode"); // Large buffer for slow encoding
393
394 // Check if display has render_file configured (will be checked in encode thread)
395 // We store a flag to know whether to enqueue frames for encoding
396 double render_fps = session_display_get_render_fps(display);
397 bool has_file = session_display_has_render_file(display);
398
399 // Enable encode thread if render_file is available
400 // Note: has_file can be true even if render_file creation failed (ctx->render_file exists)
401 // OR if render_fps is explicitly set for playback speed control
402 p->has_render_file = render_fps > 0 || has_file;
403 log_info("[PIPELINE_CREATE] render_fps=%.1f, has_file=%d, has_render_file=%d",
404 render_fps, has_file ? 1 : 0, p->has_render_file ? 1 : 0);
405
406 atomic_store_bool(&p->stop, false);
407 atomic_store_u64(&p->first_frame_ns, 0);
408
409 // Start capture thread
410 if (asciichat_thread_create(&p->capture_tid, "pipeline_capture", pipeline_capture_thread, p) != 0) {
411 frame_queue_destroy(p->display_queue);
412 frame_queue_destroy(p->encode_queue);
413 SAFE_FREE(p);
414 return SET_ERRNO(ERROR_INIT, "session_pipeline_create: failed to start capture thread");
415 }
416
417 // Start encode thread only if render_file is active
418 if (p->has_render_file) {
419 if (asciichat_thread_create(&p->encode_tid, "pipeline_encode", pipeline_encode_thread, p) != 0) {
420 atomic_store_bool(&p->stop, true);
421 asciichat_thread_join(&p->capture_tid, NULL);
422 frame_queue_destroy(p->display_queue);
423 frame_queue_destroy(p->encode_queue);
424 SAFE_FREE(p);
425 return SET_ERRNO(ERROR_INIT, "session_pipeline_create: failed to start encode thread");
426 }
427 }
428
429 *out = p;
430 return ASCIICHAT_OK;
431}

433asciichat_error_t session_pipeline_run_main(
434 session_pipeline_t *pipeline,
435 session_should_exit_fn should_exit,
436 session_keyboard_handler_fn keyboard_handler,
437 void *user_data)
438{
439 if (!pipeline || !should_exit) {
440 return SET_ERRNO(ERROR_INVALID_PARAM, "session_pipeline_run_main: NULL argument");
441 }
442
443 log_info("[PIPELINE_MAIN] Starting main thread loop");
444
445 bool snapshot_mode = GET_OPTION(snapshot_mode);
446 bool snapshot_done = false;
447
448 while (!should_exit(user_data) && !atomic_load_bool(&pipeline->stop) && !snapshot_done) {
449 pipeline_frame_t *frame = (pipeline_frame_t *)frame_queue_pop(pipeline->display_queue, 100 * NS_PER_MS_INT);
450
451 if (!frame) continue; // timeout, check should_exit again
452
453 // Log frame details IMMEDIATELY after popping
454 log_debug("[PIPELINE_MAIN_POP] frame=%p, pixels=%p, w=%d, h=%d",
455 (void *)frame, (void *)frame->pixels, frame->w, frame->h);
456
457 if (!frame->pixels) {
458 // EOF sentinel
459 log_info("[PIPELINE_MAIN_EOF] Received EOF sentinel, stopping");
460 free_frame(frame);
461 break;
462 }
463
464 // Validate frame before processing (catch memory corruption)
465 if (!frame_is_valid(frame)) {
466 log_error("Skipping frame with corrupted dimensions: w=%d, h=%d", frame->w, frame->h);
467 free_frame(frame);
468 continue;
469 }
470
471 log_info("[PIPELINE_MAIN_RENDER] Converting frame to ASCII: %dx%d", frame->w, frame->h);
472 // Convert to ASCII
473 image_t tmp = { .w = frame->w, .h = frame->h, .pixels = (rgb_pixel_t *)frame->pixels };
474 char *ascii = session_display_convert_to_ascii(pipeline->display, &tmp);
475 free_frame(frame);
476
477 if (ascii) {
478 // Write ASCII to terminal
479 session_display_write_ascii(pipeline->display, ascii);
480 SAFE_FREE(ascii);
481 }
482
483 // Keyboard polling
484 if (keyboard_handler) {
485 keyboard_key_t key = keyboard_read_nonblocking();
486 if (key > 0) {
487 keyboard_handler(NULL, (int)key, user_data); // NULL capture ctx
488 }
489 }
490
491 // Snapshot mode: check if elapsed time has reached snapshot_delay duration
492 // This check runs every iteration after first frame is rendered (when display.c sets g_snapshot_first_frame_rendered)
493 if (snapshot_mode && !snapshot_done) {
494 if (g_snapshot_first_frame_rendered && g_snapshot_first_frame_rendered_ns > 0) {
495 double snapshot_delay = GET_OPTION(snapshot_delay);
496 uint64_t now_ns = time_get_ns();
497 uint64_t elapsed_ns = now_ns - g_snapshot_first_frame_rendered_ns;
498 double elapsed_sec = (double)elapsed_ns / (double)NS_PER_SEC_INT;
499
500 if ((unsigned long)(elapsed_sec * 10) % 10 == 0 || elapsed_sec < 0.2) {
501 log_info("[SNAPSHOT_PIPELINE] CHECK: elapsed=%.3f target=%.2f", elapsed_sec, snapshot_delay);
502 }
503
504 // snapshot_delay=0 means exit after first frame
505 // snapshot_delay>0 means wait that many seconds before exiting
506 bool should_snapshot_exit = (snapshot_delay == 0.0) || (elapsed_sec >= snapshot_delay);
507
508 if (should_snapshot_exit) {
509 log_info("[SNAPSHOT_PIPELINE] EXITING - elapsed=%.3f target=%.2f", elapsed_sec, snapshot_delay);
510 snapshot_done = true;
511 }
512 }
513 }
514 }
515
516 log_info("[PIPELINE_MAIN] Main loop exiting, signaling threads to stop");
517 atomic_store_bool(&pipeline->stop, true);
518
519 return ASCIICHAT_OK;
520}

522asciichat_error_t session_pipeline_destroy(session_pipeline_t *pipeline) {
523 if (!pipeline) return ASCIICHAT_OK;
524
525 log_info("[PIPELINE] Destroying pipeline, waiting for threads...");
526
527 // Signal threads to stop (may already be stopped)
528 atomic_store_bool(&pipeline->stop, true);
529
530 // Drain and wait for capture thread
531 if (asciichat_thread_is_initialized(&pipeline->capture_tid)) {
532 asciichat_thread_join(&pipeline->capture_tid, NULL);
533 }
534
535 // Drain and wait for encode thread
536 if (asciichat_thread_is_initialized(&pipeline->encode_tid)) {
537 asciichat_thread_join(&pipeline->encode_tid, NULL);
538 }
539
540 // Flush queues
541 frame_queue_flush(pipeline->display_queue, free_frame_generic);
542 frame_queue_flush(pipeline->encode_queue, free_frame_generic);
543
544 frame_queue_destroy(pipeline->display_queue);
545 frame_queue_destroy(pipeline->encode_queue);
546 SAFE_FREE(pipeline);
547
548 log_info("[PIPELINE] Pipeline destroyed");
549 return ASCIICHAT_OK;
550}
