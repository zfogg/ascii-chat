102static asciichat_error_t client_send_sdp(const uint8_t session_id[16], const uint8_t recipient_id[16],
103 const char *sdp_type, const char *sdp, void *user_data) {
104 (void)user_data;
105
106 if (!session_id || !recipient_id || !sdp_type || !sdp) {
107 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid SDP parameters");
108 }
109
110 ensure_mutex_initialized();
111 mutex_lock(&g_signaling_mutex);
112
113 // Check if ACDS connection is active
114 if (!g_acds_transport) {
115 mutex_unlock(&g_signaling_mutex);
116 return SET_ERRNO(ERROR_INVALID_STATE, "ACDS transport not available");
117 }
118
119 // Check if session context is set
120 if (!g_session_context.is_set) {
121 mutex_unlock(&g_signaling_mutex);
122 return SET_ERRNO(ERROR_INVALID_STATE, "Session context not set");
123 }
124
125 // Calculate SDP length
126 size_t sdp_len = strlen(sdp);
127 if (sdp_len == 0 || sdp_len >= 8192) {
128 mutex_unlock(&g_signaling_mutex);
129 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid SDP length: %zu", sdp_len);
130 }
131
132 // Allocate packet buffer (header + SDP string)
133 size_t total_len = sizeof(acip_webrtc_sdp_t) + sdp_len;
134 uint8_t *packet = SAFE_MALLOC(total_len, uint8_t *);
135 if (!packet) {
136 mutex_unlock(&g_signaling_mutex);
137 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate SDP packet");
138 }
139
140 // Fill header
141 acip_webrtc_sdp_t *header = (acip_webrtc_sdp_t *)packet;
142 memcpy(header->session_id, session_id, 16);
143 memcpy(header->sender_id, g_session_context.participant_id, 16);
144 memcpy(header->recipient_id, recipient_id, 16);
145 header->sdp_type = (strcmp(sdp_type, "offer") == 0) ? 0 : 1;
146 header->sdp_len = HOST_TO_NET_U16((uint16_t)sdp_len);
147
148 log_debug("SDP packet: session_id=%02x%02x%02x%02x..., sender=%02x%02x%02x%02x..., recipient=%02x%02x%02x%02x...",
149 session_id[0], session_id[1], session_id[2], session_id[3], g_session_context.participant_id[0],
150 g_session_context.participant_id[1], g_session_context.participant_id[2],
151 g_session_context.participant_id[3], recipient_id[0], recipient_id[1], recipient_id[2], recipient_id[3]);
152
153 // Copy SDP string after header
154 memcpy(packet + sizeof(acip_webrtc_sdp_t), sdp, sdp_len);
155
156 log_info("Sending WebRTC SDP %s to participant (%.8s...) via ACDS", sdp_type, (const char *)recipient_id);
157
158 // Send via ACDS transport using generic packet sender
159 asciichat_error_t result =
160 packet_send_via_transport(g_acds_transport, PACKET_TYPE_ACIP_WEBRTC_SDP, packet, total_len, 0);
161
162 mutex_unlock(&g_signaling_mutex);
163
164 SAFE_FREE(packet);
165
166 if (result != ASCIICHAT_OK) {
167 return SET_ERRNO(result, "Failed to send SDP via ACDS");
168 }
169
170 return ASCIICHAT_OK;
171}

182static asciichat_error_t client_send_ice(const uint8_t session_id[16], const uint8_t recipient_id[16],
183 const char *candidate, const char *mid, void *user_data) {
184 (void)user_data;
185
186 if (!session_id || !recipient_id || !candidate || !mid) {
187 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid ICE parameters");
188 }
189
190 ensure_mutex_initialized();
191 mutex_lock(&g_signaling_mutex);
192
193 // Check if ACDS connection is active
194 if (!g_acds_transport) {
195 mutex_unlock(&g_signaling_mutex);
196 return SET_ERRNO(ERROR_INVALID_STATE, "ACDS transport not available");
197 }
198
199 // Check if session context is set
200 if (!g_session_context.is_set) {
201 mutex_unlock(&g_signaling_mutex);
202 return SET_ERRNO(ERROR_INVALID_STATE, "Session context not set");
203 }
204
205 // Calculate payload length (candidate + null + mid + null)
206 size_t candidate_len = strlen(candidate);
207 size_t mid_len = strlen(mid);
208 size_t payload_len = candidate_len + 1 + mid_len + 1;
209
210 if (payload_len >= 8192) {
211 mutex_unlock(&g_signaling_mutex);
212 return SET_ERRNO(ERROR_INVALID_PARAM, "ICE payload too large: %zu", payload_len);
213 }
214
215 // Allocate packet buffer (header + payload)
216 size_t total_len = sizeof(acip_webrtc_ice_t) + payload_len;
217 uint8_t *packet = SAFE_MALLOC(total_len, uint8_t *);
218 if (!packet) {
219 mutex_unlock(&g_signaling_mutex);
220 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate ICE packet");
221 }
222
223 // Fill header
224 acip_webrtc_ice_t *header = (acip_webrtc_ice_t *)packet;
225 memcpy(header->session_id, session_id, 16);
226 memcpy(header->sender_id, g_session_context.participant_id, 16);
227 memcpy(header->recipient_id, recipient_id, 16);
228 header->candidate_len = HOST_TO_NET_U16((uint16_t)candidate_len); // FIXED: Use candidate length, not total payload
229
230 // Copy candidate and mid after header
231 uint8_t *payload = packet + sizeof(acip_webrtc_ice_t);
232 memcpy(payload, candidate, candidate_len);
233 payload[candidate_len] = '\0';
234 memcpy(payload + candidate_len + 1, mid, mid_len);
235 payload[candidate_len + 1 + mid_len] = '\0';
236
237 log_debug("Sending WebRTC ICE candidate to participant (%.8s..., mid=%s) via ACDS", (const char *)recipient_id, mid);
238
239 // Send via ACDS transport using generic packet sender
240 asciichat_error_t result =
241 packet_send_via_transport(g_acds_transport, PACKET_TYPE_ACIP_WEBRTC_ICE, packet, total_len, 0);
242
243 mutex_unlock(&g_signaling_mutex);
244
245 SAFE_FREE(packet);
246
247 if (result != ASCIICHAT_OK) {
248 return SET_ERRNO(result, "Failed to send ICE via ACDS");
249 }
250
251 return ASCIICHAT_OK;
252}
