27char *image_print(const image_t *p, const char *palette) {
28 if (!p || !palette) {
29 SET_ERRNO(ERROR_INVALID_PARAM, "image_print: p=%p or palette=%p is NULL", p, palette);
30 return NULL;
31 }
32 if (!p->pixels) {
33 SET_ERRNO(ERROR_INVALID_PARAM, "image_print: p->pixels is NULL");
34 return NULL;
35 }
36
37 const int h = p->h;
38 const int w = p->w;
39
40 if (h <= 0 || w <= 0) {
41 SET_ERRNO(ERROR_INVALID_PARAM, "image_print: invalid dimensions h=%d, w=%d", h, w);
42 return NULL;
43 }
44
45 // Get UTF-8 character cache for proper multi-byte character support
46 utf8_palette_cache_t *utf8_cache = get_utf8_palette_cache(palette);
47 if (!utf8_cache) {
48 SET_ERRNO(ERROR_INVALID_STATE, "Failed to get UTF-8 palette cache for scalar rendering");
49 return NULL;
50 }
51
52 // Character index ramp is now part of UTF-8 cache - no separate cache needed
53
54 // Need space for h rows with UTF-8 characters, plus h-1 newlines, plus null terminator
55 const size_t max_char_bytes = 4; // Max UTF-8 character size
56
57 const rgb_pixel_t *pix = p->pixels;
58
59 // Use outbuf_t for efficient UTF-8 RLE emission (same as SIMD renderers)
60 outbuf_t ob = {0};
61
62 // Calculate buffer size with overflow checking
63 size_t w_times_bytes;
64 if (checked_size_mul((size_t)w, max_char_bytes, &w_times_bytes) != ASCIICHAT_OK) {
65 SET_ERRNO(ERROR_INVALID_PARAM, "Buffer size overflow: width too large for UTF-8 encoding");
66 return NULL;
67 }
68
69 size_t w_times_bytes_plus_one;
70 if (checked_size_add(w_times_bytes, 1, &w_times_bytes_plus_one) != ASCIICHAT_OK) {
71 SET_ERRNO(ERROR_INVALID_PARAM, "Buffer size overflow: width * bytes + 1 overflow");
72 return NULL;
73 }
74
75 if (checked_size_mul((size_t)h, w_times_bytes_plus_one, &ob.cap) != ASCIICHAT_OK) {
76 SET_ERRNO(ERROR_INVALID_PARAM, "Buffer size overflow: height * (width * bytes + 1) overflow");
77 return NULL;
78 }
79
80 ob.buf = SAFE_MALLOC(ob.cap ? ob.cap : 1, char *);
81 if (!ob.buf) {
82 SET_ERRNO(ERROR_MEMORY, "Failed to allocate output buffer for scalar rendering");
83 return NULL;
84 }
85
86 // Process pixels with UTF-8 RLE emission (same approach as SIMD)
87 for (int y = 0; y < h; y++) {
88 const int row_offset = y * w;
89
90 for (int x = 0; x < w;) {
91 const rgb_pixel_t pixel = pix[row_offset + x];
92 // Use same luminance formula as SIMD: ITU-R BT.601 with rounding
93 const int luminance = (77 * pixel.r + 150 * pixel.g + 29 * pixel.b + 128) >> 8;
94
95 // Use same 6-bit precision as SIMD: map luminance (0-255) to bucket (0-63) then to character
96 uint8_t safe_luminance = clamp_rgb(luminance);
97 uint8_t luma_idx = (uint8_t)(safe_luminance >> 2); // 0-63 index (same as SIMD)
98 uint8_t char_idx = utf8_cache->char_index_ramp[luma_idx]; // Map to character index (same as SIMD)
99
100 // Use same 64-entry cache as SIMD for consistency
101 const utf8_char_t *char_info = &utf8_cache->cache64[luma_idx];
102
103 // Find run length for same character (RLE optimization)
104 int j = x + 1;
105 while (j < w) {
106 const rgb_pixel_t next_pixel = pix[row_offset + j];
107 const int next_luminance = (77 * next_pixel.r + 150 * next_pixel.g + 29 * next_pixel.b + 128) >> 8;
108 uint8_t next_safe_luminance = clamp_rgb(next_luminance);
109 uint8_t next_luma_idx = (uint8_t)(next_safe_luminance >> 2); // 0-63 index (same as SIMD)
110 uint8_t next_char_idx = utf8_cache->char_index_ramp[next_luma_idx]; // Map to character index (same as SIMD)
111 if (next_char_idx != char_idx)
112 break;
113 j++;
114 }
115 uint32_t run = (uint32_t)(j - x);
116
117 // Emit UTF-8 character with RLE (same as SIMD)
118 ob_write(&ob, (const char *)char_info->utf8_bytes, char_info->byte_len);
119 if (rep_is_profitable(run)) {
120 emit_rep(&ob, run - 1);
121 } else {
122 for (uint32_t k = 1; k < run; k++) {
123 ob_write(&ob, (const char *)char_info->utf8_bytes, char_info->byte_len);
124 }
125 }
126 x = j;
127 }
128
129 // Add newline between rows (except last row)
130 if (y != h - 1) {
131 ob_putc(&ob, '\n');
132 }
133 }
134
135 ob_term(&ob);
136 return ob.buf;
137}

293 // Check for incomplete ANSI sequences line by line

297 while (*line_start) {
298 const char *line_end = line_start;

302 // Check for incomplete ANSI sequence at end of this line

305 const char *seq_start = p;
306 p += 2;
307 // Scan for terminator (@ through ~, i.e., 0x40-0x7E)
308 while (p < line_end && !(*p >= '@' && *p <= '~'))
309 p++;
310 if (p >= line_end) {
311 // Incomplete sequence found
312 size_t incomplete_len = (size_t)(line_end - seq_start);
313 log_error("RENDER_INCOMPLETE_ANSI: Line %d (w=%d h=%d) has incomplete ANSI sequence (%zu bytes)", line_num,
314 w, h, incomplete_len);
315 // Show the incomplete sequence
316 char debug_buf[128] = {0};
317 size_t debug_pos = 0;
318 for (const char *d = seq_start; d < line_end && debug_pos < sizeof(debug_buf) - 10; d++) {
319 unsigned char c = (unsigned char)*d;
320 if (c == '\033') {
321 debug_buf[debug_pos++] = '<';
322 debug_buf[debug_pos++] = 'E';
323 debug_buf[debug_pos++] = 'S';
324 debug_buf[debug_pos++] = 'C';
325 debug_buf[debug_pos++] = '>';
326 } else if (c >= 0x20 && c < 0x7F) {
327 debug_buf[debug_pos++] = (char)c;
328 } else {
329 debug_pos += (size_t)safe_snprintf(debug_buf + debug_pos, sizeof(debug_buf) - debug_pos, "<%02X>", c);
330 }
331 }
332 log_error(" Incomplete seq: %s", debug_buf);
333 break;
334 } else {
335 p++; // Skip terminator
336 }
337 } else {
338 p++;
339 }
340 }
341
342 line_num++;
343 line_start = *line_end ? line_end + 1 : line_end;
344 }
345 }
346
347 return lines;
348}

363void rgb_to_ansi_8bit(int r, int g, int b, int *fg_code, int *bg_code) {
364 if (!fg_code || !bg_code) {
365 SET_ERRNO(ERROR_INVALID_PARAM, "rgb_to_ansi_8bit: fg_code or bg_code is NULL");
366 return;
367 }
368
369 // Convert RGB to 8-bit color code (216 color cube + 24 grayscale)
370 if (r == g && g == b) {
371 // Grayscale
372 if (r < 8) {
373 *fg_code = 16;
374 } else if (r > 248) {
375 *fg_code = 231;
376 } else {
377 *fg_code = 232 + (r - 8) / 10;
378 }
379 } else {
380 // Color cube: 16 + 36*r + 6*g + b where r,g,b are 0-5
381 int r_level = (r * 5) / 255;
382 int g_level = (g * 5) / 255;
383 int b_level = (b * 5) / 255;
384 *fg_code = 16 + 36 * r_level + 6 * g_level + b_level;
385 }
386 *bg_code = *fg_code; // Same logic for background
387}

673char *image_print_16color_dithered_with_background(const image_t *image, bool use_background, const char *palette) {
674 if (!image || !image->pixels || !palette) {
675 SET_ERRNO(ERROR_INVALID_PARAM, "image=%p or image->pixels=%p or palette=%p is NULL", image, image->pixels, palette);

677 }
678
679 int h = image->h;
680 int w = image->w;
681
682 if (h <= 0 || w <= 0) {
683 SET_ERRNO(ERROR_INVALID_STATE, "image_print_16color_dithered_with_background: invalid dimensions h=%d, w=%d", h, w);
684 return NULL;
685 }
686
687 // Initialize 16-color lookup table
688 ansi_fast_init_16color();
689
690 // Allocate error buffer for Floyd-Steinberg dithering
691 size_t pixel_count = (size_t)h * (size_t)w;
692 rgb_error_t *error_buffer;
693 error_buffer = SAFE_CALLOC(pixel_count, sizeof(rgb_error_t), rgb_error_t *);
694
695 // Calculate buffer size (larger for background mode due to more ANSI sequences)
696 size_t buffer_size =
697 (size_t)h * (size_t)w * (use_background ? 24 : 12) + (size_t)h; // Space for ANSI codes + newlines
698 char *buffer;
699 buffer = SAFE_MALLOC(buffer_size, char *);

702 const char *reset_code = "\033[0m";
703

706 rgb_pixel_t pixel = image->pixels[y * w + x];

714 get_16color_rgb(color_index, &bg_r, &bg_g, &bg_b);
715
716 // Calculate luminance to choose contrasting foreground
717 int bg_luminance = (bg_r * 77 + bg_g * 150 + bg_b * 29) / 256;
718 uint8_t fg_color = (bg_luminance < 127) ? 15 : 0; // White on dark, black on bright

721 ptr = append_16color_fg(ptr, fg_color); // Set contrasting foreground

734 return NULL;

738 uint8_t safe_luminance = clamp_rgb(luminance);
739 uint8_t luma_idx = (uint8_t)(safe_luminance >> 2); // 0-63 index (same as SIMD)
740 uint8_t char_idx = utf8_cache->char_index_ramp[luma_idx]; // Map to character index (same as SIMD)
741
742 // Use direct palette character lookup (same as SIMD would do)
743 char ascii_char = palette[char_idx];
744 const utf8_char_t *char_info = &utf8_cache->cache[(unsigned char)ascii_char];
745
746 if (char_info) {
747 // Copy UTF-8 character bytes
748 for (int byte_idx = 0; byte_idx < char_info->byte_len; byte_idx++) {
749 *ptr++ = char_info->utf8_bytes[byte_idx];

760 SAFE_STRNCPY(ptr, reset_code, 5);

763 *ptr++ = '\n';
764 }
765 }
