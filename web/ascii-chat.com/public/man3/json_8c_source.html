34static size_t format_timestamp_microseconds(uint64_t time_nanoseconds, char *buf, size_t buf_size) {
35 if (buf_size < 32) {
36 return 0;
37 }
38
39 /* Extract seconds and nanoseconds */
40 time_t seconds = (time_t)(time_nanoseconds / NS_PER_SEC_INT);
41 long nanoseconds = (long)(time_nanoseconds % NS_PER_SEC_INT);
42
43 /* Convert seconds to struct tm */
44 struct tm tm_info;
45 platform_localtime(&seconds, &tm_info);
46
47 /* Format the time part (HH:MM:SS) */
48 size_t len = strftime(buf, buf_size, "%H:%M:%S", &tm_info);
49 if (len == 0 || len >= buf_size) {
50 return 0;
51 }
52
53 /* Convert nanoseconds to microseconds for display */
54 long microseconds = nanoseconds / 1000;
55 if (microseconds < 0) {
56 microseconds = 0;
57 }
58 if (microseconds > 999999) {
59 microseconds = 999999;
60 }
61
62 /* Append microseconds */
63 int result = safe_snprintf(buf + len, buf_size - len, ".%06ld", microseconds);
64 if (result < 0 || result >= (int)(buf_size - len)) {
65 return 0;
66 }
67
68 return len + (size_t)result;
69}

186static size_t json_escape_async_safe(const char *src, char *dest, size_t dest_size) {
187 if (!src || !dest || dest_size < 2) {
188 return 0;
189 }
190
191 size_t pos = 0;
192 for (const char *p = src; *p && pos < dest_size - 1; p++) {
193 char c = *p;
194 if (c == '"' || c == '\\') {
195 if (pos + 1 < dest_size - 1) {
196 dest[pos++] = '\\';
197 dest[pos++] = c;
198 } else {
199 break;
200 }
201 } else if (c == '\n') {
202 if (pos + 1 < dest_size - 1) {
203 dest[pos++] = '\\';
204 dest[pos++] = 'n';
205 } else {
206 break;
207 }
208 } else if (c == '\r') {
209 if (pos + 1 < dest_size - 1) {
210 dest[pos++] = '\\';
211 dest[pos++] = 'r';
212 } else {
213 break;
214 }
215 } else if (c == '\t') {
216 if (pos + 1 < dest_size - 1) {
217 dest[pos++] = '\\';
218 dest[pos++] = 't';
219 } else {
220 break;
221 }
222 } else if ((unsigned char)c < 32) {
223 /* Skip other control characters */
224 continue;
225 } else {
226 dest[pos++] = c;
227 }
228 }
229 dest[pos] = '\0';
230 return pos;
231}

274 format_timestamp_microseconds(time_ns, timestamp_buf, sizeof(timestamp_buf));

276 /* Get thread ID (note: asciichat_thread_current_id is assumed to be async-safe) */
277 uint64_t tid = (uint64_t)asciichat_thread_current_id();
278
279 /* Format complete JSON object with timestamp */
280 int written = snprintf(json_buffer, sizeof(json_buffer),
281 "{\"header\":{\"timestamp\":\"%s\",\"level\":\"%s\",\"tid\":%lu,"
282 "\"file\":\"%s\",\"line\":%d,\"func\":\"%s\"},"
