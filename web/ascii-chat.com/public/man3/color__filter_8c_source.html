169void color_filter_calculate_rainbow(float time, uint8_t *r, uint8_t *g, uint8_t *b) {
170 // Cycle period: 3.5 seconds
171 const float cycle_period = 3.5f;
172
173 // Normalize time to [0, 1] range within cycle
174 float phase = fmodf(time, cycle_period) / cycle_period;
175
176 // Convert to hue angle [0, 360)
177 float hue = phase * 360.0f;
178
179 // HSV to RGB conversion (S=1, V=1 for full saturation and brightness)
180 // Hue divided into 6 segments
181 float h = hue / 60.0f;
182 int i = (int)floorf(h);
183 float f = h - (float)i;
184
185 // Calculate RGB based on hue segment
186 float q = 1.0f - f;
187 float t = f;
188
189 switch (i % 6) {
190 case 0:
191 *r = 255;
192 *g = (uint8_t)(t * 255.0f + 0.5f);
193 *b = 0;
194 break; // Red to Yellow
195 case 1:
196 *r = (uint8_t)(q * 255.0f + 0.5f);
197 *g = 255;
198 *b = 0;
199 break; // Yellow to Green
200 case 2:
201 *r = 0;
202 *g = 255;
203 *b = (uint8_t)(t * 255.0f + 0.5f);
204 break; // Green to Cyan
205 case 3:
206 *r = 0;
207 *g = (uint8_t)(q * 255.0f + 0.5f);
208 *b = 255;
209 break; // Cyan to Blue
210 case 4:
211 *r = (uint8_t)(t * 255.0f + 0.5f);
212 *g = 0;
213 *b = 255;
214 break; // Blue to Magenta
215 case 5:
216 *r = 255;
217 *g = 0;
218 *b = (uint8_t)(q * 255.0f + 0.5f);
219 break; // Magenta to Red
220 default:
221 *r = 255;
222 *g = 0;
223 *b = 0;
224 break; // Fallback to red
225 }
226
227 // Ensure minimum perceived brightness by boosting luminance
228 // Pure blue (0,0,255) has luminance ~18, which appears very dark
229 // We need minimum luminance of ~120 for bright, vibrant colors
230 const float min_luminance = 120.0f;
231
232 // Calculate current luminance using ITU-R BT.709 coefficients
233 float luminance = 0.2126f * *r + 0.7152f * *g + 0.0722f * *b;
234
235 if (luminance < min_luminance) {
236 // Boost luminance by adding white (increasing all channels proportionally)
237 // This desaturates the color but makes it much brighter
238 float boost = (min_luminance - luminance) / 3.0f;
239 *r = (uint8_t)fminf(255.0f, *r + boost);
240 *g = (uint8_t)fminf(255.0f, *g + boost);
241 *b = (uint8_t)fminf(255.0f, *b + boost);
242 }
243}

257static inline void colorize_grayscale_pixel(uint8_t gray, const color_filter_def_t *filter, uint8_t *out_r,
258 uint8_t *out_g, uint8_t *out_b) {
259 if (filter->foreground_on_bg) {
260 // Black-on-white mode: dark pixels get full color, light pixels become white
261 // Blend: color * (255 - gray) + white * gray, all / 255
262 // When gray=0 (black), we get full color. When gray=255 (white), we get white.
263 *out_r = (uint8_t)((filter->r * (255U - gray) + 255U * gray) / 255U);
264 *out_g = (uint8_t)((filter->g * (255U - gray) + 255U * gray) / 255U);
265 *out_b = (uint8_t)((filter->b * (255U - gray) + 255U * gray) / 255U);
266 } else {
267 // White-on-color mode: scale color intensity by grayscale
268 *out_r = (uint8_t)((filter->r * gray) / 255U);
269 *out_g = (uint8_t)((filter->g * gray) / 255U);
270 *out_b = (uint8_t)((filter->b * gray) / 255U);
271 }
272}

274int apply_color_filter(uint8_t *pixels, uint32_t width, uint32_t height, uint32_t stride, color_filter_t filter,
275 float time) {
276 if (!pixels || width == 0 || height == 0 || stride == 0) {
277 return -1;
278 }
279
280 if (filter == COLOR_FILTER_NONE) {
281 return 0; // No-op for none filter
282 }
283
284 // Special handling for rainbow filter
285 if (filter == COLOR_FILTER_RAINBOW) {
286 uint8_t rainbow_r, rainbow_g, rainbow_b;
287 color_filter_calculate_rainbow(time, &rainbow_r, &rainbow_g, &rainbow_b);
288
289 // Create temporary filter def with current rainbow color
290 color_filter_def_t rainbow_filter = {
291 .name = "rainbow",
292 .cli_name = "rainbow",
293 .r = rainbow_r,
294 .g = rainbow_g,
295 .b = rainbow_b,
296 .foreground_on_bg = false,
297 };
298
299 // Process each pixel with minimum brightness to prevent black
300 // Minimum brightness is 70% to ensure rainbow colors stay vibrant even for dark input
301 const uint8_t min_brightness = 179; // 70% of 255
302
303 for (uint32_t y = 0; y < height; y++) {
304 uint8_t *row = pixels + y * stride;
305 for (uint32_t x = 0; x < width; x++) {
306 uint8_t *pixel = row + x * 3;
307 uint8_t r = pixel[0];
308 uint8_t g = pixel[1];
309 uint8_t b = pixel[2];
310 uint8_t gray = rgb_to_grayscale(r, g, b);
311
312 // Boost grayscale to maintain minimum brightness for rainbow
313 // Formula: adjusted = min + gray * (1 - min/255)
314 // This ensures: black pixels -> min_brightness, white pixels -> 255
315 uint8_t adjusted_gray = min_brightness + (uint8_t)((gray * (255U - min_brightness)) / 255U);
316
317 colorize_grayscale_pixel(adjusted_gray, &rainbow_filter, &pixel[0], &pixel[1], &pixel[2]);
318 }
319 }
320 return 0;
321 }
322
323 const color_filter_def_t *filter_def = color_filter_get_metadata(filter);
324 if (!filter_def) {
325 return -1; // Invalid filter
326 }
327
328 // Process each pixel in the image
329 for (uint32_t y = 0; y < height; y++) {
330 uint8_t *row = pixels + y * stride;
331 for (uint32_t x = 0; x < width; x++) {
332 uint8_t *pixel = row + x * 3; // Assume RGB24 (3 bytes per pixel)
333 uint8_t r = pixel[0];
334 uint8_t g = pixel[1];
335 uint8_t b = pixel[2];
336
337 // Convert RGB to grayscale using ITU-R BT.601
338 uint8_t gray = rgb_to_grayscale(r, g, b);
339
340 // Colorize the grayscale value
341 colorize_grayscale_pixel(gray, filter_def, &pixel[0], &pixel[1], &pixel[2]);
342 }
343 }
344
345 return 0;
346}

348char *rainbow_replace_ansi_colors(const char *ansi_string, float time_seconds) {
349 if (!ansi_string) {
350 return NULL;
351 }
352
353 // Calculate rainbow color once for this frame
354 uint8_t rainbow_r = 0, rainbow_g = 0, rainbow_b = 0;
355 color_filter_calculate_rainbow(time_seconds, &rainbow_r, &rainbow_g, &rainbow_b);
356
357 // Build the replacement ANSI code string once
358 char rainbow_code[32];
359 snprintf(rainbow_code, sizeof(rainbow_code), "\x1b[38;2;%d;%d;%dm", rainbow_r, rainbow_g, rainbow_b);
360 size_t rainbow_code_len = strlen(rainbow_code);
361
362 // Check if there are any ANSI codes to replace
363 if (!strstr(ansi_string, "\x1b[38;2;")) {
364 return NULL; // No ANSI codes found, no replacement needed
365 }
366
367 // Allocate buffer for result (2x size for safety with many replacements)
368 size_t input_len = strlen(ansi_string);
369 char *new_result = SAFE_MALLOC(input_len * 2, char *);
370 if (!new_result) {
371 return NULL;
372 }
373
374 // Replace all RGB color codes with rainbow color
375 const char *src = ansi_string;
376 char *dst = new_result;
377
378 while (*src) {
379 const char *ansi_start = strstr(src, "\x1b[38;2;");
380 if (ansi_start) {
381 // Copy everything before this ANSI code
382 size_t before_len = (size_t)(ansi_start - src);
383 memcpy(dst, src, before_len);
384 dst += before_len;
385
386 // Skip the old ANSI code (find the 'm' that ends it)
387 const char *ansi_end = strchr(ansi_start + 7, 'm');
388 if (ansi_end) {
389 // Copy the rainbow replacement code
390 memcpy(dst, rainbow_code, rainbow_code_len);
391 dst += rainbow_code_len;
392
393 // Move source past the old code
394 src = ansi_end + 1;
395 } else {
396 // Malformed ANSI code, just copy it
397 *dst++ = *src++;
398 }
399 } else {
400 // No more ANSI codes, copy the rest
401 memcpy(dst, src, strlen(src) + 1);
402 break;
403 }
404 }
405
406 return new_result;
407}
