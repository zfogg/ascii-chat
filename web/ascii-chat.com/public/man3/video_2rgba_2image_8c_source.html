38image_t *image_new(size_t width, size_t height) {
39 image_t *p;
40
41 p = SAFE_MALLOC(sizeof(image_t), image_t *);
42
43 // Validate dimensions are non-zero and within bounds
44 if (image_validate_dimensions(width, height) != ASCIICHAT_OK) {
45 SET_ERRNO(ERROR_INVALID_PARAM, "Image dimensions invalid or too large: %zu x %zu", width, height);
46 SAFE_FREE(p);
47 return NULL;
48 }
49
50 // Calculate pixel count with overflow checking
51 size_t total_pixels;
52 if (checked_size_mul(width, height, &total_pixels) != ASCIICHAT_OK) {
53 SET_ERRNO(ERROR_INVALID_PARAM, "Image dimensions would cause overflow: %zu x %zu", width, height);
54 SAFE_FREE(p);
55 return NULL;
56 }
57
58 // Calculate total pixel buffer size with overflow checking
59 size_t pixels_size;
60 if (checked_size_mul(total_pixels, sizeof(rgb_pixel_t), &pixels_size) != ASCIICHAT_OK) {
61 SET_ERRNO(ERROR_INVALID_PARAM, "Image pixel buffer size would cause overflow");
62 SAFE_FREE(p);
63 return NULL;
64 }
65
66 if (pixels_size > IMAGE_MAX_PIXELS_SIZE) {
67 SET_ERRNO(ERROR_INVALID_PARAM, "Image size exceeds maximum allowed: %zu x %zu (%zu bytes)", width, height,
68 pixels_size);
69 SAFE_FREE(p);
70 return NULL;
71 }
72
73 // Use SIMD-aligned allocation for optimal NEON/AVX performance with vld3q_u8
74 p->pixels = SAFE_MALLOC_SIMD(pixels_size, rgb_pixel_t *);
75 if (!p->pixels) {
76 SET_ERRNO(ERROR_MEMORY, "Failed to allocate image pixels: %zu bytes", pixels_size);
77 SAFE_FREE(p);
78 return NULL;
79 }
80
81 p->w = (int)width;
82 p->h = (int)height;
83 p->alloc_method = IMAGE_ALLOC_SIMD; // Track allocation method for correct deallocation
84 return p;
85}

87void image_destroy(image_t *p) {
88 if (!p) {
89 SET_ERRNO(ERROR_INVALID_PARAM, "image_destroy: p is NULL");
90 return;
91 }
92
93 // Check allocation method and deallocate appropriately
94 if (p->alloc_method == IMAGE_ALLOC_POOL) {
95 // Pool-allocated images: free entire contiguous buffer (image + pixels)
96 // Validate dimensions before calculating size (guard against corruption)
97 if (p->w <= 0 || p->h <= 0) {
98 SET_ERRNO(ERROR_INVALID_PARAM, "image_destroy: invalid dimensions %dx%d (pool-allocated)", p->w, p->h);
99 return;
100 }
101
102 // Calculate original allocation size for proper buffer pool free
103 size_t w = (size_t)p->w;
104 size_t h = (size_t)p->h;
105 size_t pixels_size;
106 if (checked_size_mul3(w, h, sizeof(rgb_pixel_t), &pixels_size) != ASCIICHAT_OK) {
107 SET_ERRNO(ERROR_INVALID_STATE, "image_destroy: dimensions would overflow: %dx%d", p->w, p->h);
108 return;
109 }
110 size_t total_size;
111 if (checked_size_add(sizeof(image_t), pixels_size, &total_size) != ASCIICHAT_OK) {
112 SET_ERRNO(ERROR_INVALID_STATE, "image_destroy: total size would overflow: %dx%d", p->w, p->h);
113 return;
114 }
115
116 // Free the entire contiguous buffer back to pool
117 buffer_pool_free(NULL, p, total_size);
118 } else {
119 // SIMD-allocated images: free pixels and structure separately
120 // SAFE_MALLOC_SIMD allocates with aligned allocation (posix_memalign on macOS, aligned_alloc on Linux)
121 // These can be freed with standard free() / SAFE_FREE()
122 SAFE_FREE(p->pixels);
123 SAFE_FREE(p);
124 }

206 if (!p || !p->pixels) {
207 SET_ERRNO(ERROR_INVALID_PARAM, "image_clear: p or p->pixels is NULL");
208 return;
209 }
210 // Check for integer overflow before multiplication.
211 unsigned long w_ul = (unsigned long)p->w;
212 unsigned long h_ul = (unsigned long)p->h;
213 if (w_ul > 0 && h_ul > ULONG_MAX / w_ul) {
214 SET_ERRNO(ERROR_INVALID_PARAM, "image_clear: dimensions overflow: %d x %d", p->w, p->h);
215 return;
216 }

219 SET_ERRNO(ERROR_INVALID_PARAM, "image_clear: buffer size overflow");
220 return;
221 }
222 size_t clear_size = pixel_count * sizeof(rgb_pixel_t);
223 SAFE_MEMSET(p->pixels, clear_size, 0, clear_size);

267void image_resize_interpolation(const image_t *source, image_t *dest) {
268 if (!source || !dest || !source->pixels || !dest->pixels) {
269 SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters to image_resize_interpolation");
270 return;
271 }
272
273 const int src_w = source->w;
274 const int src_h = source->h;
275 const int dst_w = dest->w;
276 const int dst_h = dest->h;
277
278 // Handle edge cases
279 if (src_w <= 0 || src_h <= 0 || dst_w <= 0 || dst_h <= 0) {
280 SET_ERRNO(ERROR_INVALID_PARAM, "Invalid image dimensions for resize: src=%dx%d dst=%dx%d", src_w, src_h, dst_w,
281 dst_h);
282 return;
283 }
284
285 // Defensive checks to prevent invalid shift (UBSan protection)
286 if (src_w < 1 || src_h < 1 || dst_w < 1 || dst_h < 1) {
287 SET_ERRNO(ERROR_INVALID_PARAM, "Invalid dimensions detected after first check");
288 return;
289 }
290
291 // Use fixed-point arithmetic for better performance
292 // Use uint64_t intermediate to prevent overflow when shifting large dimensions
293 const uint32_t x_ratio = (uint32_t)((((uint64_t)src_w << 16) / (uint64_t)dst_w) + 1);
294 const uint32_t y_ratio = (uint32_t)((((uint64_t)src_h << 16) / (uint64_t)dst_h) + 1);
295
296 const rgb_pixel_t *src_pixels = source->pixels;
297 rgb_pixel_t *dst_pixels = dest->pixels;
298
299 for (int y = 0; y < dst_h; y++) {
300 const uint32_t src_y = ((uint32_t)(unsigned int)y * y_ratio) >> 16;
301 // Explicitly clamp to valid range [0, src_h-1]
302 const uint32_t safe_src_y = src_y >= (uint32_t)(unsigned int)src_h ? (uint32_t)(src_h - 1) : src_y;
303
304 // Bounds check: ensure safe_src_y is valid
305 if (safe_src_y >= (uint32_t)(unsigned int)src_h) {
306 SET_ERRNO(ERROR_INVALID_PARAM, "safe_src_y out of bounds: %u >= %d", safe_src_y, src_h);
307 return;
308 }
309
310 const rgb_pixel_t *src_row = src_pixels + (safe_src_y * (size_t)src_w);
311
312 rgb_pixel_t *dst_row = dst_pixels + ((size_t)y * (size_t)dst_w);
313
314 for (int x = 0; x < dst_w; x++) {
315 const uint32_t src_x = ((uint32_t)(unsigned int)x * x_ratio) >> 16;
316 // Explicitly clamp to valid range [0, src_w-1]
317 const uint32_t safe_src_x = src_x >= (uint32_t)(unsigned int)src_w ? (uint32_t)(src_w - 1) : src_x;
318
319 // Bounds check: ensure safe_src_x is valid
320 if (safe_src_x >= (uint32_t)(unsigned int)src_w) {
321 SET_ERRNO(ERROR_INVALID_PARAM, "safe_src_x out of bounds: %u >= %d", safe_src_x, src_w);
322 return;
323 }
324
325 dst_row[x] = src_row[safe_src_x];
326 }
327 }
328}

332void precalc_rgb_palettes(const float red, const float green, const float blue) {
333 // Validate input parameters to prevent overflow
334 // Luminance weights should typically be in range 0.0-1.0
335 // But allow slightly larger values for brightness adjustment
336 const float max_weight = 255.0f; // Maximum value that won't overflow when multiplied by 255
337 const float min_weight = -255.0f; // Allow negative for color correction
338
339 // Note: isfinite() checks are skipped in Release builds due to -ffinite-math-only flag
340 // which disables NaN/infinity support for performance. Range checks below are sufficient.
341#if !defined(NDEBUG) && !defined(__FAST_MATH__)
342 if (!isfinite(red) || !isfinite(green) || !isfinite(blue)) {
343 log_error("Invalid weight values (non-finite): red=%f, green=%f, blue=%f", red, green, blue);
344 SET_ERRNO(ERROR_INVALID_PARAM, "precalc_rgb_palettes: non-finite weight values");
345 return;
346 }
347#endif
348
349 if (red < min_weight || red > max_weight || green < min_weight || green > max_weight || blue < min_weight ||
350 blue > max_weight) {
351 log_warn(
352 "precalc_rgb_palettes: Weight values out of expected range: red=%f, green=%f, blue=%f (clamping to safe range)",
353 red, green, blue);
354 }
355
356 // Clamp weights to safe range to prevent overflow
357 const float safe_red = (red < min_weight) ? min_weight : ((red > max_weight) ? max_weight : red);
358 const float safe_green = (green < min_weight) ? min_weight : ((green > max_weight) ? max_weight : green);
359 const float safe_blue = (blue < min_weight) ? min_weight : ((blue > max_weight) ? max_weight : blue);
360
361 const unsigned short max_ushort = 65535;
362 const unsigned short min_ushort = 0;
363
364 for (int n = 0; n < ASCII_LUMINANCE_LEVELS; ++n) {
365 // Compute with float, then clamp to unsigned short range
366 float red_val = (float)n * safe_red;
367 float green_val = (float)n * safe_green;
368 float blue_val = (float)n * safe_blue;
369
370 // Clamp to unsigned short range before assignment
371 if (red_val < (float)min_ushort) {
372 red_val = (float)min_ushort;
373 } else if (red_val > (float)max_ushort) {
374 red_val = (float)max_ushort;
375 }
376
377 if (green_val < (float)min_ushort) {
378 green_val = (float)min_ushort;
379 } else if (green_val > (float)max_ushort) {
380 green_val = (float)max_ushort;
381 }
382
383 if (blue_val < (float)min_ushort) {
384 blue_val = (float)min_ushort;
385 } else if (blue_val > (float)max_ushort) {
386 blue_val = (float)max_ushort;
387 }
388
389 RED[n] = (unsigned short)red_val;
390 GREEN[n] = (unsigned short)green_val;
391 BLUE[n] = (unsigned short)blue_val;
392 GRAY[n] = (unsigned short)n;
393 }
394}
