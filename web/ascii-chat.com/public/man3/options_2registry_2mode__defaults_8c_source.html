64const mode_descriptor_t *get_modes_by_group(const char *group, size_t *out_count) {
65 if (!group || !out_count) {
66 return NULL;
67 }
68
69 // Count matching modes
70 size_t count = 0;
71 for (size_t i = 0; g_mode_descriptors[i].name; i++) {
72 if (strcmp(g_mode_descriptors[i].group, group) == 0) {
73 count++;
74 }
75 }
76
77 if (count == 0) {
78 *out_count = 0;
79 return NULL;
80 }
81
82 // Allocate result array
83 mode_descriptor_t *result = SAFE_MALLOC((count + 1) * sizeof(mode_descriptor_t), mode_descriptor_t *);
84 size_t idx = 0;
85
86 // Fill result array
87 for (size_t i = 0; g_mode_descriptors[i].name && idx < count; i++) {
88 if (strcmp(g_mode_descriptors[i].group, group) == 0) {
89 result[idx++] = g_mode_descriptors[i];
90 }
91 }
92
93 // Add terminator
94 result[idx].mode = MODE_INVALID;
95 result[idx].name = NULL;
96 result[idx].description = NULL;
97 result[idx].group = NULL;
98
99 *out_count = count;
100 return result;
101}

192void apply_mode_specific_defaults(options_t *opts) {
193 if (!opts) {
194 return;
195 }
196
197 // Apply port default based on mode
198 const int *mode_port = (const int *)get_default_port(opts->detected_mode);
199 if (mode_port) {
200 // Only update if still at a generic default (27224 or 27225)
201 if (opts->port == OPT_PORT_INT_DEFAULT || opts->port == OPT_ACDS_PORT_INT_DEFAULT) {
202 opts->port = *mode_port;
203 }
204 }
205
206 // Apply websocket-port default based on mode
207 const int *mode_websocket_port = (const int *)get_default_websocket_port(opts->detected_mode);
208 if (mode_websocket_port) {
209 // Only update if still at a generic default (27226 or 27227)
210 if (opts->websocket_port == OPT_WEBSOCKET_PORT_SERVER_DEFAULT ||
211 opts->websocket_port == OPT_WEBSOCKET_PORT_ACDS_DEFAULT) {
212 opts->websocket_port = *mode_websocket_port;
