769 if (!buffer || buffer_size < 1 || !nouns_count) {
770 return -1;
771 }
772
773 // Select a random noun
774 srand((unsigned int)time(NULL) + rand()); // Better randomness with seed
775 int noun_idx = rand() % (int)nouns_count;
776 const char *noun = nouns[noun_idx];
777
778 if (!noun) {

780 }

784 // Try to generate a unique name by appending a counter

786 for (int counter = 0; counter < 100; counter++) {
787 int written = snprintf(buffer, buffer_size, "%s.%d (%s:%d)", noun, counter, transport_name, port);
788 if (written < 0 || (size_t)written >= buffer_size) {
789 return -1; // Buffer too small

804 // Name is unique, return it
