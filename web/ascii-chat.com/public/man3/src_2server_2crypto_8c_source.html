161int server_crypto_init(void) {
162 // Only skip handshake if BOTH encryption and authentication are disabled
163 bool no_encrypt = GET_OPTION(no_encrypt);
164 bool no_auth = GET_OPTION(no_auth);
165
166 if (no_encrypt && no_auth) {
167 log_info("ACIP handshake disabled (both encryption and authentication disabled)");
168 return 0;
169 }
170
171 log_debug("Server crypto system initialized (per-client contexts will be created on demand)");
172 return 0;

646 client->crypto_initialized = true;

649 STOP_TIMER_AND_LOG(debug, 100 * NS_PER_MS_INT, "server_crypto_handshake_client_%s",
