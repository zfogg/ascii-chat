203static void on_datachannel_adapter(int pc_id, int dc_id, void *user_data) {
204 (void)pc_id; // Unused - we get peer connection from user_data
205 log_info("on_datachannel_adapter: pc_id=%d, dc_id=%d, user_data=%p", pc_id, dc_id, user_data);
206
207 webrtc_peer_connection_t *pc = (webrtc_peer_connection_t *)user_data;
208 if (!pc) {
209 log_error("on_datachannel_adapter: peer connection user_data is NULL!");
210 return;
211 }
212
213 log_info("on_datachannel_adapter: received DataChannel (dc_id=%d) from remote peer", dc_id);
214
215 // Allocate data channel wrapper
216 webrtc_data_channel_t *dc = SAFE_MALLOC(sizeof(webrtc_data_channel_t), webrtc_data_channel_t *);
217 if (!dc) {
218 log_error("Failed to allocate data channel wrapper");
219 rtcDeleteDataChannel(dc_id);
220 return;
221 }
222
223 dc->rtc_id = dc_id;
224 dc->pc = pc;
225 dc->is_open = false;
226 log_debug("Initialized DataChannel wrapper: dc=%p, is_open=false", (void *)dc);
227
228 // Store in peer connection
229 pc->dc = dc;
230
231 // Set up data channel callbacks for incoming channel
232 rtcSetUserPointer(dc_id, dc);
233 rtcSetOpenCallback(dc_id, on_datachannel_open_adapter);
234 rtcSetMessageCallback(dc_id, on_datachannel_message_adapter);
235 rtcSetErrorCallback(dc_id, on_datachannel_error_adapter);
236
237 log_info("Set up callbacks for incoming DataChannel (dc_id=%d, dc=%p)", dc_id, (void *)dc);
238
239 // Check if DataChannel is already open (can happen if it opened before callbacks were set)
240 // In libdatachannel, negotiated DataChannels may be open immediately when received
241 // We manually trigger the open callback if that's the case
242 if (rtcIsOpen(dc_id)) {
243 log_info("DataChannel was already open when received, manually triggering open callback");
244 on_datachannel_open_adapter(dc_id, dc);
245 }

381asciichat_error_t webrtc_create_peer_connection(const webrtc_config_t *config, webrtc_peer_connection_t **pc_out) {
382 if (!config || !pc_out) {
383 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid config or output parameter");
384 }
385
386 if (!lifecycle_is_initialized(&g_webrtc_lc)) {
387 return SET_ERRNO(ERROR_INIT, "WebRTC library not initialized");
388 }
389
390 // Allocate peer connection wrapper
391 webrtc_peer_connection_t *pc = SAFE_MALLOC(sizeof(webrtc_peer_connection_t), webrtc_peer_connection_t *);
392 if (!pc) {
393 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate peer connection");
394 }
395
396 pc->config = *config; // Copy config
397 pc->state = WEBRTC_STATE_NEW;
398 pc->dc = NULL;
399
400 // Build ICE server list for libdatachannel
401 const char **ice_servers = NULL;
402 size_t ice_count = 0;
403
404 if (config->stun_count > 0 || config->turn_count > 0) {
405 ice_count = config->stun_count + config->turn_count;
406 ice_servers = SAFE_MALLOC(ice_count * sizeof(char *), const char **);
407 if (!ice_servers) {
408 SAFE_FREE(pc);
409 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate ICE server list");
410 }
411
412 // Add STUN servers
413 for (size_t i = 0; i < config->stun_count; i++) {
414 ice_servers[i] = config->stun_servers[i].host;
415 }
416
417 // Add TURN servers
418 for (size_t i = 0; i < config->turn_count; i++) {
419 ice_servers[config->stun_count + i] = config->turn_servers[i].url;
420 }
421 }
422
423 // Create libdatachannel configuration
424 rtcConfiguration rtc_config;
425 memset(&rtc_config, 0, sizeof(rtc_config));
426 rtc_config.iceServers = ice_servers;
427 rtc_config.iceServersCount = (int)ice_count;
428 rtc_config.iceTransportPolicy = RTC_TRANSPORT_POLICY_ALL;
429
430 // Create peer connection
431 int pc_id = rtcCreatePeerConnection(&rtc_config);
432
433 // Free ICE server list (libdatachannel makes a copy)
434 if (ice_servers) {
435 SAFE_FREE(ice_servers);
436 }
437
438 if (pc_id < 0) {
439 SAFE_FREE(pc);
440 return SET_ERRNO(ERROR_NETWORK, "Failed to create peer connection (rtc error %d)", pc_id);
441 }
442
443 pc->rtc_id = pc_id;
444
445 // Initialize gathering state
446 pc->gathering_state = WEBRTC_GATHERING_NEW;
447 pc->gathering_start_time_ms = 0;
448
449 // Set up callbacks
450 rtcSetUserPointer(pc_id, pc);
451 rtcSetStateChangeCallback(pc_id, on_state_change_adapter);
452 rtcSetGatheringStateChangeCallback(pc_id, on_gathering_state_change_adapter);
453 rtcSetLocalDescriptionCallback(pc_id, on_local_description_adapter);
454 rtcSetLocalCandidateCallback(pc_id, on_local_candidate_adapter);
455 rtcSetDataChannelCallback(pc_id, on_datachannel_adapter);
456
457 /* Register peer connection with named registry */
458 char pc_name[64];
459 snprintf(pc_name, sizeof(pc_name), "webrtc_peer_connection:%d", pc_id);
460 NAMED_REGISTER(pc, pc_name, "webrtc_peer_connection_t", "0x%tx", NULL);
461
462 *pc_out = pc;
463 log_debug("Created WebRTC peer connection (id=%d)", pc_id);
464 return ASCIICHAT_OK;
465}

502bool webrtc_is_gathering_timed_out(webrtc_peer_connection_t *pc, uint32_t timeout_ms) {
503 if (!pc) {
504 return false;
505 }
506
507 // Only check timeout if we're actively gathering
508 if (pc->gathering_state != WEBRTC_GATHERING_GATHERING) {
509 return false;
510 }
511
512 // If gathering hasn't started yet (start_time == 0), no timeout
513 if (pc->gathering_start_time_ms == 0) {
514 return false;
515 }
516
517 // Check if elapsed time exceeds timeout
518 uint64_t current_time_ms = platform_get_monotonic_time_us() / 1000;
519 uint64_t elapsed_ms = current_time_ms - pc->gathering_start_time_ms;
520
521 if (elapsed_ms > timeout_ms) {
522 log_warn("ICE gathering timeout: %llu ms elapsed (timeout: %u ms)", (unsigned long long)elapsed_ms, timeout_ms);
523 return true;
524 }
525
526 return false;
527}

648 "%02x%02x%02x%02x %02x%02x%02x%02x %02x%02x%02x%02x",
649 dc->rtc_id, size, pkt[0], pkt[1], pkt[2], pkt[3], pkt[4], pkt[5], pkt[6], pkt[7], pkt[8], pkt[9], pkt[10],
650 pkt[11], pkt[12], pkt[13], pkt[14], pkt[15], pkt[16], pkt[17], pkt[18], pkt[19]);
651 } else {
652 log_debug("★ RTCSENDMESSAGE_BEFORE: dc_id=%d, size=%zu (too small to log content)", dc->rtc_id, size);
653 }
654
655 int result = rtcSendMessage(dc->rtc_id, (const char *)data, (int)size);
656
657 log_debug("★ RTCSENDMESSAGE_AFTER: dc_id=%d, rtcSendMessage returned %d for size=%zu", dc->rtc_id, result, size);
658
659 if (result < 0) {
660 log_error("★ WEBRTC_DATACHANNEL_SEND: FAILED with error code %d", result);
661 return SET_ERRNO(ERROR_NETWORK, "Failed to send data (rtc error %d)", result);
662 }
663
664 log_debug("★ WEBRTC_DATACHANNEL_SEND: SUCCESS - sent %zu bytes", size);
665 return ASCIICHAT_OK;
666}

718asciichat_error_t webrtc_datachannel_set_callbacks(webrtc_data_channel_t *dc,
719 const webrtc_datachannel_callbacks_t *callbacks) {
720 if (!dc) {
721 return SET_ERRNO(ERROR_INVALID_PARAM, "DataChannel is NULL");
722 }
723
724 if (!callbacks) {
725 return SET_ERRNO(ERROR_INVALID_PARAM, "Callbacks struct is NULL");
726 }
727
728 // Store user callbacks in the data channel struct
729 dc->user_on_open = callbacks->on_open;
730 dc->user_on_close = callbacks->on_close;
731 dc->user_on_error = callbacks->on_error;
732 dc->user_on_message = callbacks->on_message;
733 dc->user_data = callbacks->user_data;
734
735 // Register adapter functions with libdatachannel (adapters use correct signature)
736 // The adapters will look up our stored callbacks and invoke them with proper types
737 if (callbacks->on_open) {
738 rtcSetOpenCallback(dc->rtc_id, on_datachannel_open_adapter);
739 }
740
741 if (callbacks->on_close) {
742 rtcSetClosedCallback(dc->rtc_id, on_datachannel_closed_adapter);
743 }
744
745 if (callbacks->on_error) {
746 rtcSetErrorCallback(dc->rtc_id, on_datachannel_error_adapter);
747 }
748
749 if (callbacks->on_message) {
750 rtcSetMessageCallback(dc->rtc_id, on_datachannel_message_adapter);
751 }
752
753 // Set user pointer to the data channel so adapters can retrieve it
754 rtcSetUserPointer(dc->rtc_id, dc);
755
756 log_debug("Set DataChannel callbacks (dc_id=%d)", dc->rtc_id);
757 return ASCIICHAT_OK;
758}
