180int display_init() {
181 // Build display configuration from options
182 session_display_config_t config = {
183 .snapshot_mode = GET_OPTION(snapshot_mode),
184 .palette_type = GET_OPTION(palette_type),
185 .custom_palette = GET_OPTION(palette_custom_set) ? GET_OPTION(palette_custom) : NULL,
186 .color_mode = TERM_COLOR_AUTO // Will be overridden by command-line options
187 };
188
189 // Create display context using session library
190 g_display_ctx = session_display_create(&config);
191 if (!g_display_ctx) {
192 SET_ERRNO(ERROR_DISPLAY, "Failed to initialize display");
193 return -1;
194 }
195
196 // Check if client mode should load local media (--file or --url)
197 const char *media_url = GET_OPTION(media_url);
198 const char *media_file = GET_OPTION(media_file);
199
200 if ((media_url && strlen(media_url) > 0) || (media_file && strlen(media_file) > 0)) {
201 // Client mode with local media file/URL - create capture context for keyboard controls
202 session_capture_config_t capture_config = {0};
203 capture_config.target_fps = 60;
204 capture_config.resize_for_network = false;
205 capture_config.should_exit_callback = NULL;
206 capture_config.callback_data = NULL;
207
208 if (media_url && strlen(media_url) > 0) {
209 capture_config.type = MEDIA_SOURCE_FILE;
210 capture_config.path = media_url;
211 } else {
212 if (strcmp(media_file, "-") == 0) {
213 capture_config.type = MEDIA_SOURCE_STDIN;
214 capture_config.path = NULL;
215 } else {
216 capture_config.type = MEDIA_SOURCE_FILE;
217 capture_config.path = media_file;
218 }
219 }
220
221 capture_config.loop = GET_OPTION(media_loop);
222 capture_config.initial_seek_timestamp = GET_OPTION(media_seek_timestamp);
223
224 g_display_capture_ctx = session_capture_create(&capture_config);
225 if (!g_display_capture_ctx) {
226 log_warn("Failed to create capture context for local media - keyboard seek/pause disabled");
227 g_display_capture_ctx = NULL;
228 }
229 }
230
231 // Keyboard is pre-initialized from asciichat_shared_init()
232 // Only enable keyboard input if stdin is a TTY (avoid interfering with piped/redirected I/O)
233 g_keyboard_enabled = terminal_is_stdin_tty();
234
235 // Register display atomics with named debug registry
236 static bool display_atomics_registered = false;
237 if (!display_atomics_registered) {
238 NAMED_REGISTER_ATOMIC(&g_is_first_frame_of_connection, "first_frame_of_display_connection", NULL);
239 display_atomics_registered = true;
240 }
241
242 return 0;
243}

290void display_disable_logging_for_first_frame() {
291 // Atomically check if this is the first frame and mark as not-first
292 // Avoids TOCTOU race where multiple threads could both think they're first
293 bool was_first_frame = atomic_exchange_bool(&g_is_first_frame_of_connection, false);
294
295 if (was_first_frame) {
296 // Disable terminal logging before clearing display and rendering first frame
297 log_set_terminal_output(false);
298
299 // Signal the intro splash screen to stop - first frame is ready to render
300 splash_intro_done();
301 }

316void display_render_frame(const char *frame_data) {
317 if (!frame_data) {
318 SET_ERRNO(ERROR_INVALID_PARAM, "Attempted to render NULL frame data");
319 return;
320 }
321
322 // Stop splash screen animation on first frame
323 // This must be called BEFORE any frame rendering to prevent splash/frame flickering
324 display_disable_logging_for_first_frame();
325
326 // Render help screen if active (suppresses frame rendering)
327 if (keyboard_help_is_active(g_display_ctx)) {
328 keyboard_help_render(g_display_ctx);
329
330 // Still poll keyboard for interactive controls while help is visible
331 if (g_keyboard_enabled) {
332 keyboard_key_t key = keyboard_read_nonblocking();
333 if (key != KEY_NONE) {
334 session_handle_keyboard_input(g_display_capture_ctx, g_display_ctx, key);
335 }
336 }
337 return; // Skip normal frame rendering
338 }
339
340 // Use session display library for frame rendering
341 session_display_render_frame(g_display_ctx, frame_data);
342
343 // Poll for keyboard input and handle interactive controls
344 // - If client mode has local media (--file/--url), pass capture context for full controls
345 // (seek, pause, play, volume, color mode, flip)
346 // - If client mode is network-only, pass NULL (volume, color mode, flip work; seek/pause ignored)
347 if (g_keyboard_enabled) {
348 keyboard_key_t key = keyboard_read_nonblocking();
349 if (key != KEY_NONE) {
350 session_handle_keyboard_input(g_display_capture_ctx, g_display_ctx, key);
351 }
352 }
353}

363void display_cleanup() {
364 // Cleanup keyboard input if it was initialized
365 if (g_keyboard_enabled) {
366 keyboard_destroy();
367 g_keyboard_enabled = false;
368 }
369
370 // Cleanup optional local media capture context if it was created
371 if (g_display_capture_ctx) {
372 session_capture_destroy(g_display_capture_ctx);
373 g_display_capture_ctx = NULL;
374 }
375
376 // Display context is now managed by session_client_like framework
377 // Do not destroy it here - the framework will handle cleanup
378 g_display_ctx = NULL;
379}
