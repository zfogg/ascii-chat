39static bool is_official_server(const char *acds_server) {
40 if (!acds_server) {
41 return false;
42 }
43
44 // Case-insensitive comparison
45 char server_lower[BUFFER_SIZE_SMALL];
46 SAFE_STRNCPY(server_lower, acds_server, sizeof(server_lower));
47 for (char *p = server_lower; *p; p++) {
48 *p = (char)tolower((unsigned char)*p);
49 }
50
51 return strcmp(server_lower, "discovery.ascii-chat.com") == 0;
52}

188asciichat_error_t discovery_keys_save_cached(const char *acds_server, const uint8_t pubkey[32]) {
189 if (!acds_server || !pubkey) {
190 return SET_ERRNO(ERROR_INVALID_PARAM, "NULL parameter in discovery_keys_save_cached");
191 }
192
193 char cache_path[PLATFORM_MAX_PATH_LENGTH];
194 asciichat_error_t result = discovery_keys_get_cache_path(acds_server, cache_path, sizeof(cache_path));
195 if (result != ASCIICHAT_OK) {
196 return result;
197 }
198
199 // Create cache directory if it doesn't exist
200 char dir_path[PLATFORM_MAX_PATH_LENGTH];
201 safe_snprintf(dir_path, sizeof(dir_path), "%s", cache_path);
202
203 // Find the last path separator (handle both / and \ for cross-platform compatibility)
204 char *last_sep = strrchr(dir_path, '\\');
205 if (!last_sep) {
206 last_sep = strrchr(dir_path, '/');
207 }
208
209 if (last_sep) {
210 *last_sep = '\0';
211 if (!platform_is_directory(dir_path)) {
212 asciichat_error_t result = platform_mkdir(dir_path, 0700);
213 if (result != ASCIICHAT_OK) {
214 return SET_ERRNO(ERROR_FILE_OPERATION, "Failed to create ACDS key cache directory: %s", dir_path);
215 }
216 }
217 }
218
219 // Save key in OpenSSH public key format
220 FILE *f = platform_fopen("file_stream", cache_path, "w");
221 if (!f) {
222 return SET_ERRNO_SYS(ERROR_FILE_OPERATION, "Failed to create cache file: %s", cache_path);
223 }
224
225 // Convert binary Ed25519 key to base64 for OpenSSH format
226 char base64_key[128];
227 sodium_bin2base64(base64_key, sizeof(base64_key), pubkey, 32, sodium_base64_VARIANT_ORIGINAL);
228
229 fprintf(f, "ssh-ed25519 %s acds-cached-key\n", base64_key);
230 fclose(f);
231
232 log_debug("Cached ACDS key for server: %s", acds_server);
233 return ASCIICHAT_OK;
234}

303asciichat_error_t discovery_keys_verify(const char *acds_server, const char *key_spec, uint8_t pubkey_out[32]) {
304 if (!acds_server || !pubkey_out) {
305 return SET_ERRNO(ERROR_INVALID_PARAM, "NULL parameter in discovery_keys_verify");
306 }
307
308 uint8_t new_pubkey[32];
309 asciichat_error_t result;
310 bool is_official = is_official_server(acds_server);
311
312 // ===========================================================================
313 // Step 1: Obtain the public key
314 // ===========================================================================
315
316 if (!key_spec && is_official) {
317 // Automatic trust for official server: try SSH key first, then GPG
318 log_info("Attempting automatic HTTPS key trust for official ACDS server");
319
320 public_key_t key;
321 result = parse_public_key(ACDS_OFFICIAL_KEY_SSH_URL, &key);
322 if (result != ASCIICHAT_OK) {
323 log_debug("SSH key download failed, trying GPG key");
324 result = parse_public_key(ACDS_OFFICIAL_KEY_GPG_URL, &key);
325 }
326
327 if (result != ASCIICHAT_OK) {
328 return SET_ERRNO(ERROR_NETWORK, "Failed to download key from official ACDS server");
329 }
330
331 memcpy(new_pubkey, key.key, 32);
332
333 } else if (!key_spec) {
334 // No key_spec and not official server = error
335 return SET_ERRNO(ERROR_INVALID_PARAM,
336 "Third-party ACDS servers require explicit --discovery-service-key configuration."
337 "Only %s has automatic trust.",
338 ACDS_OFFICIAL_SERVER);
339
340 } else {
341 // Use parse_public_key for all other cases (handles HTTPS, files, github:, gitlab:, etc.)
342 public_key_t key;
343 result = parse_public_key(key_spec, &key);
344 if (result != ASCIICHAT_OK) {
345 return SET_ERRNO(ERROR_CRYPTO_KEY, "Failed to load/download ACDS key from: %s", key_spec);
346 }
347
348 memcpy(new_pubkey, key.key, 32);
349 }
350
351 // ===========================================================================
352 // Step 2: Check cache and handle key changes
353 // ===========================================================================
354
355 uint8_t cached_pubkey[32];
356 result = discovery_keys_load_cached(acds_server, cached_pubkey);
357
358 if (result == ASCIICHAT_OK) {
359 // Cached key exists - compare
360 if (memcmp(cached_pubkey, new_pubkey, 32) != 0) {
361 // Key changed - require user verification
362 result = discovery_keys_verify_change(acds_server, cached_pubkey, new_pubkey);
363 if (result != ASCIICHAT_OK) {
364 return result; // User rejected or error
365 }
366
367 // User accepted - update cache
368 result = discovery_keys_save_cached(acds_server, new_pubkey);
369 if (result != ASCIICHAT_OK) {
370 log_warn("Failed to update cached key, continuing anyway");
371 }
372 } else {
373 log_debug("ACDS key matches cached key for: %s", acds_server);
374 }
375
376 } else {
377 // No cached key - save it for next time
378 log_info("First connection to ACDS server: %s, caching key", acds_server);
379 result = discovery_keys_save_cached(acds_server, new_pubkey);
380 if (result != ASCIICHAT_OK) {
381 log_warn("Failed to cache key, continuing anyway");
382 }
383 }
384
385 // ===========================================================================
386 // Step 3: Return verified key
387 // ===========================================================================
388
389 memcpy(pubkey_out, new_pubkey, 32);
390 log_debug("ACDS key verification successful for: %s", acds_server);
391 return ASCIICHAT_OK;
392}
