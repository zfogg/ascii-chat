78static void webrtc_on_message(webrtc_data_channel_t *channel, const uint8_t *data, size_t len, void *user_data) {
79 (void)channel; // Unused
80 webrtc_transport_data_t *wrtc = (webrtc_transport_data_t *)user_data;
81
82 if (!wrtc || !data || len == 0) {
83 return;
84 }
85
86 // Allocate message buffer using buffer pool (will be freed by acip_client_receive_and_dispatch)
87 webrtc_recv_msg_t msg;
88 msg.data = buffer_pool_alloc(NULL, len);
89 if (!msg.data) {
90 return;
91 }
92
93 // Copy data
94 memcpy(msg.data, data, len);
95 msg.len = len;
96
97 // Push to receive queue
98 mutex_lock(&wrtc->queue_mutex);
99
100 bool success = ringbuffer_write(wrtc->recv_queue, &msg);
101 if (!success) {
102 // Queue full - drop oldest message to make room
103 webrtc_recv_msg_t dropped_msg;
104 if (ringbuffer_read(wrtc->recv_queue, &dropped_msg)) {
105 buffer_pool_free(NULL, dropped_msg.data, dropped_msg.len);
106 }
107
108 // Try again
109 success = ringbuffer_write(wrtc->recv_queue, &msg);
110 if (!success) {
111 buffer_pool_free(NULL, msg.data, len);
112 mutex_unlock(&wrtc->queue_mutex);
113 return;
114 }
115 }
116
117 // Signal waiting recv() call
118 cond_signal(&wrtc->queue_cond);
119 mutex_unlock(&wrtc->queue_mutex);
120}

186static asciichat_error_t webrtc_send(acip_transport_t *transport, const void *data, size_t len) {
187 webrtc_transport_data_t *wrtc = (webrtc_transport_data_t *)transport->impl_data;
188
189 mutex_lock(&wrtc->state_mutex);
190 bool connected = wrtc->is_connected;
191 mutex_unlock(&wrtc->state_mutex);
192
193 if (!connected) {
194 return SET_ERRNO(ERROR_NETWORK, "WebRTC transport not connected");
195 }
196
197 // Send via DataChannel
198 asciichat_error_t result = webrtc_datachannel_send(wrtc->data_channel, data, len);
199
200 if (result != ASCIICHAT_OK) {
201 return SET_ERRNO(ERROR_NETWORK, "Failed to send on WebRTC DataChannel");
202 }
203
204 return ASCIICHAT_OK;
205}

207static asciichat_error_t webrtc_recv(acip_transport_t *transport, void **buffer, size_t *out_len,
208 void **out_allocated_buffer) {
209 webrtc_transport_data_t *wrtc = (webrtc_transport_data_t *)transport->impl_data;
210
211 mutex_lock(&wrtc->queue_mutex);
212
213 // Block until message arrives or connection closes
214 while (ringbuffer_is_empty(wrtc->recv_queue)) {
215 mutex_lock(&wrtc->state_mutex);
216 bool connected = wrtc->is_connected;
217 mutex_unlock(&wrtc->state_mutex);
218
219 if (!connected) {
220 mutex_unlock(&wrtc->queue_mutex);
221 return SET_ERRNO(ERROR_NETWORK, "Connection closed while waiting for data");
222 }
223
224 // Wait for message arrival or connection close
225 cond_wait(&wrtc->queue_cond, &wrtc->queue_mutex);
226 }
227
228 // Read message from queue
229 webrtc_recv_msg_t msg;
230 bool success = ringbuffer_read(wrtc->recv_queue, &msg);
231 mutex_unlock(&wrtc->queue_mutex);
232
233 if (!success) {
234 return SET_ERRNO(ERROR_NETWORK, "Failed to read from receive queue");
235 }
236
237 // Return message to caller (caller owns the buffer)
238 *buffer = msg.data;
239 *out_len = msg.len;
240 *out_allocated_buffer = msg.data;
241
242 return ASCIICHAT_OK;
243}

245static asciichat_error_t webrtc_close(acip_transport_t *transport) {
246 webrtc_transport_data_t *wrtc = (webrtc_transport_data_t *)transport->impl_data;
247
248 mutex_lock(&wrtc->state_mutex);
249
250 if (!wrtc->is_connected) {
251 mutex_unlock(&wrtc->state_mutex);
252 return ASCIICHAT_OK; // Already closed
253 }
254
255 wrtc->is_connected = false;
256 mutex_unlock(&wrtc->state_mutex);
257
258 // Close DataChannel
259 if (wrtc->data_channel) {
260 webrtc_datachannel_close(wrtc->data_channel);
261 }
262
263 // Close peer connection
264 if (wrtc->peer_conn) {
265 webrtc_peer_connection_close(wrtc->peer_conn);
266 }
267
268 // Wake any blocking recv() calls
269 cond_broadcast(&wrtc->queue_cond);
270
271 log_debug("WebRTC transport closed");
272 return ASCIICHAT_OK;
273}

308static void webrtc_destroy_impl(acip_transport_t *transport) {
309 if (!transport || !transport->impl_data) {
310 return;
311 }
312
313 webrtc_transport_data_t *wrtc = (webrtc_transport_data_t *)transport->impl_data;
314
315 // Destroy peer connection and data channel
316 if (wrtc->data_channel) {
317 webrtc_datachannel_destroy(wrtc->data_channel);
318 wrtc->data_channel = NULL;
319 }
320
321 if (wrtc->peer_conn) {
322 webrtc_peer_connection_destroy(wrtc->peer_conn);
323 wrtc->peer_conn = NULL;
324 }
325
326 // Clear receive queue and free buffered messages
327 if (wrtc->recv_queue) {
328 mutex_lock(&wrtc->queue_mutex);
329
330 webrtc_recv_msg_t msg;
331 while (ringbuffer_read(wrtc->recv_queue, &msg)) {
332 if (msg.data) {
333 SAFE_FREE(msg.data);
334 }
335 }
336
337 mutex_unlock(&wrtc->queue_mutex);
338 ringbuffer_destroy(wrtc->recv_queue);
339 wrtc->recv_queue = NULL;
340 }
341
342 // Destroy synchronization primitives
343 mutex_destroy(&wrtc->state_mutex);
344 cond_destroy(&wrtc->queue_cond);
345 mutex_destroy(&wrtc->queue_mutex);
346
347 log_debug("Destroyed WebRTC transport resources");
348}

368acip_transport_t *acip_webrtc_transport_create(webrtc_peer_connection_t *peer_conn, webrtc_data_channel_t *data_channel,
369 crypto_context_t *crypto_ctx) {
370 if (!peer_conn || !data_channel) {
371 SET_ERRNO(ERROR_INVALID_PARAM, "peer_conn and data_channel are required");
372 return NULL;
373 }
374
375 // Allocate transport structure
376 acip_transport_t *transport = SAFE_MALLOC(sizeof(acip_transport_t), acip_transport_t *);
377 if (!transport) {
378 SET_ERRNO(ERROR_MEMORY, "Failed to allocate WebRTC transport");
379 return NULL;
380 }
381
382 // Allocate WebRTC-specific data
383 webrtc_transport_data_t *wrtc_data = SAFE_MALLOC(sizeof(webrtc_transport_data_t), webrtc_transport_data_t *);
384 if (!wrtc_data) {
385 SAFE_FREE(transport);
386 SET_ERRNO(ERROR_MEMORY, "Failed to allocate WebRTC transport data");
387 return NULL;
388 }
389
390 // Create receive queue
391 wrtc_data->recv_queue = ringbuffer_create(sizeof(webrtc_recv_msg_t), WEBRTC_RECV_QUEUE_SIZE);
392 if (!wrtc_data->recv_queue) {
393 SAFE_FREE(wrtc_data);
394 SAFE_FREE(transport);
395 SET_ERRNO(ERROR_MEMORY, "Failed to create receive queue");
396 return NULL;
397 }
398
399 // Initialize synchronization primitives
400 if (mutex_init(&wrtc_data->queue_mutex, "webrtc_queue") != 0) {
401 ringbuffer_destroy(wrtc_data->recv_queue);
402 SAFE_FREE(wrtc_data);
403 SAFE_FREE(transport);
404 SET_ERRNO(ERROR_INTERNAL, "Failed to initialize queue mutex");
405 return NULL;
406 }
407
408 if (cond_init(&wrtc_data->queue_cond, "queue") != 0) {
409 mutex_destroy(&wrtc_data->queue_mutex);
410 ringbuffer_destroy(wrtc_data->recv_queue);
411 SAFE_FREE(wrtc_data);
412 SAFE_FREE(transport);
413 SET_ERRNO(ERROR_INTERNAL, "Failed to initialize queue condition variable");
414 return NULL;
415 }
416
417 if (mutex_init(&wrtc_data->state_mutex, "webrtc_state") != 0) {
418 cond_destroy(&wrtc_data->queue_cond);
419 mutex_destroy(&wrtc_data->queue_mutex);
420 ringbuffer_destroy(wrtc_data->recv_queue);
421 SAFE_FREE(wrtc_data);
422 SAFE_FREE(transport);
423 SET_ERRNO(ERROR_INTERNAL, "Failed to initialize state mutex");
424 return NULL;
425 }
426
427 // Initialize WebRTC data
428 wrtc_data->peer_conn = peer_conn;
429 wrtc_data->data_channel = data_channel;
430 wrtc_data->is_connected = false; // Will be set to true in on_open callback
431
432 // Register DataChannel callbacks
433 webrtc_datachannel_callbacks_t callbacks = {
434 .on_open = webrtc_on_open,
435 .on_close = webrtc_on_close,
436 .on_error = webrtc_on_error,
437 .on_message = webrtc_on_message,
438 .user_data = wrtc_data,
439 };
440
441 asciichat_error_t result = webrtc_datachannel_set_callbacks(data_channel, &callbacks);
442 if (result != ASCIICHAT_OK) {
443 mutex_destroy(&wrtc_data->state_mutex);
444 cond_destroy(&wrtc_data->queue_cond);
445 mutex_destroy(&wrtc_data->queue_mutex);
446 ringbuffer_destroy(wrtc_data->recv_queue);
447 SAFE_FREE(wrtc_data);
448 SAFE_FREE(transport);
449 SET_ERRNO(ERROR_INTERNAL, "Failed to set DataChannel callbacks");
450 return NULL;
451 }
452
453 // IMPORTANT: The transport is always created from peer_manager's on_datachannel_open callback,
454 // which means the DataChannel is ALREADY OPEN when we get here. However, by setting our own
455 // callbacks above (webrtc_datachannel_set_callbacks), we replaced the callbacks that would
456 // have set dc->is_open=true. So we need to manually mark both the transport AND the DataChannel
457 // as open/connected now.
458 //
459 // We cannot rely on webrtc_on_open being called later because:
460 // 1. The DataChannel is already open
461 // 2. libdatachannel won't fire the open event again
462 // 3. Setting callbacks after open doesn't trigger a retroactive open event
463
464 // Mark DataChannel as open (needed for webrtc_datachannel_send() check)
465 webrtc_datachannel_set_open_state(data_channel, true);
466
467 // Mark transport as connected
468 mutex_lock(&wrtc_data->state_mutex);
469 wrtc_data->is_connected = true;
470 mutex_unlock(&wrtc_data->state_mutex);
471 log_debug("Transport and DataChannel marked as connected/open (already open from peer_manager callback)");
472
473 // Initialize transport
474 transport->methods = &webrtc_methods;
475 transport->crypto_ctx = crypto_ctx;
476 transport->impl_data = wrtc_data;
477
478 log_info("Created WebRTC transport (crypto: %s)", crypto_ctx ? "enabled" : "disabled");
479
480 // Register WebRTC transport implementation data with transport as parent
481 NAMED_REGISTER(wrtc_data, "impl", "webrtc_impl", "0x%tx", (uintptr_t)(const void *)(transport));
482
483 return transport;
