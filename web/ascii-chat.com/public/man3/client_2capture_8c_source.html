189static void *webcam_capture_thread_func(void *arg) {
190 (void)arg;
191
192 // FPS tracking for webcam capture thread
193 static fps_t fps_tracker = {0};
194 static bool fps_tracker_initialized = false;
195 static uint64_t capture_frame_count = 0;
196 static uint64_t last_capture_frame_time_ns = 0;
197 static image_t *last_frame = NULL; // Cache last frame to render when paused
198 if (!fps_tracker_initialized) {
199 fps_init(&fps_tracker, CAPTURE_TARGET_FPS, "WEBCAM_TX");
200 fps_tracker_initialized = true;
201 }
202
203 while (!should_exit() && !server_connection_is_lost()) {
204 // Check connection status
205 if (!server_connection_is_active()) {
206 log_debug_every(LOG_RATE_NORMAL, "Capture thread: waiting for connection to become active");
207 platform_sleep_us(100 * US_PER_MS_INT); // Wait for connection
208 continue;
209 }
210
211 // Frame rate limiting using session capture adaptive sleep
212 session_capture_sleep_for_fps(g_capture_capture_ctx);
213
214 // Re-check connection after FPS sleep in case it was lost during sleep (reconnection)
215 if (!server_connection_is_active() || server_connection_is_lost()) {
216 log_debug("Capture thread: connection lost after FPS sleep, exiting");
217 break;
218 }
219
220 // Read frame using session capture library
221 image_t *image = session_capture_read_frame(g_capture_capture_ctx);
222
223 // Check if media is paused and we have a last frame - render last frame to keep keyboard polling active
224 if (!image) {
225 media_source_t *source = (media_source_t *)session_capture_get_media_source(g_capture_capture_ctx);
226 if (source && media_source_get_type(source) == MEDIA_SOURCE_FILE && media_source_is_paused(source) &&
227 last_frame) {
228 // Use last frame when paused (keeps display thread rendering and keyboard polling active)
229 image = last_frame;
230 log_debug_every(LOG_RATE_SLOW, "Using cached frame while paused");
231 } else if (session_capture_at_end(g_capture_capture_ctx)) {
232 // Check if we've reached end of file for media sources
233 log_debug("Media source reached end of file");
234 image_destroy(last_frame);
235 last_frame = NULL;
236 break; // Exit capture loop - end of media
237 } else {
238 log_debug_every(LOG_RATE_SLOW, "No frame available from media source yet (returned NULL)");
239 platform_sleep_us(10 * US_PER_MS_INT); // 10ms delay before retry
240 continue;
241 }
242 }
243
244 // Track frame for FPS reporting
245 fps_frame_ns(&fps_tracker, time_get_ns(), "webcam frame captured");
246
247 // Process frame for network transmission using session library
248 // session_capture_process_for_transmission() returns a new image that we own
249 // NOTE: The original 'image' is owned by media_source - do NOT free it!
250 image_t *processed_image = session_capture_process_for_transmission(g_capture_capture_ctx, image);
251 if (!processed_image) {
252 SET_ERRNO(ERROR_INVALID_STATE, "Failed to process frame for transmission");
253 // NOTE: Do NOT free 'image' - it's owned by capture context
254 continue;
255 }
256
257 // Check connection before sending
258 if (!server_connection_is_active()) {
259 log_warn("Connection lost before sending, stopping video transmission");
260 image_destroy(processed_image);
261 break;
262 }
263
264 // Send frame packet to server using proper packet format
265 acip_transport_t *transport = server_connection_get_transport();
266 if (!transport || !server_connection_is_active()) {
267 log_warn("Transport became unavailable during capture, stopping transmission");
268 image_destroy(processed_image);
269 break;
270 }
271
272 uint64_t send_start_ns = time_get_ns();
273
274 // Determine which codec to use based on --video-codec option
275 const char *video_codec = GET_OPTION(video_codec);
276 bool use_hevc = video_codec && strcmp(video_codec, "raw") != 0; // Default to HEVC unless explicitly "raw"
277
278 // Request keyframe periodically to force encoder flush (every 30 frames)
279 // This ensures we get some encoded output even when the encoder is buffering inter-frames
280 static uint32_t keyframe_interval = 30;
281 static uint32_t frames_since_keyframe = 0;
282 if (use_hevc && (frames_since_keyframe++ >= keyframe_interval)) {
283 h265_encoder_request_keyframe(g_h265_encoder);
284 frames_since_keyframe = 0;
285 log_debug("Requesting H.265 keyframe for encoder flush");
286 }
287
288 asciichat_error_t send_result;
289 if (use_hevc) {
290 log_debug_every(LOG_RATE_SLOW, "Capture thread: sending IMAGE_FRAME_H265 %ux%u", processed_image->w,
291 processed_image->h);
292 send_result =
293 threaded_send_image_frame_h265((const void *)processed_image->pixels, (uint32_t)processed_image->w,
294 (uint32_t)processed_image->h);
295 } else {
296 log_debug_every(LOG_RATE_SLOW, "Capture thread: sending IMAGE_FRAME (raw) %ux%u", processed_image->w,
297 processed_image->h);
298 send_result = threaded_send_image_frame((const void *)processed_image->pixels, (uint32_t)processed_image->w,
299 (uint32_t)processed_image->h, 1); // pixel_format = 1 (RGB24)
300 }
301
302 // If send failed due to connection loss, break out of loop
303 if (send_result != ASCIICHAT_OK && !server_connection_is_active()) {
304 log_debug("Connection lost during send, stopping transmission");
305 image_destroy(processed_image);
306 break;
307 }
308 uint64_t send_duration_ns = time_elapsed_ns(send_start_ns, time_get_ns());
309
310 if (send_result != ASCIICHAT_OK) {
311 const char *codec_name = use_hevc ? "H265" : "RAW";
312 log_error("🔴 CAPTURE_SEND_FAILED: IMAGE_FRAME_%s send error=%d (%s) after %.1fms, closing connection",
313 codec_name, send_result, asciichat_error_string(send_result), (double)send_duration_ns / 1e6);
314 server_connection_lost();
315 image_destroy(processed_image);
316 break;
317 }
318
319 if (send_duration_ns > 500 * NS_PER_MS_INT) {
320 const char *codec_name = use_hevc ? "H.265" : "RAW";
321 log_warn("⚠️ SLOW_FRAME_SEND: %.1fms to send %ux%u %s frame (may indicate full send buffer)",
322 (double)send_duration_ns / 1e6, processed_image->w, processed_image->h, codec_name);
323 }
324
325 const char *codec_name = use_hevc ? "H265" : "RAW";
326 log_info("✅ CAPTURE_FRAME_SENT: IMAGE_FRAME_%s delivered to server in %.1fms", codec_name,
327 (double)send_duration_ns / 1e6);
328
329 // Cache last frame for rendering when paused
330 // Make a copy since the original is owned by media_source
331 if (last_frame) {
332 image_destroy(last_frame);
333 }
334 last_frame = image_new(processed_image->w, processed_image->h);
335 if (last_frame) {
336 memcpy(last_frame->pixels, processed_image->pixels,
337 (size_t)processed_image->w * (size_t)processed_image->h * sizeof(rgb_pixel_t));
338 }
339
340 // FPS tracking - frame successfully captured and sent
341 capture_frame_count++;
342
343 // Calculate time since last frame for lag detection (using nanosecond precision internally)
344 uint64_t frame_capture_time_ns = time_get_ns();
345
346 uint64_t frame_interval_ns = time_elapsed_ns(last_capture_frame_time_ns, frame_capture_time_ns);
347 last_capture_frame_time_ns = frame_capture_time_ns;
348
349 // Expected frame interval in nanoseconds
350 uint64_t expected_interval_ns = NS_PER_SEC_INT / (uint64_t)session_capture_get_target_fps(g_capture_capture_ctx);
351 uint64_t lag_threshold_ns = expected_interval_ns + (expected_interval_ns / 2); // 50% over expected
352
353 // Log warning if frame took too long to capture (display in milliseconds for readability)
354 if (capture_frame_count > 1 && frame_interval_ns > lag_threshold_ns) {
355 double late_ms = (double)(frame_interval_ns - expected_interval_ns) / 1e6;
356 double expected_ms = (double)expected_interval_ns / 1e6;
357 double actual_ms = (double)frame_interval_ns / 1e6;
358 double actual_fps = 1e9 / (double)frame_interval_ns;
359 log_warn_every(LOG_RATE_FAST,
360 "CLIENT CAPTURE LAG: Frame captured %.1fms late (expected %.1fms, got %.1fms, actual fps: %.1f)",
361 late_ms, expected_ms, actual_ms, actual_fps);
362 }
363
364 // Clean up processed frame
365 image_destroy(processed_image);
366 processed_image = NULL;
367
368 // Yield to reduce CPU usage
369 platform_sleep_us(1 * US_PER_MS_INT); // 1ms
370 }
371
372#ifdef DEBUG_THREADS
373 log_debug("Webcam capture thread stopped");
374#endif
375
376 // Clean up cached frame before thread exit
377 if (last_frame) {
378 image_destroy(last_frame);
379 last_frame = NULL;
380 }
381
382 log_debug("CAPTURE_THREAD_EXIT: About to mark thread as exited");
383 atomic_store_bool(&g_capture_thread_exited, true);
384 log_debug("CAPTURE_THREAD_EXIT: Thread marked as exited, cleaning up errno");
385
386 // Clean up thread-local error context before exit
387 asciichat_errno_destroy();
388
389 log_debug("CAPTURE_THREAD_EXIT: Exiting capture thread");
390 return NULL;
391}

405int capture_init() {
406 // Build capture configuration from options
407 session_capture_config_t config = {0};
408 const char *media_url = GET_OPTION(media_url);
409 const char *media_file = GET_OPTION(media_file);
410 bool media_from_stdin = GET_OPTION(media_from_stdin);
411
412 if (media_url && media_url[0] != '\0') {
413 // Network URL streaming (takes priority over --file)
414 // Don't open webcam when streaming from URL
415 config.type = MEDIA_SOURCE_FILE;
416 config.path = media_url;
417 config.loop = false; // Network URLs cannot be looped
418 log_debug("Using network URL: %s (webcam disabled)", media_url);
419 } else if (media_file && media_file[0] != '\0') {
420 // File or stdin streaming - don't open webcam
421 config.type = media_from_stdin ? MEDIA_SOURCE_STDIN : MEDIA_SOURCE_FILE;
422 config.path = media_file;
423 config.loop = GET_OPTION(media_loop) && !media_from_stdin;
424 log_debug("Using media %s: %s (webcam disabled)", media_from_stdin ? "stdin" : "file", media_file);
425 } else if (GET_OPTION(test_pattern)) {
426 // Test pattern mode - don't open real webcam
427 config.type = MEDIA_SOURCE_TEST;
428 config.path = NULL;
429 log_debug("Using test pattern mode");
430 } else {
431 // Webcam mode (default)
432 static char webcam_index_str[32];
433 safe_snprintf(webcam_index_str, sizeof(webcam_index_str), "%u", GET_OPTION(webcam_index));
434 config.type = MEDIA_SOURCE_WEBCAM;
435 config.path = webcam_index_str;
436 log_debug("Using webcam device %u", GET_OPTION(webcam_index));
437 }
438 config.target_fps = CAPTURE_TARGET_FPS;
439 config.resize_for_network = true; // Client always resizes for network transmission
440
441 // Configure audio capture with fallback to microphone
442 config.enable_audio = true;
443 config.audio_fallback_to_mic = true;
444 config.mic_audio_ctx = audio_get_context();
445
446 // Add seek timestamp if specified
447 config.initial_seek_timestamp = GET_OPTION(media_seek_timestamp);
448
449 // Create capture context using session library
450 g_capture_capture_ctx = session_capture_create(&config);
451 if (!g_capture_capture_ctx) {
452 // Check if there's already an error set (e.g., ERROR_WEBCAM_IN_USE)
453 asciichat_error_t existing_error = GET_ERRNO();
454 log_debug("session_capture_create failed, GET_ERRNO() returned: %d", existing_error);
455 if (existing_error != ASCIICHAT_OK) {
456 log_debug("Returning existing error code %d", existing_error);
457 return existing_error;
458 }
459 SET_ERRNO(ERROR_MEDIA_INIT, "Failed to initialize capture source");
460 return -1;
461 }
462
463 return 0;
464}

475int capture_start_thread() {
476 if (THREAD_IS_CREATED(g_capture_thread_created)) {
477 log_warn("Capture thread already created");
478 return 0;
479 }
480
481 // Register capture atomics with named debug registry
482 static bool capture_atomics_registered = false;
483 if (!capture_atomics_registered) {
484 NAMED_REGISTER_ATOMIC(&g_capture_thread_exited, "capture_webcam_thread_exit_confirmation", NULL);
485 capture_atomics_registered = true;
486 }
487
488 // Start webcam capture thread
489 atomic_store_bool(&g_capture_thread_exited, false);
490 if (thread_pool_spawn(g_client_worker_pool, webcam_capture_thread_func, NULL, 2, "webcam_capture") != ASCIICHAT_OK) {
491 SET_ERRNO(ERROR_THREAD, "Webcam capture thread creation failed");
492 LOG_ERRNO_IF_SET("Webcam capture thread creation failed");
493 return -1;
494 }
495
496 g_capture_thread_created = true;
497 log_debug("Webcam capture thread created successfully");
498
499 return 0;

509void capture_stop_thread() {
510 if (!THREAD_IS_CREATED(g_capture_thread_created)) {
511 return;
512 }
513
514 // Wait for thread to exit gracefully
515 int wait_count = 0;
516 while (wait_count < 20 && !atomic_load_bool(&g_capture_thread_exited)) {
517 platform_sleep_us(100 * US_PER_MS_INT); // 100ms
518 wait_count++;
519 }
520
521 if (!atomic_load_bool(&g_capture_thread_exited)) {
522 log_warn("Capture thread not responding after 2 seconds - will be joined by thread pool");
523 }
524
525 // Thread will be joined by thread_pool_stop_all() in protocol_stop_connection()
526 g_capture_thread_created = false;
527}
