30static bool is_marker_line(const char *line, const char **type_out, char **section_out) {
31 if (!line) {
32 return false;
33 }
34
35 // Skip leading whitespace
36 const char *p = line;
37 while (isspace((unsigned char)*p)) {
38 p++;
39 }
40
41 // Check for groff comment: .\ " or ."
42 if (strncmp(p, ".\\\"", 3) != 0 && strncmp(p, ".\"", 2) != 0) {
43 return false;
44 }
45
46 // Skip comment marker
47 if (p[0] == '.' && p[1] == '\\' && p[2] == '"') {
48 p += 3;
49 } else if (p[0] == '.' && p[1] == '"') {
50 p += 2;
51 }
52
53 // Skip whitespace after comment marker
54 while (isspace((unsigned char)*p)) {
55 p++;
56 }
57
58 // Check for marker pattern: TYPE-START: SECTION or TYPE-END: SECTION
59 const char *type = NULL;
60 const char *section_start = NULL;
61
62 if (strncmp(p, "AUTO-START:", 11) == 0) {
63 type = "AUTO";
64 section_start = p + 11;
65 } else if (strncmp(p, "AUTO-END:", 9) == 0) {
66 type = "AUTO";
67 section_start = p + 9;
68 } else if (strncmp(p, "MANUAL-START:", 13) == 0) {
69 type = "MANUAL";
70 section_start = p + 13;
71 } else if (strncmp(p, "MANUAL-END:", 11) == 0) {
72 type = "MANUAL";
73 section_start = p + 11;
74 } else if (strncmp(p, "MERGE-START:", 12) == 0) {
75 type = "MERGE";
76 section_start = p + 12;
77 } else if (strncmp(p, "MERGE-END:", 10) == 0) {
78 type = "MERGE";
79 section_start = p + 10;
80 } else {
81 return false;
82 }
83
84 // Skip whitespace before section name
85 while (isspace((unsigned char)*section_start)) {
86 section_start++;
87 }
88
89 // Extract section name (until end of line)
90 const char *section_end = section_start;
91 while (*section_end && *section_end != '\n' && *section_end != '\r') {
92 section_end++;
93 }
94
95 if (section_end > section_start) {
96 if (type_out) {
97 *type_out = type;
98 }
99 if (section_out) {
100 size_t len = section_end - section_start;
101 *section_out = SAFE_MALLOC(len + 1, char *);
102 memcpy(*section_out, section_start, len);
103 (*section_out)[len] = '\0';
104 // Trim trailing whitespace
105 char *s = *section_out + len - 1;
106 while (s >= *section_out && isspace((unsigned char)*s)) {
107 *s-- = '\0';
108 }
109 }
110 return true;
111 }
112
113 return false;
114}

119static bool is_section_header(const char *line, char **section_name_out) {
120 if (!line) {
121 return false;
122 }
123
124 // Skip leading whitespace
125 const char *p = line;
126 while (isspace((unsigned char)*p)) {
127 p++;
128 }
129
130 // Check for .SH directive
131 if (strncmp(p, ".SH", 3) != 0) {
132 return false;
133 }
134
135 p += 3;
136
137 // Skip whitespace
138 while (isspace((unsigned char)*p)) {
139 p++;
140 }
141
142 // Extract section name (until end of line)
143 const char *name_start = p;
144 const char *name_end = name_start;
145 while (*name_end && *name_end != '\n' && *name_end != '\r') {
146 name_end++;
147 }
148
149 if (name_end > name_start) {
150 size_t len = name_end - name_start;
151 *section_name_out = SAFE_MALLOC(len + 1, char *);
152 memcpy(*section_name_out, name_start, len);
153 (*section_name_out)[len] = '\0';
154 // Trim trailing whitespace
155 char *s = *section_name_out + len - 1;
156 while (s >= *section_name_out && isspace((unsigned char)*s)) {
157 *s-- = '\0';
158 }
159 // Remove quotes if present
160 len = strlen(*section_name_out);
161 if (len >= 2 && (*section_name_out)[0] == '"' && (*section_name_out)[len - 1] == '"') {
162 memmove(*section_name_out, *section_name_out + 1, len - 2);
163 (*section_name_out)[len - 2] = '\0';
164 }
165 return true;
166 }
167
168 return false;
169}

193static asciichat_error_t parse_sections_internal(FILE *f, parsed_section_t **out_sections, size_t *out_count) {
194 if (!f || !out_sections || !out_count) {
195 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters for parse_sections_internal");
196 }
197
198 parsed_section_t *sections = NULL;
199 size_t capacity = 16;
200 size_t count = 0;
201 sections = SAFE_MALLOC(capacity * sizeof(parsed_section_t), parsed_section_t *);
202
203 char *line = NULL;
204 size_t line_len = 0;
205 size_t line_num = 0;
206
207 parsed_section_t *current_section = NULL;
208 char *current_content = NULL;
209 size_t current_content_capacity = 4096;
210 size_t current_content_len = 0;
211 current_content = SAFE_MALLOC(current_content_capacity, char *);
212
213 // Track markers for current section
214 const char *current_type = NULL;
215 char *current_marker_section = NULL;
216 bool in_marked_section = false;
217
218 while (platform_getline(&line, &line_len, f) != -1) {
219 line_num++;
220
221 const char *marker_type = NULL;
222 char *marker_section = NULL;
223 bool is_marker = is_marker_line(line, &marker_type, &marker_section);
224
225 char *section_header_name = NULL;
226 bool is_header = is_section_header(line, &section_header_name);
227
228 if (is_marker) {
229 // Check if this is a START marker
230 if (strstr(line, "-START:") != NULL) {
231 // Start of a marked section
232 current_type = marker_type;
233 if (current_marker_section) {
234 SAFE_FREE(current_marker_section);
235 }
236 current_marker_section = marker_section;
237 in_marked_section = true;
238 marker_section = NULL; // Ownership transferred
239 } else if (strstr(line, "-END:") != NULL) {
240 // End of marked section
241 if (marker_section) {
242 SAFE_FREE(marker_section);
243 }
244 in_marked_section = false;
245 current_type = NULL;
246 if (current_marker_section) {
247 SAFE_FREE(current_marker_section);
248 current_marker_section = NULL;
249 }
250 } else {
251 if (marker_section) {
252 SAFE_FREE(marker_section);
253 }
254 }
255 }
256
257 if (is_header) {
258 // Finalize previous section if exists
259 if (current_section) {
260 if (current_content && current_content_len > 0) {
261 current_section->content = current_content;
262 current_section->content_len = current_content_len;
263 current_content = NULL;
264 current_content_capacity = 4096;
265 current_content_len = 0;
266 current_content = SAFE_MALLOC(current_content_capacity, char *);
267 } else {
268 SAFE_FREE(current_content);
269 current_content = NULL;
270 current_content_capacity = 4096;
271 current_content_len = 0;
272 current_content = SAFE_MALLOC(current_content_capacity, char *);
273 }
274 }
275
276 // Start new section
277 if (count >= capacity) {
278 capacity *= 2;
279 sections = SAFE_REALLOC(sections, capacity * sizeof(parsed_section_t), parsed_section_t *);
280 }
281
282 current_section = &sections[count++];
283 memset(current_section, 0, sizeof(parsed_section_t));
284 current_section->section_name = section_header_name;
285 current_section->start_line = line_num;
286 current_section->end_line = line_num;
287 current_section->type = in_marked_section ? type_string_to_enum(current_type) : SECTION_TYPE_UNMARKED;
288 current_section->has_markers = in_marked_section;
289
290 // Append header line to content
291 size_t line_strlen = strlen(line);
292 if (current_content_len + line_strlen + 1 >= current_content_capacity) {
293 current_content_capacity = (current_content_len + line_strlen + 1) * 2;
294 current_content = SAFE_REALLOC(current_content, current_content_capacity, char *);
295 }
296 memcpy(current_content + current_content_len, line, line_strlen);
297 current_content_len += line_strlen;
298 current_content[current_content_len] = '\0';
299 } else if (current_section) {
300 // Append line to current section content
301 size_t line_strlen = strlen(line);
302 if (current_content_len + line_strlen + 1 >= current_content_capacity) {
303 current_content_capacity = (current_content_len + line_strlen + 1) * 2;
304 current_content = SAFE_REALLOC(current_content, current_content_capacity, char *);
305 }
306 memcpy(current_content + current_content_len, line, line_strlen);
307 current_content_len += line_strlen;
308 current_content[current_content_len] = '\0';
309 current_section->end_line = line_num;
310 }
311 }
312
313 // Finalize last section
314 if (current_section && current_content && current_content_len > 0) {
315 current_section->content = current_content;
316 current_section->content_len = current_content_len;
317 } else if (current_content) {
318 SAFE_FREE(current_content);
319 }
320
321 if (current_marker_section) {
322 SAFE_FREE(current_marker_section);
323 }
324 if (line) {
325 free(line);
326 }
327
328 *out_sections = sections;
329 *out_count = count;
330 log_debug("Parsed %zu sections from file", count);
331 return ASCIICHAT_OK;
332}

346asciichat_error_t manpage_parser_parse_memory(const char *content, size_t content_len, parsed_section_t **out_sections,
347 size_t *out_count) {
348 if (!content || content_len == 0 || !out_sections || !out_count) {
349 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters for manpage_parser_parse_memory");
350 }
351
352 // Create temporary file from memory
353 FILE *tmp = platform_tmpfile();
354 if (!tmp) {
355 return SET_ERRNO_SYS(ERROR_CONFIG, "Failed to create temporary file for memory parsing");
356 }
357
358 // Write content to temporary file
359 // Write content_len + 1 to include the null terminator.
360 // The embedded string is content_len + 1 bytes (content_len chars + null terminator).
361 // platform_getline() needs the null terminator to properly handle EOF.
362 size_t bytes_to_write = content_len + 1;
363 size_t written = fwrite(content, 1, bytes_to_write, tmp);
364 if (written != bytes_to_write) {
365 fclose(tmp);
366 return SET_ERRNO_SYS(ERROR_CONFIG, "Failed to write complete content to temporary file");
367 }
368
369 // Rewind to beginning for reading
370 rewind(tmp);
371
372 // Parse using common function
373 asciichat_error_t err = parse_sections_internal(tmp, out_sections, out_count);
374 fclose(tmp);
375
376 return err;
377}
