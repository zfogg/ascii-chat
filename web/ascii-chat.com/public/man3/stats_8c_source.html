343#ifndef NDEBUG

372 int clients_with_audio = 0;

379 active_clients++;
380 if (g_client_manager.clients[i].audio_queue) {

382 }

384 clients_with_video++;

405 if (len > 0 && client_details_len + len < (int)sizeof(client_details)) {
406 client_details_len += len;
407 }
408 }
409 }
410 // Check video buffer stats
411 if (client->outgoing_video_buffer) {
412 video_frame_stats_t stats;
413 video_frame_get_stats(client->outgoing_video_buffer, &stats);
414 if (stats.total_frames > 0) {

473void update_server_stats(void) {
474 // Aggregate statistics from all active clients
475 uint64_t total_frames_sent = 0;
476 uint64_t total_frames_dropped = 0;
477
478 // Read-lock to safely iterate over all clients
479 rwlock_rdlock(&g_client_manager_rwlock);
480
481 for (int i = 0; i < MAX_CLIENTS; i++) {
482 client_info_t *client = &g_client_manager.clients[i];
483 if (client->client_id[0] != '\0' && atomic_load_bool(&client->active)) {
484 // Aggregate frames sent to all clients
485 total_frames_sent += client->frames_sent;
486
487 // Get dropped frame statistics from outgoing video buffer
488 if (client->outgoing_video_buffer) {
489 video_frame_stats_t stats;
490 video_frame_get_stats(client->outgoing_video_buffer, &stats);
491 total_frames_dropped += stats.dropped_frames;
492 }
493 }
494 }
495
496 rwlock_rdunlock(&g_client_manager_rwlock);
497
498 // Update global statistics atomically
499 mutex_lock(&g_stats_mutex);
500 g_stats.frames_sent = total_frames_sent;
501 g_stats.frames_dropped = total_frames_dropped;
502
503 // Calculate frames_captured from frames_sent and frames_dropped
504 // frames_captured = frames_sent + frames_dropped (total output frames)
505 if (total_frames_sent > 0 || total_frames_dropped > 0) {
506 g_stats.frames_captured = total_frames_sent + total_frames_dropped;
507 }
508
509 mutex_unlock(&g_stats_mutex);
510}

554void log_server_stats(void) {
555 // Check if stats mutex is initialized (safe in debug builds where stats_init() may not be called)
556 if (!lifecycle_is_initialized(&g_stats_lc)) {
557 return; // Mutex not initialized, skip logging
558 }
559
560 // Now safe to use the mutex
561 mutex_lock(&g_stats_mutex);
562 log_info("Server Statistics:\n"
563 " frames_captured=%llu\n"
564 " frames_sent=%llu\n"
565 " frames_dropped=%llu\n"
566 " Average FPS: capture=%.2f, send=%.2f",
567 (unsigned long long)g_stats.frames_captured, (unsigned long long)g_stats.frames_sent,
568 (unsigned long long)g_stats.frames_dropped, g_stats.avg_capture_fps, g_stats.avg_send_fps);
569 mutex_unlock(&g_stats_mutex);
570}
