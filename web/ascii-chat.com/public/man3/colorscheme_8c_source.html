257static const color_scheme_t *find_builtin_scheme(const char *name) {
258 if (!name)
259 return NULL;
260
261 /* Handle "default" alias */
262 if (strcmp(name, "default") == 0) {
263 name = "pastel";
264 }
265
266 for (size_t i = 0; i < NUM_BUILTIN_SCHEMES; i++) {
267 if (strcmp(BUILTIN_SCHEMES[i]->name, name) == 0) {

269 }
270 }
271

279asciichat_error_t colorscheme_init(void) {
280 if (lifecycle_is_initialized(&g_colorscheme_lifecycle)) {
281 return ASCIICHAT_OK;
282 }
283
284 /* NOTE: Mutex is already statically initialized on POSIX with PTHREAD_MUTEX_INITIALIZER.
285 * On Windows and Emscripten/WASM, we initialize it here. Do NOT call mutex_init() on native POSIX
286 * because double-initialization of pthread_mutex_t causes undefined behavior and deadlocks.
287 * Emscripten with pthreads requires explicit initialization because PTHREAD_MUTEX_INITIALIZER
288 * doesn't work correctly in threaded WASM builds. */
289#if defined(_WIN32) || defined(__EMSCRIPTEN__)
290 static bool mutex_initialized = false;
291 if (!mutex_initialized) {
292#ifdef __EMSCRIPTEN__
293#else
294#endif
295 mutex_init(&g_colorscheme_mutex, "colorscheme");
296 mutex_initialized = true;
297 } else {
298 }
299#else
300#endif
301
302 /* Load default scheme */
303 const color_scheme_t *pastel = find_builtin_scheme("pastel");
304 if (!pastel) {
305 return SET_ERRNO(ERROR_CONFIG, "Failed to load default pastel scheme");
306 }
307
308 memcpy(&g_active_scheme, pastel, sizeof(color_scheme_t));
309
310 /* Mark as initialized */
311 lifecycle_init(&g_colorscheme_lifecycle, "colorscheme");
312
313 return ASCIICHAT_OK;
314}

316void colorscheme_cleanup_compiled(compiled_color_scheme_t *compiled) {

320
321 /* Free allocated color code strings */

324 char *str_256 = (char *)compiled->codes_256[i];
325 char *str_truecolor = (char *)compiled->codes_truecolor[i];
326 /* Only free non-NULL pointers - protects against uninitialized data */
327 if (str_16 != NULL) {
328 SAFE_FREE(str_16);
329 }
330 if (str_256 != NULL) {
331 SAFE_FREE(str_256);
332 }
333 if (str_truecolor != NULL) {
334 SAFE_FREE(str_truecolor);
335 }

340void colorscheme_destroy(void) {
341 if (!lifecycle_is_initialized(&g_colorscheme_lifecycle)) {
342 return;
343 }
344
345 mutex_lock(&g_colorscheme_mutex);
346 memset(&g_active_scheme, 0, sizeof(color_scheme_t));
347
348 /* Mark as uninitialized */
349 lifecycle_shutdown(&g_colorscheme_lifecycle);
350
351 mutex_unlock(&g_colorscheme_mutex);
352
353 /* NOTE: Do NOT call mutex_destroy() on native POSIX because the mutex is statically
354 * initialized with PTHREAD_MUTEX_INITIALIZER. Destroying a statically-initialized
355 * mutex is undefined behavior. On Windows and Emscripten, we must destroy the mutex
356 * because it was explicitly initialized in colorscheme_init(). */
357#if defined(_WIN32) || defined(__EMSCRIPTEN__)
358 mutex_destroy(&g_colorscheme_mutex);
359#endif

375asciichat_error_t colorscheme_set_active_scheme(const char *name) {
376 if (!name) {
377 return SET_ERRNO(ERROR_INVALID_PARAM, "Scheme name is NULL");
378 }
379
380 /* Ensure color system is initialized */
381 if (!lifecycle_is_initialized(&g_colorscheme_lifecycle)) {
382 colorscheme_init();
383 }
384
385 color_scheme_t scheme = {0};
386 asciichat_error_t result = ASCIICHAT_OK;
387
388 /* Try loading as built-in scheme first */
389 const color_scheme_t *builtin = find_builtin_scheme(name);
390 if (builtin) {
391 memcpy(&scheme, builtin, sizeof(color_scheme_t));
392 } else if (strchr(name, '/') || strchr(name, '.')) {
393 /* Try loading as file path if it contains / or . */
394 result = colorscheme_load_from_file(name, &scheme);
395 if (result != ASCIICHAT_OK) {
396 return result;
397 }
398 } else {
399 return SET_ERRNO(ERROR_CONFIG, "Unknown color scheme: %s (not a built-in scheme or valid file path)", name);
400 }
401
402 mutex_lock(&g_colorscheme_mutex);
403 memcpy(&g_active_scheme, &scheme, sizeof(color_scheme_t));
404 mutex_unlock(&g_colorscheme_mutex);
405
406 log_debug("Switched to color scheme: %s", name);
407 return ASCIICHAT_OK;
408}

424asciichat_error_t colorscheme_load_from_file(const char *path, color_scheme_t *scheme) {
425 if (!path || !scheme) {
426 return SET_ERRNO(ERROR_INVALID_PARAM, "NULL path or scheme pointer");
427 }
428
429 /* Check if file exists and is readable */
430 struct stat sb;
431 if (stat(path, &sb) != 0 || !S_ISREG(sb.st_mode)) {
432 return SET_ERRNO(ERROR_FILE_NOT_FOUND, "Color scheme file not found or not readable: %s", path);
433 }
434
435 /* Parse TOML file */
436 toml_result_t result = toml_parse_file_ex(path);
437 if (!result.ok) {
438 toml_free(result);
439 return SET_ERRNO(ERROR_CONFIG, "Failed to parse color scheme file '%s': %s", path, result.errmsg);
440 }
441
442 /* Extract scheme information */
443 memset(scheme, 0, sizeof(color_scheme_t));
444
445 /* Get scheme section */
446 toml_datum_t scheme_section = toml_get(result.toptab, "scheme");
447 if (scheme_section.type == TOML_TABLE) {
448 /* Get scheme name */
449 toml_datum_t name_datum = toml_get(scheme_section, "name");
450 if (name_datum.type == TOML_STRING) {
451 SAFE_STRNCPY(scheme->name, name_datum.u.s, sizeof(scheme->name));
452 }
453
454 /* Get scheme description */
455 toml_datum_t desc_datum = toml_get(scheme_section, "description");
456 if (desc_datum.type == TOML_STRING) {
457 SAFE_STRNCPY(scheme->description, desc_datum.u.s, sizeof(scheme->description));
458 }
459 }
460
461 /* Get colors section */
462 toml_datum_t colors_section = toml_get(result.toptab, "colors");
463 if (colors_section.type == TOML_TABLE) {
464 /* Parse dark mode colors */
465 toml_datum_t dark_section = toml_get(colors_section, "dark");
466 if (dark_section.type == TOML_TABLE) {
467 const char *color_names[] = {"dev", "debug", "warn", "info", "error", "fatal", "grey", "reset"};
468 for (int i = 0; i < 8; i++) {
469 toml_datum_t color_value = toml_get(dark_section, color_names[i]);
470 if (color_value.type == TOML_STRING) {
471 parse_hex_color(color_value.u.s, &scheme->log_colors_dark[i].r, &scheme->log_colors_dark[i].g,
472 &scheme->log_colors_dark[i].b);
473 }
474 }
475 }
476
477 /* Parse light mode colors (optional) */
478 toml_datum_t light_section = toml_get(colors_section, "light");
479 if (light_section.type == TOML_TABLE) {
480 scheme->has_light_variant = true;
481 const char *color_names[] = {"dev", "debug", "warn", "info", "error", "fatal", "grey", "reset"};
482 for (int i = 0; i < 8; i++) {
483 toml_datum_t color_value = toml_get(light_section, color_names[i]);
484 if (color_value.type == TOML_STRING) {
485 parse_hex_color(color_value.u.s, &scheme->log_colors_light[i].r, &scheme->log_colors_light[i].g,
486 &scheme->log_colors_light[i].b);
487 }
488 }
489 }
490 }
491
492 scheme->is_builtin = false;
493 SAFE_STRNCPY(scheme->source_file, path, sizeof(scheme->source_file));
494
495 toml_free(result);
496 return ASCIICHAT_OK;
497}

499/* ============================================================================
500 * Public API: Color Conversion
501 * ============================================================================ */
502
503asciichat_error_t parse_hex_color(const char *hex, uint8_t *r, uint8_t *g, uint8_t *b) {
504 if (!hex || !r || !g || !b) {
505 return SET_ERRNO(ERROR_INVALID_PARAM, "NULL parameter");
506 }
507
508 /* Skip '#' prefix if present */
509 if (hex[0] == '#') {
510 hex++;
511 }
512
513 /* Validate hex string length */
514 if (strlen(hex) != 6) {
515 return SET_ERRNO(ERROR_CONFIG, "Invalid hex color (must be #RRGGBB): %s", hex);
516 }
517
518 /* Parse hex digits */
519 unsigned int rgb = 0;
520 if (SAFE_SSCANF(hex, "%6x", &rgb) != 1) {
521 return SET_ERRNO(ERROR_CONFIG, "Invalid hex color format: %s", hex);
522 }
523
524 *r = (rgb >> 16) & 0xFF;
525 *g = (rgb >> 8) & 0xFF;
526 *b = rgb & 0xFF;
527
528 return ASCIICHAT_OK;
529}

541/* ============================================================================
542 * Public API: Scheme Compilation
543 * ============================================================================ */
544
545asciichat_error_t colorscheme_compile_scheme(const color_scheme_t *scheme, terminal_color_mode_t mode,
546 terminal_background_t background, compiled_color_scheme_t *compiled) {
547 if (!scheme || !compiled) {
548 return SET_ERRNO(ERROR_INVALID_PARAM, "NULL scheme or compiled pointer");
549 }
550
551 /* Note: mode parameter reserved for future use */
552 (void)mode;
553
554 /* Free any previously compiled strings before recompiling */
555 /* This prevents memory leaks when the color scheme is recompiled */
556 /* Only cleanup if we've actually compiled before (codes_16[0] != NULL means already compiled) */
557 if (compiled->codes_16[0] != NULL) {
558 colorscheme_cleanup_compiled(compiled);
559 } else {
560 }
561
562 /* Select color array based on background */
563 const rgb_pixel_t *colors = (background == TERM_BACKGROUND_LIGHT && scheme->has_light_variant)
564 ? scheme->log_colors_light
565 : scheme->log_colors_dark;
566 /* Helper: allocate and format a color code string */
567 char temp_buf[128];
568
569 /* Compile for 16-color mode */
570 for (int i = 0; i < 8; i++) {
571 if (i == 7) {
572 /* RESET */
573 SAFE_STRNCPY(temp_buf, "\x1b[0m", sizeof(temp_buf));
574 } else {
575 uint8_t color_idx = rgb_to_16color(colors[i].r, colors[i].g, colors[i].b);
576 /* ANSI color codes: 30-37 for normal, 90-97 for bright */
577 if (color_idx < 8) {
578 safe_snprintf(temp_buf, sizeof(temp_buf), "\x1b[%dm", 30 + color_idx);
579 } else {
580 safe_snprintf(temp_buf, sizeof(temp_buf), "\x1b[%dm", 90 + (color_idx - 8));
581 }
582 }
583 /* Allocate string with SAFE_MALLOC and copy */
584 size_t len = strlen(temp_buf) + 1;
585 char *allocated = SAFE_MALLOC(len, char *);
586 if (!allocated) {
587 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate color code string");
588 }
589 memcpy(allocated, temp_buf, len);
590 compiled->codes_16[i] = allocated;
591 }
592 /* Compile for 256-color mode */
593 for (int i = 0; i < 8; i++) {
594 if (i == 7) {
595 SAFE_STRNCPY(temp_buf, "\x1b[0m", sizeof(temp_buf));
596 } else {
597 uint8_t color_idx = rgb_to_256color(colors[i].r, colors[i].g, colors[i].b);
598 safe_snprintf(temp_buf, sizeof(temp_buf), "\x1b[38;5;%dm", color_idx);
599 }
600 /* Allocate string with SAFE_MALLOC and copy */
601 size_t len = strlen(temp_buf) + 1;
602 char *allocated = SAFE_MALLOC(len, char *);
603 if (!allocated) {
604 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate color code string");
605 }
606 memcpy(allocated, temp_buf, len);
607 compiled->codes_256[i] = allocated;
608 }
609
610 /* Compile for truecolor mode */
611 for (int i = 0; i < 8; i++) {
612 if (i == 7) {
613 SAFE_STRNCPY(temp_buf, "\x1b[0m", sizeof(temp_buf));
614 } else {
615 rgb_to_truecolor_ansi(colors[i].r, colors[i].g, colors[i].b, temp_buf, sizeof(temp_buf));
616 }
617 /* Allocate string with SAFE_MALLOC and copy */
618 size_t len = strlen(temp_buf) + 1;
619 char *allocated = SAFE_MALLOC(len, char *);
620 if (!allocated) {
621 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate color code string");
622 }
623 memcpy(allocated, temp_buf, len);
624 compiled->codes_truecolor[i] = allocated;
625 }
626
627 return ASCIICHAT_OK;
628}

634asciichat_error_t colorscheme_export_scheme(const char *scheme_name, const char *file_path) {
635 if (!scheme_name) {
636 return SET_ERRNO(ERROR_INVALID_PARAM, "Scheme name is NULL");
637 }
638
639 color_scheme_t scheme = {0};
640 asciichat_error_t result = colorscheme_load_builtin(scheme_name, &scheme);
641 if (result != ASCIICHAT_OK) {
642 return result;
643 }
644
645 /* Generate TOML content */
646 char toml_content[8192] = {0};
647 size_t offset = 0;
648
649 /* Scheme header section */
650 offset += safe_snprintf(toml_content + offset, sizeof(toml_content) - offset,
651 "[scheme]\n"
652 "name = \"%s\"\n"
653 "description = \"%s\"\n\n",
654 scheme.name, scheme.description);
655
656 /* Dark mode colors */
657 const char *color_names[] = {"dev", "debug", "warn", "info", "error", "fatal", "grey", "reset"};
658 offset += safe_snprintf(toml_content + offset, sizeof(toml_content) - offset, "[colors.dark]\n");
659
660 for (int i = 0; i < 8; i++) {
661 offset +=
662 safe_snprintf(toml_content + offset, sizeof(toml_content) - offset, "%s = \"#%02X%02X%02X\"\n", color_names[i],
663 scheme.log_colors_dark[i].r, scheme.log_colors_dark[i].g, scheme.log_colors_dark[i].b);
664 }
665
666 /* Light mode colors if available */
667 if (scheme.has_light_variant) {
668 offset += safe_snprintf(toml_content + offset, sizeof(toml_content) - offset, "\n[colors.light]\n");
669 for (int i = 0; i < 8; i++) {
670 offset += safe_snprintf(toml_content + offset, sizeof(toml_content) - offset, "%s = \"#%02X%02X%02X\"\n",
671 color_names[i], scheme.log_colors_light[i].r, scheme.log_colors_light[i].g,
672 scheme.log_colors_light[i].b);
673 }
674 }
675
676 /* Write to file or stdout */
677 if (file_path) {
678 FILE *fp = platform_fopen("file_stream", file_path, "w");
679 if (!fp) {
680 return SET_ERRNO_SYS(ERROR_FILE_OPERATION, "Cannot open %s for writing", file_path);
681 }
682 if (fputs(toml_content, fp) < 0) {
683 fclose(fp);
684 return SET_ERRNO_SYS(ERROR_FILE_OPERATION, "Failed to write to %s", file_path);
685 }
686 fclose(fp);
687 } else {
688 /* Write to stdout */
689 if (fputs(toml_content, stdout) < 0) {
690 return SET_ERRNO_SYS(ERROR_FILE_OPERATION, "Failed to write to stdout");
691 }
692 /* Flush to ensure piped output is written immediately */
693 (void)fflush(stdout);
694 }
695
696 return ASCIICHAT_OK;
697}
