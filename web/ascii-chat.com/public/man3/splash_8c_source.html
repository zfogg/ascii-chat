125static rgb_pixel_t interpolate_color(rgb_pixel_t color1, rgb_pixel_t color2, double t) {
126 rgb_pixel_t result;
127 result.r = (uint8_t)(color1.r * (1.0 - t) + color2.r * t);
128 result.g = (uint8_t)(color1.g * (1.0 - t) + color2.g * t);
129 result.b = (uint8_t)(color1.b * (1.0 - t) + color2.b * t);
130 return result;
131}

138static rgb_pixel_t get_rainbow_color_rgb(double position) {
139 // Normalize position to 0-1 range
140 double norm_pos = position - (long)position;
141 if (norm_pos < 0) {
142 norm_pos += 1.0;
143 }
144
145 // Scale position to color range
146 double color_pos = norm_pos * (RAINBOW_COLOR_COUNT - 1);
147 int color_idx = (int)color_pos;
148 double blend = color_pos - color_idx;
149
150 // Wrap around at the end
151 if (color_idx >= RAINBOW_COLOR_COUNT - 1) {
152 color_idx = RAINBOW_COLOR_COUNT - 1;
153 blend = 0;
154 }
155
156 int next_idx = (color_idx + 1) % RAINBOW_COLOR_COUNT;
157
158 return interpolate_color(g_rainbow_colors[color_idx], g_rainbow_colors[next_idx], blend);
159}

179static void build_connection_target(char *buffer, size_t buffer_size) {
180 if (!buffer || buffer_size == 0) {
181 return;
182 }
183 buffer[0] = '\0';
184
185 // Mirror mode - no network connection
186 asciichat_mode_t mode = GET_OPTION(detected_mode);
187 if (mode == MODE_MIRROR) {
188 const char *media_url = GET_OPTION(media_url);
189 const char *media_file = GET_OPTION(media_file);
190
191 if (media_url && media_url[0] != '\0') {
192 snprintf(buffer, buffer_size, "Loading from URL...");
193 } else if (media_file && media_file[0] != '\0') {
194 snprintf(buffer, buffer_size, "Loading from file...");
195 } else {
196 snprintf(buffer, buffer_size, "Initializing...");
197 }
198 return;
199 }
200
201 // Check if we have a session string (discovery mode)
202 const char *session = GET_OPTION(session_string);
203 if (session && session[0] != '\0') {
204 snprintf(buffer, buffer_size, "Connecting to session: %s", session);
205 return;
206 }
207
208 // Check if we have an address (client mode)
209 const char *addr = GET_OPTION(address);
210
211 if (addr && addr[0] != '\0') {
212 // Use IP utility to classify the connection type
213 const char *ip_type = get_ip_type_string(addr);
214
215 if (strcmp(ip_type, "Localhost") == 0) {
216 snprintf(buffer, buffer_size, "Connecting to localhost...");
217 } else if (strcmp(ip_type, "LAN") == 0) {
218 snprintf(buffer, buffer_size, "Connecting to %s (LAN)", addr);
219 } else if (strcmp(ip_type, "Internet") == 0) {
220 snprintf(buffer, buffer_size, "Connecting to %s (Internet)", addr);
221 } else if (strcmp(addr, "localhost") == 0) {
222 // Hostname "localhost" not detected by IP util
223 snprintf(buffer, buffer_size, "Connecting to localhost...");
224 } else {
225 // Unknown or hostname
226 snprintf(buffer, buffer_size, "Connecting to %s", addr);
227 }
228 return;
229 }
230
231 // Fallback if no connection info available
232 snprintf(buffer, buffer_size, "Initializing...");
233}

249static void render_splash_header(frame_buffer_t *buf, terminal_size_t term_size, void *user_data) {
250 const splash_header_ctx_t *ctx = (const splash_header_ctx_t *)user_data;
251 if (!ctx) {
252 return;
253 }
254
255 // ASCII logo lines (same as help output)
256 const char *ascii_logo[4] = {
257 " __ _ ___ ___(_|_) ___| |__ __ _| |_ ", " / _` / __|/ __| | |_____ / __| '_ \\ / _` | __| ",
258 "| (_| \\__ \\ (__| | |_____| (__| | | | (_| | |_ ", " \\__,_|___/\\___|_|_| \\___|_| |_|\\__,_|\\__| "};
259 const char *tagline = "Video chat in your terminal";
260 const int logo_width = 52;
261
262 // Line 1: Top border
263 frame_buffer_render_border(buf, term_size.cols, "\033[1;36m");
264
265 // Lines 2-5: ASCII logo (centered, truncated if too long)
266 for (int logo_line = 0; logo_line < 4; logo_line++) {
267 // Build plain text line first (for width calculation)
268 char plain_line[512];
269 int horiz_pad = (term_size.cols - logo_width) / 2;
270 if (horiz_pad < 0) {
271 horiz_pad = 0;
272 }
273
274 int pos = 0;
275 for (int j = 0; j < horiz_pad && pos < (int)sizeof(plain_line) - 1; j++) {
276 plain_line[pos++] = ' ';
277 }
278 snprintf(plain_line + pos, sizeof(plain_line) - pos, "%s", ascii_logo[logo_line]);
279
280 // Check visible width and truncate if needed
281 int visible_width = display_width(plain_line);
282 if (visible_width < 0) {
283 visible_width = (int)strlen(plain_line);
284 }
285 if (term_size.cols > 0 && visible_width >= term_size.cols) {
286 plain_line[term_size.cols - 1] = '\0';
287 }
288
289 // Print with rainbow colors
290 int char_idx = 0;
291 for (int i = 0; plain_line[i] != '\0'; i++) {
292 char ch = plain_line[i];
293 if (ch == ' ') {
294 frame_buffer_printf(buf, " ");
295 } else if (ctx->use_colors) {
296 double char_pos = (char_idx + ctx->frame / 5.0) / 30.0;
297 rgb_pixel_t color = get_rainbow_color_rgb(char_pos);
298 frame_buffer_printf(buf, "\x1b[38;2;%u;%u;%um%c\x1b[0m", color.r, color.g, color.b, ch);

301 frame_buffer_printf(buf, "%c", ch);
302 char_idx++;
303 }
304 }
305 frame_buffer_printf(buf, "\n");
306 }
307
308 // Line 6: Tagline (centered, truncated if too long)
309 char plain_tagline[512];
310 int tagline_len = (int)strlen(tagline);
311 int tagline_pad = (term_size.cols - tagline_len) / 2;
312 if (tagline_pad < 0) {
313 tagline_pad = 0;
314 }

316 int tpos = 0;
317 for (int j = 0; j < tagline_pad && tpos < (int)sizeof(plain_tagline) - 1; j++) {
318 plain_tagline[tpos++] = ' ';

320 snprintf(plain_tagline + tpos, sizeof(plain_tagline) - tpos, "%s", tagline);
321
322 // Check visible width and truncate if needed
323 int tagline_visible_width = display_width(plain_tagline);
324 if (tagline_visible_width < 0) {
325 tagline_visible_width = (int)strlen(plain_tagline);
326 }
327 if (term_size.cols > 0 && tagline_visible_width >= term_size.cols) {
328 plain_tagline[term_size.cols - 1] = '\0';
329 }
330
331 frame_buffer_printf(buf, "%s\n", plain_tagline);
332
333 // Line 7: Update notification (if available)
334 if (ctx->update_notification[0] != '\0') {
335 char plain_update[1024];
336 int update_len = (int)strlen(ctx->update_notification);
337 int update_pad = (term_size.cols - update_len) / 2;
338 if (update_pad < 0) {
339 update_pad = 0;
340 }

354 plain_update[term_size.cols - 1] = '\0';
355 }
356
357 // Print in yellow/warning color
358 frame_buffer_printf(buf, "%s\n", colored_string(LOG_COLOR_WARN, plain_update));
359 }
360
361 // Line 8 (or 7 if no update): Connection target (centered, truncated if too long)
362 char connection_target[512];
363 build_connection_target(connection_target, sizeof(connection_target));
364

366 int connection_len = (int)strlen(connection_target);

397bool splash_should_display(bool is_intro) {
398 // Check option flags (display splash if enabled, regardless of TTY for testing)
399 if (is_intro) {
400 // Allow splash in snapshot mode if loading from URL/file (has loading time)
401 // Skip splash only for quick webcam snapshots (unless explicitly enabled)
402 bool splash_screen_opt = GET_OPTION(splash_screen);
403 bool splash_explicitly_set = GET_OPTION(splash_screen_explicitly_set);
404 bool is_snapshot = GET_OPTION(snapshot_mode);
405 bool has_media = (GET_OPTION(media_url) && strlen(GET_OPTION(media_url)) > 0) ||
406 (GET_OPTION(media_file) && strlen(GET_OPTION(media_file)) > 0);
407
408 // Show splash if enabled, but in snapshot mode only if:
409 // 1. Has media file/URL, OR
410 // 2. Explicitly enabled with --splash-screen=true
411 bool should_display = splash_screen_opt && (!is_snapshot || has_media || splash_explicitly_set);
412 // Show splash if:
413 // 1. Not in snapshot mode, OR
414 // 2. In snapshot mode but loading from URL/file (needs splash during load)
415 return should_display;
416 } else {
417 return GET_OPTION(status_screen);
418 }
419}

425static void *splash_animation_thread(void *arg) {
426 (void)arg;
427
428 // Check if colors should be used (TTY check)
429 bool use_colors = terminal_should_color_output(STDOUT_FILENO);
430
431 // Keyboard is pre-initialized from asciichat_shared_init()
432 // Only enable if terminal is interactive (keyboard won't work in non-TTY)
433 bool keyboard_enabled = terminal_is_interactive();
434
435 // Animate with rainbow wave effect - TIME-BASED, not frame-based
436 // This ensures animation speed is consistent regardless of FPS
437 int fps = GET_OPTION(fps);
438 if (fps <= 0)
439 fps = 60; // Default to 60 FPS if not specified
440 const int anim_speed = 1000 / fps; // milliseconds per frame
441 uint64_t loop_start_ns = time_get_ns();
442 int iteration_count = 0; // Just for logging actual FPS
443
444 log_dev("[SPLASH_ANIM_INIT] fps=%d anim_speed=%dms", fps, anim_speed);
445
446 // Don't log until after first frame to avoid startup flicker
447 bool first_frame = true;
448
449 while (!atomic_load_bool(&g_splash_state.should_stop) && !shutdown_is_requested()) {
450 // Calculate animation frame based on ELAPSED TIME, not iteration count
451 // This ensures the animation plays at the same speed regardless of FPS variation
452 uint64_t now_ns = time_get_ns();
453 uint64_t elapsed_ns = now_ns - loop_start_ns;
454 uint64_t elapsed_ms = elapsed_ns / 1000000;
455
456 // Check if we should stop the splash (from splash_intro_done signal)
457 uint64_t intro_done_ns = atomic_load_u64(&g_splash_state.intro_done_time_ns);
458 if (intro_done_ns > 0) {
459 // splash_intro_done() was called - keep splash running for a minimum display time
460 // The first ASCII frame will be rendered after the splash exits
461 const uint64_t min_display_ns = 2000000000ULL; // 2 seconds in nanoseconds
462 const uint64_t max_display_ns = 30000000000ULL; // 30 seconds maximum (safety limit)
463 uint64_t elapsed_since_start_ns = now_ns - g_splash_state.start_time_ns;
464
465 // Stop when: minimum time reached OR maximum time reached
466 // We don't wait for has_first_frame because the first frame is rendered
467 // by the main thread after splash_wait_for_animation() returns
468 bool min_display_reached = elapsed_since_start_ns >= min_display_ns;
469 bool max_time_reached = elapsed_since_start_ns >= max_display_ns;
470
471 if (min_display_reached || max_time_reached) {
472 atomic_store_bool(&g_splash_state.should_stop, true);
473 }
474 }
475
476 // Convert elapsed time to animation frame (at target FPS)
477 // For 60 FPS target: frame = elapsed_ms / 16.67
478 int frame = (int)(elapsed_ms * fps / 1000);
479
480 if (!first_frame) {
481 log_dev("[SPLASH_ANIM] Iter %d: elapsed=%llums frame=%d should_stop=%d", iteration_count,
482 (unsigned long long)elapsed_ms, frame, atomic_load_bool(&g_splash_state.should_stop));
483 }
484
485 // Poll keyboard for interactive grep and Escape to cancel
486 if (!first_frame) {
487 log_dev("[SPLASH_ANIM] Iter %d: keyboard_enabled=%d", iteration_count, keyboard_enabled);
488 }
489 if (keyboard_enabled) {
490 keyboard_key_t key = keyboard_read_nonblocking();
491 if (key == KEY_ESCAPE) {
492 // Escape key: cancel grep if active, otherwise cancel splash
493 if (log_search_is_active()) {
494 log_search_exit_mode(false); // Cancel grep without applying
495 } else {
496 atomic_store_bool(&g_splash_state.should_stop, true); // Exit splash screen
497 }
498 } else if (key != KEY_NONE && log_search_should_handle(key)) {
499 log_search_handle_key(key);
500 // Continue to render immediately with grep active
501 }
502 }
503 if (!first_frame) {
504 log_dev("[SPLASH_ANIM] Iter %d: keyboard check done", iteration_count);
505 }
506
507 // Set up splash header context for this frame (using TIME-BASED frame value)
508 splash_header_ctx_t header_ctx = {
509 .frame = frame,
510 .use_colors = use_colors,
511 };
512
513 if (!first_frame) {
514 log_dev("[SPLASH_ANIM] Frame %d: Setting up header context", frame);
515 }
516
517 // Initialize update notification lifecycle and mutex once (safe to call multiple times)
518 if (lifecycle_init_once(&g_update_notification_lifecycle)) {
519 if (!first_frame) {
520 log_dev("[SPLASH_ANIM] Initializing lifecycle for update_notification");
521 }
522 // Initialize the mutex (already configured in the lifecycle structure)
523 mutex_init(&g_update_notification_mutex, "update_notification");
524 lifecycle_init_commit(&g_update_notification_lifecycle);
525 }
526
527 // Copy update notification from global state (thread-safe)
528 if (!first_frame) {
529 log_dev("[SPLASH_ANIM] Frame %d: About to lock update_notification_mutex", frame);
530 }
531 mutex_lock(&g_update_notification_mutex);
532 if (!first_frame) {
533 log_dev("[SPLASH_ANIM] Frame %d: Locked update_notification_mutex", frame);
534 }
535 SAFE_STRNCPY(header_ctx.update_notification, g_update_notification, sizeof(header_ctx.update_notification));
536 mutex_unlock(&g_update_notification_mutex);
537 if (!first_frame) {
538 log_dev("[SPLASH_ANIM] Frame %d: Unlocked update_notification_mutex", frame);
539 }
540
541 // Calculate header lines: 8 base lines + 1 if update notification present
542 int header_lines = 8;
543 if (header_ctx.update_notification[0] != '\0') {
544 header_lines = 9;
545 }
546
547 // Configure terminal screen with splash header callback
548 terminal_screen_config_t screen_config = {
549 .fixed_header_lines = header_lines,
550 .render_header = render_splash_header,
551 .user_data = &header_ctx,
552 .show_logs = true, // Show live log feed below splash
553 };
554
555 // Render the screen (header + logs) only in interactive mode
556 // OR if splash screen was explicitly requested
557 // In non-interactive mode without explicit flag, logs flow to stdout/stderr normally
558 const options_t *opts_render = options_get();
559 bool should_render = terminal_is_interactive() || (opts_render && opts_render->splash_screen_explicitly_set);
560
561 // Log startup phase timing
562 if (elapsed_ms < 500) {
563 log_dev("[STARTUP_PHASE] elapsed_ms=%llu frame=%d (time-based) header_lines=%d should_render=%d",
564 (unsigned long long)elapsed_ms, frame, header_lines, should_render);
565 }
566
567 if (should_render) {
568 terminal_screen_render(&screen_config);
569 if (!first_frame) {
570 log_dev("[SPLASH_ANIM] Iter %d: terminal_screen_render() completed", iteration_count);
571 }
572 first_frame = false; // Mark first frame as complete
573 } else {
574 if (!first_frame) {
575 log_dev("[SPLASH_ANIM] Iter %d: Skipping render (should_render=false)", iteration_count);
576 }
577 }
578
579 iteration_count++;
580
581 // Log progress every 1 second of actual time (not frame count)
582 if (elapsed_ms > 0 && (int)(elapsed_ms % 1000) < anim_speed) {
583 uint64_t now_ns = time_get_ns();
584 uint64_t total_elapsed_ns = now_ns - loop_start_ns;
585 double elapsed_sec = total_elapsed_ns / 1e9;
586 double actual_fps = iteration_count / elapsed_sec;
587 log_info("[SPLASH_ANIM] %llu ms elapsed, iteration %d, actual FPS: %.1f", (unsigned long long)elapsed_ms,
588 iteration_count, actual_fps);
589 }
590
591 // Sleep to control frame rate (unless grep needs immediate rerender)
592 if (!log_search_needs_rerender()) {
593 uint64_t sleep_start_ns = time_get_ns();
594 platform_sleep_ms(anim_speed);
595 uint64_t sleep_end_ns = time_get_ns();
596 uint64_t actual_sleep_ms = (sleep_end_ns - sleep_start_ns) / 1000000;
597 if (iteration_count % 60 == 0) {
598 log_debug("[SPLASH_SLEEP] Iter %d: requested=%dms actual=%llums", iteration_count, anim_speed,
599 (unsigned long long)actual_sleep_ms);
600 }
601 } else {
602 if (iteration_count % 60 == 0) {
603 log_debug("[SPLASH_GREP_RERENDER] Iter %d: skipping sleep for grep rerender", iteration_count);
604 }
605 }
606 }
607
608 // Log final iteration count
609 uint64_t final_ns = time_get_ns();
610 uint64_t total_elapsed_ns = final_ns - loop_start_ns;
611 double total_elapsed_sec = total_elapsed_ns / 1e9;
612 double final_fps = total_elapsed_sec > 0 ? iteration_count / total_elapsed_sec : 0;
613 log_info(
614 "[SPLASH_ANIM] === ANIMATION LOOP EXITED === %d iterations in %.3f seconds (%.1f actual FPS). should_stop=%d, "
615 "shutdown=%d",
616 iteration_count, total_elapsed_sec, final_fps, atomic_load_bool(&g_splash_state.should_stop), shutdown_is_requested());
617
618 // If shutdown was requested, clear the screen immediately to prevent splash from
619 // appearing briefly during exit
620 if (shutdown_is_requested()) {
621 log_dev("[SPLASH_ANIM] Shutdown detected, clearing screen before exit");
622 terminal_clear_screen();
623 terminal_cursor_home(STDOUT_FILENO);
624 terminal_flush(STDOUT_FILENO);
625 }
626
627 // Cleanup keyboard
628 if (keyboard_enabled) {
629 log_dev("[SPLASH_ANIM] Destroying keyboard");
630 keyboard_destroy();
631 }
632
633 atomic_store_bool(&g_splash_state.is_running, false);
634 log_dev("[SPLASH_ANIM] Animation thread exiting");
635 return NULL;
636}

638int splash_intro_start(session_display_ctx_t *ctx) {
639 (void)ctx; // Not needed - display context is not actively used
640
641 // Check if splash should display
642 if (!splash_should_display(true)) {
643 return 0;
644 }
645
646 // Check terminal size
647 int width = (int)terminal_get_effective_width();
648 int height = (int)terminal_get_effective_height();
649 if (width < 50 || height < 20) {
650 return 0;
651 }
652
653 // Redirect splash screen to stderr when stdout is piped (not a TTY)
654 // This keeps stdout clean for frame data while splash goes to stderr
655 if (!isatty(STDOUT_FILENO)) {
656 terminal_screen_set_output_fd(STDERR_FILENO);
657 }
658
659 // Initialize log buffer (same pattern as server_status)
660 splash_log_init();

691int splash_intro_done(void) {
692 // Record when intro_done was called - animation thread will use this to decide when to stop
693 // Animation thread needs time to check the flag and perform cleanup, so this must happen
694 // BEFORE we try to access the terminal for frame rendering
695 atomic_store_u64(&g_splash_state.intro_done_time_ns, time_get_ns());
696
697 // NOTE: Do NOT modify thread_created flag here!
698 // splash_wait_for_animation() needs to see thread_created==true to know whether to join the thread.
699 // If we set thread_created=false here, splash_wait_for_animation() won't join, and the render loop
700 // will start while the animation thread is still running, causing frame output to conflict with splash.
701 //
702 // Instead, just signal the animation thread to stop via intro_done_time_ns, and let
703 // splash_wait_for_animation() handle the actual thread join.
704
705 // NOTE: DO NOT set is_running = false here! The animation thread sets it to false
706 // when it actually exits (line 633). If we set it to false here, the render loop
707 // guard will think splash is done while the animation thread is still running for
708 // up to 2 more seconds, allowing ASCII frames to render simultaneously with splash.
709 // The is_running flag MUST only be cleared by the animation thread when it finishes.
710
711 return 0;
712}

729void splash_wait_for_animation(void) {
730 // Wait for animation thread to fully exit before rendering ASCII art
731 // This prevents the splash and ASCII art from appearing simultaneously
732 //
733 // The animation thread will exit gracefully when:
734 // - Intro done signal received AND 2 seconds have elapsed, OR
735 // - 30 seconds have elapsed (safety timeout)
736 // - OR a shutdown signal is received
737 //
738 // We block here indefinitely to ensure the splash animation completes before ASCII art starts
739
740 // Only join if we successfully created the thread
741 if (atomic_load_bool(&g_splash_state.thread_created)) {
742 log_dev("[SPLASH_WAIT] Waiting for animation thread to exit...");
743
744 // Check if shutdown was requested - handle it specially
745 bool is_shutting_down = shutdown_is_requested();
746 if (is_shutting_down) {
747 log_dev("[SPLASH_WAIT] Shutdown requested, signaling animation thread to stop");
748 atomic_store_bool(&g_splash_state.should_stop, true);
749 // During shutdown, use a short timeout (100ms)
750 uint64_t timeout_ns = 100LL * NS_PER_MS_INT;
751 (void)asciichat_thread_join_timeout(&g_splash_state.anim_thread, NULL, timeout_ns);
752 } else {
753 // Normal operation: wait indefinitely for the animation thread to finish
754 // This ensures splash animation is 100% done before ASCII art rendering starts
755 asciichat_error_t err = asciichat_thread_join(&g_splash_state.anim_thread, NULL);
756 if (err == ASCIICHAT_OK) {
757 log_dev("[SPLASH_WAIT] Animation thread exited cleanly");
758 } else {
759 log_warn("[SPLASH_WAIT] Animation thread join failed: %s", asciichat_error_string(err));
760 // Force stop the animation thread if join failed
761 atomic_store_bool(&g_splash_state.should_stop, true);
762 // Clear any partial splash output left on screen
763 terminal_clear_screen();
764 terminal_cursor_home(STDOUT_FILENO);
765 terminal_flush(STDOUT_FILENO);
766 }
767 }
768
769 // Mark that we've joined (safe to call multiple times - only joins once)
770 atomic_store_bool(&g_splash_state.thread_created, false);
771 }
772}
