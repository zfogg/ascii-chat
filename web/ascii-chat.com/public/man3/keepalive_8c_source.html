142static void *ping_thread_func(void *arg) {
143 (void)arg;
144
145#ifdef DEBUG_THREADS
146 log_debug("Ping thread started");
147#endif
148
149 // FPS tracking for keepalive thread (ping sent every 1 second)
150 static fps_t fps_tracker = {0};
151 static bool fps_tracker_initialized = false;
152 if (!fps_tracker_initialized) {
153 fps_init_with_interval(&fps_tracker, 1, "KEEPALIVE",
154 100 * NS_PER_MS_INT); // Report keepalive metrics every 100ms
155 fps_tracker_initialized = true;
156 }
157
158 // Startup grace period: Allow connection initialization to complete before checking state
159 // Prevents race condition where thread spawns before transport is fully configured
160 platform_sleep_ms(100);
161
162 while (!should_exit() && !server_connection_is_lost()) {
163 // Check if connection is still active before sending
164 if (!server_connection_is_active()) {
165 log_debug("Connection inactive, exiting ping thread");
166 break;
167 }
168
169 // Check if session rekeying should be triggered
170 if (crypto_client_should_rekey()) {
171 log_debug("Rekey threshold reached, initiating session rekey");
172 if (crypto_client_initiate_rekey() < 0) {
173 log_error("Failed to initiate rekey");
174 // Don't break - continue with keepalive, rekey will be retried
175 }
176 }
177
178 // Send ping packet every PING_INTERVAL_SECONDS to keep connection alive
179 // Server timeout is 5 seconds, so 3-second pings provide safety margin
180 if (threaded_send_ping_packet() < 0) {
181 log_debug("Failed to send ping packet");
182 // Set connection lost flag so main loop knows to reconnect
183 server_connection_lost();
184 break;
185 }
186
187 // Track ping for FPS reporting
188 uint64_t ping_start_ns = time_get_ns();
189 fps_frame_ns(&fps_tracker, ping_start_ns, "ping sent");
190
191 // Sleep with early wake capability for responsive shutdown
192 // Break sleep into intervals to check shutdown flags and rekey triggers
193 while (!should_exit() && !server_connection_is_lost() && server_connection_is_active()) {
194 uint64_t now_ns = time_get_ns();
195 uint64_t elapsed_ns = time_elapsed_ns(ping_start_ns, now_ns);
196
197 if (elapsed_ns >= PING_INTERVAL_NS) {
198 break; // Time for next ping
199 }
200
201 // Sleep for one interval or until next ping, whichever is sooner
202 uint64_t remaining_ns = PING_INTERVAL_NS - elapsed_ns;
203 uint64_t sleep_ns = (remaining_ns > PING_SLEEP_INTERVAL_NS) ? PING_SLEEP_INTERVAL_NS : remaining_ns;
204 platform_sleep_ns(sleep_ns);
205 }
206 }
207
208#ifdef DEBUG_THREADS
209 log_debug("Ping thread stopped");
210#endif
211
212 atomic_store_bool(&g_ping_thread_exited, true);
213
214 // Clean up thread-local error context before exit
215 asciichat_errno_destroy();
216
217 return NULL;
218}

234int keepalive_start_thread() {
235 if (g_ping_thread_created) {
236 log_warn("Ping thread already created");
237 return 0;
238 }
239
240 // Register keepalive atomics with named debug registry
241 static bool keepalive_atomics_registered = false;
242 if (!keepalive_atomics_registered) {
243 NAMED_REGISTER_ATOMIC(&g_ping_thread_exited, "keepalive_ping_thread_exit_confirmation", NULL);
244 keepalive_atomics_registered = true;
245 }
246
247 // Start ping thread for keepalive
248 atomic_store_bool(&g_ping_thread_exited, false);
249 if (thread_pool_spawn(g_client_worker_pool, ping_thread_func, NULL, 3, "keepalive_ping") != ASCIICHAT_OK) {
250 log_error("Failed to spawn ping thread in worker pool");
251 LOG_ERRNO_IF_SET("Ping thread creation failed");
252 return -1;
253 }
254
255 g_ping_thread_created = true;
256 return 0;
257}

267void keepalive_stop_thread() {
268 if (!g_ping_thread_created) {
269 return;
270 }
271
272 // Don't call signal_exit() here - that's for global shutdown only!
273 // The ping thread monitors connection state and will exit when connection is lost
274
275 // Wait for thread to exit gracefully
276 int wait_count = 0;
277 while (wait_count < 20 && !atomic_load_bool(&g_ping_thread_exited)) {
278 platform_sleep_ns(100 * NS_PER_MS_INT); // 100ms
279 wait_count++;
280 }
281
282 if (!atomic_load_bool(&g_ping_thread_exited)) {
283 log_warn("Ping thread not responding - will be joined by thread pool");
284 }
285
286 // Thread will be joined by thread_pool_stop_all() in protocol_stop_connection()
287 g_ping_thread_created = false;
288
289 log_debug("Ping thread stopped and joined");
290}
