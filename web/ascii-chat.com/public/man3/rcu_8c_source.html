275#define MAX_DEFERRED_FREES 64
276static options_t *g_deferred_frees[MAX_DEFERRED_FREES];
277static size_t g_deferred_free_count = 0;
278static mutex_t g_deferred_free_mutex;
279
285static void deferred_free_add(options_t *old_opts) {
286 if (!old_opts) {
287 return;
288 }
289
290 mutex_lock(&g_deferred_free_mutex);
291
292 if (g_deferred_free_count < MAX_DEFERRED_FREES) {
293 g_deferred_frees[g_deferred_free_count++] = old_opts;
294 log_debug("Added options struct %p to deferred free list (count=%zu)", (void *)old_opts, g_deferred_free_count);
295 } else {
296 // List full - free immediately (risky but unlikely with infrequent updates)
297 log_warn("Deferred free list full (%d entries), freeing options struct %p immediately", MAX_DEFERRED_FREES,
298 (void *)old_opts);
299 SAFE_FREE(old_opts);
300 }
301
302 mutex_unlock(&g_deferred_free_mutex);
303}
304
310static void deferred_free_all(void) {
311 mutex_lock(&g_deferred_free_mutex);
312
313 log_debug("Freeing %zu deferred options structs", g_deferred_free_count);
314
315 for (size_t i = 0; i < g_deferred_free_count; i++) {
316 SAFE_FREE(g_deferred_frees[i]);
317 }
318
319 g_deferred_free_count = 0;
320 mutex_unlock(&g_deferred_free_mutex);
321}

334static void options_rcu_atfork_child(void) {
335 /* Reset the lifecycle state to allow reinitialization after fork */
336 /* This is safe because lifecycle_t is just an atomic int, can be reset directly */
337 atomic_store_int(&g_options_lifecycle.state, LIFECYCLE_UNINITIALIZED);
338
339 /* Reset the atomic options pointer to avoid dangling references */
340 /* The parent's allocated memory is not accessible/valid in the child */
341 atomic_ptr_store(&g_options, NULL);
342
343 /* Don't reset g_init_pid - we'll detect the fork in options_state_init */
344}

352__attribute__((constructor)) static void register_fork_handlers_constructor(void) {
353 // Register the child handler that will be called after fork() in the child process
354 pthread_atfork(NULL, NULL, options_rcu_atfork_child);
355}

362asciichat_error_t options_state_init(void) {
363 // Detect fork: after fork(), child process must reinitialize
364 pid_t current_pid = platform_get_pid();
365 if ((lifecycle_is_initialized(&g_options_lifecycle) || g_init_pid != -1) && g_init_pid != current_pid) {
366 log_debug("Detected fork in options_state_init (parent PID %d, child PID %d) - resetting lifecycle", g_init_pid,
367 current_pid);
368 /* Reset lifecycle to allow reinitialization after fork */
369 lifecycle_shutdown(&g_options_lifecycle);
370 g_init_pid = current_pid;
371 }
372
373 // Check if already initialized in this process using lifecycle
374 if (lifecycle_is_initialized(&g_options_lifecycle) && g_init_pid == current_pid) {
375 log_warn("Options state already initialized");
376 return ASCIICHAT_OK;
377 }
378
379 /* Use lifecycle to serialize initialization */
380 if (!lifecycle_init(&g_options_lifecycle, "options")) {
381 /* Another thread is initializing or already initialized */
382 if (lifecycle_is_initialized(&g_options_lifecycle)) {
383 return ASCIICHAT_OK;
384 }
385 return SET_ERRNO(ERROR_INVALID_STATE, "Failed to acquire options initialization lock");
386 }
387
388 // Initialize write mutex
389 if (mutex_init(&g_options_write_mutex, "options_write") != 0) {
390 lifecycle_shutdown(&g_options_lifecycle);
391 return SET_ERRNO(ERROR_THREAD, "Failed to initialize options write mutex");
392 }
393
394 // Initialize deferred free mutex
395 if (mutex_init(&g_deferred_free_mutex, "deferred_free") != 0) {
396 mutex_destroy(&g_options_write_mutex);
397 lifecycle_shutdown(&g_options_lifecycle);
398 return SET_ERRNO(ERROR_THREAD, "Failed to initialize deferred free mutex");
399 }
400
401 // Allocate initial options struct (will be populated later by options_state_populate_from_globals)
402 options_t *initial_opts = SAFE_MALLOC(sizeof(options_t), options_t *);
403 if (!initial_opts) {
404 mutex_destroy(&g_options_write_mutex);
405 mutex_destroy(&g_deferred_free_mutex);
406 lifecycle_shutdown(&g_options_lifecycle);
407 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate initial options struct");
408 }
409
410 // Initialize all defaults using options_t_new()
411 *initial_opts = options_t_new();
412
413 // Publish initial struct (release semantics - make all fields visible to readers)
414 atomic_ptr_store(&g_options, initial_opts);
415
416 g_init_pid = current_pid; // Record PID to detect fork later
417 log_dev("Options state initialized with RCU pattern (PID %d)", current_pid);
418
419 return ASCIICHAT_OK;
420}

422asciichat_error_t options_state_set(const options_t *opts) {
423 if (!opts) {
424 return SET_ERRNO(ERROR_INVALID_PARAM, "opts is NULL");
425 }
426
427 // Check initialization state using lifecycle
428 if (!lifecycle_is_initialized(&g_options_lifecycle)) {
429 return SET_ERRNO(ERROR_INVALID_STATE, "Options state not initialized (call options_state_init first)");
430 }
431
432 // Get current struct (should be the initial zero-initialized one during startup)
433 options_t *current = atomic_ptr_load(&g_options);
434
435 // Copy provided options into current struct
436 // Note: No need for RCU update here since this is called during initialization
437 // before any reader threads exist
438 memcpy(current, opts, sizeof(options_t));
439
440 log_debug("Options state set from parsed struct");
441 return ASCIICHAT_OK;
442}

444void options_state_destroy(void) {
445 // Check if initialized using lifecycle
446 if (!lifecycle_is_initialized(&g_options_lifecycle)) {
447 return;
448 }
449
450 /* Shutdown lifecycle to mark as no longer initialized */
451 lifecycle_shutdown(&g_options_lifecycle);
452
453 // Get current options pointer
454 options_t *current = atomic_ptr_load(&g_options);
455
456 // Clear the atomic pointer (prevent further reads)
457 atomic_ptr_store(&g_options, NULL);
458
459 // Free current struct
460 SAFE_FREE(current);
461
462 // Free all deferred structs
463 deferred_free_all();
464
465 // Cleanup schema (frees all dynamically allocated config strings)
466 config_schema_destroy();
467
468 // Destroy dynamically created mutexes
469 mutex_destroy(&g_options_write_mutex);
470 mutex_destroy(&g_deferred_free_mutex);
471
472 log_debug("Options state shutdown complete");
473}

479const options_t *options_get(void) {
480 // Lock-free read with acquire semantics
481 // Guarantees we see all writes made before the pointer was published
482 options_t *current = atomic_ptr_load(&g_options);
483
484 // If options not yet published or after destruction, return safe static default.
485 // This is a critical fallback that:
486 // 1. Allows early startup code (before options_state_init) to work safely
487 // 2. Allows atexit handlers and cleanup code to call GET_OPTION() safely
488 // 3. Never crashes due to NULL pointer access
489 // 4. Provides sensible defaults for all option fields
490 //
491 // The static default options are:
492 // - Allocated once at startup (static storage)
493 // - Never freed (outlives all dynamically allocated options)
494 // - Thread-safe to read (immutable const data)
495 // - Complete with all OPT_*_DEFAULT values matching options_t_new()
496 if (!current) {
497 return (const options_t *)&g_default_options;
498 }
499
500 return current;
501}

503static asciichat_error_t options_update(void (*updater)(options_t *, void *), void *context) {
504 if (!updater) {
505 return SET_ERRNO(ERROR_INVALID_PARAM, "updater function is NULL");
506 }
507
508 if (!lifecycle_is_initialized(&g_options_lifecycle)) {
509 return SET_ERRNO(ERROR_INVALID_STATE, "Options state not initialized");
510 }
511
512 // Serialize writers with mutex
513 mutex_lock(&g_options_write_mutex);
514
515 // 1. Load current options (acquire semantics)
516 options_t *old_opts = atomic_ptr_load(&g_options);
517
518 // 2. Allocate new options struct
519 options_t *new_opts = SAFE_MALLOC(sizeof(options_t), options_t *);
520 if (!new_opts) {
521 mutex_unlock(&g_options_write_mutex);
522 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate new options struct");
523 }
524
525 // 3. Copy current values to new struct
526 memcpy(new_opts, old_opts, sizeof(options_t));
527
528 // 4. Call user updater to modify the new struct
529 updater(new_opts, context);
530
531 // 5. Atomically swap global pointer (release semantics)
532 // This makes the new struct visible to all readers
533 atomic_ptr_store(&g_options, new_opts);
534
535 // 6. Add old struct to deferred free list
536 deferred_free_add(old_opts);
537
538 mutex_unlock(&g_options_write_mutex);
539
540 log_debug("Options updated via RCU (old=%p, new=%p)", (void *)old_opts, (void *)new_opts);
541 return ASCIICHAT_OK;
542}
