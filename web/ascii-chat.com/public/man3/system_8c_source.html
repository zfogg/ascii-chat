136static bool is_executable_file(const char *path) {
137#ifdef _WIN32
138 // On Windows, check if file exists and is readable
139 // Windows doesn't have execute permission bits like Unix
140 DWORD attrs = GetFileAttributesA(path);
141 if (attrs == INVALID_FILE_ATTRIBUTES) {
142 return false;
143 }
144 // Must be a regular file (not a directory)
145 if (attrs & FILE_ATTRIBUTE_DIRECTORY) {
146 return false;
147 }
148 // Try to open for reading to verify access
149 HANDLE h = CreateFileA(path, GENERIC_READ, FILE_SHARE_READ, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
150 if (h == INVALID_HANDLE_VALUE) {
151 return false;
152 }
153 CloseHandle(h);
154 return true;
155#else
156 // On Unix, use access() to check if file exists and is executable
157 return (access(path, X_OK) == 0);
158#endif
159}

169static bool convert_unix_path_to_windows(const char *unix_path, char *win_path, size_t win_path_size) {
170 if (!unix_path || !win_path || win_path_size < 4) {
171 return false;
172 }
173
174 // Check for Unix-style path like /c/foo or /d/bar (Git Bash format)
175 if (unix_path[0] == '/' && unix_path[1] != '\0' && unix_path[2] == '/') {
176 // Convert /x/... to X:\...
177 char drive_letter = (char)toupper((unsigned char)unix_path[1]);
178 safe_snprintf(win_path, win_path_size, "%c:%s", drive_letter, unix_path + 2);
179 // Convert remaining forward slashes to backslashes
180 for (char *p = win_path; *p; p++) {
181 if (*p == '/')
182 *p = '\\';
183 }
184 return true;
185 }
186
187 // Already Windows-style or relative path
188 SAFE_STRNCPY(win_path, unix_path, win_path_size);
189 return false;
190}
