47pcre2_singleton_t *asciichat_pcre2_singleton_compile(const char *pattern, uint32_t flags) {
48 if (!pattern) {
49 log_error("PCRE2 singleton: pattern is NULL");
50 return NULL;
51 }
52
53 /* Allocate singleton structure */
54 pcre2_singleton_t *singleton = SAFE_MALLOC(sizeof(pcre2_singleton_t), pcre2_singleton_t *);
55 if (!singleton) {
56 log_error("PCRE2 singleton: failed to allocate singleton structure");
57 return NULL;
58 }
59
60 /* Copy pattern string to avoid stack-use-after-return bugs */
61 size_t pattern_len = strlen(pattern);
62 singleton->pattern = SAFE_MALLOC(pattern_len + 1, char *);
63 if (!singleton->pattern) {
64 log_error("PCRE2 singleton: failed to allocate pattern buffer");
65 SAFE_FREE(singleton);
66 return NULL;
67 }
68 memcpy(singleton->pattern, pattern, pattern_len + 1);
69
70 /* Initialize fields */
71 singleton->lc = (lifecycle_t)LIFECYCLE_INIT;
72 atomic_store_u64(&singleton->code, 0);
73 singleton->jit_stack = NULL;
74 singleton->flags = flags;
75
76 /* Register singleton in global list for automatic cleanup */
77 singleton->next = g_singleton_registry;
78 g_singleton_registry = singleton;
79
80 /* Mark registry as initialized (one-time per process) */
81 lifecycle_init(&g_registry_lc, "pcre2_registry");
82
83 return singleton;
84}

98pcre2_code *asciichat_pcre2_singleton_get_code(pcre2_singleton_t *singleton) {
99 if (!singleton) {
100 return NULL;
101 }
102
103 /* Fast path: check if already compiled (lock-free atomic load, no mutex) */
104 pcre2_code *code = (pcre2_code *)(uintptr_t)atomic_load_u64(&singleton->code);
105 if (code != NULL) {
106 return code;
107 }
108
109 /* Check if compilation already attempted via lifecycle state */
110 if (lifecycle_is_initialized(&singleton->lc)) {
111 /* Already initialized but code is NULL - compilation failed previously */
112 return NULL;
113 }
114
115 /* Try to win the compilation race using lifecycle_t */
116 if (!lifecycle_init(&singleton->lc, "pcre2_pattern")) {
117 /* Lost race - someone else is compiling or already compiled. Spin for their result. */
118 while ((pcre2_code *)(uintptr_t)atomic_load_u64(&singleton->code) == NULL && lifecycle_is_initialized(&singleton->lc)) {
119 /* Spin-wait for compiler to finish */
120 }
121 return (pcre2_code *)(uintptr_t)atomic_load_u64(&singleton->code);
122 }
123
124 /* Slow path: we won the race, need to compile. */
125 int errornumber;
126 PCRE2_SIZE erroroffset;
127
128 code = pcre2_compile((PCRE2_SPTR8)singleton->pattern, PCRE2_ZERO_TERMINATED, singleton->flags, &errornumber,
129 &erroroffset, NULL);
130
131 if (!code) {
132 PCRE2_UCHAR error_buf[256];
133 pcre2_get_error_message(errornumber, error_buf, sizeof(error_buf));
134 log_warn("Failed to compile PCRE2 regex at offset %zu: %s", erroroffset, (const char *)error_buf);
135 /* Mark compilation as attempted (failed) by moving to INITIALIZED state */
136 lifecycle_init_commit(&singleton->lc);
137 return NULL;
138 }
139
140 /* Attempt JIT compilation (non-fatal if fails) */
141 int jit_rc = pcre2_jit_compile(code, PCRE2_JIT_COMPLETE);
142 if (jit_rc < 0) {
143 log_debug("PCRE2 JIT compilation not available (code %d), using interpreted mode", jit_rc);
144 } else {
145 /* Create JIT stack for pattern execution */
146 singleton->jit_stack = pcre2_jit_stack_create(32 * 1024, 512 * 1024, NULL);
147 if (!singleton->jit_stack) {
148 log_warn("Failed to create JIT stack for PCRE2 regex");
149 }
150 }
151
152 /* Store compiled code (atomic, so subsequent calls see it immediately) */
153 atomic_store_u64(&singleton->code, (uint64_t)(uintptr_t)code);
154 /* Mark compilation as completed by moving to INITIALIZED state */
155 lifecycle_init_commit(&singleton->lc);
156
157 return code;
158}

210void asciichat_pcre2_cleanup_all(void) {
211 if (!lifecycle_shutdown(&g_registry_lc)) {
212 return; /* No singletons were ever created or already cleaned up */
213 }
214
215 /* Check if already cleaned up */
216 if (g_singleton_registry == NULL) {
217 return;
218 }
219
220 pcre2_singleton_t *current = g_singleton_registry;
221
222 /* Clear registry first to prevent re-entry */
223 g_singleton_registry = NULL;
224
225 /* Now free all singletons */
226 while (current) {
227 pcre2_singleton_t *next = current->next;
228
229 /* Free compiled code */
230 pcre2_code *code = (pcre2_code *)(uintptr_t)atomic_load_u64(&current->code);
231 if (code) {
232 pcre2_code_free(code);
233 atomic_store_u64(&current->code, 0);
234 }
235
236 /* Free JIT stack */
237 if (current->jit_stack) {
238 pcre2_jit_stack_free(current->jit_stack);
239 current->jit_stack = NULL;
240 }
241
242 /* Free pattern string */
243 SAFE_FREE(current->pattern);
244
245 /* Free singleton structure */
246 SAFE_FREE(current);
247
248 current = next;
249 }
