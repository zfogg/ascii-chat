75static void tls_mutex_stack_destructor(void *arg) {
76 if (!arg) {
77 return;
78 }
79
80 // Check if this stack was already freed by cleanup()
81 // This happens if: (1) thread is in registry AND (2) registry entry matches this stack
82 bool already_freed_by_cleanup = false;
83 thread_id_t current_thread = asciichat_thread_self();
84 int count = atomic_load_int(&g_thread_registry_count);
85 for (int i = 0; i < count; i++) {
86 if (asciichat_thread_equal(g_thread_registry[i].thread_id, current_thread)) {
87 // Found this thread's entry in registry
88 if (g_thread_registry[i].stack == arg) {
89 // Registry entry matches this stack - cleanup() already freed it
90 already_freed_by_cleanup = true;
91 // Clear the entry (optional, cleanup might have done this already)
92 g_thread_registry[i].stack = NULL;
93 }
94 break;
95 }
96 }
97
98 // Free if cleanup() didn't already do it
99 // This includes "orphaned" stacks that were allocated after cleanup() ran
100 // but before the thread's destructor was called
101 if (!already_freed_by_cleanup) {
102 // Use raw free() - stacks are allocated with raw malloc(), not SAFE_CALLOC()
103 // This avoids recursive mutex allocation during destructor execution
104 free(arg);
105 }
106}

129static thread_lock_stack_t *get_thread_local_stack(void) {
130 // Skip allocation during shutdown (prevents leaks from threads killed mid-shutdown)
131 if (atomic_load_bool(&g_shutting_down)) {
132 return NULL;
133 }
134
135 // Ensure TLS key is initialized
136 ensure_tls_initialized();
137
138 // Get current thread's stack from TLS
139 thread_lock_stack_t *stack = (thread_lock_stack_t *)ascii_tls_get(g_tls_mutex_stack);
140
141 if (stack == NULL) {
142 // Allocate stack on heap for this thread
143 // Use raw malloc() to avoid recursive mutex locking in debug subsystem
144 stack = (thread_lock_stack_t *)malloc(sizeof(thread_lock_stack_t));
145 if (stack) {
146 memset(stack, 0, sizeof(thread_lock_stack_t));
147 // Store in TLS (destructor will free it when thread exits)
148 ascii_tls_set(g_tls_mutex_stack, stack);
149 register_thread_if_needed();
150 }
151 }
152 return stack;
153}

163static void register_thread_if_needed(void) {
164 static __thread bool registered = false;
165
166 // Once registered, skip immediately
167 if (registered) {
168 return;
169 }
170
171 thread_lock_stack_t *local_stack = get_thread_local_stack();
172 if (!local_stack) {
173 return;
174 }
175
176 thread_id_t current_thread = asciichat_thread_self();
177
178 // Check if thread is already in registry (without lock)
179 int current_count = atomic_load_int(&g_thread_registry_count);
180 for (int i = 0; i < current_count; i++) {
181 if (asciichat_thread_equal(g_thread_registry[i].thread_id, current_thread)) {
182 registered = true;
183 return; // Already registered
184 }
185 }
186
187 // Try to claim a slot in the registry atomically
188 int slot;
189 while (1) {
190 int old_count = atomic_load_int(&g_thread_registry_count);
191 if (old_count >= MAX_THREADS) {
192 registered = true; // Registry is full, give up
193 return;
194 }
195
196 // Try to atomically increment the count and claim a slot
197 if (atomic_cas_int(&g_thread_registry_count, &old_count, old_count + 1)) {
198 slot = old_count;
199 break; // Successfully claimed slot
200 }
201 // If compare-exchange failed, loop and try again
202 }
203
204 // Write thread data to claimed slot (release semantics for visibility)
205 g_thread_registry[slot].thread_id = current_thread;
206 g_thread_registry[slot].stack = local_stack;
207
208 // Memory barrier to ensure writes are visible to other threads
209 // Memory ordering is implicit in atomic operations
210
211 registered = true;
212}

218void mutex_stack_push_pending(uintptr_t mutex_key, const char *mutex_name) {
219 thread_lock_stack_t *stack = get_thread_local_stack();
220 if (!stack || stack->depth >= MUTEX_STACK_MAX_DEPTH) {
221 return;
222 }
223
224 // Register this thread in the global registry on first mutex use
225 register_thread_if_needed();
226
227 stack->stack[stack->depth].mutex_key = mutex_key;
228 stack->stack[stack->depth].mutex_name = mutex_name;
229 stack->stack[stack->depth].state = MUTEX_STACK_STATE_PENDING;
230 stack->stack[stack->depth].timestamp_ns = time_get_ns();
231 stack->depth++;
232}

265int mutex_stack_get_current(mutex_stack_entry_t *out_entries, int max_entries) {
266 thread_lock_stack_t *stack = get_thread_local_stack();
267 if (!stack || !out_entries) {
268 return 0;

270

280int mutex_stack_get_all_threads(mutex_stack_entry_t ***out_stacks, int **out_stack_counts, int *out_thread_count) {
281
282 if (!out_stacks || !out_stack_counts || !out_thread_count) {
283 return -1;
284 }
285
286 // Read the thread count with acquire semantics (lock-free, no mutex needed)
287 // Use memory_order_seq_cst to ensure we get a consistent snapshot
288 int thread_count = atomic_load_int(&g_thread_registry_count);
289
290 // Allocate arrays for threads in the registry
291 // Note: SAFE_MALLOC takes bytes as first parameter, not count
292 *out_stacks = SAFE_MALLOC(thread_count * sizeof(mutex_stack_entry_t *), mutex_stack_entry_t **);
293 *out_stack_counts = SAFE_MALLOC(thread_count * sizeof(int), int *);
294
295 if (!*out_stacks || !*out_stack_counts) {
296 SAFE_FREE(*out_stacks);
297 SAFE_FREE(*out_stack_counts);
298 return -1;
299 }
300
301 // Copy each thread's stack from the registry
302 // Note: the thread count can change concurrently, so we re-read it in the loop
303 // to avoid accessing out-of-bounds memory if threads exit during iteration
304 int actual_count = 0;
305 for (int i = 0; i < thread_count; i++) {
306 // Re-check thread count in case registry shrank
307 int current_registry_count = atomic_load_int(&g_thread_registry_count);
308 if (i >= current_registry_count) {
309 break;
310 }
311
312 thread_lock_stack_t *src = g_thread_registry[i].stack;
313 if (!src) {
314 // Stack not yet initialized for this thread, skip it
315 (*out_stack_counts)[i] = 0;
316 (*out_stacks)[i] = NULL;
317 continue;
318 }
319
320 // Defensively get depth - it could change or be freed by another thread
321 int depth = 0;
322 if (src && src->depth >= 0 && src->depth < MUTEX_STACK_MAX_DEPTH) {
323 depth = src->depth;
324 }
325
326 (*out_stack_counts)[i] = depth;
327
328 if (depth > 0) {
329 (*out_stacks)[i] = SAFE_MALLOC(depth * sizeof(mutex_stack_entry_t), mutex_stack_entry_t *);
330 if ((*out_stacks)[i]) {
331 memcpy((*out_stacks)[i], src->stack, depth * sizeof(mutex_stack_entry_t));
332 actual_count++;
333 }
334 } else {
335 (*out_stacks)[i] = NULL;
336 }
337 }
338
339 *out_thread_count = actual_count;
340 return 0;
341}

363static bool thread_holds_mutex(thread_lock_stack_t *stack, uintptr_t mutex_key) {
364 if (!stack) {
365 return false;
366 }
367 for (int i = 0; i < stack->depth; i++) {
368 if (stack->stack[i].mutex_key == mutex_key && stack->stack[i].state == MUTEX_STACK_STATE_LOCKED) {
369 return true;
370 }
371 }
372 return false;
373}

378static uintptr_t thread_waiting_for_mutex(thread_lock_stack_t *stack) {
379 if (!stack) {
380 return 0;
381 }
382 if (stack->depth > 0) {
383 int top = stack->depth - 1;
384 if (stack->stack[top].state == MUTEX_STACK_STATE_PENDING) {
385 return stack->stack[top].mutex_key;
386 }
387 }
388 return 0;
389}

409#define MAX_CYCLE_LEN 64
410static int detect_cycle_dfs(int thread_count, int start_thread, int *cycle_path, int *cycle_len) {
411 int visited[MAX_THREADS];
412 int path[MAX_CYCLE_LEN];
413 int path_len = 0;
414
415 // Initialize visited array
416 for (int i = 0; i < MAX_THREADS; i++) {
417 visited[i] = -1; // -1 = not visited, >= 0 = index in path
418 }
419
420 // DFS starting from start_thread
421 int current = start_thread;
422 while (path_len < MAX_CYCLE_LEN && path_len < thread_count) {
423 if (current < 0 || current >= thread_count) {
424 break; // Invalid thread
425 }
426
427 // Check if current is already in path (cycle found!)
428 for (int i = 0; i < path_len; i++) {
429 if (path[i] == current) {
430 // Found a cycle! Extract the cycle portion
431 *cycle_len = path_len - i;
432 for (int j = 0; j < *cycle_len; j++) {
433 cycle_path[j] = path[i + j];
434 }
435 return i; // Return where the cycle starts
436 }
437 }
438
439 // Add current to path
440 path[path_len++] = current;
441
442 // Find next thread in the waits-for graph
443 thread_lock_stack_t *stack = g_thread_registry[current].stack;
444 uintptr_t waiting_for = thread_waiting_for_mutex(stack);
445
446 if (waiting_for == 0) {
447 break; // No waiting, path ends
448 }
449
450 // Find who holds the mutex we're waiting for
451 current = find_thread_holding_mutex(thread_count, waiting_for);
452 }
453
454 *cycle_len = 0;
455 return -1; // No cycle found
456}

461static int compare_uintptr(const void *a, const void *b) {
462 uintptr_t ua = *(const uintptr_t *)a;
463 uintptr_t ub = *(const uintptr_t *)b;
464 if (ua < ub)
465 return -1;
466 if (ua > ub)
467 return 1;
468 return 0;

475static bool deadlock_mutexes_changed(const uintptr_t *current_mutexes, int count) {
476 if (count != g_last_deadlock.count)
477 return true;
478
479 // Sort both arrays for order-independent comparison
480 uintptr_t current_sorted[MAX_CYCLE_MUTEXES];
481 uintptr_t last_sorted[MAX_CYCLE_MUTEXES];
482
483 for (int i = 0; i < count; i++) {
484 current_sorted[i] = current_mutexes[i];
485 last_sorted[i] = g_last_deadlock.mutexes[i];
486 }
487
488 qsort(current_sorted, count, sizeof(uintptr_t), compare_uintptr);
489 qsort(last_sorted, count, sizeof(uintptr_t), compare_uintptr);
490
491 for (int i = 0; i < count; i++) {
492 if (current_sorted[i] != last_sorted[i])
493 return true;
494 }
495 return false;
496}

520void mutex_stack_detect_deadlocks(void) {
521 mutex_stack_entry_t **all_stacks = NULL;
522 int *stack_counts = NULL;
523 int thread_count = 0;
524
525 if (mutex_stack_get_all_threads(&all_stacks, &stack_counts, &thread_count) != 0) {
526 return;
527 }
528
529 // Check each thread for deadlock conditions
530 for (int i = 0; i < thread_count; i++) {
531 thread_lock_stack_t *stack_a = g_thread_registry[i].stack;
532 uintptr_t waiting_for = thread_waiting_for_mutex(stack_a);
533
534 if (waiting_for == 0)
535 continue; // Thread not waiting
536
537 // Same-thread deadlock: thread trying to acquire a mutex it already holds
538 if (thread_holds_mutex(stack_a, waiting_for)) {
539 log_error("%s", colored_string(LOG_COLOR_ERROR, "╔═══════════════════════════════════════════════════════════╗"));
540 log_error("%s", colored_string(LOG_COLOR_ERROR, "║ ⚠️ DEADLOCK DETECTED: Same-thread Recursive Lock ⚠️ ║"));
541 log_error("%s", colored_string(LOG_COLOR_ERROR, "╚═══════════════════════════════════════════════════════════╝"));
542 log_error(" Thread Address: 0x%lx", (unsigned long)g_thread_registry[i].thread_id);
543 log_error(" Mutex: 0x%lx", waiting_for);
544 log_error(" Issue: Thread attempts recursive lock on non-recursive mutex");
545 continue;
546 }
547
548 // Multi-thread circular wait: use DFS to detect cycles
549 int cycle_path[MAX_CYCLE_LEN];
550 int cycle_len = 0;
551 int cycle_start = detect_cycle_dfs(thread_count, i, cycle_path, &cycle_len);
552
553 if (cycle_start >= 0 && cycle_len > 1) {
554 // Collect mutexes involved in this deadlock
555 uintptr_t cycle_mutexes[MAX_CYCLE_MUTEXES];
556 int mutex_count = 0;
557 for (int k = 0; k < cycle_len && mutex_count < MAX_CYCLE_MUTEXES; k++) {
558 int thread_idx = cycle_path[k];
559 thread_lock_stack_t *stack = g_thread_registry[thread_idx].stack;
560 uintptr_t waiting_for = thread_waiting_for_mutex(stack);
561 if (waiting_for != 0) {
562 cycle_mutexes[mutex_count++] = waiting_for;
563 }
564 }
565
566 // Check if mutexes are different from last deadlock
567 bool is_new_deadlock = deadlock_mutexes_changed(cycle_mutexes, mutex_count);
568 if (is_new_deadlock) {
569 update_deadlock_state(cycle_mutexes, mutex_count);
570 }
571
572 // Cycle detected! Build complete message in one string
573 char cycle_msg[4096];
574 int msg_len = 0;
575
576 // Leading newline and header box
577 msg_len += snprintf(cycle_msg + msg_len, sizeof(cycle_msg) - msg_len, "\n%s\n",
578 colored_string(LOG_COLOR_ERROR, "╔═════════════════════════════════╗"));
579 msg_len += snprintf(cycle_msg + msg_len, sizeof(cycle_msg) - msg_len, "%s\n",
580 colored_string(LOG_COLOR_ERROR, "║ DEADLOCK: Circular Wait Cycle ║"));
581 msg_len += snprintf(cycle_msg + msg_len, sizeof(cycle_msg) - msg_len, "%s\n",
582 colored_string(LOG_COLOR_ERROR, "╚═════════════════════════════════╝"));
583
584 // Print each thread in the cycle
585 for (int k = 0; k < cycle_len; k++) {
586 int thread_idx = cycle_path[k];
587 int next_thread_idx = cycle_path[(k + 1) % cycle_len];
588
589 thread_lock_stack_t *current_stack = g_thread_registry[thread_idx].stack;
590 uintptr_t current_waiting = thread_waiting_for_mutex(current_stack);
591
592 char thread_name[256], mutex_name[256], held_by_name[256];
593 NAMED_GET_BY_PTR(g_thread_registry[thread_idx].thread_id, thread_name, sizeof(thread_name));
594 NAMED_GET_BY_PTR(current_waiting, mutex_name, sizeof(mutex_name));
595 NAMED_GET_BY_PTR(g_thread_registry[next_thread_idx].thread_id, held_by_name, sizeof(held_by_name));
596
597 msg_len +=
598 snprintf(cycle_msg + msg_len, sizeof(cycle_msg) - msg_len, " T%d: %s waits for %s (held by %s)%s",
599 k + 1, thread_name, mutex_name, held_by_name, k < cycle_len - 1 ? "\n" : "");
600 }
601
602 // Log repeated deadlock detections (skip first call, throttle subsequent ones)
603 if (!is_new_deadlock) {
604 log_error_every(1000000, "%s", cycle_msg); // 1000000 µs = 1 second
605 }
606 }
607 }
608
609 mutex_stack_free_all_threads(all_stacks, stack_counts, thread_count);
610}

616#define COND_DEADLOCK_THRESHOLD_NS (5ULL * 1000000000ULL) // 5 seconds
617
624static void cond_deadlock_check_callback(uintptr_t key, const char *name, void *user_data) {
625 (void)user_data; // Unused
626
627 const char *type = named_get_type(key);
628 if (!type || strcmp(type, "cond") != 0) {
629 return;
630 }
631
632 const cond_t *cond = (const cond_t *)key;
633 if (atomic_load_u64(&cond->waiting_count) == 0) {
634 return; // No threads waiting, nothing to check
635 }
636
637 // Skip deadlock checks for thread pool work queues - it's normal for worker threads
638 // to wait idly when there's no work to process
639 if (name && strstr(name, "task_available") != NULL) {
640 return;
641 }
642
643 uint64_t now = time_get_ns();
644 uint64_t stuck_ns = now - cond->last_wait_time_ns;
645 bool no_signal_since_wait = (cond->last_signal_time_ns == 0 || cond->last_signal_time_ns < cond->last_wait_time_ns);
646
647 if (stuck_ns < COND_DEADLOCK_THRESHOLD_NS || !no_signal_since_wait) {
648 return; // Not stuck yet or was signaled recently
649 }
650
651 // Condition variable appears to be stuck - log detailed diagnostic info
652 char stuck_str[64];
653 time_pretty(stuck_ns, -1, stuck_str, sizeof(stuck_str));
654
655 char cond_buf[1024] = {0};
656 int cond_written =
657 safe_snprintf(cond_buf, sizeof(cond_buf),
658 "Stuck cond '%s': %lu thread(s) waiting %s with no signal (most recent waiter: 0x%lx)\n", name,
659 cond->waiting_count, stuck_str, (unsigned long)cond->last_waiting_key);
660
661 cond_written +=
662 safe_snprintf(cond_buf + cond_written, sizeof(cond_buf) - cond_written, " wait entered at %s:%d %s()\n",
663 extract_project_relative_path(cond->last_wait_file), cond->last_wait_line, cond->last_wait_func);
664
665 if (cond_written >= 0 && cond->last_wait_mutex) {
666 safe_snprintf(cond_buf + cond_written, sizeof(cond_buf) - cond_written,
667 " associated mutex: %p (cannot safely inspect without lock ownership)",
668 (void *)cond->last_wait_mutex);
669 }
670
671 log_warn_every(500 * NS_PER_MS_INT, "%s", cond_buf);
672}

704void mutex_stack_cleanup_current_thread(void) {
705 // Explicitly free the current thread's TLS stack
706 // This is used to prevent leaks when TLS destructors might not run reliably
707 // (e.g., debug threads exiting before mutex_stack_cleanup() deletes the TLS key)
708
709 if (!atomic_load_bool(&g_tls_initialized)) {
710 return; // TLS not initialized, nothing to clean up
711 }
712
713 thread_lock_stack_t *stack = (thread_lock_stack_t *)ascii_tls_get(g_tls_mutex_stack);
714 if (!stack) {
715 return; // No stack allocated for this thread
716 }
717
718 // Clear from TLS
719 ascii_tls_set(g_tls_mutex_stack, NULL);
720
721 // Update registry if this thread is registered
722 thread_id_t current_thread = asciichat_thread_self();
723 int count = atomic_load_int(&g_thread_registry_count);
724 for (int i = 0; i < count; i++) {
725 if (asciichat_thread_equal(g_thread_registry[i].thread_id, current_thread) && g_thread_registry[i].stack == stack) {
726 g_thread_registry[i].stack = NULL; // Mark as freed in registry
727 break;
728 }
729 }
730
731 // Free the stack (raw free to match raw malloc above)
732 free(stack);
733}

735void mutex_stack_cleanup(void) {
736 // Signal shutdown to prevent new allocations from threads still running
737 atomic_store_bool(&g_shutting_down, true);
738
