86int mutex_format_state(const mutex_t *mutex, char *buffer, size_t size) {
87 if (!mutex || !buffer || size == 0) return 0;
88
89 int offset = 0;
90 uint64_t now_ns = time_get_ns();
91
92 // Format timing information
93 char lock_str[64] = "";
94 char unlock_str[64] = "";
95 char held_str[64] = "";
96 char count_str[256] = "";
97
98 if (mutex->last_lock_time_ns > 0 && mutex->last_lock_time_ns <= now_ns) {
99 char elapsed_str[64];
100 time_pretty(now_ns - mutex->last_lock_time_ns, -1, elapsed_str, sizeof(elapsed_str));
101 snprintf(lock_str, sizeof(lock_str), "lock=%s", elapsed_str);
102 }
103
104 if (mutex->last_unlock_time_ns > 0 && mutex->last_unlock_time_ns <= now_ns) {
105 char elapsed_str[64];
106 time_pretty(now_ns - mutex->last_unlock_time_ns, -1, elapsed_str, sizeof(elapsed_str));
107 snprintf(unlock_str, sizeof(unlock_str), "unlock=%s", elapsed_str);
108 }
109
110 if (mutex->currently_held_by_key != 0) {
111 snprintf(held_str, sizeof(held_str), "[LOCKED_BY=thread.%lu]", (unsigned long)mutex->currently_held_by_key);
112 }
113
114 if (mutex->lock_count > 0 || mutex->unlock_count > 0 || mutex->trylock_count > 0) {
115 snprintf(count_str, sizeof(count_str), "[ops: lock=%lu unlock=%lu trylock=%lu/%lu]",
116 mutex->lock_count, mutex->unlock_count, mutex->trylock_success_count, mutex->trylock_count);
117 }
118
119 offset += snprintf(buffer + offset, size - offset, "%s %s %s %s", lock_str, unlock_str, held_str, count_str);
120 return offset;
121}
