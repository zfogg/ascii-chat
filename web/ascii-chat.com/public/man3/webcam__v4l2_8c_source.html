80static void *webcam_camera_thread_func(void *arg) {
81 webcam_context_t *ctx = (webcam_context_t *)arg;
82
83 log_debug("Camera background thread started");
84
85 while (lifecycle_is_initialized(&ctx->async_lifecycle)) {
86 // Read frame (blocking on camera)
87 image_t *frame = webcam_read_context(ctx);
88
89 if (frame) {
90 // Copy the frame data (webcam_read_context returns ptr to module-level static buffer)
91 image_t *frame_copy = image_new_copy(frame);
92 if (frame_copy) {
93 // Atomic swap: replace latest_frame with new frame
94 image_t *old_frame = (image_t *)atomic_ptr_exchange(&ctx->latest_frame, frame_copy);
95 if (old_frame) {
96 image_destroy(old_frame);
97 }
98 }
99 }
100 }
101
102 log_debug("Camera background thread stopped");
103 return NULL;
104}

145static int webcam_v4l2_init_mjpeg_decoder(webcam_context_t *ctx) {
146 // Find JPEG codec
147 const AVCodec *codec = avcodec_find_decoder(AV_CODEC_ID_MJPEG);
148 if (!codec) {
149 log_error("MJPEG codec not found in FFmpeg");
150 return -1;
151 }
152
153 // Allocate codec context
154 ctx->mjpeg_codec_ctx = avcodec_alloc_context3(codec);
155 if (!ctx->mjpeg_codec_ctx) {
156 log_error("Failed to allocate MJPEG codec context");
157 return -1;
158 }
159
160 // Set expected frame dimensions (V4L2 tells us these)
161 ctx->mjpeg_codec_ctx->width = ctx->width;
162 ctx->mjpeg_codec_ctx->height = ctx->height;
163
164 // Open codec
165 if (avcodec_open2(ctx->mjpeg_codec_ctx, codec, NULL) < 0) {
166 log_error("Failed to open MJPEG codec");
167 avcodec_free_context(&ctx->mjpeg_codec_ctx);
168 return -1;
169 }
170
171 // Allocate frame for decoded output
172 ctx->mjpeg_decoded_frame = av_frame_alloc();
173 if (!ctx->mjpeg_decoded_frame) {
174 log_error("Failed to allocate MJPEG decoded frame");
175 avcodec_free_context(&ctx->mjpeg_codec_ctx);
176 return -1;
177 }
178
179 // Initialize swscale for converting decoded MJPEG (typically YUV420P) to RGB24
180 if (webcam_v4l2_init_swscale(ctx, AV_PIX_FMT_YUV420P) != 0) {
181 log_error("Failed to initialize swscale for MJPEG decoded frames");
182 av_frame_free(&ctx->mjpeg_decoded_frame);
183 avcodec_free_context(&ctx->mjpeg_codec_ctx);
184 return -1;
185 }
186
187 return 0;
188}

208static int webcam_v4l2_set_format(webcam_context_t *ctx, int width, int height) {
209 struct v4l2_format fmt = {0};
210 fmt.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
211 fmt.fmt.pix.width = width;
212 fmt.fmt.pix.height = height;
213 fmt.fmt.pix.field = V4L2_FIELD_ANY;
214
215 // Try RGB24 first (ideal - no conversion needed)
216 fmt.fmt.pix.pixelformat = V4L2_PIX_FMT_RGB24;
217 if (ioctl(ctx->fd, VIDIOC_S_FMT, &fmt) == 0 && fmt.fmt.pix.pixelformat == V4L2_PIX_FMT_RGB24) {
218 ctx->pixelformat = V4L2_PIX_FMT_RGB24;
219 ctx->width = fmt.fmt.pix.width;
220 ctx->height = fmt.fmt.pix.height;
221 ctx->sws_ctx = NULL;
222 log_debug("V4L2 format set to RGB24 %dx%d", ctx->width, ctx->height);
223 return 0;
224 }
225
226 // Try NV12 (Raspberry Pi cameras, modern USB cameras)
227 fmt.fmt.pix.pixelformat = V4L2_PIX_FMT_NV12;
228 if (ioctl(ctx->fd, VIDIOC_S_FMT, &fmt) == 0 && fmt.fmt.pix.pixelformat == V4L2_PIX_FMT_NV12) {
229 ctx->pixelformat = V4L2_PIX_FMT_NV12;
230 ctx->width = fmt.fmt.pix.width;
231 ctx->height = fmt.fmt.pix.height;
232 if (webcam_v4l2_init_swscale(ctx, AV_PIX_FMT_NV12) == 0) {
233 log_debug("V4L2 format set to NV12 %dx%d (will convert to RGB with swscale)", ctx->width, ctx->height);
234 return 0;
235 }
236 }
237
238 // Try I420 (planar YUV)
239 fmt.fmt.pix.pixelformat = V4L2_PIX_FMT_YUV420;
240 if (ioctl(ctx->fd, VIDIOC_S_FMT, &fmt) == 0 && fmt.fmt.pix.pixelformat == V4L2_PIX_FMT_YUV420) {
241 ctx->pixelformat = V4L2_PIX_FMT_YUV420;
242 ctx->width = fmt.fmt.pix.width;
243 ctx->height = fmt.fmt.pix.height;
244 if (webcam_v4l2_init_swscale(ctx, AV_PIX_FMT_YUV420P) == 0) {
245 log_debug("V4L2 format set to I420 %dx%d (will convert to RGB with swscale)", ctx->width, ctx->height);
246 return 0;
247 }
248 }
249
250 // Try MJPEG (Motion-JPEG - compressed format supporting 60fps on many cameras)
251 fmt.fmt.pix.pixelformat = V4L2_PIX_FMT_MJPEG;
252 int mjpeg_ret = ioctl(ctx->fd, VIDIOC_S_FMT, &fmt);
253 if (mjpeg_ret == 0 && fmt.fmt.pix.pixelformat == V4L2_PIX_FMT_MJPEG) {
254 ctx->pixelformat = V4L2_PIX_FMT_MJPEG;
255 ctx->width = fmt.fmt.pix.width;
256 ctx->height = fmt.fmt.pix.height;
257 if (webcam_v4l2_init_mjpeg_decoder(ctx) == 0) {
258 log_debug("V4L2 format set to MJPEG %dx%d (60fps compressed - will decompress JPEG with FFmpeg)", ctx->width,
259 ctx->height);
260 log_info("MJPEG format selected: pixelformat=0x%x", ctx->pixelformat);
261 return 0;
262 } else {
263 log_warn("MJPEG decoder init failed, trying next format");
264 }
265 } else {
266 log_debug("MJPEG format not supported (ioctl returned %d, got format 0x%x)", mjpeg_ret, fmt.fmt.pix.pixelformat);
267 }
268
269 // Try YUYV (YUV 4:2:2 - most webcams support this)
270 fmt.fmt.pix.pixelformat = V4L2_PIX_FMT_YUYV;
271 if (ioctl(ctx->fd, VIDIOC_S_FMT, &fmt) == 0 && fmt.fmt.pix.pixelformat == V4L2_PIX_FMT_YUYV) {
272 ctx->pixelformat = V4L2_PIX_FMT_YUYV;
273 ctx->width = fmt.fmt.pix.width;
274 ctx->height = fmt.fmt.pix.height;
275 if (webcam_v4l2_init_swscale(ctx, AV_PIX_FMT_YUYV422) == 0) {
276 log_debug("V4L2 format set to YUYV %dx%d (will convert to RGB with swscale)", ctx->width, ctx->height);
277 return 0;
278 }
279 }
280
281 // Try UYVY (YUV 4:2:2 variant)
282 fmt.fmt.pix.pixelformat = V4L2_PIX_FMT_UYVY;
283 if (ioctl(ctx->fd, VIDIOC_S_FMT, &fmt) == 0 && fmt.fmt.pix.pixelformat == V4L2_PIX_FMT_UYVY) {
284 ctx->pixelformat = V4L2_PIX_FMT_UYVY;
285 ctx->width = fmt.fmt.pix.width;
286 ctx->height = fmt.fmt.pix.height;
287 if (webcam_v4l2_init_swscale(ctx, AV_PIX_FMT_UYVY422) == 0) {
288 log_debug("V4L2 format set to UYVY %dx%d (will convert to RGB with swscale)", ctx->width, ctx->height);
289 return 0;
290 }
291 }
292
293 // Save errno before log_error() clears it
294 int saved_errno = errno;
295
296 // Check if format setting failed because device is busy
297 if (saved_errno == EBUSY) {
298 log_error("Failed to set V4L2 format: device is busy (another application is using it)");
299 errno = saved_errno;
300 return -1;
301 }
302
303 log_error("Failed to set V4L2 format: device supports none of (RGB24, NV12, I420, YUYV, UYVY) (errno=%d: %s)",
304 saved_errno, SAFE_STRERROR(saved_errno));
305 errno = saved_errno;
306 return -1;
307}

309static int webcam_v4l2_init_buffers(webcam_context_t *ctx) {
310 struct v4l2_requestbuffers req = {0};
311 req.count = WEBCAM_BUFFER_COUNT_DEFAULT;
312 req.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
313 req.memory = V4L2_MEMORY_MMAP;
314
315 if (ioctl(ctx->fd, VIDIOC_REQBUFS, &req) == -1) {
316 log_error("Failed to request V4L2 buffers: %s", SAFE_STRERROR(errno));
317 return -1;
318 }
319
320 if (req.count < 2) {
321 log_error("Insufficient buffer memory");
322 return -1;
323 }
324
325 // Ensure we don't exceed our maximum buffer count
326 if (req.count > WEBCAM_BUFFER_COUNT_MAX) {
327 log_warn("Driver requested %d buffers, limiting to %d", req.count, WEBCAM_BUFFER_COUNT_MAX);
328 req.count = WEBCAM_BUFFER_COUNT_MAX;
329 }
330
331 ctx->buffer_count = req.count;
332
333 // Allocate buffer array
334 ctx->buffers = SAFE_MALLOC(sizeof(webcam_buffer_t) * ctx->buffer_count, webcam_buffer_t *);
335 if (!ctx->buffers) {
336 log_error("Failed to allocate buffer array");
337 return -1;
338 }
339
340 // Initialize buffer array
341 memset(ctx->buffers, 0, sizeof(webcam_buffer_t) * ctx->buffer_count);
342
343 for (int i = 0; i < ctx->buffer_count; i++) {
344 struct v4l2_buffer buf = {0};
345 buf.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
346 buf.memory = V4L2_MEMORY_MMAP;
347 buf.index = i;
348
349 if (ioctl(ctx->fd, VIDIOC_QUERYBUF, &buf) == -1) {
350 log_error("Failed to query buffer %d: %s", i, SAFE_STRERROR(errno));
351 return -1;
352 }
353
354 ctx->buffers[i].length = buf.length;
355 ctx->buffers[i].start = mmap(NULL, buf.length, PROT_READ | PROT_WRITE, MAP_SHARED, ctx->fd, buf.m.offset);
356
357 if (ctx->buffers[i].start == MAP_FAILED) {
358 log_error("Failed to mmap buffer %d: %s", i, SAFE_STRERROR(errno));
359 return -1;
360 }
361 }
362
363 return 0;
364}

366static int webcam_v4l2_start_streaming(webcam_context_t *ctx) {
367 // Queue all buffers
368 for (int i = 0; i < ctx->buffer_count; i++) {
369 struct v4l2_buffer buf = {0};
370 buf.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
371 buf.memory = V4L2_MEMORY_MMAP;
372 buf.index = i;
373
374 if (ioctl(ctx->fd, VIDIOC_QBUF, &buf) == -1) {
375 log_error("Failed to queue buffer %d: %s", i, SAFE_STRERROR(errno));
376 return -1;
377 }
378 }
379
380 enum v4l2_buf_type type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
381 if (ioctl(ctx->fd, VIDIOC_STREAMON, &type) == -1) {
382 log_error("Failed to start V4L2 streaming: %s", SAFE_STRERROR(errno));
383 return -1;
384 }
385
386 log_dev("V4L2 streaming started");
387 return 0;
388}

390asciichat_error_t webcam_init_context(webcam_context_t **ctx, unsigned short int device_index) {
391 webcam_context_t *context;
392 context = SAFE_MALLOC(sizeof(webcam_context_t), webcam_context_t *);
393 if (!context) {
394 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate webcam context");
395 }
396
397 memset(context, 0, sizeof(webcam_context_t));
398
399 // Validate device index
400 if (device_index > WEBCAM_DEVICE_INDEX_MAX) {
401 SAFE_FREE(context);
402 return SET_ERRNO(ERROR_WEBCAM, "Invalid device index: %d (max: %d)", device_index, WEBCAM_DEVICE_INDEX_MAX);
403 }
404
405 // Open V4L2 device
406 char device_path[32];
407 SAFE_SNPRINTF(device_path, sizeof(device_path), "/dev/video%d", device_index);
408
409 // Check if device file exists before trying to open it
410 context->fd = platform_open("webcam_device", device_path, PLATFORM_O_RDWR | O_NONBLOCK);
411 if (context->fd == -1) {
412 // Provide more helpful error messages based on errno
413 if (errno == ENOENT) {
414 SAFE_FREE(context);
415 return SET_ERRNO(ERROR_WEBCAM,
416 "V4L2 device %s does not exist.\n"
417 "No webcam found. Try:\n"
418 " 1. Check if camera is connected: ls /dev/video*\n"
419 " 2. Use test pattern instead: --test-pattern",
420 device_path);
421 } else if (errno == EACCES) {
422 SAFE_FREE(context);
423 return SET_ERRNO(ERROR_WEBCAM_PERMISSION,
424 "Permission denied accessing %s.\n"
425 "Try: sudo usermod -a -G video $USER\n"
426 "Then log out and log back in.",
427 device_path);
428 } else if (errno == EBUSY) {
429 SAFE_FREE(context);
430 return SET_ERRNO(ERROR_WEBCAM_IN_USE, "V4L2 device %s is already in use by another application.", device_path);
431 } else {
432 SAFE_FREE(context);
433 return SET_ERRNO_SYS(ERROR_WEBCAM, "Failed to open V4L2 device %s", device_path);
434 }
435 }
436
437 // Check if it's a video capture device
438 struct v4l2_capability cap;
439 if (ioctl(context->fd, VIDIOC_QUERYCAP, &cap) == -1) {
440 close(context->fd);
441 SAFE_FREE(context);
442 return SET_ERRNO_SYS(ERROR_WEBCAM, "Failed to query V4L2 capabilities");
443 }
444
445 if (!(cap.capabilities & V4L2_CAP_VIDEO_CAPTURE)) {
446 close(context->fd);
447 SAFE_FREE(context);
448 return SET_ERRNO(ERROR_WEBCAM, "Device is not a video capture device");
449 }
450
451 // Set format (try 320x240 first for better performance, fallback to whatever the device supports)
452 if (webcam_v4l2_set_format(context, 320, 240) != 0) {
453 int saved_errno = errno; // Save errno before close() potentially changes it
454 close(context->fd);
455 SAFE_FREE(context);
456 // If format setting failed because device is busy, return ERROR_WEBCAM_IN_USE
457 if (saved_errno == EBUSY) {
458 return SET_ERRNO(ERROR_WEBCAM_IN_USE, "V4L2 device %s is in use - cannot set format", device_path);
459 }
460 return SET_ERRNO(ERROR_WEBCAM, "Failed to set V4L2 format for device %s", device_path);
461 }
462
463 // Initialize buffers
464 if (webcam_v4l2_init_buffers(context) != 0) {
465 close(context->fd);
466 SAFE_FREE(context);
467 return SET_ERRNO(ERROR_WEBCAM, "Failed to initialize V4L2 buffers for device %s", device_path);
468 }
469
470 // Start streaming
471 if (webcam_v4l2_start_streaming(context) != 0) {
472 // Cleanup buffers
473 for (int i = 0; i < context->buffer_count; i++) {
474 if (context->buffers[i].start != MAP_FAILED) {
475 munmap(context->buffers[i].start, context->buffers[i].length);
476 }
477 }
478 SAFE_FREE(context->buffers);
479 close(context->fd);
480 SAFE_FREE(context);
481 return SET_ERRNO(ERROR_WEBCAM, "Failed to start V4L2 streaming for device %s", device_path);
482 }
483
484 // Request target FPS via VIDIOC_S_PARM AFTER streaming starts
485 // Some devices require streaming to be active before frame rate control works
486 uint32_t target_fps = (uint32_t)GET_OPTION(fps);
487 if (target_fps == 0)
488 target_fps = 60; // Default to 60 if not set
489
490 struct v4l2_streamparm parm = {0};
491 parm.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
492 parm.parm.capture.timeperframe.numerator = 1;
493 parm.parm.capture.timeperframe.denominator = target_fps;
494
495 if (ioctl(context->fd, VIDIOC_S_PARM, &parm) == 0) {
496 // Get back what the driver actually set
497 if (ioctl(context->fd, VIDIOC_G_PARM, &parm) == 0) {
498 int actual_fps = parm.parm.capture.timeperframe.denominator /
499 (parm.parm.capture.timeperframe.numerator ? parm.parm.capture.timeperframe.numerator : 1);
500 log_info("V4L2 frame rate set: requested %u FPS, got %d FPS", target_fps, actual_fps);
501 }
502 } else {
503 log_debug("V4L2 device does not support VIDIOC_S_PARM (frame rate control)");
504 }
505
506 // Initialize async lifecycle for camera thread
507 if (!lifecycle_init(&context->async_lifecycle, "webcam_camera")) {
508 // Cleanup on lifecycle init failure
509 for (int i = 0; i < context->buffer_count; i++) {
510 if (context->buffers[i].start != MAP_FAILED) {
511 munmap(context->buffers[i].start, context->buffers[i].length);
512 }
513 }
514 SAFE_FREE(context->buffers);
515 close(context->fd);
516 SAFE_FREE(context);
517 return SET_ERRNO(ERROR_INIT, "Failed to initialize camera thread lifecycle");
518 }
519
520 // Start background camera thread
521 asciichat_error_t thread_err =
522 asciichat_thread_create(&context->camera_thread, "webcam_camera", webcam_camera_thread_func, context);
523 if (thread_err != ASCIICHAT_OK) {
524 lifecycle_shutdown(&context->async_lifecycle);
525 for (int i = 0; i < context->buffer_count; i++) {
526 if (context->buffers[i].start != MAP_FAILED) {
527 munmap(context->buffers[i].start, context->buffers[i].length);
528 }
529 }
530 SAFE_FREE(context->buffers);
531 close(context->fd);
532 SAFE_FREE(context);
533 return thread_err;
534 }
535
536 *ctx = context;
537 log_dev("V4L2 webcam initialized successfully on %s with async camera thread", device_path);
538
539 /* Register webcam context with named registry */
540 NAMED_REGISTER(context, device_path, "webcam", "0x%tx", NULL);
541
542 return 0;
543}

558void webcam_cleanup_context(webcam_context_t *ctx) {
559 if (!ctx)
560 return;
561
562 NAMED_UNREGISTER(ctx);
563
564 // Stop camera thread by shutting down lifecycle
565 if (lifecycle_shutdown(&ctx->async_lifecycle)) {
566 // This caller won the shutdown race, join the thread
567 asciichat_thread_join(&ctx->camera_thread, NULL);
568 log_debug("Camera background thread joined");
569 }
570
571 // Clean up any remaining frames in the atomic and cached buffers
572 image_t *leftover_frame = (image_t *)atomic_ptr_exchange(&ctx->latest_frame, NULL);
573 if (leftover_frame) {
574 image_destroy(leftover_frame);
575 }
576
577 if (ctx->async_cached_frame) {
578 image_destroy(ctx->async_cached_frame);
579 ctx->async_cached_frame = NULL;
580 }
581
582 // Free swscale context if it was allocated
583 if (ctx->sws_ctx) {
584 sws_freeContext(ctx->sws_ctx);
585 ctx->sws_ctx = NULL;
586 }
587
588 // Free MJPEG codec context if it was allocated
589 if (ctx->mjpeg_codec_ctx) {
590 avcodec_free_context(&ctx->mjpeg_codec_ctx);
591 ctx->mjpeg_codec_ctx = NULL;
592 }
593
594 // Free MJPEG decoded frame if it was allocated
595 if (ctx->mjpeg_decoded_frame) {
596 av_frame_free(&ctx->mjpeg_decoded_frame);
597 ctx->mjpeg_decoded_frame = NULL;
598 }
599
600 // Free cached frame if it was allocated
601 if (v4l2_cached_frame) {
602 image_destroy(v4l2_cached_frame);
603 v4l2_cached_frame = NULL;
604 }
605
606 // Stop streaming
607 enum v4l2_buf_type type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
608 ioctl(ctx->fd, VIDIOC_STREAMOFF, &type);
609
610 // Unmap buffers
611 if (ctx->buffers) {
612 for (int i = 0; i < ctx->buffer_count; i++) {
613 if (ctx->buffers[i].start != MAP_FAILED) {
614 munmap(ctx->buffers[i].start, ctx->buffers[i].length);
615 }
616 }
617 SAFE_FREE(ctx->buffers);
618 }
619
620 close(ctx->fd);
621 SAFE_FREE(ctx);
622 log_debug("V4L2 webcam cleaned up");
623}

625image_t *webcam_read_context(webcam_context_t *ctx) {
626 if (!ctx)
627 return NULL;
628
629 struct v4l2_buffer buf = {0};
630 buf.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
631 buf.memory = V4L2_MEMORY_MMAP;
632
633 // Use poll() to efficiently wait for frame available (500ms timeout)
634 // At 30fps, frames arrive every ~33ms; at 60fps every ~16ms
635 struct pollfd pfd = {.fd = ctx->fd, .events = POLLIN};
636
637 int poll_ret = poll(&pfd, 1, 500); // 500ms timeout for slower cameras
638
639 if (poll_ret < 0) {
640 log_error("poll() failed on V4L2 device: %s", SAFE_STRERROR(errno));
641 return NULL;
642 }
643
644 if (poll_ret == 0) {
645 // Timeout - no frame available within 100ms
646 return NULL;
647 }
648
649 // poll() returned > 0: data is available
650 if (!(pfd.revents & POLLIN)) {
651 // Some other event occurred (error/disconnect)
652 log_error("V4L2 device error: poll revents=0x%x", pfd.revents);
653 return NULL;
654 }
655
656 // Dequeue the buffer (should succeed now)
657 if (ioctl(ctx->fd, VIDIOC_DQBUF, &buf) != 0) {
658 log_error("Failed to dequeue V4L2 buffer after poll(): %s", SAFE_STRERROR(errno));
659 return NULL;
660 }
661
662 // Validate buffer index to prevent crashes
663 if (buf.index >= (unsigned int)ctx->buffer_count) {
664 log_error("V4L2 returned invalid buffer index %u (max: %d)", buf.index, ctx->buffer_count - 1);
665 return NULL;
666 }
667
668 // Validate buffer pointer
669 if (!ctx->buffers || !ctx->buffers[buf.index].start) {
670 log_error("V4L2 buffer %u not initialized (start=%p, buffers=%p)", buf.index,
671 ctx->buffers ? ctx->buffers[buf.index].start : NULL, ctx->buffers);
672 return NULL;
673 }
674
675 // Allocate or reallocate cached frame if needed
676 if (!v4l2_cached_frame || v4l2_cached_frame->w != ctx->width || v4l2_cached_frame->h != ctx->height) {
677 if (v4l2_cached_frame) {
678 image_destroy(v4l2_cached_frame);
679 }
680 v4l2_cached_frame = image_new(ctx->width, ctx->height);
681 if (!v4l2_cached_frame) {
682 log_error("Failed to allocate image buffer");
683 // Re-queue the buffer - use safe error handling
684 if (ioctl(ctx->fd, VIDIOC_QBUF, &buf) == -1) {
685 log_error("Failed to re-queue buffer after image allocation failure: %s", SAFE_STRERROR(errno));
686 }
687 return NULL;
688 }
689 }
690
691 image_t *img = v4l2_cached_frame;
692
693 // Handle MJPEG decompression if using MJPEG format
694 if (ctx->pixelformat == V4L2_PIX_FMT_MJPEG) {
695 if (!ctx->mjpeg_codec_ctx) {
696 log_error("MJPEG codec context not initialized");
697 if (ioctl(ctx->fd, VIDIOC_QBUF, &buf) == -1) {
698 log_error("Failed to re-queue buffer after MJPEG codec error: %s", SAFE_STRERROR(errno));
699 }
700 return NULL;
701 }
702
703 // Create packet from MJPEG frame data
704 AVPacket pkt = {0};
705 av_new_packet(&pkt, buf.bytesused);
706 memcpy(pkt.data, ctx->buffers[buf.index].start, buf.bytesused);
707
708 // Send packet to decoder
709 if (avcodec_send_packet(ctx->mjpeg_codec_ctx, &pkt) < 0) {
710 log_warn_every(1000000000LL, "Failed to send MJPEG packet (size=%u) to decoder", buf.bytesused);
711 av_packet_unref(&pkt);
712 if (ioctl(ctx->fd, VIDIOC_QBUF, &buf) == -1) {
713 log_error("Failed to re-queue buffer after MJPEG decode error: %s", SAFE_STRERROR(errno));
714 }
715 return NULL;
716 }
717
718 // Receive decoded frame (JPEG decompression happens here)
719 if (avcodec_receive_frame(ctx->mjpeg_codec_ctx, ctx->mjpeg_decoded_frame) < 0) {
720 log_warn_every(1000000000LL, "Failed to decode MJPEG frame from %u bytes", buf.bytesused);
721 av_packet_unref(&pkt);
722 if (ioctl(ctx->fd, VIDIOC_QBUF, &buf) == -1) {
723 log_error("Failed to re-queue buffer after MJPEG receive error: %s", SAFE_STRERROR(errno));
724 }
725 return NULL;
726 }
727
728 av_packet_unref(&pkt);
729
730 // Convert decoded frame to RGB24 using swscale
731 if (!ctx->sws_ctx) {
732 log_error("Swscale context not initialized for MJPEG decoded frame");
733 if (ioctl(ctx->fd, VIDIOC_QBUF, &buf) == -1) {
734 log_error("Failed to re-queue buffer after swscale init error: %s", SAFE_STRERROR(errno));
735 }
736 return NULL;
737 }
738
739 uint8_t *dst_data[1] = {(uint8_t *)img->pixels};
740 int dst_linesize[1] = {ctx->width * 3};
741
742 sws_scale(ctx->sws_ctx, (const uint8_t *const *)ctx->mjpeg_decoded_frame->data, ctx->mjpeg_decoded_frame->linesize,
743 0, ctx->height, dst_data, dst_linesize);
744
745 // Re-queue the buffer for future use
746 if (ioctl(ctx->fd, VIDIOC_QBUF, &buf) == -1) {
747 log_error("Failed to re-queue V4L2 buffer %u after MJPEG decode: %s", buf.index, SAFE_STRERROR(errno));
748 }
749
750 return img;
751 }
752
753 // Copy and convert frame data based on pixel format
754 if (ctx->sws_ctx) {
755 // Use libswscale for format conversion (NV12, I420, YUYV, UYVY)
756 const uint8_t *src_data[1] = {(const uint8_t *)ctx->buffers[buf.index].start};
757 int src_linesize[1] = {0};
758
759 // Calculate source linesize based on format
760 switch (ctx->pixelformat) {
761 case V4L2_PIX_FMT_NV12:
762 case V4L2_PIX_FMT_YUV420:
763 // For planar formats, linesize equals width
764 src_linesize[0] = ctx->width;
765 break;
766 case V4L2_PIX_FMT_YUYV:
767 case V4L2_PIX_FMT_UYVY:
768 // For packed YUV 4:2:2, each pixel is 2 bytes
769 src_linesize[0] = ctx->width * 2;
770 break;
771 default:
772 src_linesize[0] = ctx->width;
773 }
774
775 uint8_t *dst_data[1] = {(uint8_t *)img->pixels};
776 int dst_linesize[1] = {ctx->width * 3};
777
778 sws_scale(ctx->sws_ctx, src_data, src_linesize, 0, ctx->height, dst_data, dst_linesize);
779 } else {
780 // RGB24 - direct copy with overflow checking
781 size_t frame_size;
782 if (image_calc_rgb_size((size_t)ctx->width, (size_t)ctx->height, &frame_size) != ASCIICHAT_OK) {
783 log_error("Failed to calculate frame size: width=%d, height=%d (would overflow)", ctx->width, ctx->height);
784 image_destroy(img);
785 return NULL;
786 }
787 memcpy(img->pixels, ctx->buffers[buf.index].start, frame_size);
788 }
789
790 // Re-queue the buffer for future use
791 if (ioctl(ctx->fd, VIDIOC_QBUF, &buf) == -1) {
792 log_error("Failed to re-queue V4L2 buffer %u: %s (fd=%d, type=%d, memory=%d)", buf.index, SAFE_STRERROR(errno),
793 ctx->fd, buf.type, buf.memory);
794 // Still return the image - the frame data was already copied
795 }
796
797 return img;
798}

809image_t *webcam_read_async(webcam_context_t *ctx) {
810 if (!ctx || !lifecycle_is_initialized(&ctx->async_lifecycle)) {
811 return NULL;
812 }
813
814 // Check if there's a new frame from the camera thread
815 image_t *new_frame = (image_t *)atomic_ptr_exchange(&ctx->latest_frame, NULL);
816
817 if (new_frame) {
818 // New frame available - update cache and return it
819 // (The background thread allocated this frame, we take ownership)
820 if (ctx->async_cached_frame) {
821 image_destroy(ctx->async_cached_frame);
822 }
823 ctx->async_cached_frame = new_frame;
824 return new_frame;
825 }
826
827 // No new frame - return the last cached frame (smooth playback without gaps)
828 return ctx->async_cached_frame;
829}

840asciichat_error_t webcam_list_devices(webcam_device_info_t **out_devices, unsigned int *out_count) {
841 if (!out_devices || !out_count) {
842 return SET_ERRNO(ERROR_INVALID_PARAM, "webcam_list_devices: invalid parameters");
843 }
844
845 *out_devices = NULL;
846 *out_count = 0;
847
848 // First pass: count valid video devices
849 unsigned int device_count = 0;
850 for (int i = 0; i <= WEBCAM_DEVICE_INDEX_MAX; i++) {
851 char device_path[32];
852 safe_snprintf(device_path, sizeof(device_path), "/dev/video%d", i);
853
854 int fd = open(device_path, O_RDONLY);
855 if (fd < 0) {
856 continue;
857 }
858
859 struct v4l2_capability cap;
860 if (ioctl(fd, VIDIOC_QUERYCAP, &cap) == 0 && (cap.device_caps & V4L2_CAP_VIDEO_CAPTURE)) {
861 device_count++;
862 }
863 close(fd);
864 }
865
866 if (device_count == 0) {
867 // No devices found - not an error
868 return ASCIICHAT_OK;
869 }
870
871 // Allocate device array
872 webcam_device_info_t *devices = SAFE_CALLOC(device_count, sizeof(webcam_device_info_t), webcam_device_info_t *);
873 if (!devices) {
874 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate device info array");
875 }
876
877 // Second pass: populate device info
878 unsigned int idx = 0;
879 for (int i = 0; i <= WEBCAM_DEVICE_INDEX_MAX && idx < device_count; i++) {
880 char device_path[32];
881 safe_snprintf(device_path, sizeof(device_path), "/dev/video%d", i);
882
883 int fd = open(device_path, O_RDONLY);
884 if (fd < 0) {
885 continue;
886 }
887
888 struct v4l2_capability cap;
889 if (ioctl(fd, VIDIOC_QUERYCAP, &cap) == 0 && (cap.device_caps & V4L2_CAP_VIDEO_CAPTURE)) {
890 devices[idx].index = (unsigned int)i;
891 SAFE_STRNCPY(devices[idx].name, (const char *)cap.card, WEBCAM_DEVICE_NAME_MAX);
892 idx++;
893 }
894 close(fd);
895 }
896
897 *out_devices = devices;
898 *out_count = idx;
899
900 return ASCIICHAT_OK;
901}
