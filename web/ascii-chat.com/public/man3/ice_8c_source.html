72static uint16_t ice_calculate_local_preference_by_ip(const char *ip_address) {
73 if (!ip_address || ip_address[0] == '\0') {
74 return 0; // Invalid/empty
75 }
76
77 // Check for wildcard addresses (should not appear in candidates, but handle gracefully)
78 if (strcmp(ip_address, "0.0.0.0") == 0 || strcmp(ip_address, "::") == 0) {
79 return 0; // Invalid candidate
80 }
81
82 // LAN addresses: Highest priority (same network = lowest latency)
83 if (is_lan_ipv4(ip_address) || is_lan_ipv6(ip_address)) {
84 return 65535;
85 }
86
87 // Localhost: High priority but less useful than LAN for peer-to-peer
88 if (is_localhost_ipv4(ip_address) || is_localhost_ipv6(ip_address)) {
89 return 65000;
90 }
91
92 // Internet (public): Medium priority
93 if (is_internet_ipv4(ip_address) || is_internet_ipv6(ip_address)) {
94 return 32768;
95 }
96
97 // Unknown/unclassified: Lowest priority
98 return 0;
99}

101uint32_t ice_calculate_priority(ice_candidate_type_t type, uint16_t local_preference, uint8_t component_id) {
102 // RFC 5245: priority = (2^24 * typePreference) + (2^8 * localPreference) + (256 - componentID)
103
104 // Type preference (higher is better):
105 // - Host: 126 (most preferred, lowest latency)
106 // - SRFLX: 100 (STUN-discovered address)
107 // - PRFLX: 110 (discovered during connectivity checks)
108 // - Relay: 0 (least preferred, uses bandwidth)
109
110 uint8_t type_preference = 0;
111 switch (type) {
112 case ICE_CANDIDATE_HOST:
113 type_preference = 126;
114 break;
115 case ICE_CANDIDATE_SRFLX:
116 type_preference = 100;
117 break;
118 case ICE_CANDIDATE_PRFLX:
119 type_preference = 110;
120 break;
121 case ICE_CANDIDATE_RELAY:
122 type_preference = 0;
123 break;
124 default:
125 type_preference = 0;
126 break;
127 }
128
129 uint32_t priority =
130 (((uint32_t)type_preference) << 24) | (((uint32_t)local_preference) << 8) | (256 - (uint32_t)component_id);
131
132 return priority;
133}

135uint32_t ice_calculate_priority_for_candidate(const ice_candidate_t *candidate) {
136 if (!candidate) {
137 return 0;
138 }
139
140 // Calculate local preference based on IP classification
141 uint16_t local_pref = ice_calculate_local_preference_by_ip(candidate->ip_address);
142
143 // Use standard RFC 5245 priority formula with IP-aware local preference
144 return ice_calculate_priority(candidate->type, local_pref, (uint8_t)candidate->component_id);
145}

158static const char *ICE_CANDIDATE_PATTERN = "^([^ ]+)\\s+" // 1: foundation
159 "(\\d+)\\s+" // 2: component_id
160 "(udp|tcp)\\s+" // 3: protocol
161 "(\\d+)\\s+" // 4: priority
162 "([a-fA-F0-9.:]+)\\s+" // 5: ip_address (IPv4 or IPv6)
163 "(\\d+)\\s+" // 6: port
164 "typ\\s+" // literal "typ"
165 "(host|srflx|prflx|relay)" // 7: candidate_type
166 "(?:\\s+raddr\\s+([a-fA-F0-9.:]+))?" // 8: optional raddr
167 "(?:\\s+rport\\s+(\\d+))?" // 9: optional rport
168 "(?:\\s+tcptype\\s+(active|passive|so))?" // 10: optional tcptype
169 "\\s*$";

194static asciichat_error_t ice_parse_candidate_pcre2(const char *line, ice_candidate_t *candidate) {
195 if (!line || !candidate) {
196 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid candidate parameters");
197 }
198
199 // Get compiled regex (lazy initialization)
200 pcre2_code *regex = ice_candidate_regex_get();
201
202 // If PCRE2 not available, return error to fall back to manual parser
203 if (!regex) {
204 return ERROR_MEMORY; // Signal to use fallback
205 }
206
207 pcre2_match_data *match_data = pcre2_match_data_create_from_pattern(regex, NULL);
208 if (!match_data) {
209 return ERROR_MEMORY;
210 }
211
212 int rc = pcre2_jit_match(regex, (PCRE2_SPTR8)line, strlen(line), 0, 0, match_data, NULL);
213
214 if (rc < 7) {
215 // Must have at least foundation, component, protocol, priority, ip, port, type (7 groups)
216 pcre2_match_data_free(match_data);
217 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid ICE candidate format");
218 }
219
220 memset(candidate, 0, sizeof(ice_candidate_t));
221
222 // Extract foundation (group 1)
223 char *foundation = asciichat_pcre2_extract_group(match_data, 1, line);
224 if (foundation && strlen(foundation) < sizeof(candidate->foundation)) {
225 SAFE_STRNCPY(candidate->foundation, foundation, sizeof(candidate->foundation));
226 }
227 SAFE_FREE(foundation);
228
229 // Extract component_id (group 2)
230 unsigned long component_id;
231 if (asciichat_pcre2_extract_group_ulong(match_data, 2, line, &component_id)) {
232 candidate->component_id = (uint32_t)component_id;
233 }
234
235 // Extract protocol (group 3)
236 char *protocol_str = asciichat_pcre2_extract_group(match_data, 3, line);
237 if (protocol_str) {
238 if (strcmp(protocol_str, "udp") == 0) {
239 candidate->protocol = ICE_PROTOCOL_UDP;
240 } else if (strcmp(protocol_str, "tcp") == 0) {
241 candidate->protocol = ICE_PROTOCOL_TCP;
242 } else {
243 SAFE_FREE(protocol_str);
244 pcre2_match_data_free(match_data);
245 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid protocol");
246 }
247 SAFE_FREE(protocol_str);
248 }
249
250 // Extract priority (group 4)
251 unsigned long priority;
252 if (asciichat_pcre2_extract_group_ulong(match_data, 4, line, &priority)) {
253 candidate->priority = (uint32_t)priority;
254 }
255
256 // Extract IP address (group 5)
257 char *ip_address = asciichat_pcre2_extract_group(match_data, 5, line);
258 if (ip_address && strlen(ip_address) < sizeof(candidate->ip_address)) {
259 SAFE_STRNCPY(candidate->ip_address, ip_address, sizeof(candidate->ip_address));
260 }
261 SAFE_FREE(ip_address);
262
263 // Extract port (group 6)
264 unsigned long port;
265 if (asciichat_pcre2_extract_group_ulong(match_data, 6, line, &port)) {
266 candidate->port = (uint16_t)port;
267 }
268
269 // Extract candidate type (group 7)
270 char *type_str = asciichat_pcre2_extract_group(match_data, 7, line);
271 if (type_str) {
272 if (strcmp(type_str, "host") == 0) {
273 candidate->type = ICE_CANDIDATE_HOST;
274 } else if (strcmp(type_str, "srflx") == 0) {
275 candidate->type = ICE_CANDIDATE_SRFLX;
276 } else if (strcmp(type_str, "prflx") == 0) {
277 candidate->type = ICE_CANDIDATE_PRFLX;
278 } else if (strcmp(type_str, "relay") == 0) {
279 candidate->type = ICE_CANDIDATE_RELAY;
280 } else {
281 SAFE_FREE(type_str);
282 pcre2_match_data_free(match_data);
283 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid candidate type");
284 }
285 SAFE_FREE(type_str);
286 }
287
288 // Extract optional raddr (group 8)
289 char *raddr = asciichat_pcre2_extract_group(match_data, 8, line);
290 if (raddr && strlen(raddr) < sizeof(candidate->raddr)) {
291 SAFE_STRNCPY(candidate->raddr, raddr, sizeof(candidate->raddr));
292 }
293 SAFE_FREE(raddr);
294
295 // Extract optional rport (group 9)
296 unsigned long rport;
297 if (asciichat_pcre2_extract_group_ulong(match_data, 9, line, &rport)) {
298 candidate->rport = (uint16_t)rport;
299 }
300
301 // Extract optional tcptype (group 10)
302 char *tcptype_str = asciichat_pcre2_extract_group(match_data, 10, line);
303 if (tcptype_str) {
304 if (strcmp(tcptype_str, "active") == 0) {
305 candidate->tcp_type = ICE_TCP_TYPE_ACTIVE;
306 } else if (strcmp(tcptype_str, "passive") == 0) {
307 candidate->tcp_type = ICE_TCP_TYPE_PASSIVE;
308 } else if (strcmp(tcptype_str, "so") == 0) {
309 candidate->tcp_type = ICE_TCP_TYPE_SO;
310 }
311 SAFE_FREE(tcptype_str);
312 }
313
314 pcre2_match_data_free(match_data);
315 return ASCIICHAT_OK;
316}

322asciichat_error_t ice_format_candidate(const ice_candidate_t *candidate, char *line, size_t line_size) {
323 if (!candidate || !line || line_size == 0) {
324 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid candidate format parameters");
325 }
326
327 // Format: foundation component protocol priority ip port typ type [raddr rport] [tcptype type]
328 // Example: 1 1 udp 2130706431 192.168.1.1 54321 typ host
329 // Example with raddr: 1 1 udp 2130706431 203.0.113.45 54321 typ srflx raddr 192.168.1.1 rport 12345
330
331 int written = safe_snprintf(line, line_size, "%s %u %s %u %s %u typ %s", candidate->foundation,
332 candidate->component_id, ice_protocol_name(candidate->protocol), candidate->priority,
333 candidate->ip_address, candidate->port, ice_candidate_type_name(candidate->type));
334
335 if (written < 0 || (size_t)written >= line_size) {
336 return SET_ERRNO(ERROR_BUFFER_FULL, "Candidate line too large for buffer");
337 }
338
339 size_t remaining = line_size - (size_t)written;
340 char *pos = line + written;
341
342 // 2. If srflx/prflx/relay: append raddr and rport
343 if (candidate->type == ICE_CANDIDATE_SRFLX || candidate->type == ICE_CANDIDATE_PRFLX ||
344 candidate->type == ICE_CANDIDATE_RELAY) {
345 if (candidate->raddr[0] != '\0') {
346 int appended = safe_snprintf(pos, remaining, " raddr %s rport %u", candidate->raddr, candidate->rport);
347 if (appended < 0 || (size_t)appended >= remaining) {
348 return SET_ERRNO(ERROR_BUFFER_FULL, "Cannot append raddr/rport to candidate line");
349 }
350 pos += appended;
351 remaining -= (size_t)appended;
352 }
353 }
354
355 // 3. If TCP: append tcptype
356 if (candidate->protocol == ICE_PROTOCOL_TCP) {
357 const char *tcptype_str = "passive"; // default
358 if (candidate->tcp_type == ICE_TCP_TYPE_ACTIVE) {
359 tcptype_str = "active";
360 } else if (candidate->tcp_type == ICE_TCP_TYPE_SO) {
361 tcptype_str = "so";
362 }
363
364 int appended = safe_snprintf(pos, remaining, " tcptype %s", tcptype_str);
365 if (appended < 0 || (size_t)appended >= remaining) {
366 return SET_ERRNO(ERROR_BUFFER_FULL, "Cannot append tcptype to candidate line");
367 }
368 pos += appended;
369 remaining -= (size_t)appended;
370 }
371
372 // 4. Append any extensions
373 if (candidate->extensions[0] != '\0') {
374 int appended = safe_snprintf(pos, remaining, " %s", candidate->extensions);
375 if (appended < 0 || (size_t)appended >= remaining) {
376 return SET_ERRNO(ERROR_BUFFER_FULL, "Cannot append extensions to candidate line");
377 }
378 }
379
380 // 5. Ensure line is null-terminated (snprintf already does this)
381 return ASCIICHAT_OK;
382}

388asciichat_error_t ice_gather_candidates(const ice_config_t *config) {
389 if (!config) {
390 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid ICE config");
391 }
392
393 if (!config->send_callback) {
394 return SET_ERRNO(ERROR_INVALID_PARAM, "ICE config missing send_callback");
395 }
396
397 log_debug("ICE: Starting candidate gathering (ufrag=%s, pwd=%s)", config->ufrag, config->pwd);
398
399 // NOTE: libdatachannel (juice) handles ICE candidate gathering internally.
400 // This function is a placeholder for the callback-based API.
401 // The actual gathering happens in the peer_manager when you:
402 // 1. Create peer connection with rtcCreatePeerConnection()
403 // 2. Set ice candidate callback with rtcSetIceCandidateCallback()
404 // 3. libdatachannel automatically gathers candidates from:
405 // - Local interfaces (host candidates)
406 // - STUN servers (server-reflexive candidates)
407 // - TURN servers (relay candidates)
408 // 4. When a candidate is found, rtcSetIceCandidateCallback() is called
409 // 5. Application (webrtc.c) formats and sends via ACDS signaling
410 //
411 // This ice_gather_candidates() function may not be called directly -
412 // the gathering happens automatically when peer connection is created.
413 // Keeping this for API compatibility.
414
415 return ASCIICHAT_OK;
416}

422asciichat_error_t ice_add_remote_candidate(webrtc_peer_connection_t *pc, const ice_candidate_t *candidate,
423 const char *mid) {
424 if (!pc || !candidate || !mid) {

427
428 // Validate candidate has required fields
429 if (candidate->ip_address[0] == '\0' || candidate->port == 0) {
430 return SET_ERRNO(ERROR_INVALID_PARAM, "Candidate missing IP address or port");
431 }
432
433 log_debug("ICE: Adding remote candidate %s:%u (type=%s, foundation=%s) on mid=%s", candidate->ip_address,
434 candidate->port, ice_candidate_type_name(candidate->type), candidate->foundation, mid);
435
436 // Format candidate to SDP attribute line for libdatachannel
437 char candidate_line[512];
438 asciichat_error_t err = ice_format_candidate(candidate, candidate_line, sizeof(candidate_line));
439 if (err != ASCIICHAT_OK) {
440 return err;
441 }
442
443 // Delegate to WebRTC layer
444 return webrtc_add_remote_candidate(pc, candidate_line, mid);
