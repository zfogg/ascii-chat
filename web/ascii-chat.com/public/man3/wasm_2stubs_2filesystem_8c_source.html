82int platform_open(const char *name, const char *pathname, int flags, ...) {
83 if (!name) {
84 return -1;
85 }
86
87 // Handle optional mode parameter for O_CREAT
88 int fd = -1;
89 int mode = 0;
90 if (flags & 0x0200) { // O_CREAT flag value
91 va_list args;
92 va_start(args, flags);
93 mode = va_arg(args, int);
94 va_end(args);
95 fd = open(pathname, flags, mode);
96 } else {
97 fd = open(pathname, flags);
98 }
99
100 if (fd >= 0) {
101 NAMED_REGISTER_FD(fd, name);
102 log_dev("Opened file descriptor %d for %s at path %s", fd, name, pathname);
103 }
104
105 return fd;
106}
