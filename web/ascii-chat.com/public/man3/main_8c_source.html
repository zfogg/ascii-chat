203void signal_exit(void) {
204 // Note: This function may be called from a signal handler context where
205 // other threads may hold mutex locks. We avoid log_console() here to prevent
206 // deadlock (render loop holds terminal lock, signal handler can't acquire it).
207 // The shutdown will be logged by normal thread context later.
208 atomic_store_bool(&g_should_exit, true);
209 void (*cb)(void) = g_interrupt_callback;
210 if (cb) {
211 cb();
212 }
213}

223#ifndef _WIN32
228static void handle_sigterm(int sig) {
229 // Write to stderr about which signal was received for debugging
230 const char *sig_name = (sig == SIGTERM) ? "SIGTERM"
231 : (sig == SIGINT) ? "SIGINT"
232 : (sig == SIGHUP) ? "SIGHUP"
233 : "UNKNOWN";
234 char log_out[64];
235 safe_snprintf(log_out, sizeof(log_out), "Signal '%s' received", sig_name);
236 if (sig == SIGTERM || sig == SIGHUP || sig == SIGINT) {
237 safe_snprintf(log_out, sizeof(log_out), "%s - shutting down", log_out);
238 }
239 // log_console() is async-signal-safe - uses atomic ops and platform_write_all()
240 log_console(LOG_INFO, log_out);
241
242#ifndef NDEBUG
243 // Trigger debug sync state printing on shutdown (async-signal-safe)
244 debug_sync_trigger_print();
245#endif
246
247 // Call signal_exit() to set flag AND interrupt blocking socket operations
248 signal_exit();
249}
250#endif

256static bool console_ctrl_handler(console_ctrl_event_t event) {
257 if (event != CONSOLE_CTRL_C && event != CONSOLE_CTRL_BREAK && event != CONSOLE_CLOSE) {
258 return false;
259 }
260
261 log_console(LOG_INFO, "Ctrl+C received - shutting down");
262
263#ifndef NDEBUG
264 // Trigger debug sync state printing on shutdown (async-signal-safe)
265 debug_sync_trigger_print();
266#endif
267
268 // Double Ctrl+C forces immediate exit
269 static atomic_t ctrl_c_count = {0};
270 static bool ctrl_c_count_registered = false;
271 if (!ctrl_c_count_registered) {
272 NAMED_REGISTER_ATOMIC(&ctrl_c_count, "ctrl_c_interrupt_count", NULL);
273 ctrl_c_count_registered = true;
274 }
275 if (atomic_fetch_add_int(&ctrl_c_count, 1) + 1 > 1) {
276 platform_force_exit(1);
277 }
278
279 signal_exit();
280 return true;
281}

758 // e.g., server.log -> server.json, mirror.log -> mirror.json, etc.
759 size_t len = strlen(final_log_file);
760 if (len > 4 && strcmp(&final_log_file[len - 4], ".log") == 0) {
761 // File ends with .log - replace with .json
762 SAFE_STRNCPY(json_filename_buf, final_log_file, sizeof(json_filename_buf) - 1);
763 // Replace .log with .json
764 strcpy(&json_filename_buf[len - 4], ".json");
765 } else {
766 // File doesn't end with .log - just append .json
767 SAFE_STRNCPY(json_filename_buf, final_log_file, sizeof(json_filename_buf) - 1);
768 strncat(json_filename_buf, ".json", sizeof(json_filename_buf) - strlen(json_filename_buf) - 1);
769 }
770 }
771 final_log_file = json_filename_buf;
772 // For JSON mode: Initialize logging without file (text will be disabled)
773 // We'll set up JSON output separately below
774 // Note: log_init() was already called in asciichat_shared_init()
775 } else {
776 // Text logging mode: use the file from options (which is mode-specific default or user-specified)
777 // Note: log_init() was already called in asciichat_shared_init()
778 }
779
780 // Apply custom log template if specified (use early parsed value if available, otherwise use options)
781 const char *final_format = early_log_template ? early_log_template : GET_OPTION(log_template);
782 bool final_format_console_only =
783 early_log_template ? early_log_template_console_only : GET_OPTION(log_format_console_only);
784 if (final_format && final_format[0] != '\0') {
785 asciichat_error_t fmt_result = log_set_format(final_format, final_format_console_only);
786 if (fmt_result != ASCIICHAT_OK) {
787 log_error("Failed to apply custom log format");
788 }
789 }

791 // Configure JSON output if requested
792 if (use_json_logging) {
793 // Open the JSON file for output
794 int json_fd = platform_open("main_log_file", final_log_file, O_CREAT | O_RDWR | O_TRUNC, FILE_PERM_PRIVATE);
795 if (json_fd >= 0) {
796 log_set_json_output(json_fd);
797 } else {
798 // Failed to open JSON file - report error and exit
799 SET_ERRNO(ERROR_CONFIG, "Failed to open JSON output file: %s", final_log_file);
800 return ERROR_CONFIG;
801 }
802 }

804 // Initialize colors now that logging is fully initialized
805 // This must happen after log_init() since log_init_colors() checks if g_log.initialized
806 log_init_colors();

808 // Apply quiet mode - disables terminal output
809 // Status screen mode only disables terminal output if terminal is interactive
810 // In non-interactive mode (piped output), logs go to stdout/stderr normally
811 if (GET_OPTION(quiet) ||
812 (opts->detected_mode == MODE_SERVER && GET_OPTION(status_screen) && terminal_is_interactive())) {
813 log_set_terminal_output(false);
814 }

816 // Initialize palette based on command line options
817 const char *custom_chars = opts && opts->palette_custom_set ? opts->palette_custom : NULL;
818 if (apply_palette_config(GET_OPTION(palette_type), custom_chars) != 0) {
819 FATAL(ERROR_CONFIG, "Failed to apply palette configuration");
820 }
821
822 // Set quiet mode for memory debugging
823#if defined(DEBUG_MEMORY) && !defined(USE_MIMALLOC_DEBUG) && !defined(NDEBUG)
824 debug_memory_set_quiet_mode(GET_OPTION(quiet));
825#endif
826
827 // Truncate log if it's already too large
828 log_truncate_if_large();

830 // Handle --help and --version (these are detected and flagged by options_init)
831 // Terminal capabilities already initialized before options_init() at startup
832 if (opts->help) {
833 print_usage(opts->detected_mode);
834 fflush(NULL);
835 _Exit(0);
836 }
837
838 if (opts->version) {
839 log_set_terminal_output(true);
840 action_show_version();
841 // action_show_version() calls _Exit(), so we don't reach here
842 }
843
844#ifndef NDEBUG
845 // Handle --debug-state (debug builds only)
846 // Sleep for specified time AFTER mode initialization so locks are created
847 if (IS_OPTION_EXPLICIT(debug_sync_state_time, opts) && opts->debug_sync_state_time > 0.0) {
848 log_info("Will print sync state after %f seconds", opts->debug_sync_state_time);
849 }

852 // For server mode with status screen: disable terminal output only if interactive
853 // In non-interactive mode (piped output), logs go to stdout/stderr normally
854 // The status screen (when shown) will capture and display logs in its buffer instead
855 if (opts->detected_mode == MODE_SERVER && opts->status_screen && terminal_is_interactive()) {
856 log_set_terminal_output(false);
857 }

859 log_dev("Logging initialized to %s", final_log_file);

861 // Note: We do NOT auto-disable colors when stdout appears to be piped, because:
862 // 1. Tools like ripgrep can display ANSI colors when piped
863 // 2. Sandboxed/containerized environments may report false positives for isatty()
864 // 3. Users can explicitly disable colors with --color=false if needed
865 // Color behavior is now fully controlled by --color and --color-mode options.
866

895 sigset_t set;

897 sigaddset(&set, SIGUSR1);

919 if (opts->fps > 0) {

921 log_warn("FPS value %d out of range (1-144), using default", opts->fps);
922 } else {
923 log_debug("FPS set from command line: %d", opts->fps);
924 }
925 }

927 // Automatic update check at startup (once per week maximum)
928 if (!GET_OPTION(no_check_update)) {
929 update_check_result_t update_result;
930 asciichat_error_t update_err = update_check_startup(&update_result);
931 if (update_err == ASCIICHAT_OK && update_result.update_available) {
932 char notification[1024];
933 update_check_format_notification(&update_result, notification, sizeof(notification));
934 log_info("%s", notification);
935
936 // Set update notification for splash/status screens
937 splash_set_update_notification(notification);
938 }
939 }

942 // All modes use the same centralized exit mechanism
943 setup_signal_handlers();

946 // This connects the render loop's should_exit check to signal_exit()
947 static const app_callbacks_t app_callbacks = {
948 .should_exit = should_exit,
949 .signal_exit = signal_exit,
950 };
951 app_callbacks_register(&app_callbacks);

953 // Register shutdown callback so splash thread and other code can check for exit
954 shutdown_register_callback((shutdown_check_fn)should_exit);
955
956#ifndef NDEBUG
957 // Start debug threads now, after initialization but before mode entry
958 // This avoids lock contention during critical initialization phase
959 // Note: debug_sync_start_thread() and debug_memory_thread_start() now use
960 // direct pthread calls instead of mutex_init/cond_init to avoid named registry
961 // deadlock during thread startup
962 if (debug_sync_start_thread() != 0) {
963 LOG_ERRNO_IF_SET("Debug sync thread startup failed");
964 FATAL(ERROR_THREAD, "Debug sync thread startup failed");
965 }
966 log_debug("Debug sync thread started");
967
968#endif
969
970 // Find and dispatch to mode entry point
971 const mode_descriptor_t *mode = find_mode(opts->detected_mode);
972 if (!mode) {
973 fprintf(stderr, "Error: Mode not found for detected_mode=%d\n", opts->detected_mode);
974 return 1;
975 }
976
977 // Call the mode-specific entry point
978 // Mode entry points use options_get() to access parsed options
979 int exit_code = mode->entry_point();
980
981 // Named registry cleanup is handled by asciichat_shared_destroy()
982 // (called via atexit handler), which runs AFTER the memory report
983
984 if (exit_code == ERROR_USAGE) {
985 exit(ERROR_USAGE);
986 }
987
988 return exit_code;
