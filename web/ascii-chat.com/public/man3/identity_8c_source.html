31asciichat_error_t acds_identity_load(const char *path, uint8_t public_key[32], uint8_t secret_key[64]) {
32 if (!path || !public_key || !secret_key) {
33 return SET_ERRNO(ERROR_INVALID_PARAM, "path, public_key, and secret_key cannot be NULL");
34 }
35
36 // Open file for reading
37 FILE *fp = platform_fopen("file_stream", path, "rb");
38 if (!fp) {
39 if (errno == ENOENT) {
40 return SET_ERRNO(ERROR_CONFIG, "Identity file does not exist: %s", path);
41 }
42 return SET_ERRNO_SYS(ERROR_CONFIG, "Failed to open identity file: %s", path);
43 }
44
45 // Read secret key (64 bytes)
46 size_t read = fread(secret_key, 1, 64, fp);
47 if (read != 64) {
48 fclose(fp);
49 return SET_ERRNO(ERROR_CONFIG, "Identity file corrupted (expected 64 bytes, got %zu): %s", read, path);
50 }
51
52 // Extract public key from secret key (last 32 bytes of Ed25519 secret key)
53 memcpy(public_key, secret_key + 32, 32);
54
55 fclose(fp);
56 log_info("Loaded identity from %s", path);
57 return ASCIICHAT_OK;
58}

61 if (!path || !public_key || !secret_key) {

104 if (!public_key || !fingerprint) {
