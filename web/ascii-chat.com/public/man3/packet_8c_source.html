45static uint64_t calculate_packet_timeout(size_t packet_size) {
46 uint64_t base_timeout = SEND_TIMEOUT * NS_PER_SEC_INT;
47
48 // For large packets, increase timeout proportionally
49 if (packet_size > LARGE_PACKET_THRESHOLD) {
50 // Add extra timeout per MB above the threshold
51 uint64_t extra_timeout =
52 (uint64_t)(((double)packet_size - LARGE_PACKET_THRESHOLD) / 1000000.0 * LARGE_PACKET_EXTRA_TIMEOUT_PER_MB) +
53 NS_PER_MS_INT;
54 uint64_t total_timeout = base_timeout + extra_timeout;
55
56 // Ensure client timeout is longer than server's RECV_TIMEOUT (30s) to prevent deadlock
57 // Add 10 seconds buffer to account for server processing delays
58 if (total_timeout < MIN_CLIENT_TIMEOUT) {
59 total_timeout = MIN_CLIENT_TIMEOUT;
60 }
61
62 // Cap at maximum timeout
63 return (total_timeout > MAX_CLIENT_TIMEOUT) ? MAX_CLIENT_TIMEOUT : total_timeout;
64 }
65
66 return base_timeout;
67}

77static asciichat_error_t packet_validate_header(const packet_header_t *header, uint16_t *pkt_type, uint32_t *pkt_len,
78 uint32_t *expected_crc) {
79 if (!header || !pkt_type || !pkt_len || !expected_crc) {
80 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters: header=%p, pkt_type=%p, pkt_len=%p, expected_crc=%p",
81 header, pkt_type, pkt_len, expected_crc);
82 }
83
84 // First validate packet length BEFORE converting from network byte order
85 // This prevents potential integer overflow issues
86 uint32_t pkt_len_network = header->length;
87 if (pkt_len_network == 0xFFFFFFFF) {
88 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid packet length in network byte order: 0xFFFFFFFF");
89 }
90
91 // Convert from network byte order using safe helpers
92 uint64_t magic = endian_unpack_u64(header->magic);
93 uint16_t type = endian_unpack_u16(header->type);
94 uint32_t len = endian_unpack_u32(pkt_len_network);
95 uint32_t crc = endian_unpack_u32(header->crc32);
96
97 // Validate magic
98 if (magic != PACKET_MAGIC) {
99 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid packet magic: 0x%x (expected 0x%x)", magic, PACKET_MAGIC);
100 }
101
102 // Validate packet size with bounds checking
103 if (len > MAX_PACKET_SIZE) {
104 return SET_ERRNO(ERROR_NETWORK_SIZE, "Packet too large: %u > %d", len, MAX_PACKET_SIZE);
105 }
106
107 // Validate packet type and size constraints
108 switch (type) {
109 case PACKET_TYPE_PROTOCOL_VERSION:
110 // Protocol version packet has fixed size
111 if (len != sizeof(protocol_version_packet_t)) {
112 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid protocol version packet size: %u, expected %zu", len,
113 sizeof(protocol_version_packet_t));
114 }
115 break;
116 case PACKET_TYPE_ASCII_FRAME:
117 // ASCII frame contains header + frame data
118 if (len < sizeof(ascii_frame_packet_t)) {
119 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid ASCII frame packet size: %u, minimum %zu", len,
120 sizeof(ascii_frame_packet_t));
121 }
122 break;
123 case PACKET_TYPE_IMAGE_FRAME:
124 // Image frame contains header + pixel data
125 if (len < sizeof(image_frame_packet_t)) {
126 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid image frame packet size: %u, minimum %zu", len,
127 sizeof(image_frame_packet_t));
128 }
129 break;
130 case PACKET_TYPE_AUDIO_BATCH:
131 // Batch must have at least header + some samples
132 if (len < sizeof(audio_batch_packet_t) + sizeof(float)) {
133 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid audio batch packet size: %u", len);
134 }
135 break;
136 case PACKET_TYPE_PING:
137 case PACKET_TYPE_PONG:
138 if (len != 0) { // Ping/pong are just header packets and no payload.
139 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid ping/pong packet size: %u", len);
140 }
141 break;
142 case PACKET_TYPE_CLIENT_CAPABILITIES:
143 // Client capabilities packet can be empty or contain capability data
144 if (len > 1024) { // Reasonable limit for capability data
145 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid client capabilities packet size: %u", len);
146 }
147 break;
148 case PACKET_TYPE_CLIENT_JOIN:
149 if (len != sizeof(client_info_packet_t)) {
150 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid client join packet size: %u, expected %zu", len,
151 sizeof(client_info_packet_t));
152 }
153 break;
154 case PACKET_TYPE_CLIENT_LEAVE:
155 // Client leave packet can be empty or contain reason
156 if (len > 256) {
157 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid client leave packet size: %u", len);
158 }
159 break;
160 case PACKET_TYPE_STREAM_START:
161 case PACKET_TYPE_STREAM_STOP:
162 if (len != sizeof(uint32_t)) {
163 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid stream control packet size: %u, expected %zu", len,
164 sizeof(uint32_t));
165 }
166 break;
167 case PACKET_TYPE_SIZE_MESSAGE:
168 // Size message format: "SIZE:width,height"
169 if (len == 0 || len > 32) { // Reasonable size for "SIZE:1234,5678"
170 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid size message packet size: %u", len);
171 }
172 break;
173 case PACKET_TYPE_AUDIO_MESSAGE:
174 // Audio message format: "AUDIO:num_samples"
175 if (len == 0 || len > 32) { // Reasonable size for "AUDIO:1234"
176 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid audio message packet size: %u", len);
177 }
178 break;
179 case PACKET_TYPE_TEXT_MESSAGE:
180 // Text message can be empty or contain message
181 if (len > 1024) { // Reasonable limit for text messages
182 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid text message packet size: %u", len);
183 }
184 break;
185 case PACKET_TYPE_ERROR_MESSAGE:
186 if (len < sizeof(error_packet_t)) {
187 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid error packet size: %u (minimum %zu)", len,
188 sizeof(error_packet_t));
189 }
190 if (len > sizeof(error_packet_t) + MAX_ERROR_MESSAGE_LENGTH) {
191 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Error packet message too large: %u (max %zu)", len,
192 sizeof(error_packet_t) + (size_t)MAX_ERROR_MESSAGE_LENGTH);
193 }
194 break;
195 // All crypto handshake packet types - validate using session parameters
196 case PACKET_TYPE_CRYPTO_CAPABILITIES:
197 case PACKET_TYPE_CRYPTO_PARAMETERS:
198 case PACKET_TYPE_CRYPTO_KEY_EXCHANGE_INIT:
199 case PACKET_TYPE_CRYPTO_KEY_EXCHANGE_RESP:
200 case PACKET_TYPE_CRYPTO_AUTH_CHALLENGE:
201 case PACKET_TYPE_CRYPTO_AUTH_RESPONSE:
202 case PACKET_TYPE_CRYPTO_AUTH_FAILED:
203 case PACKET_TYPE_CRYPTO_SERVER_AUTH_RESP:
204 case PACKET_TYPE_CRYPTO_HANDSHAKE_COMPLETE:
205 case PACKET_TYPE_CRYPTO_NO_ENCRYPTION:
206 case PACKET_TYPE_ENCRYPTED:
207 // Crypto packets are validated by the crypto handshake context
208 // This is just a basic sanity check for extremely large packets
209 if (len > 65536) { // 64KB should be enough for even large post-quantum crypto packets
210 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Crypto packet too large: %u bytes (max 65536)", len);
211 }
212 // Note: Proper size validation is done in crypto_handshake_validate_packet_size()
213 // which uses the session's crypto parameters for accurate validation
214 break;
215 // ACIP protocol packets (Discovery Service)
216 case PACKET_TYPE_ACIP_SESSION_CREATE:
217 case PACKET_TYPE_ACIP_SESSION_CREATED:
218 case PACKET_TYPE_ACIP_SESSION_LOOKUP:
219 case PACKET_TYPE_ACIP_SESSION_INFO:
220 case PACKET_TYPE_ACIP_SESSION_JOIN:
221 case PACKET_TYPE_ACIP_SESSION_JOINED:
222 case PACKET_TYPE_ACIP_SESSION_LEAVE:
223 case PACKET_TYPE_ACIP_SESSION_END:
224 case PACKET_TYPE_ACIP_SESSION_RECONNECT:
225 case PACKET_TYPE_ACIP_WEBRTC_SDP:
226 case PACKET_TYPE_ACIP_WEBRTC_ICE:
227 case PACKET_TYPE_ACIP_STRING_RESERVE:
228 case PACKET_TYPE_ACIP_STRING_RESERVED:
229 case PACKET_TYPE_ACIP_STRING_RENEW:
230 case PACKET_TYPE_ACIP_STRING_RELEASE:
231 case PACKET_TYPE_ACIP_DISCOVERY_PING:
232 case PACKET_TYPE_ACIP_ERROR:
233 // ACIP packets - basic size validation
234 // Discovery service packets can vary in size (variable-length strings, etc.)
235 if (len > 65536) { // 64KB max for discovery packets
236 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "ACIP packet too large: %u bytes (max 65536)", len);
237 }
238 break;
239 default:
240 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Unknown packet type: %u", type);
241 }
242
243 // Return parsed values
244 *pkt_type = type;
245 *pkt_len = len;
246 *expected_crc = crc;
247
248 return ASCIICHAT_OK;
249}

258static asciichat_error_t packet_validate_crc32(const void *data, size_t len, uint32_t expected_crc) {
259 if (!data && len > 0) {
260 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters: data=%p but len=%zu", data, len);
261 }
262
263 if (len == 0) {
264 // Empty packets should have CRC32 of 0
265 if (expected_crc != 0) {
266 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid CRC32 for empty packet: 0x%x (expected 0)", expected_crc);
267 }
268 return 0;
269 }
270
271 uint32_t calculated_crc = asciichat_crc32(data, len);
272 if (calculated_crc != expected_crc) {
273 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "CRC32 mismatch: calculated 0x%x, expected 0x%x", calculated_crc,
274 expected_crc);
275 }
276
277 return ASCIICHAT_OK;
278}

288asciichat_error_t packet_send(socket_t sockfd, packet_type_t type, const void *data, size_t len) {
289 if (sockfd == INVALID_SOCKET_VALUE) {
290 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid socket descriptor");
291 }
292
293 if (len > MAX_PACKET_SIZE) {
294 return SET_ERRNO(ERROR_NETWORK_SIZE, "Packet too large: %zu > %d", len, MAX_PACKET_SIZE);
295 }
296
297 packet_header_t header = {.magic = HOST_TO_NET_U64(PACKET_MAGIC),
298 .type = HOST_TO_NET_U16((uint16_t)type),
299 .length = HOST_TO_NET_U32((uint32_t)len),
300 .crc32 = HOST_TO_NET_U32(len > 0 ? asciichat_crc32(data, len) : 0),
301 .client_id = HOST_TO_NET_U32(0)}; // Always initialize client_id to 0 in network byte order
302
303 // Calculate timeout based on packet size (in nanoseconds)
304 uint64_t timeout = calculate_packet_timeout(len);
305
306 // Send header first
307 ssize_t sent = send_with_timeout(sockfd, &header, sizeof(header), timeout);
308 if (sent < 0) {
309 // Error context is already set by send_with_timeout
310 return ERROR_NETWORK;
311 }
312 if ((size_t)sent != sizeof(header)) {
313 return SET_ERRNO(ERROR_NETWORK, "Failed to fully send packet header. Sent %zd/%zu bytes", sent, sizeof(header));
314 }
315
316 // Send payload if present
317 if (len > 0 && data) {
318 // Check socket validity before sending payload to avoid race conditions
319 if (!socket_is_valid(sockfd)) {
320 return SET_ERRNO(ERROR_NETWORK, "Socket became invalid between header and payload send");
321 }
322 sent = send_with_timeout(sockfd, data, len, timeout);
323 // Check for error first to avoid signed/unsigned comparison issues
324 if (sent < 0) {
325 // Error context is already set by send_with_timeout
326 return ERROR_NETWORK;
327 }
328 if ((size_t)sent != len) {
329 return SET_ERRNO(ERROR_NETWORK, "Failed to fully send packet payload. Sent %zd/%zu bytes", sent, len);
330 }
331 }
332
333#ifdef DEBUG_NETWORK
334 log_debug("Sent packet type=%d, len=%zu, errno=%d (%s)", type, len, errno, SAFE_STRERROR(errno));
335#endif
336
337 return 0;
338}

348asciichat_error_t packet_receive(socket_t sockfd, packet_type_t *type, void **data, size_t *len) {
349 if (sockfd == INVALID_SOCKET_VALUE) {
350 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid socket descriptor");
351 }
352 if (!type || !data || !len) {
353 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters: type=%p, data=%p, len=%p", type, data, len);
354 }
355
356 // Read packet header into memory from network socket
357 packet_header_t header;
358 uint64_t header_timeout_ns = RECV_TIMEOUT * NS_PER_SEC_INT;
359 ssize_t received = recv_with_timeout(sockfd, &header, sizeof(header), header_timeout_ns);
360 if (received < 0) {
361 // Error context is already set by recv_with_timeout
362 return ERROR_NETWORK;
363 }
364 if ((size_t)received != sizeof(header)) {
365 if (received == 0) {
366 log_warn("Connection closed while reading packet header");
367 return SET_ERRNO(ERROR_NETWORK, "Connection closed by peer while reading packet header");
368 }
369
370 return SET_ERRNO(ERROR_NETWORK, "Partial packet header received: %zd/%zu bytes", received, sizeof(header));
371 }
372
373 // Validate packet header
374 uint16_t pkt_type;
375 uint32_t pkt_len;
376 uint32_t expected_crc;
377 if (packet_validate_header(&header, &pkt_type, &pkt_len, &expected_crc) != ASCIICHAT_OK) {
378 // Error context is already set by packet_validate_header
379 return ERROR_NETWORK_PROTOCOL;
380 }
381
382 // Allocate buffer for payload
383 void *payload = NULL;
384 if (pkt_len > 0) {
385 payload = buffer_pool_alloc(NULL, pkt_len);
386
387 // Use adaptive timeout for large packets
388 uint64_t recv_timeout = calculate_packet_timeout(pkt_len);
389 received = recv_with_timeout(sockfd, payload, pkt_len, recv_timeout);
390 if (received < 0) {
391 buffer_pool_free(NULL, payload, pkt_len);
392 return SET_ERRNO_SYS(ERROR_NETWORK, "Failed to receive packet payload");
393 }
394 if (received != (ssize_t)pkt_len) {
395 buffer_pool_free(NULL, payload, pkt_len);
396 return SET_ERRNO(ERROR_NETWORK, "Partial packet payload received: %zd/%u bytes", received, pkt_len);
397 }
398
399 // Validate CRC32
400 if (packet_validate_crc32(payload, pkt_len, expected_crc) != ASCIICHAT_OK) {
401 buffer_pool_free(NULL, payload, pkt_len);
402 // Error context is already set by packet_validate_crc32
403 return ERROR_NETWORK_PROTOCOL;
404 }
405 }
406
407 // Return results
408 *type = (packet_type_t)pkt_type;
409 *data = payload;
410 *len = pkt_len;
411
412 return ASCIICHAT_OK;
413}

430asciichat_error_t send_packet_secure(socket_t sockfd, packet_type_t type, const void *data, size_t len,
431 crypto_context_t *crypto_ctx) {
432 if (len > MAX_PACKET_SIZE) {
433 return SET_ERRNO(ERROR_NETWORK_SIZE, "Packet too large: %zu > %d", len, MAX_PACKET_SIZE);
434 }
435
436 // Handshake packets are ALWAYS sent unencrypted
437 if (packet_is_handshake_type(type)) {
438 return packet_send(sockfd, type, data, len);
439 }
440
441 // Apply compression if beneficial for large packets
442 const void *final_data = data;
443 size_t final_len = len;
444 void *compressed_data = NULL;
445
446 // Skip compression for pre-compressed data (Opus audio) or if --no-compress flag is set
447 bool should_skip_compression = packet_is_precompressed(type) || GET_OPTION(no_compress);
448 if (!should_skip_compression && len > COMPRESSION_MIN_SIZE && should_compress(len, len)) {
449 void *temp_compressed = NULL;
450 size_t compressed_size = 0;
451
452 // Use configured compression level from options (default: 1 for fastest compression)
453 int compression_level = (GET_OPTION(compression_level) > 0) ? GET_OPTION(compression_level) : 1;
454 asciichat_error_t compress_result = compress_data(data, len, &temp_compressed, &compressed_size, compression_level);
455 if (compress_result == ASCIICHAT_OK) {
456 double ratio = (double)compressed_size / (double)len;
457 if (ratio < COMPRESSION_RATIO_THRESHOLD) {
458 final_data = temp_compressed;
459 final_len = compressed_size;
460 compressed_data = temp_compressed;
461 log_debug("Compressed packet: %zu -> %zu bytes (%.1f%%)", len, compressed_size, ratio * 100.0);
462 } else {
463 SAFE_FREE(temp_compressed);
464 }
465 }
466 }
467
468 // If no crypto context or crypto not ready, send unencrypted
469 bool ready = crypto_ctx ? crypto_is_ready(crypto_ctx) : false;
470 if (!crypto_ctx || !ready) {
471 log_warn_every(LOG_RATE_FAST, "CRYPTO_DEBUG: Sending packet type %d UNENCRYPTED (crypto_ctx=%p, ready=%d)", type,
472 (void *)crypto_ctx, ready);
473 asciichat_error_t result = packet_send(sockfd, type, final_data, final_len);
474 if (compressed_data) {
475 SAFE_FREE(compressed_data);
476 }
477 if (result != ASCIICHAT_OK) {
478 SET_ERRNO(ERROR_NETWORK, "Failed to send packet: %s", asciichat_error_string(result));
479 }
480 return result;
481 }
482
483 // Encrypt the packet: create header + payload, encrypt everything, wrap in PACKET_TYPE_ENCRYPTED
484 packet_header_t header = {.magic = HOST_TO_NET_U64(PACKET_MAGIC),
485 .type = HOST_TO_NET_U16((uint16_t)type),
486 .length = HOST_TO_NET_U32((uint32_t)final_len),
487 .crc32 = HOST_TO_NET_U32(final_len > 0 ? asciichat_crc32(final_data, final_len) : 0),
488 .client_id = HOST_TO_NET_U32(0)}; // Always 0 - client_id is not used in practice
489
490 // Combine header + payload for encryption
491 // Check for integer overflow before addition
492 if (final_len > SIZE_MAX - sizeof(header)) {
493 if (compressed_data) {
494 SAFE_FREE(compressed_data);
495 }
496 return SET_ERRNO(ERROR_NETWORK_SIZE, "Packet too large: would overflow plaintext buffer size");
497 }
498 size_t plaintext_len = sizeof(header) + final_len;
499 uint8_t *plaintext = buffer_pool_alloc(NULL, plaintext_len);
500 if (!plaintext) {
501 if (compressed_data) {
502 SAFE_FREE(compressed_data);
503 }
504 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate buffer for plaintext packet");
505 }
506
507 memcpy(plaintext, &header, sizeof(header));
508 if (final_len > 0 && final_data) {
509 memcpy(plaintext + sizeof(header), final_data, final_len);
510 }
511
512 // Encrypt
513 // Check for integer overflow before calculating ciphertext size
514 if (plaintext_len > SIZE_MAX - CRYPTO_NONCE_SIZE - CRYPTO_MAC_SIZE) {
515 buffer_pool_free(NULL, plaintext, plaintext_len);
516 if (compressed_data) {
517 SAFE_FREE(compressed_data);
518 }
519 return SET_ERRNO(ERROR_NETWORK_SIZE, "Packet too large: would overflow ciphertext buffer size");
520 }
521 size_t ciphertext_size = plaintext_len + CRYPTO_NONCE_SIZE + CRYPTO_MAC_SIZE;
522 uint8_t *ciphertext = buffer_pool_alloc(NULL, ciphertext_size);
523 if (!ciphertext) {
524 buffer_pool_free(NULL, plaintext, plaintext_len);
525 if (compressed_data) {
526 SAFE_FREE(compressed_data);
527 }
528 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate buffer for ciphertext");
529 }
530
531 size_t ciphertext_len;
532 crypto_result_t result =
533 crypto_encrypt(crypto_ctx, plaintext, plaintext_len, ciphertext, ciphertext_size, &ciphertext_len);
534 buffer_pool_free(NULL, plaintext, plaintext_len);
535
536 if (result != CRYPTO_OK) {
537 buffer_pool_free(NULL, ciphertext, ciphertext_size);
538 if (compressed_data) {
539 SAFE_FREE(compressed_data);
540 }
541 return SET_ERRNO(ERROR_CRYPTO, "Failed to encrypt packet: %s", crypto_result_to_string(result));
542 }
543
544 // Send as PACKET_TYPE_ENCRYPTED
545 log_debug_every(LOG_RATE_SLOW, "CRYPTO_DEBUG: Sending encrypted packet (original type %d as PACKET_TYPE_ENCRYPTED)",
546 type);
547 asciichat_error_t send_result = packet_send(sockfd, PACKET_TYPE_ENCRYPTED, ciphertext, ciphertext_len);
548 buffer_pool_free(NULL, ciphertext, ciphertext_size);
549
550 if (compressed_data) {
551 SAFE_FREE(compressed_data);
552 }
553
554 return send_result;
555}

565packet_recv_result_t receive_packet_secure(socket_t sockfd, void *crypto_ctx, bool enforce_encryption,
566 packet_envelope_t *envelope) {
567
568 if (!envelope) {
569 SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters: envelope=%p", envelope);
570 return PACKET_RECV_ERROR;
571 }
572
573 // Initialize envelope
574 memset(envelope, 0, sizeof(*envelope));
575
576 // Receive packet header
577 packet_header_t header;
578 uint64_t header_timeout_ns = RECV_TIMEOUT * NS_PER_SEC_INT;
579 ssize_t received = recv_with_timeout(sockfd, &header, sizeof(header), header_timeout_ns);
580
581 // Check for errors first (before comparing signed with unsigned)
582 if (received < 0) {
583 SET_ERRNO(ERROR_NETWORK, "Failed to receive packet header: %zd/%zu bytes", received, sizeof(header));
584 return PACKET_RECV_ERROR;
585 }
586
587 if (received == 0) {
588 return PACKET_RECV_EOF;
589 }
590
591 if ((size_t)received != sizeof(header)) {
592 SET_ERRNO(ERROR_NETWORK, "Failed to receive packet header: %zd/%zu bytes", received, sizeof(header));
593 return PACKET_RECV_ERROR;
594 }
595
596 // Convert from network byte order
597 uint64_t magic = NET_TO_HOST_U64(header.magic);
598 uint16_t pkt_type = NET_TO_HOST_U16(header.type);
599 uint32_t pkt_len = NET_TO_HOST_U32(header.length);
600 uint32_t expected_crc = NET_TO_HOST_U32(header.crc32);
601
602 // DEBUG: Log raw header bytes for debugging packet misalignment
603 char header_hex[95];
604 for (size_t i = 0; i < sizeof(header); i++) {
605 snprintf(&header_hex[i * 2], 3, "%02x", ((uint8_t *)&header)[i]);
606 }
607 header_hex[94] = '\0';
608
609 // DEBUG: Log all received packet types in detail
610 log_info("[RECV_PKT_HEADER] 🔍 Received packet: type=%u (0x%04x), len=%u, client_id=%u, raw=%s", pkt_type, pkt_type,
611 pkt_len, header.client_id, header_hex);
612
613 // Validate magic number
614 if (magic != PACKET_MAGIC) {
615 SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid packet magic: 0x%llx (expected 0x%llx)", magic, PACKET_MAGIC);
616 return PACKET_RECV_ERROR;
617 }
618
619 // Validate packet size
620 if (pkt_len > MAX_PACKET_SIZE) {
621 SET_ERRNO(ERROR_NETWORK_SIZE, "Packet too large: %u > %d", pkt_len, MAX_PACKET_SIZE);
622 return PACKET_RECV_ERROR;
623 }
624
625 // Handle encrypted packets
626 if (pkt_type == PACKET_TYPE_ENCRYPTED) {
627 if (!crypto_ctx) {
628 SET_ERRNO(ERROR_CRYPTO, "Received encrypted packet but no crypto context");
629 return PACKET_RECV_ERROR;
630 }
631
632 // Read encrypted payload
633 uint8_t *ciphertext = buffer_pool_alloc(NULL, pkt_len);
634 if (!ciphertext) {
635 SET_ERRNO(ERROR_MEMORY, "Failed to allocate buffer for ciphertext");
636 return PACKET_RECV_ERROR;
637 }
638
639 uint64_t recv_timeout = calculate_packet_timeout(pkt_len);
640 received = recv_with_timeout(sockfd, ciphertext, pkt_len, recv_timeout);
641 if (received != (ssize_t)pkt_len) {
642 SET_ERRNO(ERROR_NETWORK, "Failed to receive encrypted payload: %zd/%u bytes", received, pkt_len);
643 buffer_pool_free(NULL, ciphertext, pkt_len);
644 return PACKET_RECV_ERROR;
645 }
646
647 // Decrypt - allocate plaintext buffer with extra space
648 // pkt_len is already validated to be <= MAX_PACKET_SIZE, so adding 1024 cannot overflow
649 size_t plaintext_size = (size_t)pkt_len + 1024;
650 uint8_t *plaintext = buffer_pool_alloc(NULL, plaintext_size);
651 if (!plaintext) {
652 SET_ERRNO(ERROR_MEMORY, "Failed to allocate buffer for plaintext");
653 buffer_pool_free(NULL, ciphertext, pkt_len);
654 return PACKET_RECV_ERROR;
655 }
656
657 size_t plaintext_len;
658 crypto_result_t result = crypto_decrypt(crypto_ctx, ciphertext, pkt_len, plaintext, plaintext_size, &plaintext_len);
659 buffer_pool_free(NULL, ciphertext, pkt_len);
660
661 if (result != CRYPTO_OK) {
662 SET_ERRNO(ERROR_CRYPTO, "Failed to decrypt packet: %s", crypto_result_to_string(result));
663 buffer_pool_free(NULL, plaintext, plaintext_size);
664 return PACKET_RECV_ERROR;
665 }
666
667 if (plaintext_len < sizeof(packet_header_t)) {
668 SET_ERRNO(ERROR_CRYPTO, "Decrypted packet too small: %zu < %zu", plaintext_len, sizeof(packet_header_t));
669 buffer_pool_free(NULL, plaintext, plaintext_size);
670 return PACKET_RECV_ERROR;
671 }
672
673 // Parse decrypted header
674 packet_header_t *decrypted_header = (packet_header_t *)plaintext;
675 pkt_type = NET_TO_HOST_U16(decrypted_header->type);
676
677 // DEBUG: Log decrypted packet type
678 log_info("[RECV_PKT_DECRYPTED] 🔐 After decryption: actual_type=%u (0x%04x), len=%u", pkt_type, pkt_type,
679 NET_TO_HOST_U32(decrypted_header->length));
680 pkt_len = NET_TO_HOST_U32(decrypted_header->length);
681 expected_crc = NET_TO_HOST_U32(decrypted_header->crc32);
682
683 // Extract payload
684 size_t payload_len = plaintext_len - sizeof(packet_header_t);
685 if (payload_len != pkt_len) {
686 SET_ERRNO(ERROR_CRYPTO, "Decrypted payload size mismatch: %zu != %u", payload_len, pkt_len);
687 buffer_pool_free(NULL, plaintext, plaintext_size);
688 return PACKET_RECV_ERROR;
689 }
690
691 // Verify CRC
692 if (pkt_len > 0) {
693 uint32_t actual_crc = asciichat_crc32(plaintext + sizeof(packet_header_t), pkt_len);
694 if (actual_crc != expected_crc) {
695 SET_ERRNO(ERROR_CRYPTO, "Decrypted packet CRC mismatch: 0x%x != 0x%x", actual_crc, expected_crc);
696 buffer_pool_free(NULL, plaintext, plaintext_size);
697 return PACKET_RECV_ERROR;
698 }
699 }
700
701 // Set envelope
702 envelope->type = (packet_type_t)pkt_type;
703 envelope->data = plaintext + sizeof(packet_header_t);
704 envelope->len = pkt_len;
705 envelope->allocated_buffer = plaintext;
706 envelope->allocated_size = plaintext_size;
707
708 return PACKET_RECV_SUCCESS;
709 }
710
711 // Handle unencrypted packets
712 if (enforce_encryption && !packet_is_handshake_type(pkt_type)) {
713 SET_ERRNO(ERROR_CRYPTO, "Received unencrypted packet type %d but encryption is required", pkt_type);
714 return PACKET_RECV_ERROR;
715 }
716
717 // Read payload
718 if (pkt_len > 0) {
719 uint8_t *payload = buffer_pool_alloc(NULL, pkt_len);
720 if (!payload) {
721 SET_ERRNO(ERROR_MEMORY, "Failed to allocate buffer for payload");
722 return PACKET_RECV_ERROR;
723 }
724
725 uint64_t recv_timeout = calculate_packet_timeout(pkt_len);
726 received = recv_with_timeout(sockfd, payload, pkt_len, recv_timeout);
727 if (received != (ssize_t)pkt_len) {
728 SET_ERRNO(ERROR_NETWORK, "Failed to receive payload: %zd/%u bytes", received, pkt_len);
729 buffer_pool_free(NULL, payload, pkt_len);
730 return PACKET_RECV_ERROR;
731 }
732
733 // Verify CRC
734 uint32_t actual_crc = asciichat_crc32(payload, pkt_len);
735 if (actual_crc != expected_crc) {
736 SET_ERRNO(ERROR_NETWORK, "Packet CRC mismatch: 0x%x != 0x%x", actual_crc, expected_crc);
737 buffer_pool_free(NULL, payload, pkt_len);
738 return PACKET_RECV_ERROR;
739 }
740
741 envelope->data = payload;
742 envelope->allocated_buffer = payload;
743 envelope->allocated_size = pkt_len;
744 }
745
746 envelope->type = (packet_type_t)pkt_type;
747 envelope->len = pkt_len;
748
749 return PACKET_RECV_SUCCESS;
750}

824asciichat_error_t packet_decrypt_envelope(packet_envelope_t *envelope, void *crypto_ctx) {
825 if (!envelope) {
826 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid envelope");
827 }
828
829 // Only decrypt if this is an encrypted packet
830 if (envelope->type != PACKET_TYPE_ENCRYPTED) {
831 return ASCIICHAT_OK; // Not encrypted, nothing to do
832 }
833
834 if (!crypto_ctx) {
835 return SET_ERRNO(ERROR_CRYPTO, "Received encrypted packet but no crypto context");
836 }
837
838 uint8_t *ciphertext = (uint8_t *)envelope->data;
839 size_t ciphertext_len = envelope->len;
840
841 // Allocate plaintext buffer with extra space for decryption
842 size_t plaintext_size = ciphertext_len + 1024;
843 uint8_t *plaintext = buffer_pool_alloc(NULL, plaintext_size);
844 if (!plaintext) {
845 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate plaintext buffer for decryption");
846 }
847
848 size_t plaintext_len;
849 crypto_result_t result =
850 crypto_decrypt(crypto_ctx, ciphertext, ciphertext_len, plaintext, plaintext_size, &plaintext_len);
851
852 if (result != CRYPTO_OK) {
853 buffer_pool_free(NULL, plaintext, plaintext_size);
854 return SET_ERRNO(ERROR_CRYPTO, "Failed to decrypt packet: %s", crypto_result_to_string(result));
855 }
856
857 if (plaintext_len < sizeof(packet_header_t)) {
858 buffer_pool_free(NULL, plaintext, plaintext_size);
859 return SET_ERRNO(ERROR_CRYPTO, "Decrypted packet too small: %zu < %zu", plaintext_len, sizeof(packet_header_t));
860 }
861
862 // Parse inner packet header
863 const packet_header_t *inner_header = (const packet_header_t *)plaintext;
864 packet_type_t inner_type = NET_TO_HOST_U16(inner_header->type);
865 uint32_t inner_len = NET_TO_HOST_U32(inner_header->length);
866
867 // Validate inner packet length
868 if (inner_len != plaintext_len - sizeof(packet_header_t)) {
869 buffer_pool_free(NULL, plaintext, plaintext_size);
870 return SET_ERRNO(ERROR_CRYPTO, "Inner packet length mismatch: header=%u, actual=%zu", inner_len,
871 plaintext_len - sizeof(packet_header_t));
872 }
873
874 // Free the original encrypted buffer if it's different from plaintext
875 if (envelope->allocated_buffer && envelope->allocated_buffer != plaintext) {
876 buffer_pool_free(NULL, envelope->allocated_buffer, envelope->allocated_size);
877 }
878
879 // Update envelope with decrypted packet info
880 envelope->type = inner_type;
881 envelope->data = (uint8_t *)plaintext + sizeof(packet_header_t);
882 envelope->len = inner_len;
883 envelope->allocated_buffer = plaintext;
884 envelope->allocated_size = plaintext_size;
885
886 log_info("[PACKET_DECRYPT] 🔐 Decrypted PACKET_TYPE_ENCRYPTED: inner_type=%u (0x%04x), len=%u", inner_type,
887 inner_type, inner_len);
888
889 return ASCIICHAT_OK;
890}

892asciichat_error_t packet_send_error(socket_t sockfd, const crypto_context_t *crypto_ctx, asciichat_error_t error_code,
893 const char *message) {
894 if (sockfd == INVALID_SOCKET_VALUE) {
895 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid socket descriptor");
896 }
897
898 if (!message) {
899 message = "";
900 }
901
902 size_t message_len = strnlen(message, MAX_ERROR_MESSAGE_LENGTH);
903 if (message_len == MAX_ERROR_MESSAGE_LENGTH) {
904 log_warn("Error message truncated to %zu bytes", (size_t)MAX_ERROR_MESSAGE_LENGTH);
905 }
906
907 size_t payload_len = sizeof(error_packet_t) + message_len;
908 uint8_t *payload = SAFE_MALLOC(payload_len, uint8_t *);
909 if (!payload) {
910 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate %zu bytes for error packet", payload_len);
911 }
912
913 error_packet_t *packet = (error_packet_t *)payload;
914 packet->error_code = HOST_TO_NET_U32((uint32_t)error_code);
915 packet->message_length = HOST_TO_NET_U32((uint32_t)message_len);
916
917 if (message_len > 0) {
918 memcpy(payload + sizeof(error_packet_t), message, message_len);
919 }
920
921 bool encryption_ready = crypto_ctx && crypto_is_ready(crypto_ctx);
922 asciichat_error_t send_result;
923
924 if (encryption_ready) {
925 send_result =
926 send_packet_secure(sockfd, PACKET_TYPE_ERROR_MESSAGE, payload, payload_len, (crypto_context_t *)crypto_ctx);
927 } else {
928 send_result = packet_send(sockfd, PACKET_TYPE_ERROR_MESSAGE, payload, payload_len);
929 }
930 SAFE_FREE(payload);
931
932 if (send_result != ASCIICHAT_OK) {
933 return SET_ERRNO(ERROR_NETWORK, "Failed to send error packet: %s", asciichat_error_string(send_result));
934 }
935
936 return ASCIICHAT_OK;
937}

939asciichat_error_t packet_parse_error_message(const void *data, size_t len, asciichat_error_t *out_error_code,
940 char *message_buffer, size_t message_buffer_size,
941 size_t *out_message_length) {
942 if (!data || len < sizeof(error_packet_t) || !out_error_code || !message_buffer || message_buffer_size == 0) {
943 return SET_ERRNO(ERROR_INVALID_PARAM,
944 "Invalid parameters: data=%p len=%zu out_error_code=%p message_buffer=%p buffer_size=%zu", data,
945 len, out_error_code, message_buffer, message_buffer_size);
946 }
947
948 const error_packet_t *packet = (const error_packet_t *)data;
949 uint32_t raw_error_code = NET_TO_HOST_U32(packet->error_code);
950 uint32_t raw_message_length = NET_TO_HOST_U32(packet->message_length);
951
952 if (raw_message_length > MAX_ERROR_MESSAGE_LENGTH) {
953 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Error message length too large: %u", raw_message_length);
954 }
955
956 size_t total_required = sizeof(error_packet_t) + (size_t)raw_message_length;
957 if (total_required > len) {
958 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Error packet truncated: expected %zu bytes, have %zu", total_required,
959 len);
960 }
961
962 const uint8_t *message_bytes = (const uint8_t *)data + sizeof(error_packet_t);
963 size_t copy_len = raw_message_length;
964 if (copy_len >= message_buffer_size) {
965 copy_len = message_buffer_size - 1;
966 }
967
968 if (copy_len > 0) {
969 memcpy(message_buffer, message_bytes, copy_len);
970 }
971 message_buffer[copy_len] = '\0';
972
973 if (out_message_length) {
974 *out_message_length = raw_message_length;
975 }
976
977 *out_error_code = (asciichat_error_t)raw_error_code;
978 return ASCIICHAT_OK;
979}

981asciichat_error_t packet_send_remote_log(socket_t sockfd, const crypto_context_t *crypto_ctx, log_level_t level,
982 remote_log_direction_t direction, uint16_t flags, const char *message) {

984 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid socket descriptor");
985 }
986
987 if (level < LOG_DEV || level > LOG_FATAL) {
988 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid log level: %d", level);
989 }
990
991 const char *safe_message = message ? message : "";
992 bool truncated = false;
993 size_t message_len = strnlen(safe_message, MAX_REMOTE_LOG_MESSAGE_LENGTH);
994 if (message_len == MAX_REMOTE_LOG_MESSAGE_LENGTH && safe_message[message_len] != '\0') {
995 truncated = true;
996 }
997
998 size_t payload_len = sizeof(remote_log_packet_t) + message_len;
999 uint8_t *payload = SAFE_MALLOC(payload_len, uint8_t *);
1000 if (!payload) {
1001 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate %zu bytes for remote log packet", payload_len);
1002 }
1003
1004 remote_log_packet_t *packet = (remote_log_packet_t *)payload;
1005 packet->log_level = (uint8_t)level;
1006 packet->direction = (uint8_t)direction;
1007 uint16_t final_flags = flags;
1008 if (truncated) {
1009 final_flags |= REMOTE_LOG_FLAG_TRUNCATED;
1010 }
1011 packet->flags = HOST_TO_NET_U16(final_flags);
1012 packet->message_length = HOST_TO_NET_U32((uint32_t)message_len);
1013
1014 if (message_len > 0) {
1015 memcpy(payload + sizeof(remote_log_packet_t), safe_message, message_len);
1016 }
1017
1018 bool encryption_ready = crypto_ctx && crypto_is_ready(crypto_ctx);
1019 asciichat_error_t send_result;
1020
1021 if (encryption_ready) {
1022 int secure_result =
1023 send_packet_secure(sockfd, PACKET_TYPE_REMOTE_LOG, payload, payload_len, (crypto_context_t *)crypto_ctx);
1024 send_result = secure_result == 0 ? ASCIICHAT_OK : ERROR_NETWORK;
1025 } else {
1026 send_result = packet_send(sockfd, PACKET_TYPE_REMOTE_LOG, payload, payload_len);
1027 }
1028
1029 SAFE_FREE(payload);
1030
1031 if (send_result != ASCIICHAT_OK) {
1032 return SET_ERRNO(ERROR_NETWORK, "Failed to send remote log packet: %d", send_result);
1033 }
1034
1035 return ASCIICHAT_OK;
1036}

1149// =============================================================================
1150// Audio Batch Packet Sending
1151// =============================================================================
1152// These functions were moved from lib/network/av.c during refactoring.
1153// They provide socket-level audio packet sending with encryption support.
1154// For transport-agnostic sending, use acip_send_audio_batch() instead.
1155// =============================================================================
1156
1157asciichat_error_t av_send_audio_opus_batch(socket_t sockfd, const uint8_t *opus_data, size_t opus_size,
1158 const uint16_t *frame_sizes, int sample_rate, int frame_duration,
1159 int frame_count, crypto_context_t *crypto_ctx) {
1160 if (!opus_data || opus_size == 0 || !frame_sizes || sample_rate <= 0 || frame_duration <= 0 || frame_count <= 0) {
1161 return SET_ERRNO(ERROR_INVALID_PARAM,
1162 "Invalid Opus batch parameters: opus_data=%p, opus_size=%zu, frame_sizes=%p, sample_rate=%d, "
1163 "frame_duration=%d, frame_count=%d",
1164 (const void *)opus_data, opus_size, (const void *)frame_sizes, sample_rate, frame_duration,
1165 frame_count);
1166 }
1167
1168 // Validate frame_count to prevent integer overflow in size calculations
1169 // Max reasonable frames per batch: ~1 second of 10ms frames = 100 frames
1170 if (frame_count > 1000) {
1171 return SET_ERRNO(ERROR_INVALID_PARAM, "Too many Opus frames: %d (max 1000)", frame_count);
1172 }
1173
1174 // Allocate buffer for header + frame sizes + encoded data
1175 size_t header_size = 16; // sample_rate (4), frame_duration (4), frame_count (4), reserved (4)
1176 size_t frame_sizes_bytes = (size_t)frame_count * sizeof(uint16_t);
1177 size_t total_size = header_size + frame_sizes_bytes + opus_size;
1178 void *packet_data = buffer_pool_alloc(NULL, total_size);
1179 if (!packet_data) {
1180 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate buffer for Opus batch packet: %zu bytes", total_size);
1181 }
1182
1183 // Write header (network byte order for cross-platform compatibility)
1184 uint8_t *buf = (uint8_t *)packet_data;
1185 uint32_t sr = HOST_TO_NET_U32((uint32_t)sample_rate);
1186 uint32_t fd = HOST_TO_NET_U32((uint32_t)frame_duration);
1187 uint32_t fc = HOST_TO_NET_U32((uint32_t)frame_count);
1188 memcpy(buf, &sr, 4);
1189 memcpy(buf + 4, &fd, 4);
1190 memcpy(buf + 8, &fc, 4);
1191 memset(buf + 12, 0, 4); // Reserved
1192
1193 // Write frame sizes array (convert each to network byte order)
1194 uint16_t *frame_sizes_out = (uint16_t *)(buf + header_size);
1195 for (int i = 0; i < frame_count; i++) {
1196 frame_sizes_out[i] = HOST_TO_NET_U16(frame_sizes[i]);
1197 }
1198
1199 // Copy Opus data
1200 memcpy(buf + header_size + frame_sizes_bytes, opus_data, opus_size);
1201
1202 // Send packet (with encryption support)
1203 asciichat_error_t result =
1204 send_packet_secure(sockfd, PACKET_TYPE_AUDIO_OPUS_BATCH, packet_data, total_size, crypto_ctx);
1205
1206 // Clean up
1207 buffer_pool_free(NULL, packet_data, total_size);
1208
1209 return result;
1210}
