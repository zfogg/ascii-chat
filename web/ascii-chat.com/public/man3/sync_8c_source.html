51static int format_mutex_timing(const mutex_t *mutex, char *buffer, size_t size) {
52 if (!mutex)
53 return 0;
54
55 // If mutex was never locked, return empty
56 if (mutex->last_lock_time_ns == 0) {
57 return 0;
58 }
59
60 int offset = 0;
61 uint64_t now_ns = time_get_ns();
62 char lock_str[64] = "";
63 char unlock_str[64] = "";
64 char held_str[256] = "";
65
66 if (mutex->last_lock_time_ns > 0 && mutex->last_lock_time_ns <= now_ns) {
67 char elapsed_str[64];
68 format_elapsed(now_ns - mutex->last_lock_time_ns, elapsed_str, sizeof(elapsed_str));
69 snprintf(lock_str, sizeof(lock_str), "lock=%s", elapsed_str);
70 }
71
72 if (mutex->last_unlock_time_ns > 0 && mutex->last_unlock_time_ns <= now_ns) {
73 char elapsed_str[64];
74 format_elapsed(now_ns - mutex->last_unlock_time_ns, elapsed_str, sizeof(elapsed_str));
75 snprintf(unlock_str, sizeof(unlock_str), "unlock=%s", elapsed_str);
76 }
77
78 if (mutex->currently_held_by_key != 0) {
79 char thread_name[256];
80 NAMED_GET_BY_PTR(mutex->currently_held_by_key, thread_name, sizeof(thread_name));
81 snprintf(held_str, sizeof(held_str), "[LOCKED_BY=%s]", thread_name);
82 } else {
83 snprintf(held_str, sizeof(held_str), "[FREE]");
84 }
85
86 offset += snprintf(buffer + offset, size - offset, "%s %s %s", lock_str, unlock_str, held_str);
87
88 return offset;
89}

194 snprintf(broadcast_str, sizeof(broadcast_str), "broadcast=%s", elapsed_str);

196

198 if (waiting_count > 0) {

201 snprintf(waiting_str, sizeof(waiting_str), "[WAITING=%" PRIu64 " threads, last=%s]", waiting_count, thread_name);
202 } else {
203 snprintf(status_str, sizeof(status_str), "[IDLE]");
204 }
205
206 offset += snprintf(buffer + offset, size - offset, "%s %s %s %s %s", wait_str, signal_str, broadcast_str, waiting_str,
207 status_str);
208
209 return offset;

335static void debug_sync_print_lock_stacks(char *buffer, size_t buffer_size, size_t *offset) {
336 mutex_stack_entry_t **all_stacks = NULL;
337 int *stack_counts = NULL;
338 int thread_count = 0;
339
340 if (mutex_stack_get_all_threads(&all_stacks, &stack_counts, &thread_count) != 0) {
341 return;
342 }
343
344 if (thread_count == 0) {
345 mutex_stack_free_all_threads(all_stacks, stack_counts, thread_count);
346 return;
347 }
348
349 *offset += snprintf(buffer + *offset, buffer_size - *offset, "\nThread Lock Stacks:\n");
350
351 for (int i = 0; i < thread_count; i++) {
352 int depth = stack_counts[i];
353 if (depth == 0)
354 continue;
355
356 *offset += snprintf(buffer + *offset, buffer_size - *offset, " Thread %d: %d lock(s)\n", i, depth);
357
358 for (int j = 0; j < depth; j++) {
359 const mutex_stack_entry_t *entry = &all_stacks[i][j];
360 const char *state_str = (entry->state == MUTEX_STACK_STATE_LOCKED) ? "LOCKED" : "PENDING";
361
362 uint64_t elapsed = time_get_ns() - entry->timestamp_ns;
363 char elapsed_str[64];
364 time_pretty(elapsed, -1, elapsed_str, sizeof(elapsed_str));
365
366 *offset += snprintf(buffer + *offset, buffer_size - *offset, " [%d] mutex @ %p (%s) %s", j,
367 (void *)entry->mutex_key, state_str, elapsed_str);
368 }
369 }
370
371 mutex_stack_free_all_threads(all_stacks, stack_counts, thread_count);
372}

378void debug_sync_print_state(void) {
379// Use a single large buffer for all sync state output
380#define SYNC_BUFFER_SIZE 65536 // Increased from 8192 to handle many syncs
381 log_debug("[debug_sync_print_state] ENTRY");
382
383 char *buffer = SAFE_MALLOC(SYNC_BUFFER_SIZE, char *);
384 if (!buffer) {
385 log_debug("[debug_sync_print_state] Failed to allocate buffer");
386 return;
387 }
388
389 sync_buffer_t buf = {.buffer = buffer, .buffer_size = SYNC_BUFFER_SIZE, .offset = 0};
390
391 // Iterate through all registered syncs
392 log_debug("[debug_sync_print_state] Iterating mutexes");
393 named_registry_for_each(mutex_iter_callback, &buf);
394 log_debug("[debug_sync_print_state] Iterating rwlocks");
395 named_registry_for_each(rwlock_iter_callback, &buf);
396 log_debug("[debug_sync_print_state] Iterating conds");
397 named_registry_for_each(cond_iter_callback, &buf);
398
399 named_registry_for_each(atomic_t_iter_callback, &buf);
400 named_registry_for_each(atomic_ptr_iter_callback, &buf);
401
402 // Print lock stacks for deadlock analysis
403 log_debug("[debug_sync_print_state] Getting lock stacks");
404 debug_sync_print_lock_stacks(buf.buffer, buf.buffer_size, &buf.offset);
405
406 // Log everything in one call
407 log_debug("[debug_sync_print_state] Buffer size: %zu bytes", buf.offset);
408 if (buf.offset > 0) {
409 log_info("SYNC_STATE:\n%s", buf.buffer);
410 } else {
411 log_info("SYNC_STATE: (empty)");
412 }
413
414 SAFE_FREE(buffer);
415#undef SYNC_BUFFER_SIZE
416}

426typedef enum {
427 DEBUG_REQUEST_STATE, // Print sync state
428 DEBUG_REQUEST_BACKTRACE, // Print backtrace
429 DEBUG_REQUEST_MEMORY_REPORT, // Print memory report
430} debug_request_type_t;
431
432typedef struct {
433 debug_request_type_t request_type; // What to print
434 uint64_t delay_ns;
435 atomic_t should_run; // Atomic flag set by main thread
436 atomic_t should_exit; // Atomic flag for shutdown
437 atomic_t signal_triggered; // Flag set by SIGUSR1 handler
438 uint64_t memory_report_interval_ns; // Interval for periodic memory reports (0 = disabled)
439 uint64_t last_memory_report_time_ns; // Timestamp of last memory report
440 mutex_t mutex; // Protects access to flags during locked operations
441 cond_t cond; // Wakes thread when signal arrives
442 bool initialized; // Tracks if mutex/cond are initialized
443 bool handled_sync_state_time; // Track if --sync-state option was already processed
444 bool handled_backtrace_time; // Track if --backtrace option was already processed
445 bool handled_memory_report; // Track if --memory-report option was already processed
446} debug_state_request_t;

448static debug_state_request_t g_debug_state_request = {
449 .request_type = DEBUG_REQUEST_STATE,
450 .delay_ns = 0,
451 .should_run = {.impl = 0},
452 .should_exit = {.impl = 0},
453 .signal_triggered = {.impl = 0},
454 .memory_report_interval_ns = 0,
455 .last_memory_report_time_ns = 0,
456 .initialized = false,
457 .handled_sync_state_time = false,
458 .handled_backtrace_time = false,
459 .handled_memory_report = false,
460};
461static asciichat_thread_t g_debug_thread;
462static uint64_t g_debug_main_thread_id = 0; // Main thread ID for memory reporting
463static atomic_t g_cleanup_in_progress = {0}; // Flag to prevent deadlock checks during shutdown

475static void *debug_print_thread_fn(void *arg) {
476 (void)arg;
477
478 while (!atomic_load_bool(&g_debug_state_request.should_exit)) {
479 // Handle delayed printing
480 if (atomic_load_bool(&g_debug_state_request.should_run) && g_debug_state_request.delay_ns > 0) {
481 platform_sleep_ns(g_debug_state_request.delay_ns);
482 g_debug_state_request.delay_ns = 0;
483 }
484
485 // Handle both scheduled and signal-triggered printing
486 mutex_lock(&g_debug_state_request.mutex);
487 bool should_run = atomic_load_bool(&g_debug_state_request.should_run);
488 bool signal_triggered = atomic_load_bool(&g_debug_state_request.signal_triggered);
489 bool should_exit = atomic_load_bool(&g_debug_state_request.should_exit);
490
491 if ((should_run || signal_triggered) && !should_exit) {
492 debug_request_type_t request_type = g_debug_state_request.request_type;
493 mutex_unlock(&g_debug_state_request.mutex);
494
495 // Print based on request type
496 if (request_type == DEBUG_REQUEST_STATE) {
497 debug_sync_print_state();
498 } else if (request_type == DEBUG_REQUEST_BACKTRACE) {
499 backtrace_t bt = {0};
500 backtrace_capture_and_symbolize(&bt);
501 backtrace_print("Backtrace", &bt, 0, 0, NULL);
502 backtrace_t_free(&bt);
503 }
504
505 mutex_lock(&g_debug_state_request.mutex);
506 atomic_store_bool(&g_debug_state_request.should_run, false);
507 atomic_store_bool(&g_debug_state_request.signal_triggered, false);
508 should_exit = atomic_load_bool(&g_debug_state_request.should_exit);
509 }
510
511 // Wait for work or signal, with 100ms timeout to check should_exit
512 if (!should_exit) {
513 cond_timedwait(&g_debug_state_request.cond, &g_debug_state_request.mutex, 100000000); // 100ms
514 }
515 mutex_unlock(&g_debug_state_request.mutex);
516
517 // Periodic deadlock detection (runs every 100ms during wait timeout)
518 debug_sync_check_cond_deadlocks();
519 mutex_stack_detect_deadlocks();
520
521#ifndef NDEBUG
522 // Periodic memory report (if enabled and DEBUG_MEMORY is configured)
523#ifdef DEBUG_MEMORY
524 if (g_debug_state_request.memory_report_interval_ns > 0) {
525 uint64_t now = time_get_ns();
526 uint64_t last_time = g_debug_state_request.last_memory_report_time_ns;
527
528 // Check if enough time has passed (or first report)
529 if (last_time == 0 || (now - last_time >= g_debug_state_request.memory_report_interval_ns)) {
530 debug_memory_report();
531 g_debug_state_request.last_memory_report_time_ns = now;
532 }
533 }
534#endif
535
536 options_t *opts = options_get();
537 if (!opts) {
538 continue;
539 }
540
541 // Handle --debug-state (debug builds only)
542 // Schedule sync state printing after specified delay (execute only once, when option is first detected)
543 // Uses non-blocking scheduled printing instead of sleep to avoid blocking the debug thread
544 if (!g_debug_state_request.handled_sync_state_time && IS_OPTION_EXPLICIT(debug_sync_state_time, opts) &&
545 opts->debug_sync_state_time > 0.0) {
546 g_debug_state_request.handled_sync_state_time = true;
547 log_info("Will print sync state after %f seconds", opts->debug_sync_state_time);
548 uint64_t delay_ns = (uint64_t)(opts->debug_sync_state_time * NS_PER_SEC_INT);
549 debug_sync_print_state_delayed(delay_ns);
550 }
551
552 // Handle --backtrace (debug builds only)
553 // Schedule backtrace printing after specified delay (execute only once, when option is first detected)
554 // Uses non-blocking scheduled printing instead of sleep to avoid blocking the debug thread
555 if (!g_debug_state_request.handled_backtrace_time && IS_OPTION_EXPLICIT(debug_backtrace_time, opts) &&
556 opts->debug_backtrace_time > 0.0) {
557 g_debug_state_request.handled_backtrace_time = true;
558 log_info("Will print backtrace after %f seconds", opts->debug_backtrace_time);
559 uint64_t delay_ns = (uint64_t)(opts->debug_backtrace_time * NS_PER_SEC_INT);
560 debug_sync_print_backtrace_delayed(delay_ns);
561 }
562
563 // Handle --memory-report (debug builds only)
564 // Enable periodic memory reporting at specified interval (execute only once, when option is first detected)
565 if (!g_debug_state_request.handled_memory_report && IS_OPTION_EXPLICIT(debug_memory_report_interval, opts) &&
566 opts->debug_memory_report_interval > 0.0) {
567 g_debug_state_request.handled_memory_report = true;
568 log_info("Enabling memory reports every %f seconds", opts->debug_memory_report_interval);
569 uint64_t interval_ns = (uint64_t)(opts->debug_memory_report_interval * NS_PER_SEC_INT);
570 debug_sync_set_memory_report_interval(interval_ns);
571 }
572#else
573 options_t *opts = options_get();
574 if (!opts) {
575 continue;
576 }
577#endif
578 }
579
580 return NULL;
581}

665void debug_sync_cleanup_thread(void) {
666 log_debug("[DEBUG_SYNC_CLEANUP] Starting cleanup");
667
668 // Only join if thread was actually created
669 if (!g_debug_state_request.initialized) {
670 log_debug("[DEBUG_SYNC_CLEANUP] Thread not initialized, returning");
671 return;
672 }
673
674 // Set cleanup flag to prevent deadlock checks from accessing freed memory
675 // Do this after checking initialization so we don't set it unnecessarily
676 atomic_store_bool(&g_cleanup_in_progress, true);
677 log_debug("[DEBUG_SYNC_CLEANUP] Thread was initialized, proceeding with cleanup");
678
679 log_debug("[DEBUG_SYNC_CLEANUP] Setting initialized to false");
680 g_debug_state_request.initialized = false; // Prevent double-join
681
682 // Signal the thread to wake up immediately instead of waiting for 100ms timeout
683 log_debug("[DEBUG_SYNC_CLEANUP] Signaling thread to exit");
684 atomic_store_bool(&g_debug_state_request.should_exit, true);
685 cond_signal(&g_debug_state_request.cond);
686 log_debug("[DEBUG_SYNC_CLEANUP] Signal sent, about to join thread");
687
688 // Use a timeout join to ensure we don't deadlock, but still unregister the thread
689 // The debug thread should exit quickly after should_exit is set above
690 int join_result = asciichat_thread_join_timeout(&g_debug_thread, NULL, 1000000000ULL); // 1 second timeout
691 if (join_result == 0) {
692 log_debug("[DEBUG_SYNC_CLEANUP] Thread joined successfully");
693 } else if (join_result == -2) {
694 log_debug("[DEBUG_SYNC_CLEANUP] Thread join timed out (thread may still be running)");
695 // Don't unregister if timeout - thread is still alive
696 } else {
697 log_debug("[DEBUG_SYNC_CLEANUP] Thread join failed with error %d", join_result);
698 }
699}

701void debug_sync_trigger_print(void) {
702 // Set flag to trigger printing on debug thread (from SIGUSR1 handler).
703 // We don't call debug_sync_print_state() directly here to avoid logging
704 // in signal handler context, which could deadlock with logging mutexes.
705 // Uses atomic_store for thread-safe flag setting from signal handler.
706 //
707 // Signal the condition variable to wake up the debug thread immediately
708 // (without waiting for the 100ms timeout).
709 atomic_store_bool(&g_debug_state_request.signal_triggered, true);
710 cond_signal(&g_debug_state_request.cond);
711}
