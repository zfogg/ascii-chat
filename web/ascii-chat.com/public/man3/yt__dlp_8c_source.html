47static bool yt_dlp_cache_is_valid(void) {
48 if (!g_yt_dlp_cache.valid) {

65static bool yt_dlp_cache_get(const char *url, const char *yt_dlp_options, char *output_url, size_t output_size) {
66 if (!yt_dlp_cache_is_valid()) {
67 return false;
68 }
69
70 // Check if cache matches requested URL and options
71 if (strcmp(g_yt_dlp_cache.url, url) != 0) {
72 return false;
73 }
74
75 // For cache key, compare options (NULL == "" for consistency)
76 const char *cached_opts = g_yt_dlp_cache.yt_dlp_options;
77 const char *req_opts = yt_dlp_options ? yt_dlp_options : "";
78 if (strcmp(cached_opts, req_opts) != 0) {
79 return false;
80 }
81
82 // Return cached URL (may be empty if cached failure)
83 size_t url_len = strlen(g_yt_dlp_cache.stream_url);
84 if (url_len >= output_size) {
85 return false;
86 }
87
88 SAFE_STRNCPY(output_url, g_yt_dlp_cache.stream_url, output_size - 1);
89 output_url[output_size - 1] = '\0';
90
91 if (url_len == 0) {
92 log_debug("Using cached yt-dlp failure for URL (failed %ld seconds ago)",
93 (long)(time(NULL) - g_yt_dlp_cache.extracted_time));
94 } else {
95 log_debug("Using cached yt-dlp stream URL (extracted %ld seconds ago)",
96 (long)(time(NULL) - g_yt_dlp_cache.extracted_time));
97 }
98 return true;
99}

107static void yt_dlp_cache_set(const char *url, const char *yt_dlp_options, const char *stream_url) {
108 if (strlen(url) >= sizeof(g_yt_dlp_cache.url)) {
109 return; // URL too long to cache
110 }
111
112 const char *opts = yt_dlp_options ? yt_dlp_options : "";
113 if (strlen(opts) >= sizeof(g_yt_dlp_cache.yt_dlp_options)) {
114 return; // Options too long to cache
115 }
116
117 if (stream_url && strlen(stream_url) >= sizeof(g_yt_dlp_cache.stream_url)) {
118 return; // Stream URL too long to cache
119 }
120
121 SAFE_STRNCPY(g_yt_dlp_cache.url, url, sizeof(g_yt_dlp_cache.url) - 1);
122 g_yt_dlp_cache.url[sizeof(g_yt_dlp_cache.url) - 1] = '\0';
123
124 SAFE_STRNCPY(g_yt_dlp_cache.yt_dlp_options, opts, sizeof(g_yt_dlp_cache.yt_dlp_options) - 1);
125 g_yt_dlp_cache.yt_dlp_options[sizeof(g_yt_dlp_cache.yt_dlp_options) - 1] = '\0';
126
127 if (stream_url) {
128 SAFE_STRNCPY(g_yt_dlp_cache.stream_url, stream_url, sizeof(g_yt_dlp_cache.stream_url) - 1);
129 } else {
130 // Mark as empty (failure state)
131 g_yt_dlp_cache.stream_url[0] = '\0';
132 }
133 g_yt_dlp_cache.stream_url[sizeof(g_yt_dlp_cache.stream_url) - 1] = '\0';
134
135 g_yt_dlp_cache.extracted_time = time(NULL);
136 g_yt_dlp_cache.valid = true;
137}

151asciichat_error_t yt_dlp_extract_stream_url(const char *url, const char *yt_dlp_options, char *output_url,
152 size_t output_size) {
153 if (!url || !output_url || output_size < 256) {
154 SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters for yt-dlp URL extraction");
155 return ERROR_INVALID_PARAM;
156 }
157
158 // Check if yt-dlp is available
159 if (!yt_dlp_is_available()) {
160 SET_ERRNO(ERROR_YOUTUBE_EXTRACT_FAILED, "yt-dlp is not installed. Please install it with: "
161 "pip install yt-dlp (or: brew install yt-dlp on macOS)");
162 return ERROR_YOUTUBE_EXTRACT_FAILED;
163 }
164
165 // Check if we have a cached extraction for this URL+options combination
166 char cached_url[8192] = {0};
167 if (yt_dlp_cache_get(url, yt_dlp_options, cached_url, sizeof(cached_url))) {
168 if (cached_url[0] != '\0') {
169 // Cached success - return the URL
170 SAFE_STRNCPY(output_url, cached_url, output_size - 1);
171 output_url[output_size - 1] = '\0';
172 return ASCIICHAT_OK;
173 } else {
174 // Cached failure - return error without logging again
175 return ERROR_YOUTUBE_EXTRACT_FAILED;
176 }
177 }
178
179 // Build yt-dlp command
180 char command[4096];
181 const char *opts = yt_dlp_options ? yt_dlp_options : "";
182
183 int cmd_ret;
184 if (opts[0] != '\0') {
185 // User provided custom options
186 cmd_ret = safe_snprintf(command, sizeof(command),
187 "yt-dlp --quiet --no-warnings "
188 "--user-agent 'Mozilla/5.0 (Macintosh; Intel Mac OS X 10_15_7) "
189 "AppleWebKit/537.36' "
190 "%s "
191 "-f 'b' -O '%%(url)s' '%s' 2>&1",
192 opts, url);
193 } else {
194 // No custom options, use default
195 cmd_ret = safe_snprintf(command, sizeof(command),
196 "yt-dlp --quiet --no-warnings "
197 "--user-agent 'Mozilla/5.0 (Macintosh; Intel Mac OS X 10_15_7) "
198 "AppleWebKit/537.36' "
199 "-f 'b' -O '%%(url)s' '%s' 2>&1",
200 url);
201 }
202
203 if (cmd_ret < 0 || cmd_ret >= (int)sizeof(command)) {
204 SET_ERRNO(ERROR_INVALID_PARAM, "URL or yt-dlp options too long");
205 return ERROR_INVALID_PARAM;
206 }
207
208 log_debug("Executing yt-dlp: %s", command);
209
210 // Execute yt-dlp and capture output
211 FILE *pipe = NULL;
212 if (platform_popen("yt_dlp_extract", command, "r", &pipe) != ASCIICHAT_OK || !pipe) {
213 SET_ERRNO(ERROR_YOUTUBE_EXTRACT_FAILED, "Failed to execute yt-dlp subprocess");
214 return ERROR_YOUTUBE_EXTRACT_FAILED;
215 }
216
217 // Read output looking for stream URL (should be last line starting with http)
218 char url_buffer[8192] = {0};
219 char full_output[16384] = {0};
220 size_t ytdlp_output_len = 0;
221 size_t url_size = 0;
222 int c;
223
224 while ((c = fgetc(pipe)) != EOF && ytdlp_output_len < sizeof(full_output) - 1) {
225 full_output[ytdlp_output_len++] = (char)c;
226
227 // Track current line - looking for http:// or https:// stream URL
228 if ((url_size == 0 && c == 'h') || (url_size > 0 && c != '\n' && url_size < sizeof(url_buffer) - 1)) {
229 url_buffer[url_size++] = (char)c;
230 } else if (c == '\n') {
231 if (url_size > 0) {
232 url_buffer[url_size] = '\0';
233 if (url_is_valid(url_buffer)) {
234 break; // Found valid URL
235 }
236 }
237 url_size = 0; // Reset for next line
238 }
239 }
240 full_output[ytdlp_output_len] = '\0';
241 url_buffer[url_size] = '\0';
242
243 int pclose_ret = (platform_pclose(&pipe) == ASCIICHAT_OK) ? 0 : -1;
244 if (pclose_ret != 0) {
245 yt_dlp_cache_set(url, yt_dlp_options, NULL); // Cache failure
246
247 log_debug("yt-dlp exited with code %d", pclose_ret);
248 if (ytdlp_output_len > 0) {
249 log_error("yt-dlp output:\n%s", full_output);
250 SET_ERRNO(ERROR_YOUTUBE_EXTRACT_FAILED, "yt-dlp failed to extract stream: %s", full_output);
251 } else {
252 SET_ERRNO(ERROR_YOUTUBE_EXTRACT_FAILED, "yt-dlp failed to extract stream");
253 }
254 return ERROR_YOUTUBE_EXTRACT_FAILED;
255 }
256
257 if (url_size == 0 || (url_size == 2 && strncmp(url_buffer, "NA", 2) == 0)) {
258 yt_dlp_cache_set(url, yt_dlp_options, NULL);
259 log_error("yt-dlp returned empty output for URL: %s", url);
260 SET_ERRNO(ERROR_YOUTUBE_EXTRACT_FAILED, "yt-dlp returned no playable formats");
261 return ERROR_YOUTUBE_EXTRACT_FAILED;
262 }
263
264 if (url_buffer[0] != 'h' || strncmp(url_buffer, "http", 4) != 0) {
265 yt_dlp_cache_set(url, yt_dlp_options, NULL);
266 log_error("Invalid URL from yt-dlp: %s (full output: %s)", url_buffer, full_output);
267 SET_ERRNO(ERROR_YOUTUBE_EXTRACT_FAILED, "yt-dlp returned invalid URL");
268 return ERROR_YOUTUBE_EXTRACT_FAILED;
269 }
270
271 if (url_size >= output_size) {
272 SET_ERRNO(ERROR_INVALID_PARAM, "Stream URL too long for output buffer (%zu bytes, max %zu)", url_size, output_size);
273 return ERROR_INVALID_PARAM;
274 }
275
276 SAFE_STRNCPY(output_url, url_buffer, output_size - 1);
277 output_url[output_size - 1] = '\0';
278
279 // Cache successful result
280 yt_dlp_cache_set(url, yt_dlp_options, output_url);
281
282 log_debug("yt-dlp successfully extracted stream URL (%zu bytes)", url_size);
283 return ASCIICHAT_OK;
