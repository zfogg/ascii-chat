298asciichat_error_t parse_ssh_keys_from_response(const char *response_text, size_t response_len, char ***keys_out,
299 size_t *num_keys, size_t max_keys) {
300 if (!response_text || !keys_out || !num_keys) {
301 SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters for SSH key parsing");
302 return ERROR_INVALID_PARAM;
303 }
304
305 *num_keys = 0;
306 *keys_out = NULL;
307
308 // Count the number of SSH keys in the response
309 size_t key_count = 0;
310 const char *line_start = response_text;
311 const char *line_end;
312
313 while ((line_end = strchr(line_start, '\n')) != NULL) {
314 size_t line_len = line_end - line_start;
315 if (line_len > 0 && line_start[0] != '\r' && line_start[0] != '\n') {
316 key_count++;
317 }
318 line_start = line_end + 1;
319 }
320
321 // Handle last line if it doesn't end with newline
322 if (line_start < response_text + response_len) {
323 key_count++;
324 }
325
326 if (key_count == 0) {
327 SET_ERRNO(ERROR_CRYPTO_KEY, "No SSH keys found in response");
328 return ERROR_CRYPTO_KEY;
329 }
330
331 if (key_count > max_keys) {
332 key_count = max_keys;
333 }
334
335 // Allocate array for key strings
336 *keys_out = SAFE_MALLOC(sizeof(char *) * key_count, char **);
337 if (!*keys_out) {
338 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate SSH keys array");
339 }
340
341 // Parse each SSH key line
342 line_start = response_text;
343 size_t parsed_keys = 0;
344
345 while (parsed_keys < key_count && (line_end = strchr(line_start, '\n')) != NULL) {
346 size_t line_len = line_end - line_start;
347
348 // Skip empty lines
349 if (line_len > 0 && line_start[0] != '\r' && line_start[0] != '\n') {
350 // Allocate space for this key line
351 (*keys_out)[parsed_keys] = SAFE_MALLOC(line_len + 1, char *);
352 if (!(*keys_out)[parsed_keys]) {
353 // Cleanup previously allocated keys
354 for (size_t i = 0; i < parsed_keys; i++) {
355 SAFE_FREE((*keys_out)[i]);
356 }
357 SAFE_FREE(*keys_out);
358 *keys_out = NULL;
359 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate SSH key string");
360 }
361
362 // Copy the key line
363 memcpy((*keys_out)[parsed_keys], line_start, line_len);
364 (*keys_out)[parsed_keys][line_len] = '\0';
365
366 parsed_keys++;
367 }
368
369 line_start = line_end + 1;
370 }
371
372 // Handle last line if it doesn't end with newline
373 if (parsed_keys < key_count && line_start < response_text + response_len) {
374 size_t line_len = (response_text + response_len) - line_start;
375 if (line_len > 0) {
376 (*keys_out)[parsed_keys] = SAFE_MALLOC(line_len + 1, char *);
377 if (!(*keys_out)[parsed_keys]) {
378 // Cleanup previously allocated keys
379 for (size_t i = 0; i < parsed_keys; i++) {
380 SAFE_FREE((*keys_out)[i]);
381 }
382 SAFE_FREE(*keys_out);
383 *keys_out = NULL;
384 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate SSH key string");
385 }
386 memcpy((*keys_out)[parsed_keys], line_start, line_len);
387 (*keys_out)[parsed_keys][line_len] = '\0';
388 parsed_keys++;
389 }
390 }
391
392 *num_keys = parsed_keys;
393 return ASCIICHAT_OK;
394}

396asciichat_error_t parse_gpg_keys_from_response(const char *response_text, size_t response_len, char ***keys_out,
397 size_t *num_keys, size_t max_keys) {
398 (void)max_keys;
399 if (!response_text || !keys_out || !num_keys) {
400 SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters for GPG key parsing");
401 return ERROR_INVALID_PARAM;
402 }
403
404 *num_keys = 0;
405 *keys_out = NULL;
406
407 // Check if this looks like a GPG key (starts with -----BEGIN PGP)
408 if (strncmp(response_text, "-----BEGIN PGP", 14) != 0) {
409 SET_ERRNO(ERROR_CRYPTO_KEY, "Response does not contain a valid GPG key");
410 return ERROR_CRYPTO_KEY;
411 }
412
413 // Write armored block to temp file
414 char temp_file[PLATFORM_MAX_PATH_LENGTH];
415 int fd = -1;
416 if (platform_create_temp_file(temp_file, sizeof(temp_file), "asc_gpg_import", &fd) != 0) {
417 return SET_ERRNO(ERROR_CRYPTO_KEY, "Failed to create temp file for GPG import");
418 }
419
420#ifdef _WIN32
421 // On Windows, platform_create_temp_file returns fd=-1, need to open separately
422 fd = platform_open("https_key_temp", temp_file, O_WRONLY | O_BINARY);
423 if (fd < 0) {
424 platform_delete_temp_file(temp_file);
425 return SET_ERRNO(ERROR_CRYPTO_KEY, "Failed to open temp file for writing");
426 }
427#endif
428
429 ssize_t written = write(fd, response_text, response_len);
430 close(fd);
431
432 if (written != (ssize_t)response_len) {
433 platform_unlink(temp_file);
434 return SET_ERRNO(ERROR_CRYPTO_KEY, "Failed to write GPG key to temp file");
435 }
436
437 // Import the key using gpg --import
438 char import_cmd[BUFFER_SIZE_MEDIUM];
439 safe_snprintf(import_cmd, sizeof(import_cmd), "gpg --import '%s' 2>&1", temp_file);
440 FILE *import_fp = NULL;
441 if (platform_popen("gpg_https_import", import_cmd, "r", &import_fp) != ASCIICHAT_OK || !import_fp) {
442 platform_unlink(temp_file);
443 return SET_ERRNO(ERROR_CRYPTO_KEY, "Failed to run gpg --import");
444 }
445
446 char import_output[2048];
447 size_t import_len = fread(import_output, 1, sizeof(import_output) - 1, import_fp);
448 import_output[import_len] = '\0';
449 platform_pclose(&import_fp);
450 platform_unlink(temp_file);
451
452 // Extract ALL key IDs from import output (format: "gpg: key KEYID: ...")
453 // GitHub often returns multiple keys in one armored block
454 const char *key_marker = "gpg: key ";
455 char key_ids[16][17]; // Support up to 16 keys
456 size_t key_count = 0;
457
458 log_debug("GPG import output:\n%s", import_output);
459
460 char *search_pos = import_output;
461 while (key_count < 16) {
462 char *key_line = strstr(search_pos, key_marker);
463 if (!key_line)
464 break;
465
466 key_line += strlen(key_marker);
467 int i = 0;
468 while (i < 16 && key_line[i] != ':' && key_line[i] != ' ' && key_line[i] != '\n') {
469 key_ids[key_count][i] = key_line[i];
470 i++;
471 }
472 key_ids[key_count][i] = '\0';
473
474 if (i > 0) {
475 log_debug("Extracted GPG key ID #%zu: %s", key_count, key_ids[key_count]);
476 key_count++;
477 }
478 search_pos = key_line + i;
479 }
480
481 if (key_count == 0) {
482 return SET_ERRNO(ERROR_CRYPTO_KEY, "Failed to extract any key IDs from GPG import output");
483 }
484
485 log_debug("Total GPG keys extracted from import: %zu", key_count);
486
487 // Allocate array for results
488 *keys_out = SAFE_MALLOC(sizeof(char *) * key_count, char **);
489 if (!*keys_out) {
490 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate GPG keys array");
491 }
492
493 // Process each key ID to get full fingerprint
494 size_t valid_keys = 0;
495 for (size_t k = 0; k < key_count; k++) {
496 // Get full fingerprint from gpg --list-keys output
497 char list_cmd[BUFFER_SIZE_SMALL];
498 safe_snprintf(list_cmd, sizeof(list_cmd),
499 "gpg --list-keys --with-colons --fingerprint '%s' " PLATFORM_SHELL_NULL_REDIRECT, key_ids[k]);
500 FILE *list_fp = NULL;
501 if (platform_popen("gpg_https_list", list_cmd, "r", &list_fp) != ASCIICHAT_OK || !list_fp) {
502 continue; // Skip this key if we can't list it
503 }
504
505 char list_output[4096];
506 size_t list_len = fread(list_output, 1, sizeof(list_output) - 1, list_fp);
507 list_output[list_len] = '\0';
508 platform_pclose(&list_fp);
509
510 // Check if it contains an Ed25519 key (algorithm 22)
511 if (!strstr(list_output, ":22:") && !strstr(list_output, "ed25519")) {
512 continue; // Skip non-Ed25519 keys
513 }
514
515 // Extract full 40-character fingerprint from "fpr:" line
516 // Format: fpr:::::::::FINGERPRINT:
517 // The fingerprint is field 10 (after 9 colons)
518 char fingerprint[41] = {0};
519 const char *fpr_marker = "\nfpr:";
520 char *fpr_line = strstr(list_output, fpr_marker);
521 if (fpr_line) {
522 fpr_line += 1; // Skip the newline
523 // Count 9 colons from the start of the line
524 int colon_count = 0;
525 while (*fpr_line && colon_count < 9) {
526 if (*fpr_line == ':')
527 colon_count++;
528 fpr_line++;
529 }
530 // Now we should be at the start of the fingerprint
531 // Extract up to 40 hex characters
532 int fpr_len = 0;
533 while (fpr_len < 40 && fpr_line[fpr_len] && fpr_line[fpr_len] != ':' && fpr_line[fpr_len] != '\n') {
534 fingerprint[fpr_len] = fpr_line[fpr_len];
535 fpr_len++;
536 }
537 fingerprint[fpr_len] = '\0';
538 }
539
540 // If fingerprint extraction failed, use the short key ID
541 if (strlen(fingerprint) == 0) {
542 log_warn("Failed to extract fingerprint for key %s, using short key ID", key_ids[k]);
543 SAFE_STRNCPY(fingerprint, key_ids[k], sizeof(fingerprint));
544 }
545
546 log_debug("Key %s -> fingerprint: %s (length: %zu)", key_ids[k], fingerprint, strlen(fingerprint));
547
548 // Allocate and store this key in gpg:KEYID format
549 size_t gpg_key_len = strlen("gpg:") + strlen(fingerprint) + 1;
550 (*keys_out)[valid_keys] = SAFE_MALLOC(gpg_key_len, char *);
551 if (!(*keys_out)[valid_keys]) {
552 // Cleanup on allocation failure
553 for (size_t cleanup = 0; cleanup < valid_keys; cleanup++) {
554 SAFE_FREE((*keys_out)[cleanup]);
555 }
556 SAFE_FREE(*keys_out);
557 *keys_out = NULL;
558 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate GPG key string");
559 }
560
561 safe_snprintf((*keys_out)[valid_keys], gpg_key_len, "gpg:%s", fingerprint);
562 log_debug("Added valid Ed25519 key #%zu: %s", valid_keys, (*keys_out)[valid_keys]);
563 valid_keys++;
564 }
565
566 if (valid_keys == 0) {
567 SAFE_FREE(*keys_out);
568 *keys_out = NULL;
569 return SET_ERRNO(ERROR_CRYPTO_KEY, "No valid Ed25519 keys found in imported GPG keys");
570 }
571
572 *num_keys = valid_keys;
573 return ASCIICHAT_OK;
574}
