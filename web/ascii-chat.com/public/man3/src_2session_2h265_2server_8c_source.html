77 ctx->clients[i].client_id = client_id;
78 ctx->clients[i].decoder = decoder;
79 ctx->clients[i].in_use = true;
80 ctx->clients[i].last_width = 0;
81 ctx->clients[i].last_height = 0;
82 ctx->active_client_count++;

85 return (h265_server_client_t *)&ctx->clients[i];
86 }
87 }

107 if (!client_dec->decoder || !client_dec->in_use) {
108 return SET_ERRNO(ERROR_INTERNAL, "Invalid client decoder state");
109 }
110
111 uint8_t ascii_buf[256 * 64];

113
114 asciichat_error_t result = h265_decode(
115 client_dec->decoder,
116 h265_packet,
117 packet_size,
118 output_width,
119 output_height,
120 ascii_buf,
121 &ascii_size
122 );
123
124 if (result != ASCIICHAT_OK) {
125 return result;
126 }
127
128 size_t required_rgba_size = (size_t)(*output_width) * (*output_height) * 4;
129 if (*output_size < required_rgba_size) {
130 *output_size = required_rgba_size;
131 return SET_ERRNO(ERROR_NETWORK_SIZE, "Output buffer too small: need %zu, have %zu",
132 required_rgba_size, *output_size);
133 }
134
135 for (size_t i = 0; i < ascii_size && i < (size_t)(*output_width) * (*output_height); i++) {
136 uint8_t val = ascii_buf[i];
137 output_rgba[i * 4 + 0] = val;
138 output_rgba[i * 4 + 1] = val;
139 output_rgba[i * 4 + 2] = val;
140 output_rgba[i * 4 + 3] = 255;
141 }
142
143 *output_size = required_rgba_size;
144 client_dec->last_width = *output_width;
145 client_dec->last_height = *output_height;
