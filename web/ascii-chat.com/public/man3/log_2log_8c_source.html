216size_t get_current_time_formatted(char *time_buf) {
217 /* Get wall-clock time in nanoseconds */
218 uint64_t ts_ns = time_get_realtime_ns();
219 // Extract seconds and nanoseconds from total nanoseconds
220 time_t seconds = (time_t)(ts_ns / NS_PER_SEC_INT);
221 long nanoseconds = (long)(ts_ns % NS_PER_SEC_INT);
222 struct tm tm_info;
223 platform_localtime(&seconds, &tm_info);
224 // Format the time part first
225 // strftime returns 0 on error, not negative (and len is size_t/unsigned)
226 size_t len = strftime(time_buf, 32, "%H:%M:%S", &tm_info);
227 if (len == 0 || len >= 32) {
228 LOGGING_INTERNAL_ERROR(ERROR_INVALID_STATE, "Failed to format time");
229 return 0;
230 }
231
232 // Add microseconds manually (convert nanoseconds to microseconds for display)
233 long microseconds = nanoseconds / 1000;
234 if (microseconds < 0)
235 microseconds = 0;
236 if (microseconds > 999999)
237 microseconds = 999999;
238
239 int result = safe_snprintf(time_buf + len, 32 - len, ".%06ld", microseconds);
240 if (result < 0 || result >= (int)(32 - len)) {
241 LOGGING_INTERNAL_ERROR(ERROR_INVALID_STATE, "Failed to format microseconds");
242 return 0;
243 }
244
245 return len + (size_t)result;
246}

286static int truncate_at_whole_line(char *buffer, int current_len, size_t max_len) {
287 if (current_len < 0 || (size_t)current_len <= max_len) {
288 return current_len; // No truncation needed
289 }
290
291 // Search backwards from max_len position for last newline
292 int truncate_pos = (int)max_len - 1;
293 while (truncate_pos > 0 && buffer[truncate_pos] != '\n') {
294 truncate_pos--;
295 }
296
297 // If we found a newline, truncate after it
298 if (truncate_pos > 0 && buffer[truncate_pos] == '\n') {
299 truncate_pos++; // Include the newline
300 buffer[truncate_pos] = '\0';
301 return truncate_pos;
302 }
303
304 // No newline found, truncate at max_len to be safe
305 truncate_pos = (int)max_len - 1;
306 buffer[truncate_pos] = '\0';
307 return truncate_pos;
308}

367/* Log rotation function - keeps the tail (recent entries)
368 * REQUIRES: rotation_mutex must be held by caller
369 * This is the only operation that uses a mutex - regular logging is lock-free.
370 */
371static void rotate_log_locked(void) {
372 int file = atomic_load_int(&g_log.file);
373 size_t current_size = atomic_load_u64(&g_log.current_size);
374
375 if (file < 0 || file == STDERR_FILENO || strlen(g_log.filename) == 0) {
376 return;
377 }
378
379 if (current_size < MAX_LOG_SIZE) {
380 return;
381 }
382
383 platform_close(file);
384 atomic_store_int(&g_log.file, -1);
385
386 /* Open file for reading to get the tail */
387 int read_file = platform_open("log_file_read", g_log.filename, O_RDONLY, 0);
388 if (read_file < 0) {
389 log_error("Failed to open file descriptor for log rotation: fd=%d, path=%s", read_file, g_log.filename);
390 /* Fall back to regular truncation */
391 int fd = platform_open("log_file_write", g_log.filename, O_CREAT | O_RDWR | O_TRUNC, FILE_PERM_PRIVATE);
392 atomic_store_int(&g_log.file, fd);
393 atomic_store_u64(&g_log.current_size, 0);
394 return;
395 }
396
397 /* Seek to position where we want to start keeping data (keep last 2MB) */
398 size_t keep_size = MAX_LOG_SIZE * 2 / 3; /* Keep last 2MB of 3MB file */
399 if (current_size < keep_size) {
400 platform_close(read_file);
401 /* Fall back to truncation since we don't have enough data to rotate */
402 int fd = platform_open("log_file_write", g_log.filename, O_CREAT | O_RDWR | O_TRUNC, FILE_PERM_PRIVATE);
403 atomic_store_int(&g_log.file, fd);
404 atomic_store_u64(&g_log.current_size, 0);
405 return;
406 }
407 if (lseek(read_file, (off_t)(current_size - keep_size), SEEK_SET) == (off_t)-1) {
408 platform_close(read_file);
409 /* Fall back to truncation */
410 int fd = platform_open("log_file_write", g_log.filename, O_CREAT | O_RDWR | O_TRUNC, FILE_PERM_PRIVATE);
411 atomic_store_int(&g_log.file, fd);
412 atomic_store_u64(&g_log.current_size, 0);
413 return;
414 }
415
416 /* Skip to next line boundary to avoid partial lines */
417 char c;
418 while (platform_read(read_file, &c, 1) > 0 && c != '\n') {
419 /* Skip characters until newline */
420 }
421
422 /* Read the tail into a temporary file */
423 char temp_filename[PLATFORM_MAX_PATH_LENGTH];
424 int result = safe_snprintf(temp_filename, sizeof(temp_filename), "%s.tmp", g_log.filename);
425 if (result <= 0 || result >= (int)sizeof(temp_filename)) {
426 LOGGING_INTERNAL_ERROR(ERROR_INVALID_STATE, "Failed to format temp filename");
427 platform_close(read_file);
428 int fd = platform_open("log_file_append", g_log.filename, O_CREAT | O_RDWR | O_APPEND, FILE_PERM_PRIVATE);
429 atomic_store_int(&g_log.file, fd);
430 return;
431 }
432
433 int temp_file = platform_open("log_file_temp", temp_filename, O_CREAT | O_WRONLY | O_TRUNC, FILE_PERM_PRIVATE);
434 if (temp_file < 0) {
435 log_warn("Failed to open temporary file descriptor for log rotation: fd=%d, path=%s", temp_file, temp_filename);
436 platform_close(read_file);
437 /* Fall back to truncation */
438 int fd = platform_open("log_file_write", g_log.filename, O_CREAT | O_RDWR | O_TRUNC, FILE_PERM_PRIVATE);
439 atomic_store_int(&g_log.file, fd);
440 atomic_store_u64(&g_log.current_size, 0);
441 return;
442 }
443
444 /* Copy tail to temp file */
445 char buffer[8192];
446 ssize_t bytes_read;
447 size_t new_size = 0;
448 while ((bytes_read = platform_read(read_file, buffer, sizeof(buffer))) > 0) {
449 ssize_t written = platform_write(temp_file, buffer, (size_t)bytes_read);
450 if (written != bytes_read) {
451 platform_close(read_file);
452 platform_close(temp_file);
453 unlink(temp_filename);
454 /* Fall back to truncation */
455 int fd = platform_open("log_file_write", g_log.filename, O_CREAT | O_RDWR | O_TRUNC, FILE_PERM_PRIVATE);
456 atomic_store_int(&g_log.file, fd);
457 atomic_store_u64(&g_log.current_size, 0);
458 return;
459 }
460 new_size += (size_t)bytes_read;
461 }
462
463 platform_close(read_file);
464 platform_close(temp_file);
465
466 /* Replace original with temp file */
467 if (rename(temp_filename, g_log.filename) != 0) {
468 unlink(temp_filename); /* Clean up temp file */
469 /* Fall back to truncation */
470 int fd = platform_open("log_file_write", g_log.filename, O_CREAT | O_RDWR | O_TRUNC, FILE_PERM_PRIVATE);
471 atomic_store_int(&g_log.file, fd);
472 atomic_store_u64(&g_log.current_size, 0);
473 return;
474 }
475
476 /* Reopen for appending */
477 int new_fd = platform_open("log_file_append", g_log.filename, O_CREAT | O_RDWR | O_APPEND, FILE_PERM_PRIVATE);
478 if (new_fd < 0) {
479 LOGGING_INTERNAL_ERROR(ERROR_INVALID_STATE, "Failed to reopen rotated log file: %s", g_log.filename);
480 atomic_store_int(&g_log.file, STDERR_FILENO);
481 g_log.filename[0] = '\0';
482 atomic_store_u64(&g_log.current_size, 0);
483 return;
484 }
485 atomic_store_int(&g_log.file, new_fd);
486 atomic_store_u64(&g_log.current_size, new_size);
487
488 char time_buf[LOG_TIMESTAMP_BUFFER_SIZE];
489 get_current_time_formatted(time_buf);
490
491 char log_msg[256];
492 int log_msg_len =
493 safe_snprintf(log_msg, sizeof(log_msg), "[%s] [INFO] Log tail-rotated (kept %zu bytes)\n", time_buf, new_size);
494 if (log_msg_len > 0 && log_msg_len < (int)sizeof(log_msg)) {
495 (void)platform_write(new_fd, log_msg, (size_t)log_msg_len);
496 }
497}

499/* Check if rotation is needed and perform it (acquires mutex only if needed) */
500static void maybe_rotate_log(void) {
501 /* Rotation mutex not initialized - skip rotation */
502 if (!lifecycle_is_initialized(&g_log.rotation_mutex_lifecycle)) {
503 return;
504 }
505
506 /* Check mmap path first */
507 if (log_mmap_is_active()) {
508 size_t used = 0, capacity = 0;
509 if (log_mmap_get_usage(&used, &capacity) && capacity > 0) {
510 /* Rotate when 90% full to leave room for writes */
511 if (used > capacity * 9 / 10) {
512 mutex_lock(&g_log.rotation_mutex);
513 log_mmap_rotate();
514 mutex_unlock(&g_log.rotation_mutex);
515 }
516 }
517 return;
518 }
519
520 /* File path: quick atomic check - avoid mutex if not needed */
521 size_t current_size = atomic_load_u64(&g_log.current_size);
522 if (current_size < MAX_LOG_SIZE) {
523 return;
524 }
525
526 /* Size exceeded - acquire mutex and rotate */
527 mutex_lock(&g_log.rotation_mutex);
528 rotate_log_locked();
529 mutex_unlock(&g_log.rotation_mutex);
530}

532void log_init(const char *filename, log_level_t level, bool force_stderr, bool use_mmap) {
533 // Initialize rotation mutex (only operation that uses a mutex)
534 g_log.rotation_mutex_lifecycle.sync_type = LIFECYCLE_SYNC_MUTEX;
535 g_log.rotation_mutex_lifecycle.sync.mutex = &g_log.rotation_mutex;
536 lifecycle_init(&g_log.rotation_mutex_lifecycle, "log_rotation");
537
538 // Set basic config using atomic stores
539 atomic_store_u64(&g_log.force_stderr, force_stderr);
540 bool preserve_terminal_output = atomic_load_u64(&g_log.terminal_output_enabled);
541
542 // Force logs to stderr if stdout is piped/redirected
543 // This prevents logs from contaminating piped/redirected output in snapshot/batch modes
544 // Per terminal.h: "force stderr when piped to prevent data corruption"
545 if (terminal_is_piped_output()) {
546 atomic_store_bool(&g_log.force_stderr, true);
547 }
548
549 // Close any existing file (atomic load/store)
550 int old_file = atomic_load_int(&g_log.file);
551 if (lifecycle_is_initialized(&g_log.lifecycle) && old_file >= 0 && old_file != STDERR_FILENO) {
552 platform_close(old_file);
553 atomic_store_int(&g_log.file, -1);
554 }
555
556 // Check LOG_LEVEL environment variable
557 const char *env_level_str = SAFE_GETENV("LOG_LEVEL");
558 if (env_level_str) {
559 atomic_store_u64(&g_log.level, (int)parse_log_level_from_env());
560 } else {
561 atomic_store_u64(&g_log.level, (int)level);
562 }
563
564 atomic_store_bool(&g_log.level_manually_set, false);
565 atomic_store_u64(&g_log.current_size, 0);
566
567 if (filename) {
568 SAFE_STRNCPY(g_log.filename, filename, sizeof(g_log.filename) - 1);
569
570 if (use_mmap) {
571 // Lock-free mmap path - writes go to mmap'd file
572 asciichat_error_t mmap_result = log_mmap_init_simple(filename, 0);
573 if (mmap_result == ASCIICHAT_OK) {
574 atomic_store_int(&g_log.file, -1); // No regular fd - using mmap for file output
575 } else {
576 // Mmap failed - use stderr only (atomic writes, lock-free)
577 if (preserve_terminal_output) {
578 safe_fprintf(stderr, "Mmap logging failed for %s, using stderr only (lock-free)\n", filename);
579 }
580 atomic_store_int(&g_log.file, STDERR_FILENO);
581 g_log.filename[0] = '\0';
582 }
583 } else {
584 // Lock-free file I/O path - uses atomic write() syscalls
585 int fd = platform_open("log_file", filename, O_CREAT | O_RDWR | O_TRUNC, FILE_PERM_PRIVATE);
586 atomic_store_int(&g_log.file, (fd >= 0) ? fd : STDERR_FILENO);
587 if (fd < 0) {
588 if (preserve_terminal_output) {
589 safe_fprintf(stderr, "Failed to open log file: %s\n", filename);
590 }
591 g_log.filename[0] = '\0';
592 }
593 }
594 } else {
595 atomic_store_int(&g_log.file, STDERR_FILENO);
596 g_log.filename[0] = '\0';
597 }
598
599 /* Initialize default log format (NULL means use mode-specific default) */
600 log_set_format(NULL, false);
601
602 /* Mark logging system as initialized (state machine transition) */
603 if (!lifecycle_is_initialized(&g_log.lifecycle)) {
604 lifecycle_init(&g_log.lifecycle, "logging");
605 }
606
607 atomic_store_u64(&g_log.terminal_output_enabled, preserve_terminal_output);
608
609 // Reset terminal detection if needed
610 if (g_log.terminal_caps_initialized && !g_log.terminal_caps.detection_reliable) {

817 * POSIX guarantees write() is atomic for sizes <= PIPE_BUF (typically 4096 bytes)
818 * Strips ANSI escape codes from the buffer before writing to ensure clean log files.

847 * Returns the number of characters written to the buffer
848 */
849static int format_log_header(char *buffer, size_t buffer_size, log_level_t level, const char *timestamp,
850 const char *file, int line, const char *func, bool use_colors, uint64_t time_nanoseconds) {
851 const log_template_t *format = g_log.format;
852 if (!format) {
853 LOGGING_INTERNAL_ERROR(ERROR_INVALID_STATE, "Log format not initialized");
854 return -1;
855 }
856
857 /* Use log_template_apply() to generate the complete formatted output
858 * Safety: reload format immediately before use. Format is never freed to avoid TOCTOU races. */
859 const log_template_t *format_check = g_log.format;
860 if (!format_check) {
861 /* Format not available - return minimal output */
862 return -1;
863 }
864
865 int written = log_template_apply(format_check, buffer, buffer_size, level, timestamp, file, line, func,
866 asciichat_thread_current_id(), "", use_colors, time_nanoseconds);
867 if (written < 0 || written >= (int)buffer_size) {
868 LOGGING_INTERNAL_ERROR(ERROR_INVALID_STATE, "Failed to format log header");
869 return -1;
870 }
871
872 return written;
873}

876 * When terminal is locked, other threads' output goes to file only.

879/* Helper: Write colored log entry to terminal (lock-free)
880 * Uses atomic loads for state checks, skips output if terminal is locked by another thread
881 */
882static void write_to_terminal_atomic(log_level_t level, const char *timestamp, const char *file, int line,
883 const char *func, const char *fmt, va_list args, uint64_t time_nanoseconds) {
884 // Prevent logging during shutdown - if logging lifecycle is being destroyed,
885 // bail out to avoid use-after-free of the log template
886 if (!lifecycle_is_initialized(&g_log.lifecycle)) {
887 return; /* Logging is being shut down, skip */
888 }
889
890 // Choose output file descriptor using unified routing logic
891 int fd = terminal_choose_log_fd(level);
892 FILE *output_stream = (fd == STDERR_FILENO) ? stderr : stdout;
893 bool use_colors = terminal_should_color_output(fd);
894
895 // Format message first (needed for status screen capture even if terminal output disabled)
896 char msg_buffer[LOG_MSG_BUFFER_SIZE];
897 va_list args_copy;
898 va_copy(args_copy, args);
899 int msg_len = safe_vsnprintf(msg_buffer, sizeof(msg_buffer), fmt, args_copy);
900 va_end(args_copy);
901
902 // Use message buffer directly as clean message
903 // Stack allocation avoids heap corruption during shutdown signal handling
904 const char *clean_msg = msg_buffer;
905
906 // Use format system to generate the complete log line
907 char colored_log_line[LOG_MSG_BUFFER_SIZE + 512];
908 char plain_log_line[LOG_MSG_BUFFER_SIZE + 512];
909
910 /* Apply format without colors first (plain version for grep matching)
911 * Load format pointer once to prevent TOCTOU race condition where another
912 * thread frees the format between our check and use. */
913 const log_template_t *format = g_log.format;
914 int plain_len = 0;
915 if (format && lifecycle_is_initialized(&g_log.lifecycle)) { /* Double-check lifecycle is still active */
916 plain_len = log_template_apply(format, plain_log_line, sizeof(plain_log_line), level, timestamp, file, line, func,
917 asciichat_thread_current_id(), clean_msg, false, time_nanoseconds);
918 }
919
920 /* For colored output, apply format with colors enabled */
921 int colored_len = 0;
922 if (use_colors && plain_len > 0) {
923 const log_template_t *format_check = g_log.format;
924 if (format_check && lifecycle_is_initialized(&g_log.lifecycle)) { /* Re-verify lifecycle before use */
925 colored_len = log_template_apply(format_check, colored_log_line, sizeof(colored_log_line), level, timestamp, file,
926 line, func, asciichat_thread_current_id(), clean_msg, true, time_nanoseconds);
927 }
928 /* If applying with colors failed, fall back to plain text */
929 if (colored_len <= 0) {
930 // Applying with colors failed - use plain text (no colors)
931 colored_len = plain_len;
932 if (plain_len > 0 && plain_len < (int)sizeof(colored_log_line)) {
933 memcpy(colored_log_line, plain_log_line, (size_t)plain_len + 1);
934 }
935 }
936 } else {
937 /* No colors - use plain text */
938 colored_len = plain_len;
939 if (plain_len > 0 && plain_len < (int)sizeof(colored_log_line)) {
940 memcpy(colored_log_line, plain_log_line, (size_t)plain_len + 1);
941 }
942 }
943
944 // Feed log to registered session log buffer (use colored version if available, otherwise plain)
945 session_log_buffer_t *buf = (session_log_buffer_t *)atomic_ptr_load(&g_log.session_log_buffer);
946 if (buf) {
947 if (colored_len > 0 && colored_len < (int)sizeof(colored_log_line)) {
948 session_log_buffer_append(buf, colored_log_line);
949 } else if (plain_len > 0 && plain_len < (int)sizeof(plain_log_line)) {
950 session_log_buffer_append(buf, plain_log_line);
951 }
952 }
953
954 // Check if terminal output is enabled (atomic load)
955 bool is_enabled = atomic_load_u64(&g_log.terminal_output_enabled);
956 if (!is_enabled) {
957 // Terminal output disabled - return (status screen already captured)
958 return;
959 }
960
961 // Check if terminal is locked by another thread
962 if (atomic_load_u64(&g_log.terminal_locked)) {
963 uint64_t owner = atomic_load_u64(&g_log.terminal_owner_thread);
964 if (owner != (uint64_t)asciichat_thread_self()) {
965 // Terminal locked by another thread - return
966 return;
967 }
968 }
969
970 // Handle message formatting errors for terminal output
971 if (msg_len <= 0 || msg_len >= (int)sizeof(msg_buffer)) {
972 // Message formatting failed - skip filtering and try direct output
973 (void)vfprintf(output_stream, fmt, args);
974 safe_fprintf(output_stream, "\n");
975 (void)fflush(output_stream);
976 return;
977 }
978
979 // Handle log formatting errors
980 if (colored_len <= 0 || colored_len >= (int)sizeof(colored_log_line) || plain_len <= 0 ||
981 plain_len >= (int)sizeof(plain_log_line)) {
982 // Formatting failed - try to output something
983 safe_fprintf(output_stream, "%s\n", clean_msg);
984 (void)fflush(output_stream);
985 return;
986 }
987
988 // Apply grep filter (terminal output only - file logs are unfiltered)
989 // Match against PLAIN text (no ANSI codes) so offsets are correct
990 size_t match_start = 0, match_len = 0;
991 if (!grep_should_output(plain_log_line, &match_start, &match_len)) {
992 return; // No match - suppress terminal output
993 }
994
995 // Output with colors
996 if (use_colors) {
997 const char **colors = log_get_color_array();
998
999 // When grep matches, highlight in the colored line
1000 if (match_len > 0 && colors != NULL) {
1001 // Apply highlighting to the full line, respecting existing ANSI codes
1002 const char *highlighted_line = grep_highlight_colored(colored_log_line, plain_log_line, match_start, match_len);
1003 safe_fprintf(output_stream, "%s\n", highlighted_line);
1004 } else {
1005 // No grep match - output colored line as-is
1006 safe_fprintf(output_stream, "%s\n", colored_log_line);
1007 }
1008 } else {
1009 // No colors - output plain line
1010 safe_fprintf(output_stream, "%s\n", plain_log_line);
1011 }
1012
1013 (void)fflush(output_stream);
1014}

1016void log_msg(log_level_t level, const char *file, int line, const char *func, const char *fmt, ...) {
1017 // All state access uses atomic operations - fully lock-free
1018 if (!lifecycle_is_initialized(&g_log.lifecycle)) {
1019 return;
1020 }
1021 // Safety check: if main format template has been freed during shutdown, skip logging
1022 // This prevents heap-use-after-free when threads call logging after log_destroy() has freed g_log.format
1023 if (!g_log.format) {
1024 return;
1025 }
1026 if (level < (log_level_t)atomic_load_u64(&g_log.level)) {
1027 return;
1028 }
1029 /* =========================================================================
1030 * MMAP PATH: When mmap logging is active, writes go to mmap'd file
1031 * ========================================================================= */
1032 bool mmap_active = log_mmap_is_active();
1033 if (mmap_active) {
1034 maybe_rotate_log();
1035
1036 va_list args;
1037 va_start(args, fmt);
1038 char msg_buffer[LOG_MMAP_MSG_BUFFER_SIZE];
1039 int msg_len = safe_vsnprintf(msg_buffer, sizeof(msg_buffer), fmt, args);
1040 va_end(args);
1041
1042 // Truncate at whole line boundaries to avoid UTF-8 issues
1043 if (msg_len > 0) {
1044 msg_len = truncate_at_whole_line(msg_buffer, msg_len, sizeof(msg_buffer));
1045 }
1046
1047 // Validate UTF-8 in formatted message
1048 validate_log_message_utf8(msg_buffer, "mmap log message");
1049
1050 log_mmap_write(level, file, line, func, "%s", msg_buffer);
1051
1052 // Terminal output (check with atomic loads)
1053 if (atomic_load_u64(&g_log.terminal_output_enabled) && !atomic_load_bool(&g_log.terminal_locked)) {
1054 char time_buf[LOG_TIMESTAMP_BUFFER_SIZE];
1055 uint64_t time_ns = time_get_realtime_ns();
1056 get_current_time_formatted(time_buf);
1057
1058 // Choose output stream using unified routing logic
1059 int fd = terminal_choose_log_fd(level);
1060 FILE *output_stream = (fd == STDERR_FILENO) ? stderr : stdout;
1061 // Check if colors should be used
1062 // Priority 1: If --color was explicitly passed, force colors
1063 bool use_colors = true; // Default: enable colors
1064 if (g_color_flag_passed && !g_color_flag_value) {
1065 use_colors = false; // --color=false explicitly disables colors
1066 }
1067 // Priority 2: If --color NOT explicitly passed, enable colors by default
1068
1069 char header_buffer[512];
1070 int header_len = format_log_header(header_buffer, sizeof(header_buffer), level, time_buf, file, line, func,
1071 use_colors, time_ns);
1072
1073 if (header_len >= 0 && header_len < (int)sizeof(header_buffer)) {
1074 // Platform-specific log hook (e.g., for WASM browser console)
1075 platform_log_hook(level, msg_buffer);
1076
1077 if (use_colors) {
1078 const char *colorized_msg = colorize_log_message(msg_buffer);
1079 const char **colors = log_get_color_array();
1080 if (colors) {
1081 safe_fprintf(output_stream, "%s%s%s%s\n", header_buffer, colors[LOG_COLOR_RESET], colorized_msg,
1082 colors[LOG_COLOR_RESET]);
1083 } else {
1084 safe_fprintf(output_stream, "%s%s\n", header_buffer, colorized_msg);
1085 }
1086 } else {
1087 safe_fprintf(output_stream, "%s%s\n", header_buffer, msg_buffer);
1088 }
1089 (void)fflush(output_stream);
1090 }
1091 }
1092 return;
1093 }
1094 /* =========================================================================
1095 * FILE I/O PATH: Lock-free using atomic write() syscalls
1096 * ========================================================================= */
1097 char time_buf[LOG_TIMESTAMP_BUFFER_SIZE];
1098 uint64_t time_ns = time_get_realtime_ns();
1099 get_current_time_formatted(time_buf);
1100 // Format message for file output
1101 char log_buffer[LOG_MSG_BUFFER_SIZE];
1102 va_list args;
1103 va_start(args, fmt);
1104 int header_len = format_log_header(log_buffer, sizeof(log_buffer), level, time_buf, file, line, func, false, time_ns);
1105 if (header_len < 0 || header_len >= (int)sizeof(log_buffer)) {
1106 LOGGING_INTERNAL_ERROR(ERROR_INVALID_STATE, "Failed to format log header");
1107 va_end(args);
1108 return;
1109 }
1110 int msg_len = header_len;
1111 int formatted_len = safe_vsnprintf(log_buffer + header_len, sizeof(log_buffer) - (size_t)header_len, fmt, args);
1112 if (formatted_len < 0) {
1113 LOGGING_INTERNAL_ERROR(ERROR_INVALID_STATE, "Failed to format log message");
1114 va_end(args);
1115 return;
1116 }
1117
1118 msg_len += formatted_len;
1119 // Truncate at whole line boundaries to avoid UTF-8 issues
1120 msg_len = truncate_at_whole_line(log_buffer, msg_len, sizeof(log_buffer));
1121 // Add newline if there's room and message doesn't already end with one
1122 if (msg_len > 0 && msg_len < (int)sizeof(log_buffer) - 1) {
1123 if (log_buffer[msg_len - 1] != '\n') {
1124 log_buffer[msg_len++] = '\n';
1125 log_buffer[msg_len] = '\0';
1126 }
1127 }
1128 va_end(args);
1129 // Validate UTF-8 in formatted message
1130 validate_log_message_utf8(log_buffer, "leveled log message");
1131
1132 // Extract the user message part (without header) for JSON logging
1133 const char *user_message = log_buffer + header_len;
1134
1135 // For JSON output: strip automatic trailing newline (keep explicit ones)
1136 char json_message_buf[LOG_MSG_BUFFER_SIZE];
1137 const char *json_message = user_message;
1138 size_t user_msg_len = strlen(user_message);
1139
1140 // If message ends with newline and it was added automatically (not in original message)
1141 if (user_msg_len > 0 && user_message[user_msg_len - 1] == '\n') {
1142 // The newline was added automatically if the message part didn't end with one
1143 // before we added it on line 1067. We can check this by looking at msg_len - 1
1144 // If msg_len >= header_len + 2, then msg_len - 2 would be the char before the newline
1145 int msg_content_len = msg_len - header_len;
1146 if (msg_content_len > 1 && log_buffer[msg_len - 2] != '\n') {
1147 // Newline was added automatically - strip it for JSON
1148 if (user_msg_len - 1 < sizeof(json_message_buf)) {
1149 memcpy(json_message_buf, user_message, user_msg_len - 1);
1150 json_message_buf[user_msg_len - 1] = '\0';
1151 json_message = json_message_buf;
1152 }
1153 }
1154 }
1155
1156 // Check if JSON format is enabled
1157 int json_fd = atomic_load_int(&g_log.json_file);
1158 bool json_format_enabled = (json_fd >= 0);
1159
1160 // If JSON format is enabled, output ONLY JSON (skip text output)
1161 if (json_format_enabled) {
1162 // Output JSON to the JSON file descriptor
1163 log_json_write(json_fd, level, time_ns, file, line, func, json_message);
1164 // Also output JSON to console using unified routing logic, respecting quiet flag
1165 if (atomic_load_u64(&g_log.terminal_output_enabled)) {
1166 int console_fd = terminal_choose_log_fd(level);
1167 log_json_write(console_fd, level, time_ns, file, line, func, json_message);
1168 }
1169 } else {
1170 // Text format: output to file and terminal
1171 // No heap allocation - use log_buffer directly
1172 // Write to file (atomic write syscall) - use original buffer (with ANSI codes)
1173 int file_fd = (int)atomic_load_u64(&g_log.file);
1174 if (file_fd >= 0 && file_fd != STDERR_FILENO) {
1175 write_to_log_file_atomic(log_buffer, msg_len, NULL);
1176 }
1177
1178 // Write to terminal (atomic state checks)
1179 va_list args_terminal;
1180 va_start(args_terminal, fmt);
1181 write_to_terminal_atomic(level, time_buf, file, line, func, fmt, args_terminal, time_ns);
1182 va_end(args_terminal);
1183 }
1184}

1781 * Log Recoloring for Plain Text Entries
1782 * ============================================================================ */
1783
1790size_t log_recolor_plain_entry(const char *plain_line, char *colored_buf, size_t buf_size) {
1791 if (!plain_line || !colored_buf || buf_size < 128) {
1792 return 0;
1793 }
1794
1795 static char work_buffer[LOG_MSG_BUFFER_SIZE + 1024];
1796
1797 // Parse format FIRST, regardless of color availability
1798 // Parse format: [TIMESTAMP] [LEVEL] [tid:THREAD_ID] FILE:LINE in FUNC(): MESSAGE
1799 const char *p = plain_line;
1800
1801 // Extract timestamp [TIMESTAMP]
1802 if (*p != '[') {
1803 return 0; // Invalid format
1804 }
1805 p++;
1806 const char *timestamp_start = p;
1807
1808 // Find the closing ] for timestamp - look for pattern that ends with proper timestamp format
1809 // Valid timestamp: HH:MM:SS.UUUUUU (time with microseconds, no date)
1810 while (*p && *p != ']') {
1811 p++;
1812 }
1813 if (*p != ']') {
1814 return 0; // Malformed - no closing bracket
1815 }
1816
1817 size_t timestamp_len = p - timestamp_start;
1818 char timestamp[64];
1819 if (timestamp_len >= sizeof(timestamp) || timestamp_len == 0) {
1820 return 0; // Invalid timestamp - empty or too long
1821 }
1822 SAFE_STRNCPY(timestamp, timestamp_start, timestamp_len);
1823 timestamp[timestamp_len] = '\0';
1824 p++; // Skip ]
1825
1826 // Skip whitespace after timestamp
1827 while (*p && *p == ' ') {
1828 p++;
1829 }
1830
1831 // Extract level [LEVEL]
1832 if (*p != '[') {
1833 return 0; // Missing opening bracket for level
1834 }
1835 p++;
1836 const char *level_start = p;
1837 while (*p && *p != ']') {
1838 p++;
1839 }
1840 if (*p != ']') {
1841 return 0; // Missing closing bracket for level
1842 }
1843 size_t level_len = p - level_start;
1844 char level_str[16];
1845 if (level_len >= sizeof(level_str) || level_len == 0) {
1846 return 0; // Level string too long or empty
1847 }
1848 /* Use strncpy directly for substring extraction (not null-terminated source) */
1849 strncpy(level_str, level_start, level_len);
1850 level_str[level_len] = '\0';
1851
1852 // Determine log level for color selection
1853 log_level_t level = LOG_INFO; // Default
1854 if (strstr(level_str, "DEV") || strstr(level_str, "DEBUG")) {
1855 level = LOG_DEBUG;
1856 } else if (strstr(level_str, "INFO")) {
1857 level = LOG_INFO;
1858 } else if (strstr(level_str, "WARN")) {
1859 level = LOG_WARN;
1860 } else if (strstr(level_str, "ERROR")) {
1861 level = LOG_ERROR;

1870 p++;
1871 }

1883 p++;
1884 }
1885 if (*p == ']') {
1886 p++;
1887 }
1888 } else {
1889 p = tid_end + 1; // Skip past the ]
1890 }
1891 // Skip whitespace after tid
1892 while (*p && *p == ' ') {
1893 p++;
1894 }
1895 }
1896 // tid is optional - continue parsing if not present

1902 }

1924 }

1929 p++;
1930 }
1931
1932 // Try to find "in " - if not found, might still be valid, just harder to parse
1933 if (strncmp(p, "in ", 3) == 0) {
1934 p += 3; // Skip "in "
1935 } else if (*p == 'i' && *(p + 1) == 'n' && (*(p + 2) == ' ' || *(p + 2) == '\t')) {

1937 p += 2; // Skip "in"
1938 while (*p && (*p == ' ' || *p == '\t')) {
1939 p++;
1940 }
1941 } else {
1942 // Missing "in" keyword - this is a format error
1943 return 0;
1944 }
1945

1947 const char *func_start = p;
1948 while (*p && *p != '(') {
1949 p++;
1950 }

1952 return 0; // Missing function name or parentheses
1953 }

1958 func_len--;
1959 }
1960

1962 if (func_len >= sizeof(func_name)) {
1963 func_len = sizeof(func_name) - 1;
1964 }
1965 if (func_len > 0) {
1966 /* Use strncpy directly for substring extraction (not null-terminated source) */
1967 strncpy(func_name, func_start, func_len);
1968 }
1969 func_name[func_len] = '\0';
1970
1971 // Skip "(" and ")" - be lenient about what's between them
1972 if (*p == '(') {
1973 p++;
1974 while (*p && *p != ')') {
1975 p++;
1976 }
1977 if (*p == ')') {
1978 p++;
1979 }
1980 }
1981
1982 // Skip whitespace and optional colon(s) and other separators
1983 while (*p && (*p == ' ' || *p == ':' || *p == '\t')) {
1984 p++;
1985 }
1986
1987 // Remaining is the message
1988 const char *message = p;
1989
1990 // Format is valid, get colors from logging system
1991 const char **colors = log_get_color_array();
1992 if (!colors) {
1993 // No colors available, return plain text
1994 static bool warned_once = false;
1995 if (!warned_once) {
1996 log_debug("WARNING: log_recolor_plain_entry() called but colors not initialized - returning plain text");
1997 warned_once = true;
1998 }

2000 if (len >= buf_size) {
2001 return 0;
2002 }
2003 SAFE_STRNCPY(colored_buf, plain_line, buf_size - 1);
2004 colored_buf[buf_size - 1] = '\0';
2005 return len;
2006 }
2007

2009 const char *level_color = colors[level];
2010 const char *reset = colors[LOG_COLOR_RESET];
2011 const char *file_color = colors[1]; // DEBUG/Cyan
2012 const char *line_color = colors[6]; // GREY (matching tid)
2013 const char *func_color = colors[0]; // DEV/Orange
2014 const char *tid_color = colors[6]; // GREY
2015
2016 int len = safe_snprintf(work_buffer, sizeof(work_buffer),
2017 "[%s%s%s] [%s%s%s] [tid:%s%llu%s] %s%s%s:%s%d%s in %s%s%s(): %s", level_color, timestamp,
2018 reset, level_color, level_str, reset, tid_color, (unsigned long long)tid, reset, file_color,
2019 file_path, reset, line_color, line_num, reset, func_color, func_name, reset, message);

2022 return 0;
2023 }
2024

2027 len = safe_snprintf(work_buffer, sizeof(work_buffer),
