109void platform_cleanup_binary_path_cache(void) {
110 if (!lifecycle_shutdown(&g_cache_lc)) {
111 return;
112 }
113
114 if (g_bin_path_cache) {
115 rwlock_wrlock(&g_cache_rwlock);
116
117 bin_cache_entry_t *entry, *tmp;
118 HASH_ITER(hh, g_bin_path_cache, entry, tmp) {
119 HASH_DELETE(hh, g_bin_path_cache, entry);
120 free_cache_entry(entry);

129bool platform_is_binary_in_path(const char *bin_name) {
130 if (!bin_name || bin_name[0] == '\0') {
131 return false;
132 }
133
134 init_cache_once();
135 if (!lifecycle_is_initialized(&g_cache_lc)) {
136 SET_ERRNO(ERROR_INVALID_STATE, "Binary PATH cache not initialized, checking directly (this should never happen)");
137 return check_binary_in_path_uncached(bin_name);
138 }
139
140 rwlock_rdlock(&g_cache_rwlock);
141 bin_cache_entry_t *entry = NULL;
142 HASH_FIND_STR(g_bin_path_cache, bin_name, entry);
143 rwlock_rdunlock(&g_cache_rwlock);
144
145 if (entry) {
146 log_dev("Binary '%s' %s in PATH (%s)", bin_name, colored_string(LOG_COLOR_INFO, "found"),
147 colored_string(LOG_COLOR_WARN, "cached"));
148 return entry->in_path;
149 }
150
151 bool found = check_binary_in_path_uncached(bin_name);
152
153 entry = SAFE_MALLOC(sizeof(bin_cache_entry_t), bin_cache_entry_t *);
154 if (!entry) {
155 SET_ERRNO(ERROR_MEMORY, "Failed to allocate cache entry");
156 return found;
157 }
158
159 entry->bin_name = platform_strdup(bin_name);
160 if (!entry->bin_name) {
161 SET_ERRNO(ERROR_MEMORY, "Failed to duplicate binary name");
162 SAFE_FREE(entry);
163 return found;
164 }
165
166 entry->in_path = found;
167
168 rwlock_wrlock(&g_cache_rwlock);
169 HASH_ADD_KEYPTR(hh, g_bin_path_cache, entry->bin_name, strlen(entry->bin_name), entry);
170 rwlock_wrunlock(&g_cache_rwlock);
171
172 char *found_notfound_str =
173 found ? colored_string(LOG_COLOR_INFO, "found") : colored_string(LOG_COLOR_ERROR, "NOT found");
174 log_dev("Binary '%s' %s in PATH", bin_name, found_notfound_str);
175
176 return found;
177}

185bool platform_get_executable_path(char *exe_path, size_t path_size) {
186 if (!exe_path || path_size == 0) {
187 SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters: exe_path=%p, path_size=%zu", (void *)exe_path, path_size);
188 return false;
189 }
190
191#ifdef _WIN32
192 DWORD len = GetModuleFileNameA(NULL, exe_path, (DWORD)path_size);
193 if (len == 0) {
194 SET_ERRNO_SYS(ERROR_INVALID_STATE, "GetModuleFileNameA failed: error code %lu", GetLastError());
195 return false;
196 }
197 if (len >= path_size) {
198 SET_ERRNO(ERROR_BUFFER_OVERFLOW,
199 "Executable path exceeds buffer size (path length >= %zu bytes, buffer size = %zu bytes)", (size_t)len,
200 path_size);
201 return false;
202 }
203 return true;
204
205#elif defined(__linux__)
206 ssize_t len = readlink("/proc/self/exe", exe_path, path_size - 1);
207 if (len < 0) {
208 SET_ERRNO_SYS(ERROR_INVALID_STATE, "readlink(\"/proc/self/exe\") failed: %s", SAFE_STRERROR(errno));
209 return false;
210 }
211 if ((size_t)len >= path_size - 1) {
212 SET_ERRNO(ERROR_BUFFER_OVERFLOW,
213 "Executable path exceeds buffer size (path length >= %zu bytes, buffer size = %zu bytes)", (size_t)len,
214 path_size);
215 return false;
216 }
217 exe_path[len] = '\0';
218 return true;
219
220#elif defined(__APPLE__)
221 uint32_t bufsize = (uint32_t)path_size;
222 int result = _NSGetExecutablePath(exe_path, &bufsize);
223 if (result != 0) {
224 SET_ERRNO(ERROR_BUFFER_OVERFLOW, "_NSGetExecutablePath failed: path requires %u bytes, buffer size = %zu bytes",
225 bufsize, path_size);
226 return false;
227 }
228 return true;
229
230#else
231 SET_ERRNO(ERROR_GENERAL, "Unsupported platform - cannot get executable path");
232 return false;
233#endif
234}
