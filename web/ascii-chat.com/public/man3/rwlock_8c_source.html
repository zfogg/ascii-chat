62void rwlock_on_unlock(rwlock_t *rwlock) {
63 if (!rwlock) return;
64 rwlock->last_unlock_time_ns = time_get_ns();
65 rwlock->unlock_count++;
66 asciichat_thread_t current_thread = (asciichat_thread_t)asciichat_thread_current_id();
67 uintptr_t current_key = asciichat_thread_to_key(current_thread);
68 if (rwlock->write_held_by_key == current_key) {
69 rwlock->write_held_by_key = 0;
70 } else if (atomic_load_u64(&rwlock->read_lock_count) > 0) {
71 atomic_fetch_sub_u64(&rwlock->read_lock_count, 1);
72 }
73}

87int rwlock_format_state(const rwlock_t *rwlock, char *buffer, size_t size) {
88 if (!rwlock || !buffer || size == 0) return 0;
89
90 int offset = 0;
91 uint64_t now_ns = time_get_ns();
92
93 char rdlock_str[64] = "";
94 char wrlock_str[64] = "";
95 char unlock_str[64] = "";
96 char held_str[64] = "";
97 char count_str[256] = "";
98
99 if (rwlock->last_rdlock_time_ns > 0 && rwlock->last_rdlock_time_ns <= now_ns) {
100 char elapsed_str[64];
101 time_pretty(now_ns - rwlock->last_rdlock_time_ns, -1, elapsed_str, sizeof(elapsed_str));
102 snprintf(rdlock_str, sizeof(rdlock_str), "rdlock=%s", elapsed_str);
103 }
104
105 if (rwlock->last_wrlock_time_ns > 0 && rwlock->last_wrlock_time_ns <= now_ns) {
106 char elapsed_str[64];
107 time_pretty(now_ns - rwlock->last_wrlock_time_ns, -1, elapsed_str, sizeof(elapsed_str));
108 snprintf(wrlock_str, sizeof(wrlock_str), "wrlock=%s", elapsed_str);
109 }
110
111 if (rwlock->last_unlock_time_ns > 0 && rwlock->last_unlock_time_ns <= now_ns) {
112 char elapsed_str[64];
113 time_pretty(now_ns - rwlock->last_unlock_time_ns, -1, elapsed_str, sizeof(elapsed_str));
114 snprintf(unlock_str, sizeof(unlock_str), "unlock=%s", elapsed_str);
115 }
116
117 uint64_t read_count = atomic_load_u64(&rwlock->read_lock_count);
118 if (rwlock->write_held_by_key != 0) {
119 snprintf(held_str, sizeof(held_str), "[WRITE_LOCKED_BY=thread.%lu]", (unsigned long)rwlock->write_held_by_key);
120 } else if (read_count > 0) {
121 snprintf(held_str, sizeof(held_str), "[READ_LOCKED=%lu]", read_count);
122 }
123
124 if (rwlock->rdlock_count > 0 || rwlock->wrlock_count > 0 || rwlock->unlock_count > 0) {
125 snprintf(count_str, sizeof(count_str), "[ops: rdlock=%lu wrlock=%lu unlock=%lu]",
126 rwlock->rdlock_count, rwlock->wrlock_count, rwlock->unlock_count);
127 }
128
129 offset += snprintf(buffer + offset, size - offset, "%s %s %s %s %s",
130 rdlock_str, wrlock_str, unlock_str, held_str, count_str);
131 return offset;
132}
