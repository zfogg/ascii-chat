160static int memory_read_packet(void *opaque, uint8_t *buf, int buf_size) {
161 stdin_buffer_reader_t *reader = (stdin_buffer_reader_t *)opaque;
162
163 if (!reader || !reader->buffer || reader->pos >= reader->buffer->size)
164 return AVERROR_EOF;
165
166 int to_read = buf_size;
167 if (reader->pos + to_read > reader->buffer->size)
168 to_read = (int)(reader->buffer->size - reader->pos);
169
170 memcpy(buf, reader->buffer->data + reader->pos, to_read);
171 reader->pos += to_read;
172
173 return to_read;
174}

179static int64_t memory_seek_packet(void *opaque, int64_t offset, int whence) {
180 stdin_buffer_reader_t *reader = (stdin_buffer_reader_t *)opaque;
181
182 if (!reader || !reader->buffer)
183 return -1;
184
185 int64_t new_pos = 0;
186 if (whence == SEEK_SET) {
187 new_pos = offset;
188 } else if (whence == SEEK_CUR) {
189 new_pos = (int64_t)reader->pos + offset;
190 } else if (whence == SEEK_END) {
191 new_pos = (int64_t)reader->buffer->size + offset;
192 } else {
193 return -1;
194 }
195
196 if (new_pos < 0 || new_pos > (int64_t)reader->buffer->size)
197 return -1;
198
199 reader->pos = (size_t)new_pos;
200 return new_pos;
201}

207static stdin_buffer_t *stdin_buffer_read_all(void) {
208 stdin_buffer_t *sb = SAFE_MALLOC(sizeof(stdin_buffer_t), stdin_buffer_t *);
209 if (!sb)
210 return NULL;
211
212 sb->data = NULL;
213 sb->size = 0;
214
215 // Read stdin in 64KB chunks
216 const size_t chunk_size = 65536;
217 uint8_t chunk_buffer[chunk_size];
218 size_t total_capacity = 0;
219 int chunk_count = 0;
220
221 while (1) {
222 size_t bytes_read = fread(chunk_buffer, 1, chunk_size, stdin);
223 if (bytes_read == 0)
224 break;
225 chunk_count++;
226
227 // Grow buffer if needed
228 if (sb->size + bytes_read > total_capacity) {
229 total_capacity = (sb->size + bytes_read) * 2; // Always double capacity
230 uint8_t *new_data = SAFE_MALLOC(total_capacity, uint8_t *);
231 if (!new_data) {
232 SAFE_FREE(sb->data);
233 SAFE_FREE(sb);
234 return NULL;
235 }
236 if (sb->data) {
237 memcpy(new_data, sb->data, sb->size);
238 SAFE_FREE(sb->data);
239 }
240 sb->data = new_data;
241 }
242
243 // Copy chunk to buffer
244 memcpy(sb->data + sb->size, chunk_buffer, bytes_read);
245 sb->size += bytes_read;
246 }
247
248 log_info("Buffered stdin: %zu bytes in %d chunks (capacity: %zu)", sb->size, chunk_count, total_capacity);
249 if (sb->size > 0) {
250 log_info(" First 16 bytes of stdin: %02x %02x %02x %02x %02x %02x %02x %02x %02x %02x %02x %02x %02x %02x %02x %02x",
251 sb->data[0], sb->data[1], sb->data[2], sb->data[3], sb->data[4], sb->data[5], sb->data[6], sb->data[7],
252 sb->data[8], sb->data[9], sb->data[10], sb->data[11], sb->data[12], sb->data[13], sb->data[14], sb->data[15]);
253 }
254 return sb;
255}

316static void *ffmpeg_decoder_prefetch_thread_func(void *arg) {
317 ffmpeg_decoder_t *decoder = (ffmpeg_decoder_t *)arg;
318 if (!decoder || !decoder->prefetch_image_a || !decoder->prefetch_image_b) {
319 return NULL;

322 log_debug("Video prefetch thread started");
323 bool use_image_a = true; // Track which buffer we're using (critical for correct buffer swapping)
324
325 while (true) {
326 mutex_lock(&decoder->prefetch_mutex);
327
328 // Check if thread should stop
329 bool should_stop = decoder->prefetch_should_stop || decoder->eof_reached;
330 if (should_stop) {
331 mutex_unlock(&decoder->prefetch_mutex);
332 break;
333 }
334
335 // Pause if seek is in progress - wait for signal to continue
336 // cond_wait automatically releases and re-acquires mutex
337 while (decoder->seeking_in_progress) {
338 cond_wait(&decoder->prefetch_cond, &decoder->prefetch_mutex);
339 }
340
341 // Mutex is re-acquired after cond_wait - KEEP IT HELD
342
343 // Check if the buffer we want to use is still in use by the main thread
344 // If so, skip this iteration and try again later
345 bool buffer_in_use = use_image_a ? decoder->buffer_a_in_use : decoder->buffer_b_in_use;
346 if (buffer_in_use) {
347 // Can't use this buffer yet - main thread is still rendering it
348 mutex_unlock(&decoder->prefetch_mutex);
349 platform_sleep_us(1 * US_PER_MS_INT); // 1ms - brief sleep before retry
350 continue;
351 }
352
353 uint64_t read_start_ns = time_get_ns();
354 bool frame_decoded = false;
355
356 // Release prefetch mutex but hold read_frame_mutex to prevent concurrent av_read_frame() calls from audio thread
357 mutex_unlock(&decoder->prefetch_mutex);
358 mutex_lock(&decoder->read_frame_mutex);
359
360 // Read packets until we get a video frame - read_frame_mutex HELD
361 while (true) {
362 int ret = av_read_frame(decoder->format_ctx, decoder->packet);
363 if (ret < 0) {
364 if (ret == AVERROR_EOF) {
365 decoder->eof_reached = true;

367 break;

371 if (decoder->packet->stream_index != decoder->video_stream_idx) {
372 av_packet_unref(decoder->packet);
373 continue;

389 break;

392 // Frame decoded - mutex still held
393 // Update position tracking
394 decoder->last_video_pts =

398 int width = decoder->video_codec_ctx->width;
399 int height = decoder->video_codec_ctx->height;
400
401 // Get current decode buffer based on which one we're using
402 // (Buffer availability was already checked before entering the decode loop)
403 image_t *decode_buffer = use_image_a ? decoder->prefetch_image_a : decoder->prefetch_image_b;
404
405 // Reallocate if needed (width/height changed)
406 if (decode_buffer->w != width || decode_buffer->h != height) {

409 if (!decode_buffer) {

414 if (use_image_a) {
415 decoder->prefetch_image_a = decode_buffer;
416 } else {
417 decoder->prefetch_image_b = decode_buffer;

471 height, pix_fmt_to_use, AV_PIX_FMT_NONE);

474 }

477 // Log input YUV data
478 uint8_t *y_data = decoder->frame->data[0];
479 uint8_t *u_data = decoder->frame->data[1];
480 uint8_t *v_data = decoder->frame->data[2];

484 decoder->frame->linesize[0], decoder->frame->linesize[1], decoder->frame->linesize[2]);
485 }
486
487 int ret_scale = sws_scale(decoder->sws_ctx, (const uint8_t *const *)decoder->frame->data, decoder->frame->linesize, 0, height,
488 dst_data, dst_linesize);

490 log_info("[SWS_SCALE] Frame %d: returned %d", decode_count, ret_scale);
491 rgb_pixel_t *test_px = (rgb_pixel_t *)decode_buffer->pixels;
492 log_info("[SWS_RESULT] Frame %d: RGB[0]=(%u,%u,%u), RGB[100]=(%u,%u,%u)",

494 }

496 frame_decoded = true;
497 break; // Exit while loop
498 }
499
500 // Release read_frame_mutex now that av_read_frame() is done
501 mutex_unlock(&decoder->read_frame_mutex);
502
503 // Re-acquire mutex to update prefetch state
504 mutex_lock(&decoder->prefetch_mutex);
505
506 // Mutex now held - update prefetch state
507 if (frame_decoded) {
508 // Update shared prefetch state while mutex is held
509 }
510
511 mutex_unlock(&decoder->prefetch_mutex); // Release at end of main loop iteration
512
513 if (frame_decoded) {
514 uint64_t read_time_ns = time_elapsed_ns(read_start_ns, time_get_ns());
515
516 // Get current decode buffer
517 image_t *decode_buffer = use_image_a ? decoder->prefetch_image_a : decoder->prefetch_image_b;
518
519 // Update the current prefetch image (main thread will pull from this)
520 mutex_lock(&decoder->prefetch_mutex);
521 decoder->current_prefetch_image = decode_buffer;
522 decoder->prefetch_frame_ready = true;
523 mutex_unlock(&decoder->prefetch_mutex);

530 use_image_a = !use_image_a;
531 } else {
532 // EOF or error - exit thread
533 break;
534 }
535 }
536
537 log_debug("Video prefetch thread stopped");
538 return NULL;
539}

544static asciichat_error_t open_codec_context(AVFormatContext *fmt_ctx, enum AVMediaType type, int *stream_idx,
545 AVCodecContext **codec_ctx) {
546 int ret = av_find_best_stream(fmt_ctx, type, -1, -1, NULL, 0);
547 if (ret < 0) {
548 // Stream not found - not an error, just means no stream of this type
549 *stream_idx = -1;
550 *codec_ctx = NULL;
551 return ASCIICHAT_OK;
552 }
553
554 *stream_idx = ret;
555 AVStream *stream = fmt_ctx->streams[ret];
556
557 // Find decoder
558 const AVCodec *codec = avcodec_find_decoder(stream->codecpar->codec_id);
559 if (!codec) {
560 return SET_ERRNO(ERROR_MEDIA_DECODE, "Codec not found for stream %d", ret);
561 }
562
563 // Allocate codec context
564 *codec_ctx = avcodec_alloc_context3(codec);
565 if (!*codec_ctx) {
566 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate codec context");
567 }
568
569 // Copy codec parameters
570 if (avcodec_parameters_to_context(*codec_ctx, stream->codecpar) < 0) {
571 avcodec_free_context(codec_ctx);
572 return SET_ERRNO(ERROR_MEDIA_DECODE, "Failed to copy codec parameters");
573 }
574
575 // Open codec
576 if (avcodec_open2(*codec_ctx, codec, NULL) < 0) {
577 avcodec_free_context(codec_ctx);
578 return SET_ERRNO(ERROR_MEDIA_DECODE, "Failed to open codec");
579 }
580
581 return ASCIICHAT_OK;
582}

611 decoder->audio_stream_idx = -1;

633 // Enable HTTP persistent connection (keep-alive) for better performance
634 av_dict_set(&options, "http_persistent", "1", 0);
635 // Reduce connect timeout to fail faster if server is unreachable
636 av_dict_set(&options, "connect_timeout", "5000000", 0);

641 av_dict_free(&options); // Free options dictionary
642

646 ret = -1;

652 if (ret == -1) {

673 &decoder->video_codec_ctx);
674 if (err != ASCIICHAT_OK) {
675 log_warn("Failed to open video codec (file may be audio-only)");
676 }
677
678 // Open audio codec - audio is enabled by default (no option needed)
679 // Always try to open audio codec, don't rely on GET_OPTION(audio_enabled) which has a default issue
680 err = open_codec_context(decoder->format_ctx, AVMEDIA_TYPE_AUDIO, &decoder->audio_stream_idx,
681 &decoder->audio_codec_ctx);
682 if (err != ASCIICHAT_OK) {
683 log_debug("No audio codec found (file may be video-only or audio codec not available)");
684 decoder->audio_stream_idx = -1;

686 }

695 // Allocate frame and packet
696 decoder->frame = av_frame_alloc();
697 decoder->packet = av_packet_alloc();
698 if (!decoder->frame || !decoder->packet) {
699 SET_ERRNO(ERROR_MEMORY, "Failed to allocate frame/packet");
700 ffmpeg_decoder_destroy(decoder);
701 return NULL;
702 }
703
704 // Initialize swscale context for video if present
705 if (decoder->video_codec_ctx) {
706 // Validate codec context has valid dimensions and pixel format
707 // For HTTP streams, these might not be valid until first frame is read
708 if (decoder->video_codec_ctx->width <= 0 || decoder->video_codec_ctx->height <= 0) {
709 log_warn("Video codec has invalid dimensions (%dx%d), will initialize swscale on first frame",
710 decoder->video_codec_ctx->width, decoder->video_codec_ctx->height);
711 // Don't create swscale context yet - will create it lazily on first frame read
712 } else if (decoder->video_codec_ctx->pix_fmt == AV_PIX_FMT_NONE) {
713 log_warn("Video codec has invalid pixel format, will initialize swscale on first frame");
714 // Don't create swscale context yet - will create it lazily on first frame read
715 } else {
716 // Create swscale context with valid parameters
717 decoder->sws_ctx =
718 sws_getContext(decoder->video_codec_ctx->width, decoder->video_codec_ctx->height,
719 decoder->video_codec_ctx->pix_fmt, decoder->video_codec_ctx->width,
720 decoder->video_codec_ctx->height, AV_PIX_FMT_RGB24, SWS_BILINEAR, NULL, NULL, NULL);
721 if (!decoder->sws_ctx) {
722 SET_ERRNO(ERROR_MEDIA_DECODE, "Failed to create swscale context");
723 ffmpeg_decoder_destroy(decoder);
724 return NULL;
725 }
726 }
727 }
728
729 // Initialize swresample context for audio if present
730 if (decoder->audio_codec_ctx) {
731 // Store output sample rate for position tracking
732 decoder->audio_sample_rate = TARGET_SAMPLE_RATE;
733
734 // Allocate resampler context
735 decoder->swr_ctx = swr_alloc();
736 if (!decoder->swr_ctx) {
737 SET_ERRNO(ERROR_MEMORY, "Failed to allocate swresample context");
738 ffmpeg_decoder_destroy(decoder);
739 return NULL;
740 }
741
742 // Set options
743 av_opt_set_chlayout(decoder->swr_ctx, "in_chlayout", &decoder->audio_codec_ctx->ch_layout, 0);
744 av_opt_set_int(decoder->swr_ctx, "in_sample_rate", decoder->audio_codec_ctx->sample_rate, 0);
745 av_opt_set_sample_fmt(decoder->swr_ctx, "in_sample_fmt", decoder->audio_codec_ctx->sample_fmt, 0);
746
747 AVChannelLayout out_ch_layout = AV_CHANNEL_LAYOUT_MONO;
748 av_opt_set_chlayout(decoder->swr_ctx, "out_chlayout", &out_ch_layout, 0);
749 av_opt_set_int(decoder->swr_ctx, "out_sample_rate", TARGET_SAMPLE_RATE, 0);
750 av_opt_set_sample_fmt(decoder->swr_ctx, "out_sample_fmt", AV_SAMPLE_FMT_FLT, 0);
751
752 // Initialize
753 if (swr_init(decoder->swr_ctx) < 0) {
754 SET_ERRNO(ERROR_MEDIA_DECODE, "Failed to initialize swresample context");
755 ffmpeg_decoder_destroy(decoder);
756 return NULL;
757 }
758
759 // Allocate audio buffer (10 seconds worth)
760 decoder->audio_buffer_size = TARGET_SAMPLE_RATE * 10;
761 decoder->audio_buffer = SAFE_MALLOC(decoder->audio_buffer_size * sizeof(float), float *);
762 if (!decoder->audio_buffer) {
763 SET_ERRNO(ERROR_MEMORY, "Failed to allocate audio buffer");
764 ffmpeg_decoder_destroy(decoder);
765 return NULL;
766 }
767 }
768
769 // Initialize video frame prefetching system (for YouTube streaming)
770 if (decoder->video_stream_idx >= 0) {
771 int width = decoder->video_codec_ctx->width;
772 int height = decoder->video_codec_ctx->height;
773
774 // Create two prefetch image buffers for double-buffering
775 decoder->prefetch_image_a = image_new((size_t)width, (size_t)height);
776 decoder->prefetch_image_b = image_new((size_t)width, (size_t)height);
777 if (!decoder->prefetch_image_a || !decoder->prefetch_image_b) {
778 SET_ERRNO(ERROR_MEMORY, "Failed to allocate prefetch image buffers");
779 ffmpeg_decoder_destroy(decoder);
780 return NULL;
781 }
782
783 decoder->current_prefetch_image = decoder->prefetch_image_a;
784 decoder->prefetch_frame_ready = false;
785
786 // Initialize prefetch mutex
787 if (mutex_init(&decoder->prefetch_mutex, "ffmpeg_prefetch") != 0) {
788 SET_ERRNO(ERROR_MEMORY, "Failed to initialize prefetch mutex");
789 ffmpeg_decoder_destroy(decoder);
790 return NULL;
791 }
792
793 if (cond_init(&decoder->prefetch_cond, "ffmpeg_prefetch") != 0) {
794 SET_ERRNO(ERROR_MEMORY, "Failed to initialize prefetch condition variable");
795 mutex_destroy(&decoder->prefetch_mutex);
796 ffmpeg_decoder_destroy(decoder);
797 return NULL;
798 }
799
800 if (mutex_init(&decoder->read_frame_mutex, "ffmpeg_read_frame") != 0) {
801 SET_ERRNO(ERROR_MEMORY, "Failed to initialize read_frame mutex");
802 mutex_destroy(&decoder->prefetch_mutex);
803 cond_destroy(&decoder->prefetch_cond);
804 ffmpeg_decoder_destroy(decoder);
805 return NULL;
806 }
807
808 decoder->prefetch_thread_running = false;
809 decoder->prefetch_should_stop = false;
810 }
811
812 log_debug("FFmpeg decoder opened: %s (video=%s, audio=%s)", path, decoder->video_stream_idx >= 0 ? "yes" : "no",
813 decoder->audio_stream_idx >= 0 ? "yes" : "no");
814
815 char decoder_name[256];
816 snprintf(decoder_name, sizeof(decoder_name), "%s", path);
817 NAMED_REGISTER_FFMPEG_DECODER(decoder, decoder_name, NULL);
818
819 return decoder;
820}

822ffmpeg_decoder_t *ffmpeg_decoder_create_stdin(void) {

833 ffmpeg_decoder_t *decoder = SAFE_MALLOC(sizeof(ffmpeg_decoder_t), ffmpeg_decoder_t *);

837 }
838
839 memset(decoder, 0, sizeof(*decoder));
840 decoder->video_stream_idx = -1;
841 decoder->audio_stream_idx = -1;
842 decoder->is_stdin = true;

866 AVIO_BUFFER_SIZE, // buffer size

887 SET_ERRNO(ERROR_MEMORY, "Failed to allocate format context");
888 av_freep(&decoder->avio_ctx->buffer);
889 avio_context_free(&decoder->avio_ctx);
890 SAFE_FREE(decoder);
891 return NULL;
892 }
893
894 decoder->format_ctx->pb = decoder->avio_ctx;

896

898 int ret = 0;

900 // Open input from buffered stdin
901 ret = avformat_open_input(&decoder->format_ctx, NULL, NULL, NULL);

903 // Find stream info

905 ret = -1;
906 }

908 });

920 }

930 }

934 &decoder->audio_codec_ctx);

936 log_warn("Failed to open audio codec from stdin");
937 }
938 } else {
939 decoder->audio_stream_idx = -1;
940 decoder->audio_codec_ctx = NULL;
941 log_debug("Audio decoding disabled by user option");
942 }

945 SET_ERRNO(ERROR_MEDIA_DECODE, "No video or audio streams found in stdin");
946 ffmpeg_decoder_destroy(decoder);
947 return NULL;
948 }
949
950 // Allocate frame and packet
951 decoder->frame = av_frame_alloc();
952 decoder->packet = av_packet_alloc();
953 if (!decoder->frame || !decoder->packet) {
954 SET_ERRNO(ERROR_MEMORY, "Failed to allocate frame/packet");

956 return NULL;

972 decoder->sws_ctx =
973 sws_getContext(decoder->video_codec_ctx->width, decoder->video_codec_ctx->height,
974 decoder->video_codec_ctx->pix_fmt, decoder->video_codec_ctx->width,

1072void ffmpeg_decoder_destroy(ffmpeg_decoder_t *decoder) {
1073 if (!decoder) {
1074 return;
1075 }
1076
1077 NAMED_UNREGISTER(decoder);
1078
1079 // Stop prefetch thread (signal it to stop and wait for it to finish)
1080 if (decoder->prefetch_thread_running) {
1081 mutex_lock(&decoder->prefetch_mutex);
1082 decoder->prefetch_should_stop = true;
1083 mutex_unlock(&decoder->prefetch_mutex);
1084
1085 // Wait for thread to finish
1086 asciichat_thread_join(&decoder->prefetch_thread, NULL);
1087 decoder->prefetch_thread_running = false;
1088 }
1089
1090 // Clean up prefetch state
1091 cond_destroy(&decoder->prefetch_cond);
1092 mutex_destroy(&decoder->prefetch_mutex);
1093 mutex_destroy(&decoder->read_frame_mutex);
1094
1095 // Free prefetch image buffers
1096 if (decoder->prefetch_image_a) {
1097 image_destroy(decoder->prefetch_image_a);
1098 decoder->prefetch_image_a = NULL;
1099 }
1100 if (decoder->prefetch_image_b) {
1101 image_destroy(decoder->prefetch_image_b);
1102 decoder->prefetch_image_b = NULL;
1103 }
1104
1105 // Don't destroy current_image - it points to one of the prefetch buffers
1106 // which have already been destroyed above
1107 decoder->current_image = NULL;
1108
1109 // Free audio buffer
1110 SAFE_FREE(decoder->audio_buffer);
1111
1112 // Free swscale context
1113 if (decoder->sws_ctx) {
1114 sws_freeContext(decoder->sws_ctx);
1115 decoder->sws_ctx = NULL;
1116 }
1117
1118 // Free swresample context
1119 if (decoder->swr_ctx) {
1120 swr_free(&decoder->swr_ctx);
1121 }
1122
1123 // Free frame and packet
1124 if (decoder->frame) {
1125 av_frame_free(&decoder->frame);
1126 }
1127 if (decoder->packet) {
1128 av_packet_free(&decoder->packet);
1129 }
1130
1131 // Free codec contexts
1132 if (decoder->video_codec_ctx) {
1133 avcodec_free_context(&decoder->video_codec_ctx);
1134 }
1135 if (decoder->audio_codec_ctx) {
1136 avcodec_free_context(&decoder->audio_codec_ctx);
1137 }
1138
1139 // Free format context
1140 if (decoder->format_ctx) {
1141 avformat_close_input(&decoder->format_ctx);
1142 }
1143
1144 // Free AVIO context (stdin only)
1145 if (decoder->avio_ctx) {
1146 // Free the opaque reader for stdin
1147 if (decoder->is_stdin) {
1148 SAFE_FREE(decoder->avio_ctx->opaque);
1149 }
1150 av_freep(&decoder->avio_ctx->buffer);
1151 avio_context_free(&decoder->avio_ctx);
1152 }
1153
1154 SAFE_FREE(decoder);
1155}

1161image_t *ffmpeg_decoder_read_video_frame(ffmpeg_decoder_t *decoder) {
1162 if (!decoder || decoder->video_stream_idx < 0) {
1163 return NULL;
1164 }
1165
1166 // Try to get a prefetched frame from the background thread (preferred path)
1167 mutex_lock(&decoder->prefetch_mutex);
1168 if (decoder->prefetch_frame_ready && decoder->current_prefetch_image) {
1169 // Release the previous buffer (rendering is now complete)
1170 if (decoder->current_read_buffer == decoder->prefetch_image_a) {
1171 decoder->buffer_a_in_use = false;
1172 } else if (decoder->current_read_buffer == decoder->prefetch_image_b) {
1173 decoder->buffer_b_in_use = false;
1174 }
1175
1176 image_t *frame = decoder->current_prefetch_image;
1177 decoder->prefetch_frame_ready = false;
1178
1179 // Mark the new buffer as in use (prevent prefetch thread from overwriting it during rendering)
1180 if (frame == decoder->prefetch_image_a) {
1181 decoder->buffer_a_in_use = true;
1182 } else if (frame == decoder->prefetch_image_b) {
1183 decoder->buffer_b_in_use = true;
1184 }
1185
1186 decoder->current_read_buffer = frame;
1187 mutex_unlock(&decoder->prefetch_mutex);
1188
1189 // Use the prefetched frame
1190 decoder->current_image = frame;
1191 log_dev_every(5 * US_PER_SEC_INT, "Using prefetched frame");
1192 return frame;
1193 }
1194 mutex_unlock(&decoder->prefetch_mutex);
1195
1196 // No fallback synchronous decode - rely on background prefetch thread
1197 // Skipping frames when prefetch not ready allows audio timing to advance
1198 // This is critical for proper audio-video sync when prefetch is active
1199 log_dev_every(5 * US_PER_SEC_INT,
1200 "Prefetch frame not ready, skipping to next iteration (allow prefetch to catch up)");
1201 return NULL;
1202}

1211asciichat_error_t ffmpeg_decoder_start_prefetch(ffmpeg_decoder_t *decoder) {
1212 if (!decoder || decoder->video_stream_idx < 0) {
1213 return ERROR_INVALID_PARAM;
1214 }
1215
1216 if (!decoder->prefetch_image_a || !decoder->prefetch_image_b) {
1217 return ERROR_INVALID_PARAM;
1218 }
1219
1220 // Already running
1221 if (decoder->prefetch_thread_running) {
1222 return ASCIICHAT_OK;
1223 }
1224
1225 // Reset stop flag and create thread
1226 decoder->prefetch_should_stop = false;
1227
1228 int thread_err = asciichat_thread_create(&decoder->prefetch_thread, "ffmpeg_prefetch",
1229 ffmpeg_decoder_prefetch_thread_func, decoder);
1230 if (thread_err != 0) {
1231 return SET_ERRNO(ERROR_THREAD, "Failed to create video prefetch thread");
1232 }
1233
1234 decoder->prefetch_thread_running = true;
1235 return ASCIICHAT_OK;
1236}

1241void ffmpeg_decoder_stop_prefetch(ffmpeg_decoder_t *decoder) {
1242 if (!decoder || !decoder->prefetch_thread_running) {
1243 return;
1244 }
1245
1246 decoder->prefetch_should_stop = true;
1247 // Wait up to 2 seconds for thread to stop
1248 // The interrupt callback should cause av_read_frame to abort quickly
1249 int join_result = asciichat_thread_join_timeout(&decoder->prefetch_thread, NULL, 2000 * NS_PER_MS_INT);
1250
1251 if (join_result == 0) {
1252 // Thread exited successfully
1253 decoder->prefetch_thread_running = false;
1254 } else {
1255 // Timeout: thread is still running (blocked on I/O)
1256 // Mark as stopped anyway - we'll create a new thread on restart
1257 // The old thread will eventually finish and exit
1258 decoder->prefetch_thread_running = false;
1259 }
1260}

1297double ffmpeg_decoder_get_video_fps(ffmpeg_decoder_t *decoder) {
1298 if (!decoder || decoder->video_stream_idx < 0) {
1299 return -1.0;
1300 }
1301
1302 AVStream *stream = decoder->format_ctx->streams[decoder->video_stream_idx];
1303

1305 double fps = av_q2d_safe(stream->avg_frame_rate);

1321size_t ffmpeg_decoder_read_audio_samples(ffmpeg_decoder_t *decoder, float *buffer, size_t num_samples) {
1322 if (!decoder || decoder->audio_stream_idx < 0 || !buffer || num_samples == 0) {
1323 return 0;
1324 }
1325
1326 size_t samples_written = 0;
1327
1328 // First, drain any buffered samples
1329 if (decoder->audio_buffer_offset > 0) {
1330 size_t available = decoder->audio_buffer_offset;
1331 size_t to_copy = (available < num_samples) ? available : num_samples;
1332
1333 memcpy(buffer, decoder->audio_buffer, to_copy * sizeof(float));
1334 samples_written += to_copy;
1335
1336 // Shift buffer
1337 if (to_copy < available) {
1338 memmove(decoder->audio_buffer, decoder->audio_buffer + to_copy, (available - to_copy) * sizeof(float));
1339 }
1340 decoder->audio_buffer_offset -= to_copy;
1341
1342 if (samples_written >= num_samples) {
1343 return samples_written;
1344 }
1345 }
1346
1347 // Read more packets to fill the request
1348 static uint64_t packet_count = 0;
1349
1350 // Lock read_frame_mutex to prevent concurrent av_read_frame() calls from video prefetch thread
1351 mutex_lock(&decoder->read_frame_mutex);
1352
1353 while (samples_written < num_samples) {
1354 int ret = av_read_frame(decoder->format_ctx, decoder->packet);
1355 if (ret < 0) {
1356 if (ret == AVERROR_EOF) {
1357 decoder->eof_reached = true;
1358 }
1359 break;
1360 }
1361
1362 // Check if this is an audio packet
1363 if (decoder->packet->stream_index != decoder->audio_stream_idx) {
1364 av_packet_unref(decoder->packet);
1365 continue;
1366 }
1367
1368 log_info_every(50 * US_PER_MS_INT, "Audio packet #%lu: pts=%ld dts=%ld duration=%d size=%d", packet_count++,
1369 decoder->packet->pts, decoder->packet->dts, decoder->packet->duration, decoder->packet->size);
1370
1371 // Send packet to decoder
1372 ret = avcodec_send_packet(decoder->audio_codec_ctx, decoder->packet);
1373 av_packet_unref(decoder->packet);
1374
1375 if (ret < 0) {
1376 log_warn("Error sending audio packet to decoder");
1377 continue;
1378 }
1379
1380 // Receive all decoded frames from this packet
1381 // Important: a single packet can produce multiple frames. Must drain all before next packet.
1382 while (1) {
1383 ret = avcodec_receive_frame(decoder->audio_codec_ctx, decoder->frame);
1384 if (ret == AVERROR(EAGAIN)) {
1385 break; // No more frames from this packet, get next packet
1386 } else if (ret < 0) {
1387 log_warn("Error receiving audio frame from decoder");
1388 goto audio_read_done;
1389 }
1390
1391 // Update position tracking
1392 decoder->last_audio_pts =
1393 get_frame_pts_seconds(decoder->frame, decoder->format_ctx->streams[decoder->audio_stream_idx]->time_base);
1394
1395 // Resample to target format
1396 float *out_buf = buffer + samples_written;
1397 int out_samples = (int)(num_samples - samples_written);
1398
1399 uint8_t *out_ptr = (uint8_t *)out_buf;
1400 int converted = swr_convert(decoder->swr_ctx, &out_ptr, out_samples, (const uint8_t **)decoder->frame->data,
1401 decoder->frame->nb_samples);
1402
1403 if (converted > 0) {
1404 samples_written += (size_t)converted;
1405 }
1406
1407 if (samples_written >= num_samples) {
1408 goto audio_read_done;
1409 }
1410 }
1411 }
1412
1413audio_read_done:
1414 // Release read_frame_mutex now that av_read_frame() is done
1415 mutex_unlock(&decoder->read_frame_mutex);
1416
1417 // Flush resampler buffer if we haven't filled the full request
1418 // The resampler may have buffered samples that need to be output
1419 if (samples_written < num_samples) {
1420 int remaining_space = (int)(num_samples - samples_written);
1421 uint8_t *out_ptr = (uint8_t *)(buffer + samples_written);
1422 int flushed = swr_convert(decoder->swr_ctx, &out_ptr, remaining_space, NULL, 0);
1423 if (flushed > 0) {
1424 samples_written += (size_t)flushed;
1425 }
1426 }
1427
1428 // Update sample-based position tracking
1429 decoder->audio_samples_read += samples_written;
1430
1431 return samples_written;
1432}

1473asciichat_error_t ffmpeg_decoder_seek_to_timestamp(ffmpeg_decoder_t *decoder, double timestamp_sec) {
1474 if (!decoder) {
1475 return ERROR_INVALID_PARAM;
1476 }
1477
1478 if (decoder->is_stdin) {
1479 return ERROR_NOT_SUPPORTED; // Cannot seek stdin
1480 }
1481
1482 // Hold mutex during entire seek operation to prevent race with prefetch thread
1483 mutex_lock(&decoder->prefetch_mutex);
1484
1485 // Set flag to pause prefetch thread via condition variable
1486 decoder->seeking_in_progress = true;
1487
1488 // Convert seconds to FFmpeg time base units (AV_TIME_BASE = 1,000,000)
1489 int64_t target_ts = (int64_t)(timestamp_sec * AV_TIME_BASE);
1490
1491 // For HTTP streams, use simple keyframe seeking (faster than frame-accurate seeking)
1492 // HTTP seeking is expensive and can break stream state, so prefer speed over precision
1493 int seek_ret = av_seek_frame(decoder->format_ctx, -1, target_ts, AVSEEK_FLAG_BACKWARD);
1494 if (seek_ret < 0) {
1495 // Fallback: try without backward flag
1496 seek_ret = av_seek_frame(decoder->format_ctx, -1, target_ts, 0);
1497 }
1498
1499 if (seek_ret < 0) {
1500 decoder->seeking_in_progress = false;
1501 cond_signal(&decoder->prefetch_cond);
1502 mutex_unlock(&decoder->prefetch_mutex);
1503 return SET_ERRNO(ERROR_MEDIA_SEEK, "Failed to seek to timestamp %.2f seconds", timestamp_sec);
1504 }
1505
1506 // Flush codec buffers AFTER seeking
1507 if (decoder->video_codec_ctx) {
1508 avcodec_flush_buffers(decoder->video_codec_ctx);
1509 }
1510 if (decoder->audio_codec_ctx) {
1511 avcodec_flush_buffers(decoder->audio_codec_ctx);
1512 }
1513
1514 // Reset state
1515 decoder->eof_reached = false;
1516 decoder->audio_buffer_offset = 0;
1517 // Clear any stale audio data in buffer
1518 if (decoder->audio_buffer) {
1519 memset(decoder->audio_buffer, 0, decoder->audio_buffer_size * sizeof(float));
1520 }
1521 decoder->last_video_pts = -1.0;
1522 decoder->last_audio_pts = -1.0;
1523 decoder->prefetch_frame_ready = false;
1524 // Set audio_samples_read to match the seek target so position tracking works correctly
1525 decoder->audio_samples_read = (uint64_t)(timestamp_sec * decoder->audio_sample_rate);
1526
1527 // Reset current_read_buffer and mark both buffers as not in use
1528 // After seeking, the prefetch thread may reallocate buffers, so current_read_buffer
1529 // could point to freed memory. Clear it so the next read doesn't try to release stale pointers.
1530 decoder->current_read_buffer = NULL;
1531 decoder->buffer_a_in_use = false;
1532 decoder->buffer_b_in_use = false;
1533
1534 // Resume prefetch thread
1535 decoder->seeking_in_progress = false;
1536 cond_signal(&decoder->prefetch_cond);
1537
1538 // Release mutex - prefetch thread can resume
1539 mutex_unlock(&decoder->prefetch_mutex);
1540
1541 return ASCIICHAT_OK;
1542}
