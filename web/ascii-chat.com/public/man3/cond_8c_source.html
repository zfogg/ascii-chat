92int cond_format_state(const cond_t *cond, char *buffer, size_t size) {
93 if (!cond || !buffer || size == 0) return 0;
94
95 int offset = 0;
96 uint64_t now_ns = time_get_ns();
97
98 char wait_str[64] = "";
99 char signal_str[64] = "";
100 char broadcast_str[64] = "";
101 char waiting_str[64] = "";
102 char count_str[256] = "";
103
104 if (cond->last_wait_time_ns > 0 && cond->last_wait_time_ns <= now_ns) {
105 char elapsed_str[64];
106 time_pretty(now_ns - cond->last_wait_time_ns, -1, elapsed_str, sizeof(elapsed_str));
107 snprintf(wait_str, sizeof(wait_str), "wait=%s", elapsed_str);
108 }
109
110 if (cond->last_signal_time_ns > 0 && cond->last_signal_time_ns <= now_ns) {
111 char elapsed_str[64];
112 time_pretty(now_ns - cond->last_signal_time_ns, -1, elapsed_str, sizeof(elapsed_str));
113 snprintf(signal_str, sizeof(signal_str), "signal=%s", elapsed_str);
114 }
115
116 if (cond->last_broadcast_time_ns > 0 && cond->last_broadcast_time_ns <= now_ns) {
117 char elapsed_str[64];
118 time_pretty(now_ns - cond->last_broadcast_time_ns, -1, elapsed_str, sizeof(elapsed_str));
119 snprintf(broadcast_str, sizeof(broadcast_str), "broadcast=%s", elapsed_str);
120 }
121
122 uint64_t waiting_count = atomic_load_u64(&cond->waiting_count);
123 if (waiting_count > 0) {
124 snprintf(waiting_str, sizeof(waiting_str), "[WAITING=%lu]", waiting_count);
125 }
126
127 if (cond->wait_count > 0 || cond->signal_count > 0 || cond->broadcast_count > 0) {
128 snprintf(count_str, sizeof(count_str), "[ops: wait=%lu signal=%lu broadcast=%lu]",
129 cond->wait_count, cond->signal_count, cond->broadcast_count);
130 }
131
132 offset += snprintf(buffer + offset, size - offset, "%s %s %s %s %s",
133 wait_str, signal_str, broadcast_str, waiting_str, count_str);
134 return offset;
135}
