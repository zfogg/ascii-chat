19static void zsh_escape_desc(FILE *output, const char *text) {
20 if (!text) {
21 return;
22 }
23
24 for (const char *p = text; *p; p++) {
25 switch (*p) {
26 case '\'':
27 fprintf(output, "'\\''");
28 break;
29 case '\n':
30 case '\t':
31 fprintf(output, " ");
32 break;
33 case ':':
34 // Escape colons which are special in completion descriptions
35 fprintf(output, "\\:");
36 break;
37 default:
38 fputc(*p, output);
39 }
40 }
41}

46static const char **zsh_collect_groups(const option_descriptor_t *opts, size_t count, size_t *out_group_count) {
47 if (!opts || count == 0) {
48 *out_group_count = 0;
49 return NULL;
50 }
51
52 // Allocate array for unique groups (worst case: count groups)
53 const char **groups = SAFE_MALLOC(count * sizeof(const char *), const char **);
54 size_t group_count = 0;
55
56 // Collect unique groups
57 for (size_t i = 0; i < count; i++) {
58 const char *group = opts[i].group;
59 if (!group) continue;
60
61 // Check if we already have this group
62 bool found = false;
63 for (size_t j = 0; j < group_count; j++) {
64 if (strcmp(groups[j], group) == 0) {
65 found = true;
66 break;
67 }
68 }
69 if (!found) {
70 groups[group_count++] = group;
71 }
72 }
73
74 // Sort groups alphabetically for consistent ordering
75 for (size_t i = 0; i < group_count; i++) {
76 for (size_t j = i + 1; j < group_count; j++) {
77 if (strcmp(groups[i], groups[j]) > 0) {
78 const char *tmp = groups[i];
79 groups[i] = groups[j];
80 groups[j] = tmp;
81 }
82 }
83 }
84
85 *out_group_count = group_count;
86 return groups;
87}

92static void zsh_write_options_grouped(FILE *output, const option_descriptor_t *opts, size_t count,
93 const char *func_prefix) {
94 if (!opts || count == 0) return;
95
96 size_t group_count = 0;
97 const char **groups = zsh_collect_groups(opts, count, &group_count);
98
99 // First pass: declare arrays for each group
100 for (size_t g = 0; g < group_count; g++) {
101 const char *group = groups[g];
102 fprintf(output, " local -a %s_%s_opts=(\n", func_prefix, group);
103
104 // Write all options in this group
105 for (size_t i = 0; i < count; i++) {
106 if (!opts[i].group || strcmp(opts[i].group, group) != 0) continue;
107
108 // Short option if present
109 if (opts[i].short_name != '\0') {
110 fprintf(output, " '-%c:", opts[i].short_name);
111 zsh_escape_desc(output, opts[i].help_text);
112 fprintf(output, "'\n");
113 }
114
115 // Long option
116 fprintf(output, " '--%s:", opts[i].long_name);
117 zsh_escape_desc(output, opts[i].help_text);
118 fprintf(output, "'\n");
119 }
120
121 fprintf(output, " )\n");
122 }
123
124 // Second pass: call _describe for each group (with lowercase display)
125 for (size_t g = 0; g < group_count; g++) {
126 const char *group = groups[g];
127 fprintf(output, " _describe -t %s '", group);
128 utf8_write_lowercase(output, group);
129 fprintf(output, " options' %s_%s_opts\n", func_prefix, group);
130 }
131
132 SAFE_FREE(groups);
133}
