52asciichat_error_t ui_status_gather(tcp_server_t *server, const char *session_string, const char *ipv4_address,
53 const char *ipv6_address, uint16_t port, time_t start_time,
54 const char *mode_name, bool session_is_mdns_only, ui_status_t *out_status) {
55 if (!server || !out_status || !mode_name) {
56 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters for ui_status_gather");
57 }
58
59 memset(out_status, 0, sizeof(*out_status));
60
61 // Copy session string
62 if (session_string) {
63 SAFE_STRNCPY(out_status->session_string, session_string, sizeof(out_status->session_string));
64 }
65
66 // Format IPv4 address
67 out_status->ipv4_bound = (ipv4_address && ipv4_address[0] != '\0');
68 if (out_status->ipv4_bound) {
69 snprintf(out_status->ipv4_address, sizeof(out_status->ipv4_address), "%s:%u", ipv4_address, port);
70 }
71
72 // Format IPv6 address
73 out_status->ipv6_bound = (ipv6_address && ipv6_address[0] != '\0');
74 if (out_status->ipv6_bound) {
75 snprintf(out_status->ipv6_address, sizeof(out_status->ipv6_address), "[%s]:%u", ipv6_address, port);
76 }
77
78 out_status->port = port;
79 out_status->start_time = start_time;
80 out_status->mode_name = mode_name;
81 out_status->session_is_mdns_only = session_is_mdns_only;
82 out_status->connected_count = tcp_server_get_client_count(server);
83
84 return ASCIICHAT_OK;
85}

104static void render_ui_status_header(frame_buffer_t *buf, terminal_size_t term_size, void *user_data) {
105 const ui_status_t *status = (const ui_status_t *)user_data;
106 if (!status) {
107 return;
108 }
109
110 // Calculate uptime
111 time_t now = time(NULL);
112 time_t uptime_secs = now - status->start_time;
113 int uptime_hours = uptime_secs / SEC_PER_HOUR;
114 int uptime_mins = (uptime_secs % SEC_PER_HOUR) / SEC_PER_MIN;
115 int uptime_secs_rem = uptime_secs % 60;
116
117 // Line 1: Top border
118 frame_buffer_printf(buf, "\033[1;36m━");
119 for (int i = 1; i < term_size.cols - 1; i++) {
120 frame_buffer_printf(buf, "━");
121 }
122 frame_buffer_printf(buf, "\033[0m\n");
123
124 // Line 2: Title + Stats (centered)
125 char status_line[512];
126 char status_line_truncated[512];
127 char status_line_colored[600];
128 char uptime_str[12];
129 format_uptime_hms(uptime_hours, uptime_mins, uptime_secs_rem, uptime_str, sizeof(uptime_str));
130 snprintf(status_line, sizeof(status_line), "ascii-chat %s | 👥 %zu | ⏱️ %s", status->mode_name,
131 status->connected_count, uptime_str);
132
133 // Truncate if needed
134 truncate_utf8_with_ellipsis(status_line, status_line_truncated, sizeof(status_line_truncated), term_size.cols - 2);
135 snprintf(status_line_colored, sizeof(status_line_colored), "\033[1;36m%s\033[0m", status_line_truncated);
136
137 int padding = display_center_horizontal(status_line_truncated, term_size.cols);
138 for (int i = 0; i < padding; i++) {
139 frame_buffer_printf(buf, " ");
140 }
141 frame_buffer_printf(buf, "%s\n", status_line_colored);
142
143 // Line 3: Session + Addresses (centered)
144 char addr_line[512];
145 char addr_line_truncated[512];
146 int pos = 0;
147 if (status->session_string[0] != '\0') {
148 pos += snprintf(addr_line + pos, sizeof(addr_line) - pos, "🔗 %s", status->session_string);
149 }
150 if (status->ipv4_bound && pos < (int)sizeof(addr_line) - 30) {
151 if (pos > 0) {
152 pos += snprintf(addr_line + pos, sizeof(addr_line) - pos, " | ");
153 }
154 // Extract IP and get type
155 char ipv4_only[64];
156 if (extract_ip_from_address(status->ipv4_address, ipv4_only, sizeof(ipv4_only)) == 0) {
157 const char *type = get_ip_type_string(ipv4_only);
158 pos += snprintf(addr_line + pos, sizeof(addr_line) - pos, "%s (%s)", status->ipv4_address, type);
159 } else {
160 pos += snprintf(addr_line + pos, sizeof(addr_line) - pos, "%s", status->ipv4_address);
161 }
162 }
163 if (status->ipv6_bound && pos < (int)sizeof(addr_line) - 30) {
164 if (pos > 0) {
165 pos += snprintf(addr_line + pos, sizeof(addr_line) - pos, " | ");
166 }
167 // Extract IP and get type
168 char ipv6_only[64];
169 if (extract_ip_from_address(status->ipv6_address, ipv6_only, sizeof(ipv6_only)) == 0) {
170 const char *type = get_ip_type_string(ipv6_only);
171 snprintf(addr_line + pos, sizeof(addr_line) - pos, "%s (%s)", status->ipv6_address, type);
172 } else {
173 snprintf(addr_line + pos, sizeof(addr_line) - pos, "%s", status->ipv6_address);
174 }
175 }
176
177 // Truncate if needed
178 truncate_utf8_with_ellipsis(addr_line, addr_line_truncated, sizeof(addr_line_truncated), term_size.cols - 2);
179
180 int addr_padding = display_center_horizontal(addr_line_truncated, term_size.cols);
181 for (int i = 0; i < addr_padding; i++) {
182 frame_buffer_printf(buf, " ");
183 }
184 frame_buffer_printf(buf, "%s\n", addr_line_truncated);
185
186 // Line 4: Bottom border
187 frame_buffer_printf(buf, "\033[1;36m━");
188 for (int i = 1; i < term_size.cols - 1; i++) {
189 frame_buffer_printf(buf, "━");
190 }
191 frame_buffer_printf(buf, "\033[0m\n");
192}

194void ui_status_display(const ui_status_t *status) {
195 if (!status) {
196 return;
197 }
198
199 // Only render the status screen if:
200 // 1. Terminal is interactive (default behavior), OR
201 // 2. User explicitly set --status-screen on command line (force it regardless of terminal)
202 // In non-interactive mode without explicit flag, logs flow to stdout/stderr normally
203 if (!terminal_is_interactive() && !GET_OPTION(status_screen_explicitly_set)) {
204 return;
205 }
206
207 // Redirect status screen to stderr when stdout is piped (not a TTY)
208 // This keeps stdout clean for frame data while status goes to stderr
209 if (!isatty(STDOUT_FILENO)) {
210 terminal_screen_set_output_fd(STDERR_FILENO);
211 }
212
213 // If --grep pattern was provided, enter interactive grep mode with it pre-populated
214 // Only do this once (check if not already entering)
215 static bool grep_mode_entered = false;
216 if (!grep_mode_entered && grep_get_last_pattern() && grep_get_last_pattern()[0] != '\0') {
217 log_search_enter_mode();
218 grep_mode_entered = true;
219 }
220
221 // Use terminal_screen abstraction for rendering
222 terminal_screen_config_t config = {
223 .fixed_header_lines = 4,
224 .render_header = render_ui_status_header,
225 .user_data = (void *)status,
226 .show_logs = true,
227 };
228
229 terminal_screen_render(&config);
230}

236bool ui_status_display_interactive(const ui_status_t *status) {
237 if (!status) {
238 return true;
239 }
240
241 // Only render the status screen if:
242 // 1. Terminal is interactive (default behavior), OR
243 // 2. User explicitly set --status-screen on command line (force it regardless of terminal)
244 // In non-interactive mode without explicit flag, logs flow to stdout/stderr normally
245 if (!terminal_is_interactive() && !GET_OPTION(status_screen_explicitly_set)) {
246 return true;
247 }
248
249 // If --grep pattern was provided, enter interactive grep mode with it pre-populated
250 static bool grep_mode_entered = false;
251 if (!grep_mode_entered && grep_get_last_pattern() && grep_get_last_pattern()[0] != '\0') {
252 log_search_enter_mode();
253 grep_mode_entered = true;
254 }
255
256 // Initialize keyboard for interactive grep
257 bool keyboard_enabled = false;
258 if (keyboard_init() == ASCIICHAT_OK) {
259 keyboard_enabled = true;
260 }
261
262 // Use terminal_screen abstraction for rendering
263 terminal_screen_config_t config = {
264 .fixed_header_lines = 4,
265 .render_header = render_ui_status_header,
266 .user_data = (void *)status,
267 .show_logs = true,
268 };
269
270 terminal_screen_render(&config);
271
272 // Poll keyboard for Escape to exit or for interactive grep
273 bool should_exit_status = false;
274 if (keyboard_enabled) {
275 keyboard_key_t key = keyboard_read_nonblocking();
276 if (key == KEY_ESCAPE) {
277 // Escape key: cancel grep if active, otherwise exit status screen
278 if (log_search_is_active()) {
279 log_search_exit_mode(false); // Cancel grep without applying
280 } else {
281 should_exit_status = true; // Exit status screen
282 }
283 } else if (key != KEY_NONE && log_search_should_handle(key)) {
284 log_search_handle_key(key);
285 }
286 }
287
288 // Cleanup keyboard
289 if (keyboard_enabled) {
290 keyboard_destroy();
291 }
292
293 return !should_exit_status; // Return false if user wants to exit
294}

296void ui_status_update(tcp_server_t *server, const char *session_string, const char *ipv4_address,
297 const char *ipv6_address, uint16_t port, time_t start_time, const char *mode_name,
298 bool session_is_mdns_only, uint64_t *last_update_ns) {
299 if (!server || !last_update_ns) {
300 return;
301 }
302
303 // Update at FPS rate (60 Hz = 16.67ms by default)
304 uint32_t fps = GET_OPTION(fps);
305 if (fps == 0) {
306 fps = 60; // Default
307 }
308
309 // Calculate frame interval in microseconds
310 uint64_t frame_interval_us = US_PER_SEC_INT / fps;
311
312 // Get current time in microseconds using platform abstraction
313 uint64_t now_us = platform_get_monotonic_time_us();
314
315 // Check if enough time has passed
316 if ((now_us - *last_update_ns) < frame_interval_us) {
317 return; // Too soon, skip frame
318 }
319
320 ui_status_t status;
321 if (ui_status_gather(server, session_string, ipv4_address, ipv6_address, port, start_time, mode_name,
322 session_is_mdns_only, &status) == ASCIICHAT_OK) {
323 ui_status_display(&status);
324 *last_update_ns = now_us;
325 }
326}
