131session_display_ctx_t *session_display_create(const session_display_config_t *config) {
132 // Auto-create config from command-line options if NULL
133 session_display_config_t auto_config = {0};
134 if (!config) {
135 auto_config.snapshot_mode = GET_OPTION(snapshot_mode);
136 auto_config.palette_type = GET_OPTION(palette_type);
137 auto_config.custom_palette = GET_OPTION(palette_custom_set) ? GET_OPTION(palette_custom) : NULL;
138 auto_config.color_mode = TERM_COLOR_AUTO;
139 config = &auto_config;
140 }
141
142 // Check if we should exit before starting initialization
143 if (config->should_exit_callback && config->should_exit_callback(config->callback_data)) {
144 return NULL;
145 }
146
147 // Allocate context
148 session_display_ctx_t *ctx = SAFE_CALLOC(1, sizeof(session_display_ctx_t), session_display_ctx_t *);
149
150 // Store configuration
151 ctx->snapshot_mode = config->snapshot_mode;
152 ctx->palette_type = config->palette_type;
153 ctx->audio_playback_enabled = config->enable_audio_playback;
154 ctx->audio_ctx = config->audio_ctx;
155 ctx->render_fps = config->render_fps;
156 atomic_store_bool(&ctx->first_frame, true);
157 atomic_store_bool(&ctx->keyboard_help_active, false);
158
159 // Initialize FPS counter
160 ctx->fps_counter = fps_counter_create();
161
162 // Get TTY info for direct terminal access
163 ctx->tty_info = get_current_tty();
164
165 // Determine if we have a valid TTY
166 // Check if stdout is a TTY, not just any fd (stdin/stderr could be TTY while stdout is piped).
167 // If stdout is piped/redirected, never perform terminal operations regardless of other fds.
168 if (ctx->tty_info.fd >= 0) {
169 ctx->has_tty = (platform_isatty(ctx->tty_info.fd) != 0) && terminal_is_stdout_tty();
170 }
171
172 // In piped mode, force all logs to stderr to prevent frame data corruption
173 if (terminal_should_force_stderr()) {
174 log_set_force_stderr(true);
175 // Disable buffering for stdout to ensure frames appear immediately
176 // Each frame ends with newline, so unbuffered output is critical for real-time display
177 (void)setvbuf(stdout, NULL, _IONBF, 0);
178 }
179
180 // Also disable buffering for TTY mode to ensure smooth 40+ FPS animation
181 setvbuf(stdout, NULL, _IONBF, 0);
182
183 // Detect terminal capabilities
184 ctx->caps = detect_terminal_capabilities();
185
186 // Set wants_padding based on terminal output mode
187 // Enable padding for all rendering modes (including snapshot) to center output
188 // The padding adds newlines at the top, which is useful for visual centering
189 bool is_snapshot_mode = config->snapshot_mode;
190 ctx->caps.wants_padding = true;
191
192 // Calculate pad_height: when padding is enabled, add top padding for centering
193 // Halfblock mode uses 2 source rows per output row, so padding is halved
194 if (ctx->caps.wants_padding) {
195 ctx->caps.pad_height = 5; // Halfblock needs half the padding (10 / 2 = 5)
196 } else {
197 ctx->caps.pad_height = 0;
198 }
199
200 log_debug("Padding mode: wants_padding=%d (snapshot=%d, has_tty=%d, stdin_tty=%d, stdout_tty=%d), pad_height=%zu",
201 ctx->caps.wants_padding, is_snapshot_mode, ctx->has_tty, terminal_is_stdin_tty(), terminal_is_stdout_tty(),
202 ctx->caps.pad_height);
203
204 // Apply color mode override if specified
205 if (config->color_mode != TERM_COLOR_AUTO) {
206 ctx->caps.color_level = config->color_mode;
207 }
208
209 // Apply command-line overrides
210 ctx->caps = apply_color_mode_override(ctx->caps);
211
212 // Initialize palette
213 int palette_result = initialize_client_palette(config->palette_type, config->custom_palette, ctx->palette_chars,
214 &ctx->palette_len, ctx->luminance_palette);
215 if (palette_result != ASCIICHAT_OK) {
216 log_warn("Failed to initialize palette, using default");
217 // Fall back to standard palette
218 (void)initialize_client_palette(PALETTE_STANDARD, NULL, ctx->palette_chars, &ctx->palette_len,
219 ctx->luminance_palette);
220 }
221
222 // Pre-warm UTF-8 palette cache during initialization (not during rendering)
223 // This is expensive on first use (creates lookup tables), so warm it up now to avoid frame lag
224 // This function is imported from lib/video/simd/common.h
225 (void)get_utf8_palette_cache((const char *)ctx->palette_chars);
226 log_debug("UTF-8 palette cache pre-warmed during display initialization");
227
228 // Initialize ANSI fast lookup tables based on terminal capabilities
229 if (ctx->caps.color_level == TERM_COLOR_TRUECOLOR) {
230 ansi_fast_init();
231 } else if (ctx->caps.color_level == TERM_COLOR_256) {
232 ansi_fast_init_256color();
233 } else if (ctx->caps.color_level == TERM_COLOR_16) {
234 ansi_fast_init_16color();
235 }
236 // TERM_COLOR_MONO requires no init
237
238 // Initialize ASCII output system if we have a TTY
239 if (ctx->has_tty && ctx->tty_info.fd >= 0) {
240 ascii_write_init(ctx->tty_info.fd, false);
241 }
242
243 // Initialize digital rain effect if enabled
244 if (GET_OPTION(matrix_rain)) {
245 // Get terminal dimensions for grid size
246 unsigned short int width_us = terminal_get_effective_width();
247 unsigned short int height_us = terminal_get_effective_height();
248
249 int width = (int)width_us;
250 int height = (int)height_us;
251
252 ctx->digital_rain = digital_rain_init(width, height);
253 if (ctx->digital_rain) {
254 // Set color from color filter if active
255 color_filter_t filter = GET_OPTION(color_filter);
256 digital_rain_set_color_from_filter(ctx->digital_rain, filter);
257 log_info("Digital rain effect enabled: %dx%d grid", width, height);
258 } else {
259 log_warn("Failed to initialize digital rain effect");
260 }
261 ctx->last_frame_time_ns = time_get_ns();
262 }
263
264#ifndef _WIN32
265 // Initialize render-file if enabled (FFmpeg encodes to stdout when "-" is specified)
266 const char *render_file_opt = GET_OPTION(render_file);
267 log_info("DISPLAY_CREATE: render-file opt='%s' (len=%zu), will initialize=%s",
268 render_file_opt ? render_file_opt : "(null)",
269 render_file_opt ? strlen(render_file_opt) : 0,
270 (render_file_opt && strlen(render_file_opt) > 0) ? "YES" : "NO");
271 if (render_file_opt && strlen(render_file_opt) > 0) {
272 int width = (int)GET_OPTION(width);
273 int height = (int)GET_OPTION(height);
274 log_info("render-file: Creating encoder with cols=%d, rows=%d", width, height);
275 // Determine encoder FPS: in snapshot mode, use actual capture rate for correct duration;
276 // in normal mode, use configured fps option
277 uint32_t encoder_fps = 0;
278 if (GET_OPTION(snapshot_mode) && config->render_fps > 0) {
279 // Snapshot mode with known capture rate - use capture rate for correct duration
280 encoder_fps = config->render_fps;
281 log_debug("render-file: SNAPSHOT MODE - using capture rate FPS=%u (snapshot_delay=%.2f sec)", encoder_fps,
282 GET_OPTION(snapshot_delay));
283 } else if (config->render_fps > 0) {
284 // Normal mode with known capture rate - use it
285 encoder_fps = config->render_fps;
286 log_debug("render-file: Using capture rate FPS=%u for encoding", encoder_fps);
287 } else {
288 // No capture rate available - use option default
289 encoder_fps = (uint32_t)GET_OPTION(fps);
290 if (encoder_fps == 0)
291 encoder_fps = 60;
292 log_debug("render-file: Using option FPS=%u (no capture rate available)", encoder_fps);
293 }
294 log_info("render-file: Final encoder FPS=%u (config_render_fps=%u, option_fps=%u)", encoder_fps, config->render_fps,
295 (uint32_t)GET_OPTION(fps));
296 log_info("render-file: Calling render_file_create with path='%s', %dx%d, fps=%d",
297 render_file_opt, width, height, encoder_fps);
298 asciichat_error_t rf_err = render_file_create(render_file_opt, width, height, (int)encoder_fps,
299 GET_OPTION(render_theme), &ctx->render_file);
300 if (rf_err != ASCIICHAT_OK) {
301 log_error("render-file: FAILED to create with error %d (ctx->render_file=%p)",
302 rf_err, (void *)ctx->render_file);
303 } else {
304 log_info("render-file: SUCCESS - encoder initialized, ctx->render_file=%p", (void *)ctx->render_file);
305 // Set audio sources if render_file_set_audio_source is available
306 if (ctx->render_file && config->render_file_audio_source) {
307 render_file_set_audio_source((render_file_ctx_t *)ctx->render_file, config->render_file_audio_source, NULL);
308 log_debug("render-file: audio source set (media_source=%p)", config->render_file_audio_source);
309 }
310 if (ctx->render_file && config->render_file_audio_capture_rb) {
311 render_file_set_audio_source((render_file_ctx_t *)ctx->render_file, NULL, config->render_file_audio_capture_rb);
312 log_debug("render-file: audio capture ring buffer set");
313 }
314 }
315 } else {
316 log_info("render-file: NOT initializing (render_file_opt is %s)", render_file_opt ? "empty" : "NULL");
317 }
318#endif
319
320 ctx->initialized = true;
321 return ctx;
322}

324void session_display_destroy(session_display_ctx_t *ctx) {
325 if (!ctx) {
326 SET_ERRNO(ERROR_INVALID_PARAM, "Session display context is NULL");
327 return;
328 }
329
330 // Cleanup ASCII rendering if we had a TTY
331 // Don't reset terminal in snapshot mode to preserve the rendered output
332 if (ctx->has_tty && ctx->tty_info.fd >= 0) {
333 ascii_write_destroy(ctx->tty_info.fd, !ctx->snapshot_mode);
334 }
335
336 // Close the controlling terminal if we opened it
337 if (ctx->tty_info.owns_fd && ctx->tty_info.fd >= 0) {
338 (void)platform_close(ctx->tty_info.fd);
339 ctx->tty_info.fd = -1;
340 ctx->tty_info.owns_fd = false;
341 }
342
343 // Cleanup digital rain effect
344 if (ctx->digital_rain) {
345 digital_rain_destroy(ctx->digital_rain);
346 ctx->digital_rain = NULL;
347 }
348
349 // Cleanup FPS counter
350 if (ctx->fps_counter) {
351 fps_counter_destroy(ctx->fps_counter);
352 ctx->fps_counter = NULL;
353 }
354
355#ifndef _WIN32
356 // Cleanup render-file if active
357 if (ctx->render_file) {
358 render_file_destroy(ctx->render_file);
359 ctx->render_file = NULL;
360 }
361#endif
362
363 ctx->initialized = false;
364
365 // Clear the cached display context in splash state to prevent use-after-free
366 // when worker threads try to access the display after it's been freed
367 splash_clear_display_context();
368
369 SAFE_FREE(ctx);
370}

493char *session_display_convert_to_ascii(session_display_ctx_t *ctx, const image_t *image) {
494 if (!ctx) {
495 SET_ERRNO(ERROR_INVALID_PARAM, "session_display_convert_to_ascii: ctx is NULL");
496 return NULL;
497 }
498
499 if (!ctx->initialized) {
500 SET_ERRNO(ERROR_INVALID_STATE, "session_display_convert_to_ascii: ctx not initialized");
501 return NULL;
502 }
503
504 if (!image) {
505 SET_ERRNO(ERROR_INVALID_PARAM, "session_display_convert_to_ascii: image is NULL");
506 return NULL;
507 }
508
509 // Get conversion parameters from command-line options
510 unsigned short int width = GET_OPTION(width);
511 unsigned short int height = GET_OPTION(height);
512 bool stretch = GET_OPTION(stretch);
513 bool preserve_aspect_ratio = !stretch;
514
515 // Determine if we should apply flip_x and flip_y
516 bool flip_x_enabled = GET_OPTION(flip_x);
517 bool flip_y_enabled = GET_OPTION(flip_y);
518
519 color_filter_t color_filter = GET_OPTION(color_filter);
520
521 // Handle dynamic matrix rain effect toggle
522 bool matrix_rain_enabled = GET_OPTION(matrix_rain);
523 if (matrix_rain_enabled && !ctx->digital_rain) {
524 // Initialize digital rain if it's now enabled but wasn't before
525 unsigned short int width_us = terminal_get_effective_width();
526 unsigned short int height_us = terminal_get_effective_height();
527 int width_for_rain = (int)width_us;
528 int height_for_rain = (int)height_us;
529 ctx->digital_rain = digital_rain_init(width_for_rain, height_for_rain);
530 if (ctx->digital_rain) {
531 digital_rain_set_color_from_filter(ctx->digital_rain, color_filter);
532 log_info("Matrix rain effect: enabled");
533 }
534 } else if (!matrix_rain_enabled && ctx->digital_rain) {
535 // Disable digital rain if it's now disabled but was enabled before
536 digital_rain_destroy(ctx->digital_rain);
537 ctx->digital_rain = NULL;
538 log_info("Matrix rain effect: disabled");
539 }
540
541 // Make a mutable copy of terminal capabilities for ascii_convert_with_capabilities
542 terminal_capabilities_t caps_copy = ctx->caps;
543
544 // Re-evaluate render_mode on every frame to pick up live changes
545 caps_copy.render_mode = (render_mode_t)GET_OPTION(render_mode);
546
547 // MEASURE EVERY OPERATION - Debug systematic timing
548 uint64_t t_flip_start = time_get_ns();
549
550 // Apply horizontal and/or vertical flips if requested
551 image_t *flipped_image = NULL;
552 const image_t *display_image = image;
553
554 if ((flip_x_enabled || flip_y_enabled) && image->w > 1 && image->h > 1 && image->pixels) {
555 START_TIMER("image_flip");
556 uint64_t t_flip_alloc_start = time_get_ns();
557 flipped_image = image_new((size_t)image->w, (size_t)image->h);
558 uint64_t t_flip_alloc_end = time_get_ns();
559
560 if (flipped_image) {
561 uint64_t t_flip_memcpy_start = time_get_ns();
562 // OPTIMIZATION: Copy entire image first (sequential memory access - cache-friendly)
563 memcpy(flipped_image->pixels, image->pixels, (size_t)image->w * (size_t)image->h * sizeof(rgb_pixel_t));
564 uint64_t t_flip_memcpy_end = time_get_ns();
565
566 uint64_t t_flip_reverse_start = time_get_ns();
567
568 // Apply horizontal flip (X-axis)
569 if (flip_x_enabled) {
570#if SIMD_SUPPORT_NEON
571 // Use NEON-accelerated flip on ARM processors
572 image_flip_horizontal_neon(flipped_image);
573#else
574 // Scalar fallback for non-NEON systems
575 for (int y = 0; y < image->h; y++) {
576 rgb_pixel_t *row = &flipped_image->pixels[y * image->w];
577 for (int x = 0; x < image->w / 2; x++) {
578 rgb_pixel_t temp = row[x];
579 row[x] = row[image->w - 1 - x];
580 row[image->w - 1 - x] = temp;
581 }
582 }
583#endif
584 }
585
586 // Apply vertical flip (Y-axis)
587 if (flip_y_enabled) {
588 for (int y = 0; y < image->h / 2; y++) {
589 rgb_pixel_t *top_row = &flipped_image->pixels[y * image->w];
590 rgb_pixel_t *bottom_row = &flipped_image->pixels[(image->h - 1 - y) * image->w];
591 for (int x = 0; x < image->w; x++) {
592 rgb_pixel_t temp = top_row[x];
593 top_row[x] = bottom_row[x];
594 bottom_row[x] = temp;
595 }
596 }
597 }
598
599 uint64_t t_flip_reverse_end = time_get_ns();
600 display_image = flipped_image;
601
602 log_dev("TIMING_FLIP: alloc=%llu us, memcpy=%llu us, flip=%llu us (x=%d, y=%d)",
603 (t_flip_alloc_end - t_flip_alloc_start) / 1000, (t_flip_memcpy_end - t_flip_memcpy_start) / 1000,
604 (t_flip_reverse_end - t_flip_reverse_start) / 1000, flip_x_enabled, flip_y_enabled);
605 }
606 STOP_TIMER_AND_LOG_EVERY(dev, 3 * NS_PER_SEC_INT, 3 * NS_PER_MS_INT, "image_flip",
607 "IMAGE_FLIP: Flip complete (%.2f ms)");
608 }
609 uint64_t t_flip_end = time_get_ns();
610
611 uint64_t t_filter_start = time_get_ns();
612
613 // Apply color filter to RGB pixels before ASCII conversion
614 // Rainbow filter is handled separately via ANSI color replacement
615 image_t *filtered_image = NULL;
616 if (color_filter != COLOR_FILTER_NONE && color_filter != COLOR_FILTER_RAINBOW) {
617 // Create a copy to avoid modifying the original
618 filtered_image = image_new((size_t)display_image->w, (size_t)display_image->h);
619 if (filtered_image) {
620 memcpy(filtered_image->pixels, display_image->pixels,
621 (size_t)display_image->w * (size_t)display_image->h * sizeof(rgb_pixel_t));
622
623 // Apply filter to the copy
624 int stride = (int)display_image->w * 3;
625 float time_seconds = (float)t_filter_start / (float)NS_PER_SEC_INT;
626 apply_color_filter((uint8_t *)filtered_image->pixels, (uint32_t)display_image->w, (uint32_t)display_image->h,
627 (uint32_t)stride, color_filter, time_seconds);
628 }
629 }
630
631 const image_t *ascii_input_image = filtered_image ? filtered_image : display_image;
632 uint64_t t_filter_end = time_get_ns();
633
634 uint64_t t_convert_start = time_get_ns();
635 // Call the standard ASCII conversion using the filtered (or original) image
636 START_TIMER("ascii_convert_with_capabilities");
637 char *result = ascii_convert_with_capabilities(ascii_input_image, width, height, &caps_copy, preserve_aspect_ratio,
638 stretch, ctx->palette_chars);
639 STOP_TIMER_AND_LOG_EVERY(dev, 3 * NS_PER_SEC_INT, 5 * NS_PER_MS_INT, "ascii_convert_with_capabilities",
640 "ASCII_CONVERT: Conversion complete (%.2f ms)");
641 uint64_t t_convert_end = time_get_ns();
642
643 // Apply rainbow color filter to ANSI output by replacing RGB values
644 // This preserves character selection while applying the filter colors
645 if (result && color_filter == COLOR_FILTER_RAINBOW) {
646 uint64_t t_color_replace_start = time_get_ns();
647 float time_seconds = (float)t_filter_start / (float)NS_PER_SEC_INT;
648
649 char *rainbow_result = rainbow_replace_ansi_colors(result, time_seconds);

726void session_display_write_ascii(session_display_ctx_t *ctx, const char *ascii) {
727 if (!ctx || !ctx->initialized) {
728 SET_ERRNO(ERROR_INVALID_PARAM, "Display context is NULL or uninitialized");
729 return;
730 }
731
732 if (!ascii) {
733 SET_ERRNO(ERROR_INVALID_PARAM, "ASCII data is NULL");
734 return;
735 }
736
737 // Re-evaluate color_level on every frame to pick up live color_mode changes
738 terminal_color_mode_t color_mode = (terminal_color_mode_t)GET_OPTION(color_mode);
739 if (color_mode != TERM_COLOR_AUTO) {
740 ctx->caps.color_level = color_mode;
741 }
742
743 // Suppress frame rendering when help screen is active
744 if (atomic_load_bool(&ctx->keyboard_help_active)) {
745 return;
746 }
747
748 // Calculate frame length
749 size_t frame_len = strnlen(ascii, 1024 * 1024); // Max 1MB frame
750 if (frame_len == 0) {
751 SET_ERRNO(ERROR_INVALID_PARAM, "ASCII data is empty");
752 return;
753 }
754
755 // Apply digital rain effect if enabled
756 char *display_frame = (char *)ascii;
757 char *rain_result = NULL;
758 if (ctx->digital_rain) {
759 uint64_t t_rain_start = time_get_ns();
760 uint64_t current_time_ns = t_rain_start;
761 float delta_time = (float)(current_time_ns - ctx->last_frame_time_ns) / (float)NS_PER_SEC_INT;
762 ctx->last_frame_time_ns = current_time_ns;
763
764 // Update digital rain color from current filter
765 color_filter_t current_filter = GET_OPTION(color_filter);
766 digital_rain_set_color_from_filter(ctx->digital_rain, current_filter);
767
768 rain_result = digital_rain_apply(ctx->digital_rain, (char *)ascii, delta_time);
769 if (rain_result) {
770 display_frame = rain_result;
771 frame_len = strnlen(rain_result, 1024 * 1024);
772
773 uint64_t t_rain_end = time_get_ns();
774 char rain_str[32];
775 time_pretty(t_rain_end - t_rain_start, -1, rain_str, sizeof(rain_str));
776 log_info("DIGITAL_RAIN (write_ascii): Effect applied (%s)", rain_str);
777 }
778 }
779
780 // Handle first frame - perform initial terminal reset and splash cleanup
781 bool was_first_frame = atomic_exchange_bool(&ctx->first_frame, false);
782 if (was_first_frame) {
783 // Stop splash screen when first frame is ready
784 splash_intro_done();
785
786 // Perform initial terminal reset
787 if (ctx->has_tty && !ctx->render_file) {
788 (void)terminal_reset(STDOUT_FILENO);
789 (void)terminal_clear_screen();
790 (void)terminal_cursor_home(STDOUT_FILENO);
791 (void)terminal_clear_scrollback(STDOUT_FILENO);
792 (void)terminal_cursor_show();
793 if (!ctx->snapshot_mode) {
794 (void)terminal_cursor_hide();
795 }
796 (void)terminal_flush(STDOUT_FILENO);
797 }
798 }
799
800 // Output routing logic
801 bool use_tty_control = ctx->has_tty && (!ctx->snapshot_mode || terminal_is_interactive());
802
803 START_TIMER("frame_write");
804
805 if (use_tty_control) {
806 // TTY mode: Buffer cursor control + frame data together for atomic frame display
807 const char *cursor_home_sequence = "\033[H\033[3J"; // 7 bytes total
808 size_t cursor_seq_len = 7;
809 size_t total_size = cursor_seq_len + frame_len;
810
811 char *frame_buffer = SAFE_MALLOC(total_size, char *);
812 if (frame_buffer) {
813 memcpy(frame_buffer, cursor_home_sequence, cursor_seq_len);
814 memcpy(frame_buffer + cursor_seq_len, display_frame, frame_len);
815
816 log_debug("FRAME_WRITE_TTY: Writing %zu bytes (cursor=%zu + frame=%zu) to stdout", total_size, cursor_seq_len,
817 frame_len);
818 ssize_t written = platform_write_all(STDOUT_FILENO, frame_buffer, total_size);
819 log_debug("FRAME_WRITE_TTY: Wrote %zd bytes (requested %zu)", written, total_size);
820
821 // Start snapshot timer on first ASCII frame rendered
822 if (GET_OPTION(snapshot_mode) && !g_snapshot_first_frame_rendered) {
823 g_snapshot_first_frame_rendered = true;
824 g_snapshot_first_frame_rendered_ns = time_get_ns();
825 log_info("SNAPSHOT: FIRST ASCII FRAME RENDERED (write_ascii) - Timer started");
826 }
827
828 // Tick FPS counter
829 if (ctx->fps_counter) {
830 fps_counter_tick(ctx->fps_counter);
831 }
832
833 SAFE_FREE(frame_buffer);
834 }
835
836 // Flush buffers
837 log_debug("FRAME_FLUSH: Flushing stdout");
838 (void)fflush(stdout);
839 int flush_fd = (ctx->tty_info.fd >= 0) ? ctx->tty_info.fd : STDOUT_FILENO;
840 (void)terminal_flush(flush_fd);
841
842 // Render FPS counter overlay if enabled
843 if (ctx->fps_counter && GET_OPTION(fps_counter)) {
844 session_display_render_fps_overlay(ctx);
845 }
846 } else if (terminal_is_interactive()) {
847 // Piped to interactive terminal: output ASCII frames with newline
848 char *write_buf = SAFE_MALLOC(frame_len + 1, char *);
849 if (write_buf) {
850 memcpy(write_buf, display_frame, frame_len);
851 write_buf[frame_len] = '\n';
852 (void)platform_write_all(STDOUT_FILENO, write_buf, frame_len + 1);
853 SAFE_FREE(write_buf);
854 }
855
856 // Flush buffers
857 (void)fflush(stdout);
858 int flush_fd = (ctx->tty_info.fd >= 0) ? ctx->tty_info.fd : STDOUT_FILENO;
859 (void)terminal_flush(flush_fd);
860 } else {
861 // Non-interactive piped output
862 // BUT: Don't write ASCII frames to stdout if render_file is using stdout (--render-file="-")
863 if (!ctx->render_file) {
864 // No render-file, safe to write ASCII frames to stdout
865 char *write_buf = SAFE_MALLOC(frame_len + 1, char *);
866 if (write_buf) {
867 memcpy(write_buf, display_frame, frame_len);
868 write_buf[frame_len] = '\n';
869 (void)platform_write_all(STDOUT_FILENO, write_buf, frame_len + 1);
870
871 // Start snapshot timer on first ASCII frame rendered
872 if (GET_OPTION(snapshot_mode) && !g_snapshot_first_frame_rendered) {
873 g_snapshot_first_frame_rendered = true;
874 g_snapshot_first_frame_rendered_ns = time_get_ns();
875 log_info("SNAPSHOT: FIRST ASCII FRAME RENDERED (write_ascii piped) - Timer started");
876 }
877
878 SAFE_FREE(write_buf);
879 }
880
881 // Flush buffers (skip terminal_flush on pipes to avoid blocking in snapshot mode)
882 (void)fflush(stdout);
883 if (!GET_OPTION(snapshot_mode) || ctx->has_tty) {
884 int flush_fd = (ctx->tty_info.fd >= 0) ? ctx->tty_info.fd : STDOUT_FILENO;
885 (void)terminal_flush(flush_fd);
886 }
887 } else {
888 // render_file is using stdout: skip ASCII output to avoid mixing with video
889 if (GET_OPTION(snapshot_mode) && !g_snapshot_first_frame_rendered) {
890 g_snapshot_first_frame_rendered = true;
891 g_snapshot_first_frame_rendered_ns = time_get_ns();
892 log_info("SNAPSHOT: FIRST FRAME RENDERED (render_file piped) - Timer started");
893 }
894 }
895 }
896
897 STOP_TIMER_AND_LOG_EVERY(dev, 3 * NS_PER_SEC_INT, 5 * NS_PER_MS_INT, "frame_write",
898 "FRAME_WRITE: Write and flush complete (%.2f ms)");
899
900 // Clean up digital rain result if allocated
901 if (rain_result) {
902 SAFE_FREE(rain_result);
903 }

1056asciichat_error_t session_display_write_audio(session_display_ctx_t *ctx, const float *buffer, size_t num_samples) {
1057 if (!ctx || !ctx->initialized || !buffer || num_samples == 0) {
1058 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters: ctx=%p, buffer=%p, num_samples=%zu", ctx, buffer,
1059 num_samples);
1060 }
1061
1062 if (!ctx->audio_playback_enabled || !ctx->audio_ctx) {
1063 // Audio not enabled, but not an error - just skip silently
1064 return ASCIICHAT_OK;
1065 }
1066
1067 // For mirror mode with local files: write samples directly without jitter buffering
1068 // The jitter buffer is designed for network scenarios with irregular packet arrivals
1069 // For local playback, we just want raw samples flowing to the speakers
1070 audio_context_t *audio_ctx = (audio_context_t *)ctx->audio_ctx;
1071 if (audio_ctx && audio_ctx->playback_buffer) {
1072 audio_ring_buffer_t *rb = audio_ctx->playback_buffer;
1073
1074 // Simple direct write: append samples to ring buffer without network-oriented complexity
1075 uint32_t write_idx = (uint32_t)atomic_load_u64(&rb->write_index);
1076 uint32_t read_idx = (uint32_t)atomic_load_u64(&rb->read_index);
1077
1078 // Calculate available space in ring buffer
1079 uint32_t available = (read_idx - write_idx - 1) & (AUDIO_RING_BUFFER_SIZE - 1);
1080
1081 if ((uint32_t)num_samples > available) {
1082 // Buffer full - just skip this write to avoid distortion from overwriting old audio
1083 return ASCIICHAT_OK;
1084 }
1085
1086 // Direct memcpy to ring buffer, handling wrap-around
1087 uint32_t space_before_wrap = AUDIO_RING_BUFFER_SIZE - write_idx;
1088 if ((uint32_t)num_samples <= space_before_wrap) {
1089 memcpy(&rb->data[write_idx], buffer, num_samples * sizeof(float));
1090 } else {
1091 // Split write: first part to end of buffer, second part wraps to beginning
1092 uint32_t first_part = space_before_wrap;
1093 uint32_t second_part = num_samples - first_part;
1094 memcpy(&rb->data[write_idx], buffer, first_part * sizeof(float));
1095 memcpy(&rb->data[0], &buffer[first_part], second_part * sizeof(float));
1096 }
1097
1098 // Update write index atomically
1099 atomic_store_u64(&rb->write_index, (write_idx + num_samples) % AUDIO_RING_BUFFER_SIZE);
1100
1101 return ASCIICHAT_OK;
1102 }
1103
1104 return ASCIICHAT_OK;
1105}
