107asciichat_error_t h265_decode(
108 h265_decoder_t *decoder,
109 const uint8_t *encoded_packet,
110 size_t packet_size,
111 uint16_t *output_width,
112 uint16_t *output_height,
113 uint8_t *output_buf,
114 size_t *output_size
115) {
116 if (!decoder || !encoded_packet || !output_width || !output_height || !output_buf || !output_size) {
117 return SET_ERRNO(ERROR_INTERNAL, "Invalid decoder arguments");
118 }
119
120 if (packet_size < 5) {
121 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Packet too small (minimum 5 bytes)");
122 }
123
124 // Parse header
125 uint8_t flags = encoded_packet[0];
126 uint16_t width = ((uint16_t)encoded_packet[1] << 8) | encoded_packet[2];
127 uint16_t height = ((uint16_t)encoded_packet[3] << 8) | encoded_packet[4];
128
129 if (width == 0 || height == 0) {
130 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid frame dimensions: %ux%u", width, height);
131 }
132
133 size_t required_output = width * height;
134 if (*output_size < required_output) {
135 *output_size = required_output;
136 return SET_ERRNO(ERROR_NETWORK_SIZE, "Output buffer too small: need %zu, have %zu",
137 required_output, *output_size);
138 }
139
140 // Decode HEVC data
141 if (packet_size > 5) {
142 const uint8_t *hevc_data = encoded_packet + 5;
143 size_t hevc_size = packet_size - 5;
144
145 // Send packet to decoder
146 av_packet_from_data(decoder->packet, (uint8_t *)hevc_data, hevc_size);
147
148 if (avcodec_send_packet(decoder->codec_ctx, decoder->packet) < 0) {
149 return SET_ERRNO(ERROR_MEDIA_DECODE, "Failed to send packet to decoder");
150 }
151
152 if (avcodec_receive_frame(decoder->codec_ctx, decoder->frame) < 0) {
153 return SET_ERRNO(ERROR_MEDIA_DECODE, "Failed to receive decoded frame");
154 }
155 } else {
156 // Empty packet - just signal a frame boundary
157 if (avcodec_send_packet(decoder->codec_ctx, NULL) < 0) {
158 return SET_ERRNO(ERROR_MEDIA_DECODE, "Failed to flush decoder");
159 }
160 }
161
162 // Extract Y plane (grayscale intensity)
163 uint8_t *y_plane = decoder->frame->data[0];
164 int y_stride = decoder->frame->linesize[0];
165 int img_width = decoder->frame->width;
166 int img_height = decoder->frame->height;
167
168 if (img_width != (int)width || img_height != (int)height) {
169 log_warn("Image dimensions mismatch: expected %ux%u, got %dx%d", width, height, img_width, img_height);
170 }
171
172 // Copy Y plane to output (ASCII intensity values)
173 for (int y = 0; y < img_height && y < (int)height; y++) {
174 memcpy(output_buf + y * width, y_plane + y * y_stride, width);
175 }
176
177 *output_width = width;
178 *output_height = height;
179 *output_size = width * height;
180
181 decoder->last_width = width;
182 decoder->last_height = height;
183 decoder->total_frames++;
184
185 if (flags & H265_DECODER_FLAG_KEYFRAME) {
186 decoder->keyframes++;
187 }
188
189 return ASCIICHAT_OK;
190}
