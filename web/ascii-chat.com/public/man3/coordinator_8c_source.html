44 consensus_election_func_t election_func, void *election_context,
45 consensus_coordinator_t **out_coordinator) {
46 if (!my_id || !topology || !election_func || !out_coordinator) {
47 return SET_ERRNO(
48 ERROR_INVALID_PARAM, "Invalid parameter: my_id=%p, topology=%p, election_func=%p, out_coordinator=%p",
49 (const void *)my_id, (const void *)topology, (const void *)election_func, (const void *)out_coordinator);
50 }
51
52 consensus_coordinator_t *coordinator = SAFE_MALLOC(sizeof(consensus_coordinator_t), consensus_coordinator_t *);
53 if (!coordinator) {
54 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate coordinator");
55 }
56
57 memset(coordinator, 0, sizeof(*coordinator));
58 memcpy(coordinator->my_id, my_id, 16);
59 coordinator->topology = topology;
60 coordinator->election_func = election_func;
61 coordinator->election_context = election_context;
62
63 /* Initialize state machine */
64 asciichat_error_t err = consensus_state_create(my_id, topology, &coordinator->state);
65 if (err != ASCIICHAT_OK) {
66 SAFE_FREE(coordinator);
67 return err;
68 }
69
70 /* Initialize round scheduling */
71 coordinator->last_round_start_ns = time_get_ns();
72 coordinator->next_round_id = 1;
73 coordinator->has_stored_result = false;
74
75 /* Register coordinator with named registry */
76 char coordinator_name[64];
77 snprintf(coordinator_name, sizeof(coordinator_name), "consensus_coordinator:node_%u", my_id[0]);
78 NAMED_REGISTER(coordinator, coordinator_name, "consensus_coordinator_t", "0x%tx", NULL);
79
80 /* Register child state with coordinator as parent */
81 NAMED_REGISTER(coordinator->state, "state", "consensus_state_t", "0x%tx", (uintptr_t)(const void *)(coordinator));
82
83 log_debug("Coordinator created for node %u, first round in 5 minutes", my_id[0]);
84
85 *out_coordinator = coordinator;
86 return ASCIICHAT_OK;
87}

103static bool should_start_new_round(consensus_coordinator_t *coordinator, uint64_t now_ns) {
104 if (!coordinator || !coordinator->state) {
105 return false;
106 }
107
108 consensus_state_machine_t current_state = consensus_state_get_current_state(coordinator->state);
109 if (current_state != CONSENSUS_STATE_IDLE) {
110 return false;
111 }
112
113 uint64_t time_since_last_round = time_elapsed_ns(coordinator->last_round_start_ns, now_ns);
114 return time_since_last_round >= CONSENSUS_ROUND_INTERVAL_NS;
115}

176static asciichat_error_t complete_collection(consensus_coordinator_t *coordinator) {
177 asciichat_error_t err = consensus_state_collection_complete(coordinator->state);
178 if (err != ASCIICHAT_OK) {
179 return err;
180 }
181
182 consensus_state_machine_t current_state = consensus_state_get_current_state(coordinator->state);
183
184 if (current_state == CONSENSUS_STATE_ELECTION_START) {
185 /* Leader: Run election */
186 log_info("Leader: Computing election from %d metrics", consensus_state_get_metrics_count(coordinator->state));
187
188 err = consensus_state_compute_election(coordinator->state);
189 if (err != ASCIICHAT_OK) {
190 return err;
191 }
192
193 /* Optionally run election callback */
194 if (coordinator->election_func) {
195 err = coordinator->election_func(coordinator->election_context, coordinator->state);
196 if (err != ASCIICHAT_OK) {
197 log_warn("Election callback returned error: %d", err);
198 }
199 }
200
201 /* Store elected hosts */
202 uint8_t host_id[16], backup_id[16];
203 err = consensus_state_get_elected_host(coordinator->state, host_id);
204 if (err == ASCIICHAT_OK) {
205 err = consensus_state_get_elected_backup(coordinator->state, backup_id);
206 }
207 if (err == ASCIICHAT_OK) {
208 memcpy(coordinator->stored_host_id, host_id, 16);
209 memcpy(coordinator->stored_backup_id, backup_id, 16);
210 coordinator->has_stored_result = true;
211 log_info("Election complete: host=%u, backup=%u", host_id[0], backup_id[0]);
212 }
213 }
214
215 return ASCIICHAT_OK;
216}

221asciichat_error_t consensus_coordinator_process(consensus_coordinator_t *coordinator, uint32_t timeout_ms) {
222 (void)timeout_ms; /* Currently unused, but kept for future use */
223
224 if (!coordinator) {
225 return SET_ERRNO(ERROR_INVALID_PARAM, "Coordinator is NULL");
226 }
227
228 uint64_t now_ns = time_get_ns();
229
230 /* Check if we should start a new round */
231 if (should_start_new_round(coordinator, now_ns)) {
232 /* Only leader starts rounds */
233 if (consensus_topology_am_leader(coordinator->topology)) {
234 asciichat_error_t err = start_collection_round(coordinator, now_ns);
235 if (err != ASCIICHAT_OK) {
236 log_warn("Failed to start collection round: %d", err);
237 return err;
238 }
239 }
240 }
241
242 consensus_state_machine_t current_state = consensus_state_get_current_state(coordinator->state);
243
244 /* Check for collection timeout */
245 if (current_state == CONSENSUS_STATE_COLLECTING && has_collection_timed_out(coordinator, now_ns)) {
246 log_warn("Collection round timed out, completing early");
247 asciichat_error_t err = complete_collection(coordinator);
248 if (err != ASCIICHAT_OK) {
249 log_error("Failed to complete collection on timeout: %d", err);
250 }
251 }
252
253 return ASCIICHAT_OK;
254}

288asciichat_error_t consensus_coordinator_on_collection_start(consensus_coordinator_t *coordinator, uint32_t round_id,

295 if (current_state != CONSENSUS_STATE_IDLE) {

298
299 /* Start collection */
300 asciichat_error_t err = consensus_state_start_collection(coordinator->state);
301 if (err != ASCIICHAT_OK) {
302 return err;
303 }
304
305 coordinator->next_round_id = round_id;
306 coordinator->collection_deadline_ns = deadline_ns;
307
308 /* Measure and add our metrics */
309 err = measure_and_add_metrics(coordinator);
310 if (err != ASCIICHAT_OK) {
311 log_warn("Failed to measure metrics: %d", err);
312 }
313
314 log_dev("Collection started: round_id=%u, deadline in %u seconds", round_id,
315 (unsigned int)((deadline_ns - time_get_ns()) / NS_PER_SEC_INT));
316
317 return ASCIICHAT_OK;

379asciichat_error_t consensus_coordinator_get_current_host(const consensus_coordinator_t *coordinator,
380 uint8_t out_host_id[16], uint8_t out_backup_id[16]) {
381 if (!coordinator || !out_host_id || !out_backup_id) {
382 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameter");
383 }
384
385 /* Try to get from current state first */
386 consensus_state_machine_t current_state = consensus_state_get_current_state(coordinator->state);
387 if (current_state == CONSENSUS_STATE_ELECTION_COMPLETE) {
388 asciichat_error_t err = consensus_state_get_elected_host(coordinator->state, out_host_id);
389 if (err == ASCIICHAT_OK) {
390 err = consensus_state_get_elected_backup(coordinator->state, out_backup_id);
391 }
392 if (err == ASCIICHAT_OK) {
393 return ASCIICHAT_OK;
394 }
395 }
396
397 /* Fall back to stored result */
398 if (coordinator->has_stored_result) {
399 memcpy(out_host_id, coordinator->stored_host_id, 16);
400 memcpy(out_backup_id, coordinator->stored_backup_id, 16);
401 return ASCIICHAT_OK;
402 }
403
404 return SET_ERRNO(ERROR_INVALID_STATE, "No election result available");
405}
