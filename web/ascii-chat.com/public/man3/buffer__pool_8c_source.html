37static inline void update_peak(atomic_t *peak, uint64_t value) {
38 uint64_t old = atomic_load_u64(peak);
39 while (value > old) {
40 if (atomic_cas_u64(peak, &old, value)) {
41 break;
42 }

51buffer_pool_t *buffer_pool_create(size_t max_bytes, uint64_t shrink_delay_ns) {
52 buffer_pool_t *pool = SAFE_CALLOC(1, sizeof(buffer_pool_t), buffer_pool_t *);
53 if (!pool) {
54 SET_ERRNO(ERROR_MEMORY, "Failed to allocate buffer pool");
55 return NULL;
56 }
57
58 if (mutex_init(&pool->shrink_mutex, "buffer_pool_shrink") != 0) {
59 SET_ERRNO(ERROR_THREAD, "Failed to initialize shrink mutex");
60 SAFE_FREE(pool);
61 return NULL;
62 }
63
64 pool->max_bytes = max_bytes > 0 ? max_bytes : BUFFER_POOL_MAX_BYTES;
65 pool->shrink_delay_ns = shrink_delay_ns > 0 ? shrink_delay_ns : BUFFER_POOL_SHRINK_DELAY_NS;
66
67 // Initialize atomic pointer fields
68 atomic_ptr_store(&pool->free_list, NULL);
69
70 // Initialize atomic counter fields
71 atomic_store_u64(&pool->current_bytes, 0);
72 atomic_store_u64(&pool->used_bytes, 0);
73 atomic_store_u64(&pool->peak_bytes, 0);
74 atomic_store_u64(&pool->peak_pool_bytes, 0);
75 atomic_store_u64(&pool->hits, 0);
76 atomic_store_u64(&pool->allocs, 0);
77 atomic_store_u64(&pool->returns, 0);
78 atomic_store_u64(&pool->shrink_freed, 0);
79 atomic_store_u64(&pool->malloc_fallbacks, 0);
80
81 char pretty_max[64];
82 format_bytes_pretty(pool->max_bytes, pretty_max, sizeof(pretty_max));
83 log_dev("Created buffer pool (max: %s, shrink: %llu ns, lock-free)", pretty_max,
84 (unsigned long long)pool->shrink_delay_ns);
85
86 // Register pool atomics with named registry for debugging and --sync-state output
87 NAMED_REGISTER_ATOMIC_PTR(&pool->free_list, "buffer_pool.global_free_list_head_ptr", NULL);
88 NAMED_REGISTER_ATOMIC(&pool->current_bytes, "buffer_pool.global_current_bytes_in_pool", NULL);
89 NAMED_REGISTER_ATOMIC(&pool->used_bytes, "buffer_pool.global_bytes_currently_in_use", NULL);
90 NAMED_REGISTER_ATOMIC(&pool->peak_bytes, "buffer_pool.global_peak_bytes_in_use", NULL);
91 NAMED_REGISTER_ATOMIC(&pool->peak_pool_bytes, "buffer_pool.global_peak_bytes_in_pool", NULL);
92 NAMED_REGISTER_ATOMIC(&pool->hits, "buffer_pool.global_allocations_from_free_list", NULL);
93 NAMED_REGISTER_ATOMIC(&pool->allocs, "buffer_pool.global_new_buffer_allocations", NULL);
94 NAMED_REGISTER_ATOMIC(&pool->returns, "buffer_pool.global_buffers_returned_to_pool", NULL);
95 NAMED_REGISTER_ATOMIC(&pool->shrink_freed, "buffer_pool.global_buffers_freed_by_shrink", NULL);
96 NAMED_REGISTER_ATOMIC(&pool->malloc_fallbacks, "buffer_pool.global_allocations_bypassing_pool", NULL);
97
98 // Register sync primitives for debugging
99 NAMED_REGISTER_MUTEX(&pool->shrink_mutex, "shrink_mutex", (uintptr_t)(const void *)(pool));
100
101 return pool;
102}

104void buffer_pool_destroy(buffer_pool_t *pool) {
105 if (!pool)
106 return;
107
108 NAMED_UNREGISTER(pool);
109
110 // Drain the free list
111 buffer_node_t *node = atomic_ptr_load(&pool->free_list);
112 while (node) {
113 buffer_node_t *next = atomic_ptr_load(&node->next);
114 SAFE_FREE(node); // Node and data are one allocation
115 node = next;
116 }
117
118 mutex_destroy(&pool->shrink_mutex);
119 SAFE_FREE(pool);
120}

122void *buffer_pool_alloc(buffer_pool_t *pool, size_t size) {
123 // Use global pool if none specified
124 if (!pool) {
125 pool = buffer_pool_get_global();
126 }
127
128 // Size out of range - use malloc with node header for consistent cleanup
129 if (!pool || size < BUFFER_POOL_MIN_SIZE || size > BUFFER_POOL_MAX_SINGLE_SIZE) {
130 size_t total_size = sizeof(buffer_node_t) + size;
131 buffer_node_t *node = SAFE_MALLOC(total_size, buffer_node_t *);
132 node->magic = MAGIC_BUFFER_POOL_FALLBACK; // Different magic for fallbacks
133 node->_pad = 0;
134 node->size = size;
135 atomic_ptr_store(&node->next, NULL); // Initialize atomic pointer field
136 atomic_store_u64(&node->returned_at_ns, 0);
137 node->pool = NULL; // No pool for fallbacks
138
139 if (pool) {
140 atomic_fetch_add_u64(&pool->malloc_fallbacks, 1);
141 }
142 return data_from_node(node);
143 }
144
145 // Try to pop from lock-free stack (LIFO)
146 void *node_ptr = atomic_ptr_load(&pool->free_list);
147 buffer_node_t *node = (buffer_node_t *)node_ptr;
148 while (node) {
149 void *next_ptr = atomic_ptr_load(&node->next);
150 if (atomic_ptr_cas(&pool->free_list, &node_ptr, next_ptr)) {
151 // Successfully popped - check if it's big enough
152 if (node->size >= size) {
153 // Reuse this buffer
154 atomic_ptr_store(&node->next, NULL);
155 size_t node_size = node->size;
156 atomic_fetch_add_u64(&pool->used_bytes, node_size);
157 atomic_fetch_add_u64(&pool->hits, 1);
158 update_peak(&pool->peak_bytes, atomic_load_u64(&pool->used_bytes));
159 return data_from_node(node);
160 } else {
161 // Too small - push it back and allocate new
162 // (This is rare with LIFO - usually we get similar sizes)
163 void *head = atomic_ptr_load(&pool->free_list);
164 do {
165 atomic_ptr_store(&node->next, head);
166 } while (!atomic_ptr_cas(&pool->free_list, &head, (void *)node));
167 break; // Fall through to allocate new
168 }
169 }
170 // CAS failed - reload and retry
171 node = atomic_ptr_load(&pool->free_list);
172 }
173
174 // Check if we can allocate more
175 size_t total_size = sizeof(buffer_node_t) + size;
176 size_t current = atomic_load_u64(&pool->current_bytes);
177
178 // Atomically try to reserve space
179 while (current + total_size <= pool->max_bytes) {
180 if (atomic_cas_u64(&pool->current_bytes, &current, current + total_size)) {
181 // Reserved space - now allocate
182 // Allocate node + data in one chunk for cache efficiency
183 node = SAFE_MALLOC_ALIGNED(total_size, 64, buffer_node_t *);
184 if (!node) {
185 // Undo reservation
186 atomic_fetch_sub_u64(&pool->current_bytes, total_size);
187 atomic_fetch_add_u64(&pool->malloc_fallbacks, 1);
188 return SAFE_MALLOC(size, void *);
189 }
190
191 node->magic = MAGIC_BUFFER_POOL_VALID;
192 node->_pad = 0;
193 node->size = size;
194 atomic_ptr_store(&node->next, NULL); // Initialize atomic pointer field
195 atomic_store_u64(&node->returned_at_ns, 0);
196 node->pool = pool;
197
198 atomic_fetch_add_u64(&pool->used_bytes, size);
199 atomic_fetch_add_u64(&pool->allocs, 1);
200 update_peak(&pool->peak_bytes, atomic_load_u64(&pool->used_bytes));
201 update_peak(&pool->peak_pool_bytes, atomic_load_u64(&pool->current_bytes));
202
203 return data_from_node(node);
204 }
205 // CAS failed - someone else allocated, reload and check again
206 }
207
208 // Pool at capacity - fall back to malloc with node header for consistent cleanup
209 total_size = sizeof(buffer_node_t) + size;
210 node = SAFE_MALLOC(total_size, buffer_node_t *);
211 if (!node) {
212 return NULL;
213 }
214 node->magic = MAGIC_BUFFER_POOL_FALLBACK;
215 node->_pad = 0;
216 node->size = size;
217 atomic_store_u64(&node->returned_at_ns, 0);
218 node->pool = pool;
219
220 atomic_fetch_add_u64(&pool->malloc_fallbacks, 1);
221 return data_from_node(node);
222}

224void buffer_pool_free(buffer_pool_t *pool, const void *data, size_t size) {
225 (void)size; // Size parameter not needed with header-based detection
226
227 if (!data)
228 return;
229
230 // All buffer_pool allocations have headers, safe to check magic
231 // Cast away const since we're just reading the header, not modifying data
232 buffer_node_t *node = node_from_data((void *)data);
233
234 // If it's a malloc fallback (has fallback magic), free the node directly
235 if (IS_MAGIC_VALID(node->magic, MAGIC_BUFFER_POOL_FALLBACK)) {
236 SAFE_FREE(node); // Free the node (includes header + data)
237 return;
238 }
239
240 // If it's not a pooled buffer (no valid magic), it's external - use platform free
241 if (!IS_MAGIC_VALID(node->magic, MAGIC_BUFFER_POOL_VALID)) {
242 // Cast away const for free() which expects non-const void *
243 free((void *)data); // Unknown allocation, just free the data pointer
244 return;
245 }
246
247 // It's a pooled buffer - return to pool
248 // Use the pool stored in the node if none provided
249 if (!pool) {
250 pool = node->pool;
251 }
252
253 if (!pool) {
254 // Shouldn't happen, but safety
255 log_error("Pooled buffer has no pool reference!");
256 return;
257 }
258
259 // Update stats
260 atomic_fetch_sub_u64(&pool->used_bytes, node->size);
261 atomic_fetch_add_u64(&pool->returns, 1);
262
263 // Set return timestamp
264 atomic_store_u64(&node->returned_at_ns, time_get_ns());
265
266 // Push to lock-free stack
267 void *head = atomic_ptr_load(&pool->free_list);
268 do {
269 atomic_ptr_store(&node->next, head);
270 } while (!atomic_ptr_cas(&pool->free_list, &head, (void *)node));
271
272 // Periodically trigger shrink (every 100 returns)
273 uint64_t returns = atomic_load_u64(&pool->returns);
274 if (returns % 100 == 0) {
275 buffer_pool_shrink(pool);
276 }
277}

279void buffer_pool_shrink(buffer_pool_t *pool) {
280 if (!pool || pool->shrink_delay_ns == 0)
281 return;
282
283 // Only one thread can shrink at a time
284 if (mutex_trylock(&pool->shrink_mutex) != 0) {
285 return; // Another thread is shrinking
286 }
287
288 uint64_t now = time_get_ns();
289 uint64_t cutoff = (now > pool->shrink_delay_ns) ? (now - pool->shrink_delay_ns) : 0;
290
291 // Atomically swap out the entire free list
292 buffer_node_t *list = (buffer_node_t *)atomic_ptr_exchange(&pool->free_list, NULL);
293
294 // Partition into keep and free lists
295 buffer_node_t *keep_list = NULL;
296 buffer_node_t *free_list = NULL;
297
298 while (list) {
299 buffer_node_t *next = atomic_ptr_load(&list->next);
300 uint64_t returned_at = atomic_load_u64(&list->returned_at_ns);
301
302 if (returned_at < cutoff) {
303 // Old buffer - add to free list
304 atomic_ptr_store(&list->next, (void *)free_list);
305 free_list = list;
306 } else {
307 // Recent buffer - keep it
308 atomic_ptr_store(&list->next, (void *)keep_list);
309 keep_list = list;
310 }
311 list = next;
312 }
313
314 // Push kept buffers back to free list
315 if (keep_list) {
316 // Find tail
317 buffer_node_t *tail = keep_list;
318 void *next_ptr = atomic_ptr_load(&tail->next);
319 while (next_ptr) {
320 tail = (buffer_node_t *)next_ptr;
321 next_ptr = atomic_ptr_load(&tail->next);
322 }
323
324 // Atomically prepend to current free list
325 void *head = atomic_ptr_load(&pool->free_list);
326 while (!atomic_ptr_cas(&pool->free_list, &head, (void *)keep_list)) {
327 atomic_ptr_store(&tail->next, head);
328 }
329 }
330
331 // Free old buffers
332 while (free_list) {
333 buffer_node_t *next = atomic_ptr_load(&free_list->next);
334 size_t total_size = sizeof(buffer_node_t) + free_list->size;
335 atomic_fetch_sub_u64(&pool->current_bytes, total_size);
336 atomic_fetch_add_u64(&pool->shrink_freed, 1);
337 SAFE_FREE(free_list);
338 free_list = next;
339 }
340
341 mutex_unlock(&pool->shrink_mutex);
342}

344void buffer_pool_get_stats(buffer_pool_t *pool, size_t *current_bytes, size_t *used_bytes, size_t *free_bytes) {
345 if (!pool) {
346 if (current_bytes)
347 *current_bytes = 0;
348 if (used_bytes)
349 *used_bytes = 0;
350 if (free_bytes)
351 *free_bytes = 0;
352 return;
353 }
354
355 size_t current = atomic_load_u64(&pool->current_bytes);
356 size_t used = atomic_load_u64(&pool->used_bytes);
357
358 if (current_bytes)
359 *current_bytes = current;
360 if (used_bytes)
361 *used_bytes = used;
362 if (free_bytes)
363 *free_bytes = (current > used) ? (current - used) : 0;
364}

366void buffer_pool_log_stats(buffer_pool_t *pool, const char *name) {
367 if (!pool)
368 return;
369
370 size_t current = atomic_load_u64(&pool->current_bytes);
371 size_t used = atomic_load_u64(&pool->used_bytes);
372 size_t peak = atomic_load_u64(&pool->peak_bytes);
373 size_t peak_pool = atomic_load_u64(&pool->peak_pool_bytes);
374 uint64_t hits = atomic_load_u64(&pool->hits);
375 uint64_t allocs = atomic_load_u64(&pool->allocs);
376 uint64_t returns = atomic_load_u64(&pool->returns);
377 uint64_t shrink_freed = atomic_load_u64(&pool->shrink_freed);
378 uint64_t fallbacks = atomic_load_u64(&pool->malloc_fallbacks);
379
380 char pretty_current[64], pretty_used[64], pretty_free[64];
381 char pretty_peak[64], pretty_peak_pool[64], pretty_max[64];
382
383 format_bytes_pretty(current, pretty_current, sizeof(pretty_current));
384 format_bytes_pretty(used, pretty_used, sizeof(pretty_used));
385 format_bytes_pretty(current > used ? current - used : 0, pretty_free, sizeof(pretty_free));
386 format_bytes_pretty(peak, pretty_peak, sizeof(pretty_peak));
387 format_bytes_pretty(peak_pool, pretty_peak_pool, sizeof(pretty_peak_pool));
388 format_bytes_pretty(pool->max_bytes, pretty_max, sizeof(pretty_max));
389
390 uint64_t total_requests = hits + allocs + fallbacks;
391 double hit_rate = total_requests > 0 ? (double)hits * 100.0 / (double)total_requests : 0;
392
393 log_debug("=== Buffer Pool: %s ===", name ? name : "unnamed");
394 log_debug(" Current: %s / %s capacity (peak used: %s)", pretty_current, pretty_max, pretty_peak_pool);
395 log_debug(" Buffers: %s used, %s free (peak: %s)", pretty_used, pretty_free, pretty_peak);
396 log_debug(" Hits: %llu (%.1f%%), Allocs: %llu, Fallbacks: %llu", (unsigned long long)hits, hit_rate,
397 (unsigned long long)allocs, (unsigned long long)fallbacks);
398 log_debug(" Returns: %llu, Shrink freed: %llu", (unsigned long long)returns, (unsigned long long)shrink_freed);
399}
