166asciichat_instr_runtime_t *asciichat_instr_runtime_get(void) {
167 if (g_disable_write) {
168 return NULL;
169 }
170
171 // Initialize runtime once using lock-free lifecycle API
172 if (!lifecycle_is_initialized(&g_runtime_lc)) {
173 if (lifecycle_init_once(&g_runtime_lc)) {
174 asciichat_instr_runtime_init_once();
175 lifecycle_init_commit(&g_runtime_lc);
176 }
177 }
178
179 asciichat_instr_runtime_t *runtime = ascii_tls_get(g_runtime_key);
180 if (runtime != NULL) {
181 return runtime;
182 }
183
184 runtime = SAFE_CALLOC(1, sizeof(*runtime), asciichat_instr_runtime_t *);
185 if (runtime == NULL) {
186 return NULL;
187 }
188
189 runtime->pid = platform_get_pid();
190 runtime->thread_id = asciichat_thread_current_id();
191 runtime->sequence = 0;
192 runtime->call_counter = 0;
193 runtime->fd = -1;
194 runtime->filter_include = SAFE_GETENV("ASCII_INSTR_SOURCE_PRINT_INCLUDE");
195 runtime->filter_exclude = SAFE_GETENV("ASCII_INSTR_SOURCE_PRINT_EXCLUDE");
196 runtime->filter_thread = SAFE_GETENV("ASCII_INSTR_SOURCE_PRINT_THREAD");
197 runtime->filter_function_include = NULL;
198 runtime->filter_function_exclude = NULL;
199 runtime->filters_enabled = false;
200 runtime->rate = 1;
201 runtime->rate_enabled = false;
202
203 asciichat_instr_runtime_configure(runtime);
204
205 if (ascii_tls_set(g_runtime_key, runtime) != 0) {
206 SAFE_FREE(runtime);
207 return NULL;
208 }
209
210 return runtime;
211}

213void asciichat_instr_runtime_destroy(asciichat_instr_runtime_t *runtime) {
214 if (runtime == NULL) {
215 return;
216 }
217
218 if (runtime->fd >= 0) {
219 platform_close(runtime->fd);
220 runtime->fd = -1;
221 }
222
223#if ASCII_INSTR_SOURCE_PRINT_HAVE_REGEX
224 if (runtime->include_regex_valid) {
225 regfree(&runtime->include_regex);
226 runtime->include_regex_valid = false;
227 }
228 if (runtime->exclude_regex_valid) {
229 regfree(&runtime->exclude_regex);
230 runtime->exclude_regex_valid = false;
231 }
232 if (runtime->function_include_regex_valid) {
233 regfree(&runtime->function_include_regex);
234 runtime->function_include_regex_valid = false;
235 }
236 if (runtime->function_exclude_regex_valid) {
237 regfree(&runtime->function_exclude_regex);
238 runtime->function_exclude_regex_valid = false;
239 }
240#endif
241
242 asciichat_instr_only_list_destroy(&runtime->only_selectors);
243 SAFE_FREE(runtime);
244}

265void asciichat_instr_log_line(const char *file_path, uint32_t line_number, const char *function_name,
266 const char *snippet, uint8_t is_macro_expansion) {
267 // Instrumentation is enabled by default when the binary is built with Source Print.
268 // Set ASCII_INSTR_SOURCE_PRINT_ENABLE=0 to disable tracing at runtime.
269 if (!g_instrumentation_enabled_checked) {
270 const char *enable_env = SAFE_GETENV("ASCII_INSTR_SOURCE_PRINT_ENABLE");
271 // Default to enabled (true) - only disable if explicitly set to "0", "false", "off", or "no"
272 if (enable_env != NULL && enable_env[0] != '\0') {
273 g_instrumentation_enabled = asciichat_instr_env_is_enabled(enable_env);
274 } else {
275 g_instrumentation_enabled = true; // Default to enabled when instrumented
276 }
277 g_instrumentation_enabled_checked = true;
278 }
279
280 if (!g_instrumentation_enabled) {
281 return;
282 }
283
284 if (g_disable_write) {
285 return;
286 }
287
288 if (g_logging_reentry_guard) {
289 return;
290 }
291
292 g_logging_reentry_guard = true;
293
294 asciichat_instr_runtime_t *runtime = asciichat_instr_runtime_get();
295 if (runtime == NULL) {
296 goto cleanup;
297 }
298
299 if (!asciichat_instr_should_log(runtime, file_path, line_number, function_name)) {
300 goto cleanup;
301 }
302
303 runtime->call_counter++;
304 if (runtime->rate_enabled) {
305 const uint64_t counter = runtime->call_counter;
306 if (((counter - 1U) % runtime->rate) != 0U) {
307 goto cleanup;
308 }
309 }
310
311 if (runtime->fd < 0) {
312 if (asciichat_instr_open_log_file(runtime) != 0) {
313 runtime->stderr_fallback = true;
314 }
315 }
316
317 const int fd = runtime->stderr_fallback ? STDERR_FILENO : runtime->fd;
318 char buffer[ASCII_INSTR_SOURCE_PRINT_MAX_LINE];
319 size_t pos = 0;
320
321 uint64_t realtime_ns = time_get_realtime_ns();
322 time_t sec = (time_t)(realtime_ns / NS_PER_SEC_INT);
323 long nsec = (long)(realtime_ns % NS_PER_SEC_INT);
324 struct tm tm_now;
325 memset(&tm_now, 0, sizeof(tm_now));
326 if (platform_gtime(&sec, &tm_now) != ASCIICHAT_OK) {
327 memset(&tm_now, 0, sizeof(tm_now));
328 }
329
330 char timestamp[32];
331 size_t ts_len = strftime(timestamp, sizeof(timestamp), "%Y-%m-%dT%H:%M:%S", &tm_now);
332 if (ts_len == 0) {
333 SAFE_STRNCPY(timestamp, "1970-01-01T00:00:00", sizeof(timestamp));
334 ts_len = strlen(timestamp);
335 }
336
337 char elapsed_buf[32];
338 elapsed_buf[0] = '\0';
339 if (g_ticks_initialized) {
340 uint64_t now_ns = time_get_ns();
341 uint64_t elapsed_ns = time_elapsed_ns(g_start_ns, now_ns);
342 if (time_pretty((uint64_t)elapsed_ns, -1, elapsed_buf, sizeof(elapsed_buf)) < 0) {
343 elapsed_buf[0] = '\0';
344 }
345 }
346
347 runtime->sequence++;
348
349 const char *elapsed_field = (elapsed_buf[0] != '\0') ? elapsed_buf : "-";
350
351 const char *safe_file_path = (file_path != NULL) ? file_path : "<unknown>";
352 pos += safe_snprintf(buffer + pos, sizeof(buffer) - pos,
353 "pid=%d tid=%llu seq=%llu ts=%.*s.%09ldZ elapsed=%s file=%s line=%u func=%s macro=%u snippet=",
354 runtime->pid, (unsigned long long)runtime->thread_id, (unsigned long long)runtime->sequence,
355 (int)ts_len, timestamp, nsec, elapsed_field, safe_file_path, line_number,
356 function_name ? function_name : "<unknown>", (unsigned)is_macro_expansion);
357
358 if (snippet != NULL) {
359 size_t snippet_len = strnlen(snippet, ASCII_INSTR_SOURCE_PRINT_MAX_SNIPPET);
360 for (size_t i = 0; i < snippet_len && pos < sizeof(buffer) - 2; ++i) {
361 const char ch = snippet[i];
362 switch (ch) {
363 case '\n':
364 buffer[pos++] = '\\';
365 buffer[pos++] = 'n';
366 break;
367 case '\r':
368 buffer[pos++] = '\\';
369 buffer[pos++] = 'r';
370 break;
371 case '\t':
372 buffer[pos++] = '\\';
373 buffer[pos++] = 't';
374 break;
375 default:
376 buffer[pos++] = ch;
377 break;
378 }
379 }
380 }
381
382 if (pos >= sizeof(buffer) - 1) {
383 pos = sizeof(buffer) - 2;
384 }
385
386 buffer[pos++] = '\n';
387 buffer[pos] = '\0';
388
389 asciichat_instr_write_full(fd, buffer, pos);
390
391 // Also echo to stderr if requested via environment variable
392 if (!g_echo_to_stderr_initialized) {
393 const char *echo_env = SAFE_GETENV("ASCII_CHAT_DEBUG_SELF_SOURCE_CODE_LOG_STDERR");
394 g_echo_to_stderr = asciichat_instr_env_is_enabled(echo_env);
395 g_echo_to_stderr_initialized = true;
396 }
397
398 if (g_echo_to_stderr && !runtime->stderr_fallback) {
399 // Write to stderr in addition to the log file
400 // Suppress Windows deprecation warning for POSIX write function
401#ifdef _WIN32
402#pragma clang diagnostic push
403#pragma clang diagnostic ignored "-Wdeprecated-declarations"
404#endif
405 (void)posix_write(STDERR_FILENO, buffer, pos);
406#ifdef _WIN32
407#pragma clang diagnostic pop
408#endif
409 }
410
411cleanup:
412 g_logging_reentry_guard = false;
413}

442 // Called by lifecycle_init_once winner
443 (void)ascii_tls_key_create(&g_runtime_key, asciichat_instr_runtime_tls_destructor);
444 const char *output_dir_env = SAFE_GETENV("ASCII_INSTR_SOURCE_PRINT_OUTPUT_DIR");
445 if (output_dir_env != NULL && output_dir_env[0] != '\0') {
446 char *normalized_output_dir = NULL;
447 asciichat_error_t validation_result =
448 path_validate_user_path(output_dir_env, PATH_ROLE_LOG_FILE, &normalized_output_dir);
449 if (validation_result == ASCIICHAT_OK && normalized_output_dir != NULL) {
450 SAFE_STRNCPY(g_output_dir, normalized_output_dir, sizeof(g_output_dir));
451 g_output_dir[sizeof(g_output_dir) - 1] = '\0';
452 g_output_dir_set = true;
453 } else {
454 log_warn("Ignoring invalid ASCII_INSTR_SOURCE_PRINT_OUTPUT_DIR path: %s", output_dir_env);
455 }
456 SAFE_FREE(normalized_output_dir);
457 }
458 const char *coverage_env = SAFE_GETENV("ASCII_INSTR_SOURCE_PRINT_ENABLE_COVERAGE");
459 g_coverage_enabled = asciichat_instr_env_is_enabled(coverage_env);
460 g_start_ns = time_get_ns();
461 g_ticks_initialized = true;
462}

468static bool asciichat_instr_build_log_path(asciichat_instr_runtime_t *runtime) {
469 // Check for custom log file path first
470 const char *custom_log_file = SAFE_GETENV("ASCII_CHAT_DEBUG_SELF_SOURCE_CODE_LOG_FILE");
471 bool is_custom_path = (custom_log_file != NULL && custom_log_file[0] != '\0');
472
473 if (is_custom_path) {
474 const char *debug_env = SAFE_GETENV("ASCII_INSTR_SOURCE_PRINT_ECHO_STDERR");
475 if (debug_env && debug_env[0] == '1') {
476 fprintf(stderr, "ASCII_INSTR: Using custom log path: %s\n", custom_log_file);
477 }
478
479 // For instrumentation log files, bypass strict path validation since this is a debug feature
480 // Just normalize the path to an absolute path without security checks
481 char *expanded = expand_path(custom_log_file);
482 if (!expanded) {
483 log_warn("Failed to expand ASCII_CHAT_DEBUG_SELF_SOURCE_CODE_LOG_FILE path: %s", custom_log_file);
484 return false;
485 }
486
487 // Convert to absolute path if relative
488 char absolute_buf[PATH_MAX];
489 if (!path_is_absolute(expanded)) {
490 char cwd_buf[PATH_MAX];
491 if (!platform_get_cwd(cwd_buf, sizeof(cwd_buf))) {
492 SAFE_FREE(expanded);
493 return false;
494 }
495 // Check if expanded already starts with a separator to avoid double separators
496 if (strlen(expanded) > 0 && expanded[0] == PATH_DELIM) {
497 safe_snprintf(absolute_buf, sizeof(absolute_buf), "%s%s", cwd_buf, expanded);
498 } else {
499 safe_snprintf(absolute_buf, sizeof(absolute_buf), "%s%c%s", cwd_buf, PATH_DELIM, expanded);
500 }
501 SAFE_FREE(expanded);
502 expanded = platform_strdup(absolute_buf);
503 if (!expanded) {
504 return false;
505 }
506 }
507
508 SAFE_STRNCPY(runtime->log_path, expanded, sizeof(runtime->log_path));
509 runtime->log_path[sizeof(runtime->log_path) - 1] = '\0';
510 SAFE_FREE(expanded);
511
512 if (debug_env && debug_env[0] == '1') {
513 fprintf(stderr, "ASCII_INSTR: Resolved custom log path: %s\n", runtime->log_path);
514 }
515
516 // Don't check if file exists - allow appending to existing file
517 // Path already normalized, skip validation below
518 } else {
519 // Determine output directory
520 char output_dir_buf[PATH_MAX];
521 if (g_output_dir_set) {
522 // Use the output directory from ASCII_INSTR_SOURCE_PRINT_OUTPUT_DIR
523 SAFE_STRNCPY(output_dir_buf, g_output_dir, sizeof(output_dir_buf));
524 } else if (!platform_get_cwd(output_dir_buf, sizeof(output_dir_buf))) {
525 // Fallback to temp directory if cwd fails
526 const char *fallback = SAFE_GETENV("TMPDIR");
527 if (fallback == NULL) {
528 fallback = SAFE_GETENV("TEMP");
529 }
530 if (fallback == NULL) {
531 fallback = SAFE_GETENV("TMP");
532 }
533 if (fallback == NULL) {
534 fallback = "/tmp";
535 }
536 SAFE_STRNCPY(output_dir_buf, fallback, sizeof(output_dir_buf));
537 }
538
539 // Build log file name
540 if (g_output_dir_set) {
541 // When output directory is explicitly set, use unique naming with pid and tid
542 if (safe_snprintf(runtime->log_path, sizeof(runtime->log_path), "%s%c%s-%d-%llu.log", output_dir_buf, PATH_DELIM,
543 ASCII_INSTR_SOURCE_PRINT_DEFAULT_BASENAME, runtime->pid,
544 (unsigned long long)runtime->thread_id) >= (int)sizeof(runtime->log_path)) {
545 return false;
546 }
547 } else {
548 // Use simple "trace.log" name in current directory
549 if (safe_snprintf(runtime->log_path, sizeof(runtime->log_path), "%s%ctrace.log", output_dir_buf, PATH_DELIM) >=
550 (int)sizeof(runtime->log_path)) {
551 return false;
552 }
553 }
554 }
555
556 // Only validate auto-generated paths (custom paths already validated above)
557 if (!is_custom_path) {
558 char *validated_log_path = NULL;
559 asciichat_error_t validate_result =
560 path_validate_user_path(runtime->log_path, PATH_ROLE_LOG_FILE, &validated_log_path);
561 if (validate_result != ASCIICHAT_OK || validated_log_path == NULL) {
562 SAFE_FREE(validated_log_path);
563 log_warn("Failed to validate instrumentation log path: %s", runtime->log_path);
564 return false;
565 }
566 SAFE_STRNCPY(runtime->log_path, validated_log_path, sizeof(runtime->log_path));
567 runtime->log_path[sizeof(runtime->log_path) - 1] = '\0';
568 SAFE_FREE(validated_log_path);
569 }
570
571 // Find last path separator
572 const char *last_sep = strrchr(runtime->log_path, PATH_DELIM);
573 if (last_sep != NULL && last_sep != runtime->log_path) {
574 const size_t dir_path_len = (size_t)(last_sep - runtime->log_path);
575 char dir_path[PATH_MAX];
576 memcpy(dir_path, runtime->log_path, dir_path_len);
577 dir_path[dir_path_len] = '\0';
578 if (mkdir(dir_path, DIR_PERM_PRIVATE) != 0) {
579 if (errno != EEXIST) {
580 return false;
581 }
582 }
583 }
584
585 return true;
586}

588static int asciichat_instr_open_log_file(asciichat_instr_runtime_t *runtime) {
589 if (!asciichat_instr_build_log_path(runtime)) {
590 const char *debug_env = SAFE_GETENV("ASCII_INSTR_SOURCE_PRINT_ECHO_STDERR");
591 if (debug_env && debug_env[0] == '1') {
592 fprintf(stderr, "ASCII_INSTR: Failed to build log path\n");
593 }
594 return -1;
595 }
596
597 // Check if this is a custom log file (env var was set)
598 const char *custom_log_file = SAFE_GETENV("ASCII_CHAT_DEBUG_SELF_SOURCE_CODE_LOG_FILE");
599 const bool is_custom_file = (custom_log_file != NULL && custom_log_file[0] != '\0');
600
601 // For custom files, allow appending to existing file (no O_EXCL)
602 // For auto-generated files, use O_EXCL to avoid conflicts
603 int flags;
604 if (is_custom_file) {
605 flags = PLATFORM_O_WRONLY | PLATFORM_O_CREAT | PLATFORM_O_APPEND | PLATFORM_O_BINARY;
606 } else {
607 flags = PLATFORM_O_WRONLY | PLATFORM_O_CREAT | PLATFORM_O_EXCL | PLATFORM_O_APPEND | PLATFORM_O_BINARY;
608 }
609
610 const char *debug_env = SAFE_GETENV("ASCII_INSTR_SOURCE_PRINT_ECHO_STDERR");
611 if (debug_env && debug_env[0] == '1') {
612 fprintf(stderr, "ASCII_INSTR: Opening log file: %s (custom=%d)\n", runtime->log_path, is_custom_file);
613 }
614
615 const mode_t mode = S_IRUSR | S_IWUSR;
616 int fd = platform_open("instrumentation_log", runtime->log_path, flags, mode);

680static bool asciichat_instr_env_is_enabled(const char *value) {
681 if (value == NULL) {
682 return false;
683 }
684
685 while (*value != '\0' && isspace((unsigned char)*value) != 0) {
686 value++;
687 }
688
689 if (*value == '\0') {
690 return false;
691 }
692
693 if (strcmp(value, STR_ZERO) == 0) {
694 return false;
695 }
696
697 char lowered[8];
698 size_t len = 0;
699 while (value[len] != '\0' && len < sizeof(lowered) - 1) {
700 lowered[len] = (char)tolower((unsigned char)value[len]);
701 len++;
702 }
703 lowered[len] = '\0';
704
705 return !(strcmp(lowered, STR_FALSE) == 0 || strcmp(lowered, STR_OFF) == 0 || strcmp(lowered, STR_NO) == 0);
706}

758static bool asciichat_instr_only_list_append(asciichat_instr_only_list_t *list, asciichat_instr_selector_type_t type,
759 const char *module, const char *pattern) {
760 if (list == NULL) {
761 return false;
762 }

764 if (type == ASCII_INSTR_SOURCE_PRINT_SELECTOR_FILE_SUBSTRING || type == ASCII_INSTR_SOURCE_PRINT_SELECTOR_FILE_GLOB ||
765 type == ASCII_INSTR_SOURCE_PRINT_SELECTOR_FUNCTION_GLOB) {
766 if (pattern == NULL || pattern[0] == '\0') {
767 return false;
768 }
769 }
770
771 if (type == ASCII_INSTR_SOURCE_PRINT_SELECTOR_MODULE) {
772 if (module == NULL || module[0] == '\0') {
773 return false;
774 }
775 }

781 if (new_items == NULL) {
782 return false; // SAFE_REALLOC already called FATAL, but satisfy analyzer
783 }
784 list->items = new_items;
785 list->capacity = new_capacity;
786 }
787
788 asciichat_instr_only_selector_t *selector = &list->items[list->count];
789 memset(selector, 0, sizeof(*selector));
790 selector->type = type;
791
792 if (module != NULL && module[0] != '\0') {
793 SAFE_STRDUP(selector->module, module);
794 }
795 if (pattern != NULL && pattern[0] != '\0') {
796 SAFE_STRDUP(selector->pattern, pattern);
797 }
798
799 list->count++;
800 return true;
801}

803static void asciichat_instr_trim(char *value) {
804 if (value == NULL) {
805 return;
806 }
807
808 char *start = value;
809 while (*start != '\0' && isspace((unsigned char)*start) != 0) {
810 start++;
811 }
812
813 char *end = start + strlen(start);
814 while (end > start && isspace((unsigned char)end[-1]) != 0) {
815 end--;
816 }
817
818 const size_t length = (size_t)(end - start);
819 if (start != value && length > 0U) {
820 memmove(value, start, length);
821 }
822 value[length] = '\0';
823}

825static bool asciichat_instr_parse_only_filters(asciichat_instr_runtime_t *runtime, const char *value) {
826 if (runtime == NULL) {
827 return false;
828 }
829
830 asciichat_instr_only_list_destroy(&runtime->only_selectors);
831
832 if (value == NULL || value[0] == '\0') {
833 return true;
834 }
835
836 char *mutable_value = NULL;
837 SAFE_STRDUP(mutable_value, value);
838 if (mutable_value == NULL) {
839 return false;
840 }
841
842 char *cursor = mutable_value;
843 while (cursor != NULL && *cursor != '\0') {
844 char *token_start = cursor;
845 while (*cursor != '\0' && *cursor != ',') {
846 cursor++;
847 }
848 if (*cursor == ',') {
849 *cursor = '\0';
850 cursor++;
851 } else {
852 cursor = NULL;
853 }
854
855 asciichat_instr_trim(token_start);
856 if (*token_start == '\0') {
857 continue;
858 }
859
860 char *equal_sign = strchr(token_start, '=');
861 if (equal_sign != NULL) {
862 *equal_sign = '\0';
863 char *kind = token_start;
864 char *spec = equal_sign + 1;
865 asciichat_instr_trim(kind);
866 asciichat_instr_trim(spec);
867 if (*spec == '\0') {
868 continue;
869 }
870
871 if (strcmp(kind, "func") == 0 || strcmp(kind, "function") == 0) {
872 (void)asciichat_instr_only_list_append(&runtime->only_selectors,
873 ASCII_INSTR_SOURCE_PRINT_SELECTOR_FUNCTION_GLOB, NULL, spec);
874 } else if (strcmp(kind, "module") == 0) {
875 char *module_value = spec;
876 char *module_pattern = strchr(module_value, ':');
877 if (module_pattern != NULL) {
878 *module_pattern = '\0';
879 module_pattern++;
880 asciichat_instr_trim(module_pattern);
881 }
882 asciichat_instr_trim(module_value);
883 if (*module_value == '\0') {
884 continue;
885 }
886 const char *pattern_part = (module_pattern != NULL && *module_pattern != '\0') ? module_pattern : NULL;
887 (void)asciichat_instr_only_list_append(&runtime->only_selectors, ASCII_INSTR_SOURCE_PRINT_SELECTOR_MODULE,
888 module_value, pattern_part);
889 } else {
890 // "file" kind or unknown kinds default to FILE_GLOB
891 (void)asciichat_instr_only_list_append(&runtime->only_selectors, ASCII_INSTR_SOURCE_PRINT_SELECTOR_FILE_GLOB,
892 NULL, spec);
893 }
894 continue;
895 }
896
897 char *colon = strchr(token_start, ':');
898 if (colon != NULL) {
899 *colon = '\0';
900 char *module_name = token_start;
901 char *pattern_part = colon + 1;
902 asciichat_instr_trim(module_name);
903 asciichat_instr_trim(pattern_part);
904 if (*module_name == '\0') {
905 continue;
906 }
907 const char *pattern_spec = (*pattern_part != '\0') ? pattern_part : NULL;
908 (void)asciichat_instr_only_list_append(&runtime->only_selectors, ASCII_INSTR_SOURCE_PRINT_SELECTOR_MODULE,
909 module_name, pattern_spec);
910 continue;
911 }
912
913 (void)asciichat_instr_only_list_append(&runtime->only_selectors, ASCII_INSTR_SOURCE_PRINT_SELECTOR_FILE_SUBSTRING,
914 NULL, token_start);
915 }
916
917 SAFE_FREE(mutable_value);
918 return true;
919}

921static bool asciichat_instr_match_glob(const char *pattern, const char *value) {
922 if (pattern == NULL || value == NULL) {
923 return false;
924 }
925
926 const char *p = pattern;
927 const char *v = value;
928 const char *star = NULL;
929 const char *match = NULL;
930
931 while (*v != '\0') {
932 if (*p == '*') {
933 star = p++;
934 match = v;
935 } else if (*p == '?' || *p == *v) {
936 p++;
937 v++;
938 } else if (star != NULL) {
939 p = star + 1;
940 match++;
941 v = match;
942 } else {
943 return false;
944 }
945 }
946
947 while (*p == '*') {
948 p++;
949 }
950
951 return *p == '\0';
952}

967static bool asciichat_instr_path_contains_module(const char *file_path, const char *module_name) {
968 if (file_path == NULL || module_name == NULL || module_name[0] == '\0') {
969 return false;
970 }
971
972 const size_t module_len = strlen(module_name);
973 const char *cursor = file_path;
974 while ((cursor = strstr(cursor, module_name)) != NULL) {
975 bool left_ok = (cursor == file_path);
976 if (!left_ok) {
977 const char prev = cursor[-1];
978 left_ok = (prev == PATH_DELIM);
979 }
980
981 const char tail = cursor[module_len];
982 bool right_ok = (tail == '\0' || tail == PATH_DELIM);
983
984 if (left_ok && right_ok) {
985 return true;
986 }
987
988 cursor = cursor + 1;
989 }
990
991 return false;
992}

994static bool asciichat_instr_only_selectors_match(const asciichat_instr_runtime_t *runtime, const char *file_path,
995 const char *function_name) {
996 const asciichat_instr_only_list_t *list = &runtime->only_selectors;
997 if (list->count == 0) {
998 return true;
999 }
1000
1001 const char *base_name = asciichat_instr_basename(file_path);
1002 for (size_t i = 0; i < list->count; ++i) {
1003 const asciichat_instr_only_selector_t *selector = &list->items[i];
1004 switch (selector->type) {
1005 case ASCII_INSTR_SOURCE_PRINT_SELECTOR_FILE_SUBSTRING:
1006 if (file_path != NULL && selector->pattern != NULL && strstr(file_path, selector->pattern) != NULL) {
1007 return true;
1008 }
1009 break;
1010 case ASCII_INSTR_SOURCE_PRINT_SELECTOR_FILE_GLOB:
1011 if (file_path != NULL && selector->pattern != NULL && asciichat_instr_match_glob(selector->pattern, file_path)) {
1012 return true;
1013 }
1014 break;
1015 case ASCII_INSTR_SOURCE_PRINT_SELECTOR_FUNCTION_GLOB:
1016 if (function_name != NULL && selector->pattern != NULL &&
1017 asciichat_instr_match_glob(selector->pattern, function_name)) {
1018 return true;
1019 }
1020 break;
1021 case ASCII_INSTR_SOURCE_PRINT_SELECTOR_MODULE:
1022 if (file_path != NULL && selector->module != NULL &&
1023 asciichat_instr_path_contains_module(file_path, selector->module)) {
1024 if (selector->pattern == NULL ||
1025 (base_name != NULL && asciichat_instr_match_glob(selector->pattern, base_name))) {
1026 return true;
1027 }
1028 }
1029 break;
1030 default:
1031 break;
1032 }
1033 }
1034
1035 return false;
1036}
