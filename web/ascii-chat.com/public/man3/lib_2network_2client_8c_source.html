125void app_client_destroy(app_client_t **client_ptr) {

127 return; // No-op if NULL
128 }
129
130 app_client_t *client = *client_ptr;
131
132 log_debug("Destroying app client");

136 NAMED_UNREGISTER(&client->audio_capture_thread_exited);
137 NAMED_UNREGISTER(&client->data_thread_exited);

139 NAMED_UNREGISTER(&client->ping_thread_exited);
140 NAMED_UNREGISTER(&client->is_first_frame_of_connection);
141
142 // Destroy mutexes and condition variables

144 cond_destroy(&client->audio_send_queue_cond);
145
146 // Note: Network clients (tcp_client, ws_client) are not destroyed here.
147 // They should be destroyed separately in connection_context_cleanup()
