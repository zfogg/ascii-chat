103 printf("\n");

105 }
106
107 // Check for empty input (Enter pressed)
108 if (input[0] == '\n' || input[0] == '\r' || input[0] == '\0') {

149int ui_mdns_select(const ui_mdns_server_t *servers, int count) {
150 if (!servers || count <= 0) {
151 // No servers found - return special code
152 // Message will be printed at exit in client main
153 return -1;
154 }
155
156 // Lock terminal to prevent concurrent logging from overwriting TUI
157 bool prev_lock_state = log_lock_terminal();
158
159 // Clear terminal
160 log_plain("%s", ANSI_CLEAR);
161
162 // Display header
163 log_plain("\n");
164 log_plain("%s╭─ 🔍 ascii-chat Server Discovery %s────────────╮%s\n", ANSI_BOLD, ANSI_RESET, ANSI_BOLD);
165 log_plain("│%s\n", ANSI_RESET);
166 log_plain("%s│%s Found %d server%s on your local network:%s\n", ANSI_BOLD, ANSI_GREEN, count, count == 1 ? "" : "s",
167 ANSI_RESET);
168 log_plain("%s│%s\n", ANSI_BOLD, ANSI_RESET);
169
170 // Display server list with formatting
171 for (int i = 0; i < count; i++) {
172 const ui_mdns_server_t *srv = &servers[i];
173 const char *addr = ui_mdns_get_best_address(srv);
174
175 log_plain("%s│%s ", ANSI_BOLD, ANSI_RESET);
176 log_plain("%s[%d]%s %-30s %s%s:%u%s", ANSI_CYAN, i + 1, ANSI_RESET, srv->name, ANSI_YELLOW, addr, srv->port,
177 ANSI_RESET);
178 log_plain("\n");
179 }
180
181 log_plain("%s│%s\n", ANSI_BOLD, ANSI_RESET);
182 log_plain("%s╰────────────────────────────────────────────╯%s\n", ANSI_BOLD, ANSI_RESET);
183
184 // Prompt for selection
185 log_plain("\n");
186 log_plain("Enter server number (1-%d) or press Enter to cancel: ", count);
187 fflush(stdout);
188
189 // Unlock before waiting for input (user might take time)
190 log_unlock_terminal(prev_lock_state);
191
192 // Read user input
193 char input[32];
194 if (fgets(input, sizeof(input), stdin) == NULL) {
195 return -1;
196 }
197
198 // Check for empty input (Enter pressed)
199 if (input[0] == '\n' || input[0] == '\r' || input[0] == '\0') {
200 return -1;
201 }
202
203 // Parse input as number
204 char *endptr;
205 long selection = strtol(input, &endptr, 10);
206
207 // Validate input
208 if (selection < 1 || selection > count) {
209 printf("%sError:%s Please enter a number between 1 and %d\n\n", ANSI_YELLOW, ANSI_RESET, count);
210 return ui_mdns_select(servers, count); // Re-prompt
211 }
212
213 // Lock terminal again for final output
214 prev_lock_state = log_lock_terminal();
215
216 // Clear screen and show connection status
217 log_plain("%s", ANSI_CLEAR);
218 log_plain("\n");
219 log_plain("%s🔗 Connecting to %s...%s\n", ANSI_GREEN, servers[selection - 1].name, ANSI_RESET);
220 log_plain("\n");
221 fflush(stdout);
222
223 // Brief delay so user can see the selection before logs overwrite it
224 platform_sleep_ms(200);
225
226 // Unlock terminal now that TUI is complete
227 log_unlock_terminal(prev_lock_state);
228
229 return (int)(selection - 1); // Convert to 0-based index
230}

235const char *ui_mdns_get_best_address(const ui_mdns_server_t *server) {
236 if (!server) {
237 return "";
238 }
239
240 // Prefer IPv4 > name > IPv6
241 if (server->ipv4[0] != '\0') {
242 return server->ipv4;
243 }
244 if (server->name[0] != '\0') {
245 return server->name;
246 }
247 if (server->ipv6[0] != '\0') {
248 return server->ipv6;
249 }
250
251 return server->address; // Fallback to address field
252}
