104static bool thread_filter_list_append(thread_filter_list_t *list, uint64_t value) {
105 if (list == NULL) {
106 return false;
107 }
108
109 if (list->count == list->capacity) {
110 size_t new_capacity = list->capacity == 0 ? 4 : list->capacity * 2;
111 uint64_t *new_values = SAFE_REALLOC(list->values, new_capacity * sizeof(uint64_t), uint64_t *);
112 if (new_values == NULL) {
113 return false;
114 }
115 list->values = new_values;
116 list->capacity = new_capacity;
117 }
118
119 list->values[list->count++] = value;
120 return true;
121}

123static bool thread_filter_list_contains(const thread_filter_list_t *list, uint64_t value) {
124 if (list == NULL || list->count == 0) {
125 return true;
126 }
127
128 for (size_t i = 0; i < list->count; ++i) {
129 if (list->values[i] == value) {
130 return true;
131 }
132 }
133 return false;
134}

194static bool extract_token(const char *line, const char *key, char **out_value) {
195 const char *position = strstr(line, key);
196 if (position == NULL) {
197 return false;
198 }
199 position += strlen(key);
200 const char *end = strchr(position, ' ');
201 if (end == NULL) {
202 end = line + strlen(line);
203 }
204 size_t length = (size_t)(end - position);
205 *out_value = duplicate_segment(position, length);
206 return *out_value != NULL;
207}

209static bool extract_snippet(const char *line, char **out_value) {
210 const char *position = strstr(line, "snippet=");
211 if (position == NULL) {
212 return false;
213 }
214 position += strlen("snippet=");
215 size_t length = strlen(position);
216 while (length > 0 && (position[length - 1] == '\n' || position[length - 1] == '\r')) {
217 length--;
218 }
219 *out_value = duplicate_segment(position, length);
220 return *out_value != NULL;
221}

223static bool extract_uint64(const char *line, const char *key, uint64_t *out_value) {
224 const char *position = strstr(line, key);
225 if (position == NULL) {
226 return false;
227 }
228 position += strlen(key);
229 unsigned long long value = 0;
230 asciichat_error_t result = parse_ulonglong(position, &value, 0, ULLONG_MAX);
231 if (result != ASCIICHAT_OK) {
232 return false;
233 }
234 *out_value = (uint64_t)value;
235 return true;

238static bool extract_uint32(const char *line, const char *key, uint32_t *out_value) {
239 uint64_t tmp_value = 0;

241 return false;

243 if (tmp_value > UINT32_MAX) {
244 return false;
245 }
246 *out_value = (uint32_t)tmp_value;

250static bool parse_log_line(const char *line, log_record_t *record) {
251 if (!extract_uint64(line, "pid=", &record->pid)) {
252 return false;
253 }
254 if (!extract_uint64(line, "tid=", &record->tid)) {
255 return false;
256 }
257 if (!extract_uint64(line, "seq=", &record->seq)) {
258 return false;
259 }
260 if (!extract_token(line, "ts=", &record->timestamp)) {
261 return false;
262 }
263 if (!extract_token(line, "elapsed=", &record->elapsed)) {
264 return false;
265 }
266 if (!extract_token(line, "file=", &record->file)) {
267 return false;
268 }
269 if (!extract_uint32(line, "line=", &record->line)) {
270 return false;
271 }
272 if (!extract_token(line, "func=", &record->function)) {
273 return false;
274 }
275 if (!extract_uint32(line, "macro=", &record->macro_flag)) {
276 return false;
277 }
278 if (!extract_snippet(line, &record->snippet)) {
279 return false;
280 }
281
282 size_t len = strlen(line);
283 record->raw_line = duplicate_segment(line, len);
284 return record->raw_line != NULL;
285}

287static bool record_matches_filters(const report_config_t *config, const log_record_t *record) {
288 if (config->include_filter != NULL && config->include_filter[0] != '\0') {
289 if (record->file == NULL || strstr(record->file, config->include_filter) == NULL) {
290 return false;
291 }
292 }
293
294 if (config->exclude_filter != NULL && config->exclude_filter[0] != '\0') {
295 if (record->file != NULL && strstr(record->file, config->exclude_filter) != NULL) {
296 return false;
297 }
298 }
299
300 if (!thread_filter_list_contains(&config->threads, record->tid)) {
301 return false;
302 }
303
304 return true;
305}

307static void update_entry(thread_entry_t **entries, const log_record_t *record) {
308 thread_entry_t *entry = NULL;
309 HASH_FIND(hh, *entries, &record->tid, sizeof(uint64_t), entry);
310 if (entry == NULL) {
311 entry = SAFE_CALLOC(1, sizeof(*entry), thread_entry_t *);
312 entry->thread_id = record->tid;
313 entry->record = *record;
314 HASH_ADD(hh, *entries, thread_id, sizeof(uint64_t), entry);
315 return;
316 }
317
318 if (record->seq >= entry->record.seq) {
319 free_record(&entry->record);
320 entry->record = *record;
321 } else {
322 free_record((log_record_t *)record);
323 }
324}

326static int compare_entries(const void *lhs, const void *rhs) {
327 const thread_entry_t *const *a = lhs;
328 const thread_entry_t *const *b = rhs;
329 if ((*a)->thread_id < (*b)->thread_id) {
330 return -1;
331 }
332 if ((*a)->thread_id > (*b)->thread_id) {
333 return 1;
334 }
335 return 0;
336}

338static void print_summary(const report_config_t *config, thread_entry_t **entries) {
339 size_t count = HASH_COUNT(*entries);
340 if (count == 0) {
341 printf("No instrumentation records matched the given filters.\n");
342 return;
343 }
344
345 thread_entry_t **sorted = SAFE_CALLOC(count, sizeof(*sorted), thread_entry_t **);
346 size_t index = 0;
347 thread_entry_t *entry = NULL;
348 thread_entry_t *tmp = NULL;
349 HASH_ITER(hh, *entries, entry, tmp) {
350 sorted[index++] = entry;
351 }
352
353 qsort(sorted, count, sizeof(*sorted), compare_entries);
354
355 printf("Latest instrumentation record per thread (%zu thread%s)\n", count, count == 1 ? "" : "s");
356 printf("======================================================================\n");
357 for (size_t i = 0; i < count; ++i) {
358 const log_record_t *record = &sorted[i]->record;
359 if (config->emit_raw_line) {
360 printf("%s\n", record->raw_line);
361 continue;
362 }
363
364 printf("tid=%" PRIu64 " seq=%" PRIu64 " pid=%" PRIu64 "\n", record->tid, record->seq, record->pid);
365 printf(" timestamp : %s\n", record->timestamp);
366 printf(" elapsed : %s\n", record->elapsed);
367 printf(" location : %s:%u\n", record->file != NULL ? record->file : "<unknown>", record->line);
368 printf(" function : %s\n", record->function != NULL ? record->function : "<unknown>");
369 printf(" macro : %s (%u)\n", macro_flag_label(record->macro_flag), record->macro_flag);
370 printf(" snippet : %s\n", record->snippet != NULL ? record->snippet : "<missing>");
371 printf("----------------------------------------------------------------------\n");
372 }
373
374 SAFE_FREE(sorted);
375}

390// Simple pre-scan for --log-file to avoid corrupted optarg from instrumented getopt
391// This is a workaround for stack corruption caused by panic instrumentation
392static const char *prescan_log_file(int argc, char **argv) {
393 for (int i = 1; i < argc; i++) {
394 if (strcmp(argv[i], "--log-file") == 0 && i + 1 < argc) {
395 return argv[i + 1];
396 }
397 // Handle --log-file=path format
398 if (strncmp(argv[i], "--log-file=", 11) == 0) {
399 return argv[i] + 11;
400 }
401 }
402 return NULL;
403}

417static bool parse_arguments(int argc, char **argv, report_config_t *config) {
418 // Pre-scan for --log-file before getopt (workaround for instrumentation bug)
419 const char *log_file_arg = prescan_log_file(argc, argv);
420 if (log_file_arg != NULL) {
421 config->log_file = log_file_arg;
422 }
423
424 // If just --log-file with no other args, skip getopt entirely
425 // (instrumented getopt corrupts memory)
426 if (config->log_file != NULL && is_simple_log_file_invocation(argc, argv)) {
427 return true;
428 }

430 static const struct option kLongOptions[] = {
431 {"log-file", required_argument, NULL, 'f'}, {"log-dir", required_argument, NULL, 'd'},
432 {"thread", required_argument, NULL, 't'}, {"include", required_argument, NULL, 'i'},
433 {"exclude", required_argument, NULL, 'x'}, {"raw", no_argument, NULL, 'r'},
434 {"help", no_argument, NULL, 'h'}, {0, 0, 0, 0},
435 };
436
437 int option = 0;
438 while ((option = getopt_long(argc, argv, "", kLongOptions, NULL)) != -1) {
439 switch (option) {
440 case 'f':
441 // Already handled by prescan - skip (optarg may be corrupted)
442 break;
443 case 'd':
444 config->log_dir = optarg;
445 break;
446 case 't': {
447 unsigned long long tid_value = 0;
448 asciichat_error_t result = parse_ulonglong(optarg, &tid_value, 0, ULLONG_MAX);
449 if (result != ASCIICHAT_OK) {
450 log_error("Invalid thread id: %s", optarg);
451 return false;
452 }
453 if (!thread_filter_list_append(&config->threads, (uint64_t)tid_value)) {
454 log_error("Failed to record thread filter");
455 return false;
456 }
457 break;
458 }
459 case 'i':
460 config->include_filter = optarg;
461 break;
462 case 'x':
463 config->exclude_filter = optarg;
464 break;
465 case 'r':
466 config->emit_raw_line = true;
467 break;
468 case 'h':
469 usage(stdout, argv[0]);
470 return false;
471 default:
472 usage(stderr, argv[0]);
473 return false;
474 }
475 }
476
477 if (optind < argc) {
478 log_error("Unexpected positional argument: %s", argv[optind]);
479 usage(stderr, argv[0]);
480 return false;
481 }
482
483 // Validate --log-file and --log-dir are mutually exclusive
484 if (config->log_file != NULL && config->log_dir != NULL) {
485 log_error("Cannot specify both --log-file and --log-dir");
486 return false;
487 }
488
489 // Fall back to default log directory only if no log file specified
490 if (config->log_file == NULL && config->log_dir == NULL) {
491 config->log_dir = resolve_default_log_dir();
492 }
493
494 return true;
495}

497static bool process_file(const report_config_t *config, const char *path, thread_entry_t **entries) {
498 FILE *handle = platform_fopen("file_stream", path, "r");
499 if (handle == NULL) {
500 log_warn("Cannot open log file '%s': %s", path, SAFE_STRERROR(errno));
501 return false;
502 }
503
504 char buffer[8192];
505 while (fgets(buffer, sizeof(buffer), handle) != NULL) {
506 log_record_t record = {0};
507 if (!parse_log_line(buffer, &record)) {
508 free_record(&record);
509 continue;
510 }
511 if (!record_matches_filters(config, &record)) {
512 free_record(&record);
513 continue;
514 }
515 update_entry(entries, &record);
516 }
517
518 fclose(handle);
519 return true;
520}

523static bool build_windows_glob_pattern(char *buffer, size_t capacity, const char *directory, const char *suffix) {
524 if (buffer == NULL || directory == NULL || suffix == NULL) {
525 return false;
526 }
527
528 size_t length = 0;
529 for (const char *cursor = directory; *cursor != '\0'; ++cursor) {
530 char ch = *cursor == '/' ? '\\' : *cursor;
531 if (length + 1 >= capacity) {
532 return false;
533 }
534 buffer[length++] = ch;
535 }
536 if (length == 0) {
537 return false;
538 }
539 if (buffer[length - 1] != '\\') {
540 if (length + 1 >= capacity) {
541 return false;
542 }
543 buffer[length++] = '\\';
544 }
545 const size_t suffix_length = strlen(suffix);
546 if (length + suffix_length >= capacity) {
547 return false;
548 }
549 memcpy(buffer + length, suffix, suffix_length + 1);
550 return true;
551}

554static bool collect_entries(const report_config_t *config, thread_entry_t **entries) {
555 // If a single log file is specified, just process that file
556 if (config->log_file != NULL) {
557 return process_file(config, config->log_file, entries);
558 }
559
560#if defined(_WIN32)
561 char pattern[MAX_PATH];
562 if (!build_windows_glob_pattern(pattern, sizeof(pattern), config->log_dir, "ascii-instr-*.log")) {
563 log_error("Instrumentation log directory path is too long: %s", config->log_dir);
564 return false;
565 }
566
567 struct _finddata_t data = {0};
568 intptr_t handle = _findfirst(pattern, &data);
569 if (handle == -1) {
570 log_error("Unable to open instrumentation log directory '%s': %s", config->log_dir, SAFE_STRERROR(errno));
571 return false;
572 }
573
574 bool success = true;
575 do {
576 if ((data.attrib & _A_SUBDIR) != 0) {
577 continue;
578 }
579 size_t name_len = strlen(data.name);
580 if (name_len < 4 || strcmp(data.name + name_len - 4, ".log") != 0) {
581 continue;
582 }
583
584 char path_buffer[MAX_PATH];
585 int written = safe_snprintf(path_buffer, sizeof(path_buffer), "%s/%s", config->log_dir, data.name);
586 if (written < 0 || written >= (int)sizeof(path_buffer)) {
587 log_warn("Skipping path that exceeds buffer: %s/%s", config->log_dir, data.name);
588 continue;
589 }
590 if (!process_file(config, path_buffer, entries)) {
591 success = false;
592 }
593 } while (_findnext(handle, &data) == 0);
594
595 _findclose(handle);
596 return success;
597#else
598 DIR *directory = opendir(config->log_dir);
599 if (directory == NULL) {
600 log_error("Unable to open instrumentation log directory '%s': %s", config->log_dir, SAFE_STRERROR(errno));
601 return false;
602 }
603
604 struct dirent *entry = NULL;
605 while ((entry = readdir(directory)) != NULL) {
606 if (entry->d_name[0] == '.') {
607 continue;
608 }
609 if (strncmp(entry->d_name, "ascii-instr-", strlen("ascii-instr-")) != 0) {
610 continue;
611 }
612 size_t name_length = strlen(entry->d_name);
613 if (name_length < 5 || strcmp(entry->d_name + name_length - 4, ".log") != 0) {
614 continue;
615 }
616
617 char path_buffer[PATH_MAX];
618 int written = safe_snprintf(path_buffer, sizeof(path_buffer), "%s/%s", config->log_dir, entry->d_name);
619 if (written < 0 || written >= (int)sizeof(path_buffer)) {
620 log_warn("Skipping path that exceeds buffer: %s/%s", config->log_dir, entry->d_name);
621 continue;
622 }
623
624 process_file(config, path_buffer, entries);
625 }
626
627 closedir(directory);
628 return true;
629#endif
630}

632int main(int argc, char **argv) {
633 report_config_t config = {
634 .log_dir = NULL,
635 .log_file = NULL,
636 .include_filter = NULL,
637 .exclude_filter = NULL,

658 destroy_entries(&entries);

662 log_destroy();
