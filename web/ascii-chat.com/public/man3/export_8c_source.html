100 // Parse OpenPGP packet to extract Ed25519 public key
101 // OpenPGP public key packet format (simplified):
102 // - Packet tag (1 byte): 0x99 for public key packet (old format) or 0xC6 (new format)
103 // - Packet length (variable)
104 // - Version (1 byte): 0x04 for modern keys
105 // - Creation time (4 bytes)
106 // - Algorithm (1 byte): 22 (0x16) for EdDSA
107 // - Curve OID length + OID
108 // - Public key material (MPI format)

110 size_t offset = 0;
111
112 // Skip to public key packet (tag 6 - public key, or tag 14 - public subkey)
113 while (offset < bytes_read) {
114 uint8_t tag = packet_data[offset];
115
116 // Check if this is a public key packet (old or new format)
117 bool is_public_key = false;
118 size_t packet_len = 0;
119
120 if ((tag & 0x80) == 0) {
121 // Not a valid packet tag
122 offset++;
123 continue;
124 }
125
126 if ((tag & 0x40) == 0) {
127 // Old format packet
128 uint8_t packet_type = (tag >> 2) & 0x0F;
129 is_public_key = (packet_type == 6 || packet_type == 14); // Public key or subkey
130
131 uint8_t length_type = tag & 0x03;
132 offset++; // Move past tag
133
134 if (length_type == 0) {
135 packet_len = packet_data[offset++];
136 } else if (length_type == 1) {
137 packet_len = (packet_data[offset] << 8) | packet_data[offset + 1];
138 offset += 2;
139 } else if (length_type == 2) {
140 packet_len = (packet_data[offset] << 24) | (packet_data[offset + 1] << 16) | (packet_data[offset + 2] << 8) |
141 packet_data[offset + 3];
142 offset += 4;
143 } else {
144 // Indeterminate length - skip
145 break;
146 }
147 } else {
148 // New format packet
149 uint8_t packet_type = tag & 0x3F;
150 is_public_key = (packet_type == 6 || packet_type == 14); // Public key or subkey
151 offset++; // Move past tag
152
153 // Parse new format length
154 if (offset >= bytes_read)
155 break;
156 uint8_t first_len = packet_data[offset++];
157
158 if (first_len < 192) {
159 packet_len = first_len;
160 } else if (first_len < 224) {
161 if (offset >= bytes_read)
162 break;
163 packet_len = ((first_len - 192) << 8) + packet_data[offset++] + 192;
164 } else if (first_len == 255) {
165 if (offset + 4 > bytes_read)
166 break;
167 packet_len = (packet_data[offset] << 24) | (packet_data[offset + 1] << 16) | (packet_data[offset + 2] << 8) |
168 packet_data[offset + 3];
169 offset += 4;
170 } else {
171 // Partial body length - not expected for key packets
172 break;
173 }
174 }
175
176 if (!is_public_key || packet_len == 0 || offset + packet_len > bytes_read) {
177 offset += packet_len;
178 continue;
179 }
180
181 // Parse the public key packet content
182 size_t packet_start = offset;
183
184 // Check version (should be 4)
185 if (packet_data[offset] != 0x04) {
186 offset += packet_len;
187 continue;
188 }
189 offset++; // Skip version
190
191 offset += 4; // Skip creation time
192
193 // Check algorithm (22 = EdDSA/Ed25519)
194 if (offset >= packet_start + packet_len) {
195 offset = packet_start + packet_len;
196 continue;
197 }
198
199 uint8_t algorithm = packet_data[offset++];
200 if (algorithm != 22) { // Not EdDSA
201 offset = packet_start + packet_len;
202 continue;
203 }
204
205 // Skip curve OID (should be Ed25519 OID)
206 if (offset >= packet_start + packet_len) {
207 offset = packet_start + packet_len;
208 continue;
209 }
210
211 uint8_t oid_len = packet_data[offset++];
212 offset += oid_len; // Skip OID bytes
213
214 // Now we should have the MPI-encoded public key
215 // MPI format: 2-byte bit count, then key data
216 if (offset + 2 > packet_start + packet_len) {
217 offset = packet_start + packet_len;
218 continue;
219 }
220
221 uint16_t mpi_bits = (packet_data[offset] << 8) | packet_data[offset + 1];
222 offset += 2;
223
224 // Ed25519 public keys should be 263 bits (0x0107) - includes 0x40 prefix byte
225 // Or 256 bits for just the key without prefix
226 size_t mpi_bytes = (mpi_bits + 7) / 8;
227
228 if (offset + mpi_bytes > packet_start + packet_len) {
229 offset = packet_start + packet_len;
230 continue;
231 }
232
233 // Ed25519 keys in OpenPGP have a 0x40 prefix byte
234 if (mpi_bytes == 33 && packet_data[offset] == 0x40) {
235 // Found it! Extract the 32-byte public key (skip 0x40 prefix)
236 memcpy(public_key_out, &packet_data[offset + 1], 32);
237 log_debug("Extracted Ed25519 public key from gpg --export (fallback method)");
238 return 0;
239 } else if (mpi_bytes == 32) {
240 // Key without prefix (less common but valid)
241 memcpy(public_key_out, &packet_data[offset], 32);
242 log_debug("Extracted Ed25519 public key from gpg --export (fallback method)");
243 return 0;
244 }
245
246 offset = packet_start + packet_len;
247 }
