231static int audio_queue_packet(const uint8_t *opus_data, size_t opus_size, const uint16_t *frame_sizes,
232 int frame_count) {
233 if (!lifecycle_is_initialized(&g_audio_send_queue_lc) || !opus_data || opus_size == 0) {
234 return -1;
235 }
236
237 mutex_lock(&g_audio_send_queue_mutex);
238
239 // Check if queue is full
240 int next_head = (g_audio_send_queue_head + 1) % AUDIO_SEND_QUEUE_SIZE;
241 if (next_head == g_audio_send_queue_tail) {
242 mutex_unlock(&g_audio_send_queue_mutex);
243 log_warn_every(LOG_RATE_FAST, "Audio send queue full, dropping packet");
244 return -1;
245 }
246
247 // Copy packet to queue
248 audio_send_packet_t *packet = &g_audio_send_queue[g_audio_send_queue_head];
249 if (opus_size > sizeof(packet->data)) {
250 mutex_unlock(&g_audio_send_queue_mutex);
251 log_warn("Audio packet too large for queue: %zu > %zu bytes", opus_size, sizeof(packet->data));
252 return -1;
253 }
254
255 memcpy(packet->data, opus_data, opus_size);
256 packet->size = opus_size;
257 packet->frame_count = frame_count;
258 for (int i = 0; i < frame_count && i < 8; i++) {
259 packet->frame_sizes[i] = frame_sizes[i];
260 }
261 g_audio_send_queue_head = next_head;
262
263 // Signal sender thread (must be held while signaling)
264 cond_signal(&g_audio_send_queue_cond);
265 mutex_unlock(&g_audio_send_queue_mutex);
266
267 return 0;
268}

276static void *audio_sender_thread_func(void *arg) {
277 (void)arg;
278 log_debug("Audio sender thread started");
279
280 static int send_count = 0;
281
282 while (!atomic_load_bool(&g_audio_sender_should_exit)) {
283 mutex_lock(&g_audio_send_queue_mutex);
284
285 // Wait for packet or exit signal (cond_wait re-acquires mutex before returning)
286 while (g_audio_send_queue_head == g_audio_send_queue_tail && !atomic_load_bool(&g_audio_sender_should_exit)) {
287 cond_wait(&g_audio_send_queue_cond, &g_audio_send_queue_mutex);
288 }
289
290 // Check exit flag after waking from cond_wait
291 if (atomic_load_bool(&g_audio_sender_should_exit)) {
292 mutex_unlock(&g_audio_send_queue_mutex);
293 break;
294 }
295
296 // Safety check: ensure queue actually has data (handle spurious wakeups)
297 if (g_audio_send_queue_head == g_audio_send_queue_tail) {
298 // Spurious wakeup - queue is empty, go back to waiting
299 mutex_unlock(&g_audio_send_queue_mutex);
300 continue;
301 }
302
303 // Dequeue packet
304 audio_send_packet_t packet = g_audio_send_queue[g_audio_send_queue_tail];
305 g_audio_send_queue_tail = (g_audio_send_queue_tail + 1) % AUDIO_SEND_QUEUE_SIZE;
306
307 mutex_unlock(&g_audio_send_queue_mutex);
308
309 // Send packet (may block on network I/O - that's OK, we're not in capture thread)
310 START_TIMER("network_send_audio");
311 asciichat_error_t send_result =
312 threaded_send_audio_opus_batch(packet.data, packet.size, packet.frame_sizes, packet.frame_count);
313 double send_time_ns = STOP_TIMER("network_send_audio");
314
315 send_count++;
316 if (send_result < 0) {
317 log_debug_every(LOG_RATE_VERY_FAST, "Failed to send audio packet");
318 } else if (send_count % 50 == 0) {
319 char duration_str[32];
320 time_pretty((uint64_t)(send_time_ns), -1, duration_str, sizeof(duration_str));
321 log_debug("Audio network send #%d: %zu bytes (%d frames) in %s", send_count, packet.size, packet.frame_count,
322 duration_str);
323 }
324 }
325
326 // Clean up thread-local error context before exit
327 asciichat_errno_destroy();
328
329 // Signal that audio sender thread has exited
330 atomic_store_bool(&g_audio_sender_exited, true);
331
332 return NULL;
333}

402 cond_destroy(&g_audio_send_queue_cond);

431void audio_process_received_samples(const float *samples, int num_samples) {
432 // Validate parameters
433 if (!samples || num_samples <= 0) {
434 SET_ERRNO(ERROR_INVALID_PARAM, "Invalid audio samples: samples=%p, num_samples=%d", (void *)samples, num_samples);
435 return;
436 }
437
438 if (!GET_OPTION(audio_enabled)) {
439 log_warn_every(NS_PER_MS_INT, "Received audio samples but audio is disabled");
440 return;
441 }
442
443 // Allow both single packets and batched packets
444 if (num_samples > AUDIO_BATCH_SAMPLES) {
445 log_warn("Audio packet too large: %d samples (max %d)", num_samples, AUDIO_BATCH_SAMPLES);
446 return;
447 }
448
449 // Calculate RMS energy of received samples
450 float sum_squares = 0.0f;
451 for (int i = 0; i < num_samples; i++) {
452 sum_squares += samples[i] * samples[i];
453 }
454 float received_rms = sqrtf(sum_squares / num_samples);
455
456 // DUMP: Received audio from server (before playback processing)
457 if (g_wav_playback_received) {
458 wav_writer_write(g_wav_playback_received, samples, num_samples);
459 }
460
461 // Track samples for analysis
462 if (GET_OPTION(audio_analysis_enabled)) {
463 for (int i = 0; i < num_samples; i++) {
464 audio_analysis_track_received_sample(samples[i]);
465 }
466 }
467
468 // Copy samples to playback buffer (no processing needed - mixer already handled clipping)
469 float audio_buffer[AUDIO_BATCH_SAMPLES];
470 memcpy(audio_buffer, samples, (size_t)num_samples * sizeof(float));
471
472 // DEBUG: Log what we're writing to playback buffer (with first 4 samples to verify audio integrity)
473 static int recv_count = 0;
474 recv_count++;
475 if (recv_count <= 10 || recv_count % 50 == 0) {
476 float peak = 0.0f;
477 for (int i = 0; i < num_samples; i++) {
478 float abs_val = fabsf(samples[i]);
479 if (abs_val > peak)
480 peak = abs_val;
481 }
482 log_debug("CLIENT AUDIO RECV #%d: %d samples, RMS=%.6f, Peak=%.6f, first4=[%.4f,%.4f,%.4f,%.4f]", recv_count,
483 num_samples, received_rms, peak, num_samples > 0 ? samples[0] : 0.0f, num_samples > 1 ? samples[1] : 0.0f,
484 num_samples > 2 ? samples[2] : 0.0f, num_samples > 3 ? samples[3] : 0.0f);
485 }
486
487 // Submit to playback system (goes to jitter buffer and speakers)
488 // NOTE: AEC3's AnalyzeRender is called in output_callback() when audio actually plays,
489 // NOT here. The jitter buffer adds 50-100ms delay, so calling AnalyzeRender here
490 // would give AEC3 the wrong timing and break echo cancellation.
491 audio_write_samples(&g_audio_context, audio_buffer, num_samples);
492
493 // Log latency after writing to playback buffer
494 if (g_audio_context.playback_buffer) {
495 size_t buffer_samples = audio_ring_buffer_available_read(g_audio_context.playback_buffer);
496 float buffer_latency_ms = (float)buffer_samples / 48.0f;
497 log_dev_every(500 * US_PER_MS_INT, "LATENCY: Client playback buffer after recv: %.1fms (%zu samples)",
498 buffer_latency_ms, buffer_samples);
499 }
500
501#ifdef DEBUG_AUDIO
502 log_debug("Processed %d received audio samples", num_samples);
503#endif
504}

531static void *audio_capture_thread_func(void *arg) {
532 (void)arg;
533
534 log_debug("Audio capture thread started");
535
536 // FPS tracking for audio capture thread (tracking Opus frames, ~50 FPS at 20ms per frame)
537 static fps_t fps_tracker = {0};
538 static bool fps_tracker_initialized = false;
539 if (!fps_tracker_initialized) {
540 fps_init(&fps_tracker, 50, "AUDIO_TX");
541 fps_tracker_initialized = true;
542 }
543
544 // Detailed timing stats
545 static double total_loop_ns = 0;
546 static double total_read_ns = 0;
547 static double total_encode_ns = 0;
548 static double total_queue_ns = 0;
549 static double max_loop_ns = 0;
550 static double max_read_ns = 0;
551 static double max_encode_ns = 0;
552 static double max_queue_ns = 0;
553 static uint64_t timing_loop_count = 0;
554
555// Opus frame size: 960 samples = 20ms @ 48kHz (must match pipeline config)
556#define OPUS_FRAME_SAMPLES 960
557#define OPUS_MAX_PACKET_SIZE 500 // Max Opus packet size
558
559 // Read enough samples per iteration to drain faster than we fill
560 // Buffer holds multiple Opus frames worth to prevent overflow
561 // 4 frames = 3840 samples = 80ms, but we'll read what's available up to this
562#define CAPTURE_READ_SIZE (OPUS_FRAME_SAMPLES * 4)
563
564 float audio_buffer[CAPTURE_READ_SIZE];
565 static bool wav_dumpers_initialized = false;
566
567 // Initialize WAV dumpers only once (file handles persist)
568 if (!wav_dumpers_initialized && wav_dump_enabled()) {
569 g_wav_capture_raw = wav_writer_open("/tmp/audio_capture_raw.wav", AUDIO_SAMPLE_RATE, 1);
570 g_wav_capture_processed = wav_writer_open("/tmp/audio_capture_processed.wav", AUDIO_SAMPLE_RATE, 1);
571 log_debug("Audio debugging enabled: dumping to /tmp/audio_capture_*.wav");
572 wav_dumpers_initialized = true;
573 }
574
575 // Accumulator for building complete Opus frames
576 float opus_frame_buffer[OPUS_FRAME_SAMPLES];
577 int opus_frame_samples_collected = 0;
578
579 // Batch buffer for multiple Opus frames - send all at once to reduce blocking
580#define MAX_BATCH_FRAMES 8
581#define BATCH_TIMEOUT_NS (40LL * NS_PER_MS_INT) // Flush batch after 40ms even if not full (2 Opus frames @ 20ms each)
582 static uint8_t batch_buffer[MAX_BATCH_FRAMES * OPUS_MAX_PACKET_SIZE];
583 static uint16_t batch_frame_sizes[MAX_BATCH_FRAMES];
584 static int batch_frame_count = 0;
585 static size_t batch_total_size = 0;
586 static uint64_t batch_start_time_ns = 0;
587 static bool batch_has_data = false;
588
589 while (!should_exit() && !server_connection_is_lost()) {
590 START_TIMER("audio_capture_loop_iteration");
591 timing_loop_count++;
592
593 if (!server_connection_is_active()) {
594 STOP_TIMER("audio_capture_loop_iteration"); // Don't count sleep time
595 platform_sleep_us(100 * US_PER_MS_INT); // Wait for connection
596 continue;
597 }
598
599 // Check if pipeline is ready
600 if (!g_audio_pipeline) {
601 STOP_TIMER("audio_capture_loop_iteration"); // Don't count sleep time
602 platform_sleep_us(100 * US_PER_MS_INT);
603 continue;
604 }
605
606 // Check how many samples are available in the ring buffer
607 int available = audio_ring_buffer_available_read(g_audio_context.capture_buffer);
608 if (available <= 0) {
609 // Flush partial batch before sleeping (prevent starvation during idle periods)
610 if (batch_has_data && batch_frame_count > 0) {
611 uint64_t now_ns = time_get_ns();
612 uint64_t elapsed_ns = time_elapsed_ns(batch_start_time_ns, now_ns);
613
614 if (elapsed_ns >= BATCH_TIMEOUT_NS) {
615 long elapsed_ms = (long)time_ns_to_ms(elapsed_ns);
616 log_debug_every(LOG_RATE_FAST, "Idle timeout flush: %d frames (%zu bytes) after %ld ms", batch_frame_count,
617 batch_total_size, elapsed_ms);
618 (void)audio_queue_packet(batch_buffer, batch_total_size, batch_frame_sizes, batch_frame_count);
619 batch_frame_count = 0;
620 batch_total_size = 0;
621 batch_has_data = false;
622 }
623 }
624
625 // Sleep briefly to reduce CPU usage when idle
626 // 5ms polling = 200 times/sec, fast enough to catch audio promptly
627 // Note: 50ms was causing 872ms gaps in audio transmission!
628 STOP_TIMER("audio_capture_loop_iteration"); // Must stop before loop repeats
629 platform_sleep_us(5 * US_PER_MS_INT); // 5ms (was 50ms - caused huge gaps!)
630 continue;
631 }
632
633 // Read as many samples as possible (up to CAPTURE_READ_SIZE) to drain faster
634 // This prevents buffer overflow when processing is slower than capture
635 int to_read = (available < CAPTURE_READ_SIZE) ? available : CAPTURE_READ_SIZE;
636
637 START_TIMER("audio_read_samples");
638 asciichat_error_t read_result = audio_read_samples(&g_audio_context, audio_buffer, to_read);
639 double read_time_ns = STOP_TIMER("audio_read_samples");
640
641 total_read_ns += read_time_ns;
642 if (read_time_ns > max_read_ns)
643 max_read_ns = read_time_ns;
644
645 if (read_result != ASCIICHAT_OK) {
646 log_error("Failed to read audio samples from ring buffer");
647 STOP_TIMER("audio_capture_loop_iteration"); // Don't count sleep time
648 platform_sleep_us(5 * US_PER_MS_INT); // 5ms (error path - was 50ms, caused gaps!)
649 continue;
650 }
651
652 int samples_read = to_read;
653
654 // Log every 10 reads to see if we're getting samples
655 static int total_reads = 0;
656 total_reads++;
657 if (total_reads % 10 == 0) {
658 log_debug("Audio capture loop iteration #%d: available=%d, samples_read=%d", total_reads, available,
659 samples_read);
660 }
661
662 if (samples_read > 0) {
663 // Normalize input to prevent clipping: bring peak to ±0.99
664 // Calculate peak level first
665 float peak = 0.0f;
666 for (int i = 0; i < samples_read; i++) {
667 float abs_val = fabsf(audio_buffer[i]);
668 if (abs_val > peak)
669 peak = abs_val;
670 }
671
672 // Apply normalization if peak exceeds 1.0
673 // Use 0.99 to leave headroom for processing
674 if (peak > 1.0f) {
675 float gain = 0.99f / peak;
676 for (int i = 0; i < samples_read; i++) {
677 audio_buffer[i] *= gain;
678 }
679 static int norm_count = 0;
680 norm_count++;
681 if (norm_count <= 5 || norm_count % 100 == 0) {
682 log_debug("Input normalization #%d: peak=%.4f, gain=%.4f", norm_count, peak, gain);
683 }
684 }
685
686 // DUMP: Raw captured audio (before any processing)
687 if (g_wav_capture_raw) {
688 wav_writer_write(g_wav_capture_raw, audio_buffer, samples_read);
689 }
690
691 // DEBUG: Log EVERY read to see what we're getting from the ring buffer
692 static int read_count = 0;
693 read_count++;
694 float sum_squares = 0.0f;
695 for (int i = 0; i < samples_read && i < 10; i++) {
696 sum_squares += audio_buffer[i] * audio_buffer[i];
697 }
698 float rms = sqrtf(sum_squares / (samples_read > 10 ? 10 : samples_read));
699 if (read_count <= 5 || read_count % 20 == 0) {
700 log_debug("Audio capture read #%d: available=%d, samples_read=%d, first=[%.6f,%.6f,%.6f], RMS=%.6f", read_count,
701 available, samples_read, samples_read > 0 ? audio_buffer[0] : 0.0f,
702 samples_read > 1 ? audio_buffer[1] : 0.0f, samples_read > 2 ? audio_buffer[2] : 0.0f, rms);
703 }
704
705 // Track sent samples for analysis
706 if (GET_OPTION(audio_analysis_enabled)) {
707 for (int i = 0; i < samples_read; i++) {
708 audio_analysis_track_sent_sample(audio_buffer[i]);
709 }
710 }
711
712 // Accumulate samples into Opus frame buffer
713 int samples_to_process = samples_read;
714 int sample_offset = 0;
715
716 while (samples_to_process > 0) {
717 // How many samples can we add to current frame?
718 int space_in_frame = OPUS_FRAME_SAMPLES - opus_frame_samples_collected;
719 int samples_to_copy = (samples_to_process < space_in_frame) ? samples_to_process : space_in_frame;
720
721 // Copy samples to frame buffer
722 memcpy(&opus_frame_buffer[opus_frame_samples_collected], &audio_buffer[sample_offset],
723 (size_t)samples_to_copy * sizeof(float));
724
725 opus_frame_samples_collected += samples_to_copy;
726 sample_offset += samples_to_copy;
727 samples_to_process -= samples_to_copy;
728
729 // Do we have a complete frame?
730 if (opus_frame_samples_collected >= OPUS_FRAME_SAMPLES) {
731 // Process through pipeline: AEC, filters, AGC, noise gate, Opus encode
732 uint8_t opus_packet[OPUS_MAX_PACKET_SIZE];
733
734 START_TIMER("opus_encode");
735 int opus_len = client_audio_pipeline_capture(g_audio_pipeline, opus_frame_buffer, OPUS_FRAME_SAMPLES,
736 opus_packet, OPUS_MAX_PACKET_SIZE);
737
738 static int encode_count = 0;
739 encode_count++;
740 double opus_elapsed_ns = STOP_TIMER("opus_encode");
741 if (encode_count % 50 == 0) {
742 if (opus_elapsed_ns >= 0.0) {
743 char _duration_str[32];
744 time_pretty((uint64_t)(opus_elapsed_ns), -1, _duration_str, sizeof(_duration_str));
745 log_dev("Opus encode #%d: %d samples -> %d bytes in %s", encode_count, OPUS_FRAME_SAMPLES, opus_len,
746 _duration_str);
747 }
748 }
749
750 double encode_time_ns = 0; // Timing already logged
751
752 total_encode_ns += encode_time_ns;
753 if (encode_time_ns > max_encode_ns)
754 max_encode_ns = encode_time_ns;
755
756 if (opus_len > 0) {
757
758 log_debug_every(LOG_RATE_VERY_FAST, "Pipeline encoded: %d samples -> %d bytes (compression: %.1fx)",
759 OPUS_FRAME_SAMPLES, opus_len,
760 (float)(OPUS_FRAME_SAMPLES * sizeof(float)) / (float)opus_len);
761
762 // Add to batch buffer
763 if (batch_frame_count < MAX_BATCH_FRAMES && batch_total_size + (size_t)opus_len <= sizeof(batch_buffer)) {
764 // Mark batch start time on first frame
765 if (batch_frame_count == 0) {
766 batch_start_time_ns = time_get_ns();
767 batch_has_data = true;
768 }
769
770 memcpy(batch_buffer + batch_total_size, opus_packet, (size_t)opus_len);
771 batch_frame_sizes[batch_frame_count] = (uint16_t)opus_len;
772 batch_total_size += (size_t)opus_len;
773 batch_frame_count++;
774
775 if (GET_OPTION(audio_analysis_enabled)) {
776 audio_analysis_track_sent_packet((size_t)opus_len);
777 }
778 }
779 } else if (opus_len == 0) {
780 // DTX frame (silence) - no data to send
781 log_debug_every(LOG_RATE_VERY_FAST, "Pipeline DTX frame (silence detected)");
782 }
783
784 // Reset frame buffer
785 opus_frame_samples_collected = 0;
786 }
787 }
788
789 // Queue batch for async sending (non-blocking - sender thread handles network I/O)
790 if (batch_frame_count > 0) {
791 static int batch_send_count = 0;
792 batch_send_count++;
793
794 START_TIMER("audio_queue_packet");
795 int queue_result = audio_queue_packet(batch_buffer, batch_total_size, batch_frame_sizes, batch_frame_count);
796 double queue_time_ns = STOP_TIMER("audio_queue_packet");
797
798 total_queue_ns += queue_time_ns;
799 if (queue_time_ns > max_queue_ns)
800 max_queue_ns = queue_time_ns;
801
802 if (queue_result < 0) {
803 log_debug_every(LOG_RATE_VERY_FAST, "Failed to queue audio batch (queue full)");
804 } else {
805 if (batch_send_count <= 10 || batch_send_count % 50 == 0) {
806 char queue_duration_str[32];
807 time_pretty((uint64_t)(queue_time_ns), -1, queue_duration_str, sizeof(queue_duration_str));
808 log_debug("CLIENT: Queued Opus batch #%d (%d frames, %zu bytes) in %s", batch_send_count, batch_frame_count,
809 batch_total_size, queue_duration_str);
810 }
811 // Track audio frame for FPS reporting
812 fps_frame_ns(&fps_tracker, time_get_ns(), "audio batch queued");
813 }
814
815 // Reset batch
816 batch_frame_count = 0;
817 batch_total_size = 0;
818 batch_has_data = false;
819 }
820
821 // Log overall loop iteration time periodically
822 double loop_time_ns = STOP_TIMER("audio_capture_loop_iteration");
823 total_loop_ns += loop_time_ns;
824 if (loop_time_ns > max_loop_ns)
825 max_loop_ns = loop_time_ns;
826
827 // Comprehensive timing report every 100 iterations (~2 seconds)
828 if (timing_loop_count % 100 == 0) {
829 char avg_loop_str[32], max_loop_str[32];
830 char avg_read_str[32], max_read_str[32];
831 char avg_encode_str[32], max_encode_str[32];
832 char avg_queue_str[32], max_queue_str[32];
833
834 time_pretty((uint64_t)(total_loop_ns / timing_loop_count), -1, avg_loop_str, sizeof(avg_loop_str));
835 time_pretty((uint64_t)(max_loop_ns), -1, max_loop_str, sizeof(max_loop_str));
836 time_pretty((uint64_t)(total_read_ns / timing_loop_count), -1, avg_read_str, sizeof(avg_read_str));
837 time_pretty((uint64_t)(max_read_ns), -1, max_read_str, sizeof(max_read_str));
838 time_pretty((uint64_t)(total_encode_ns / timing_loop_count), -1, avg_encode_str, sizeof(avg_encode_str));
839 time_pretty((uint64_t)(max_encode_ns), -1, max_encode_str, sizeof(max_encode_str));
840 time_pretty((uint64_t)(total_queue_ns / timing_loop_count), -1, avg_queue_str, sizeof(avg_queue_str));
841 time_pretty((uint64_t)(max_queue_ns), -1, max_queue_str, sizeof(max_queue_str));
842
843 log_debug("CAPTURE TIMING #%lu: loop avg=%s max=%s, read avg=%s max=%s", timing_loop_count, avg_loop_str,
844 max_loop_str, avg_read_str, max_read_str);
845 log_info(" encode avg=%s max=%s, queue avg=%s max=%s", avg_encode_str, max_encode_str, avg_queue_str,
846 max_queue_str);
847 }
848
849 // Check if we have a partial batch that's been waiting too long (time-based flush)
850 // This prevents batches from sitting indefinitely when audio capture is irregular
851 if (batch_has_data && batch_frame_count > 0) {
852 uint64_t now_ns = time_get_ns();
853 uint64_t elapsed_ns = time_elapsed_ns(batch_start_time_ns, now_ns);
854
855 if (elapsed_ns >= BATCH_TIMEOUT_NS) {
856 static int timeout_flush_count = 0;
857 timeout_flush_count++;
858
859 long elapsed_ms = (long)time_ns_to_ms(elapsed_ns);
860 log_debug_every(LOG_RATE_FAST, "Timeout flush #%d: %d frames (%zu bytes) after %ld ms", timeout_flush_count,
861 batch_frame_count, batch_total_size, elapsed_ms);
862
863 // Queue partial batch
864 int queue_result = audio_queue_packet(batch_buffer, batch_total_size, batch_frame_sizes, batch_frame_count);
865 if (queue_result == 0) {
866 // Track audio frame for FPS reporting
867 fps_frame_ns(&fps_tracker, time_get_ns(), "audio batch timeout flush");
868 }
869
870 // Reset batch
871 batch_frame_count = 0;
872 batch_total_size = 0;
873 batch_has_data = false;
874 }
875 }
876
877 // Yield to reduce CPU usage - audio arrives at ~20ms per Opus frame (960 samples @ 48kHz)
878 // Without sleep, thread spins at 90-100% CPU constantly checking for new samples
879 // Even 1ms sleep reduces CPU usage from 90% to <10% with minimal latency impact
880 platform_sleep_us(1 * US_PER_MS_INT); // 1ms
881 } else {
882 // Track loop time even when no samples processed
883 double loop_time_ns = STOP_TIMER("audio_capture_loop_iteration");
884 total_loop_ns += loop_time_ns;
885 if (loop_time_ns > max_loop_ns)
886 max_loop_ns = loop_time_ns;
887
888 // Flush partial batch before sleeping on error path (prevent starvation)
889 if (batch_has_data && batch_frame_count > 0) {
890 uint64_t now_ns = time_get_ns();
891 uint64_t elapsed_ns = time_elapsed_ns(batch_start_time_ns, now_ns);
892
893 if (elapsed_ns >= BATCH_TIMEOUT_NS) {
894 long elapsed_ms = (long)time_ns_to_ms(elapsed_ns);
895 log_debug_every(LOG_RATE_FAST, "Error path timeout flush: %d frames (%zu bytes) after %ld ms",
896 batch_frame_count, batch_total_size, elapsed_ms);
897 (void)audio_queue_packet(batch_buffer, batch_total_size, batch_frame_sizes, batch_frame_count);
898 batch_frame_count = 0;
899 batch_total_size = 0;
900 batch_has_data = false;
901 }
902 }
903
904 platform_sleep_us(5 * US_PER_MS_INT); // 5ms (error path - was 50ms, caused gaps!)
905 }
906 }
907
908 log_debug("Audio capture thread stopped");
909 atomic_store_bool(&g_audio_capture_thread_exited, true);
910
911 // Clean up thread-local error context before exit
912 asciichat_errno_destroy();
913
914 return NULL;
915}

931int audio_client_init() {
932 if (!GET_OPTION(audio_enabled)) {
933 return 0; // Audio disabled - not an error
934 }
935
936 // Use lifecycle to ensure only one thread initializes audio at a time
937 // This prevents race conditions in worker thread creation
938 if (!lifecycle_init_once(&g_audio_client_init_lc)) {
939 // Another thread is already initializing, wait for it to complete
940 while (!lifecycle_is_initialized(&g_audio_client_init_lc)) {
941 platform_sleep_us(1000); // 1ms
942 }
943 return 0; // Already initialized by another thread
944 }
945
946 // Clean up previous audio context if it was already initialized
947 // This prevents leaks and thread creation issues on reconnection
948 if (g_audio_context.initialized) {
949 log_debug("Cleaning up previous audio context before reinitializing");
950 audio_destroy(&g_audio_context);
951 // NOTE: Do NOT reset lifecycle - keep it INITIALIZED so only ONE global
952 // audio context is ever created. Subsequent calls will return early.
953 }
954
955 // Initialize WAV dumper for received audio if debugging enabled
956 if (wav_dump_enabled()) {
957 g_wav_playback_received = wav_writer_open("/tmp/audio_playback_received.wav", AUDIO_SAMPLE_RATE, 1);
958 if (g_wav_playback_received) {
959 log_debug("Audio debugging enabled: dumping received audio to /tmp/audio_playback_received.wav");
960 }
961 }
962
963 // Initialize PortAudio context using library function
964 log_debug("DEBUG: About to call audio_init()...");
965 if (audio_init(&g_audio_context) != ASCIICHAT_OK) {
966 log_error("Failed to initialize audio system");
967 // Clean up WAV writer if it was opened
968 if (g_wav_playback_received) {
969 wav_writer_close(g_wav_playback_received);
970 g_wav_playback_received = NULL;
971 }
972 lifecycle_init_abort(&g_audio_client_init_lc);
973 return -1;
974 }
975 log_debug("DEBUG: audio_init() completed successfully");
976
977 lifecycle_init_commit(&g_audio_client_init_lc);
978
979 // Create unified audio pipeline (handles AEC, AGC, noise suppression, Opus)
980 client_audio_pipeline_config_t pipeline_config = client_audio_pipeline_default_config();
981 pipeline_config.opus_bitrate = 128000; // 128 kbps AUDIO mode for music quality
982
983 // Enable echo cancellation, AGC, and essential processing for clear audio
984 // Noise suppression and VAD can destroy music quality, so keep them disabled
985 pipeline_config.flags.echo_cancel = true; // ENABLE: removes echo
986 pipeline_config.flags.jitter_buffer = true; // ENABLE: needed for AEC sync
987 pipeline_config.flags.noise_suppress = false; // DISABLED: destroys music quality
988 pipeline_config.flags.agc = true; // ENABLE: boost quiet microphones (35 dB gain)
989 pipeline_config.flags.vad = false; // DISABLED: destroys music quality
990 pipeline_config.flags.compressor = true; // ENABLE: prevent clipping from AGC boost
991 pipeline_config.flags.noise_gate = false; // DISABLED: would cut quiet music passages
992 pipeline_config.flags.highpass = true; // ENABLE: remove rumble and low-frequency feedback
993 pipeline_config.flags.lowpass = false; // DISABLED: preserve high-frequency content
994
995 // Set jitter buffer margin for smooth playback without excessive delay
996 // 100ms is conservative - AEC3 will adapt to actual network delay automatically
997 // We don't tune this; let the system adapt to its actual conditions
998 pipeline_config.jitter_margin_ns = 100;
999
1000 log_debug("DEBUG: About to create audio pipeline...");
1001 g_audio_pipeline = client_audio_pipeline_create(&pipeline_config);
1002 log_debug("DEBUG: client_audio_pipeline_create() returned");
1003 if (!g_audio_pipeline) {
1004 log_error("Failed to create audio pipeline");
1005 audio_destroy(&g_audio_context);
1006 // Clean up WAV writer if it was opened
1007 if (g_wav_playback_received) {
1008 wav_writer_close(g_wav_playback_received);
1009 g_wav_playback_received = NULL;
1010 }
1011 return -1;
1012 }
1013
1014 log_debug("Audio pipeline created: %d Hz sample rate, %d bps bitrate", pipeline_config.sample_rate,
1015 pipeline_config.opus_bitrate);
1016
1017 // Associate pipeline with audio context for echo cancellation
1018 // The audio output callback will feed playback samples directly to AEC3 from the speaker output,
1019 // ensuring proper timing synchronization (not from the decode path 50-100ms earlier)
1020 audio_set_pipeline(&g_audio_context, (void *)g_audio_pipeline);
1021
1022 // Start full-duplex audio (simultaneous capture + playback for perfect AEC3 timing)
1023 if (audio_start_duplex(&g_audio_context) != ASCIICHAT_OK) {
1024 log_error("Failed to start full-duplex audio");
1025 client_audio_pipeline_destroy(g_audio_pipeline);
1026 g_audio_pipeline = NULL;
1027 audio_destroy(&g_audio_context);
1028 // Clean up WAV writer if it was opened
1029 if (g_wav_playback_received) {
1030 wav_writer_close(g_wav_playback_received);
1031 g_wav_playback_received = NULL;
1032 }
1033 return -1;
1034 }
1035
1036 // Note: audio_sender_init() is now called from protocol_start_connection()
1037 // after connection succeeds, to prevent deadlock if connection fails
1038
1039 return 0;
1040}

1084 // This is a resource leak of the stuck thread but continuing is safer than hanging

1113void audio_stop_thread() {
1114 // If audio was never initialized (e.g., in snapshot mode), nothing to stop
1115 if (!lifecycle_is_initialized(&g_audio_client_init_lc)) {
1116 log_debug("[AUDIO_STOP] Audio never initialized, nothing to stop");
1117 return;
1118 }
1119
1120 // Ensure only one thread performs cleanup - prevent double-join and double-cleanup
1121 // First, initialize the lifecycle if it hasn't been yet (moves UNINITIALIZED → INITIALIZED)
1122 if (!lifecycle_is_initialized(&g_audio_stop_thread_lc)) {
1123 if (!lifecycle_init(&g_audio_stop_thread_lc, "audio_stop_thread")) {
1124 // Already initialized by another thread or being destroyed
1125 log_debug("[AUDIO_STOP] Lifecycle already initialized by another thread");
1126 return;
1127 }
1128 log_debug("[AUDIO_STOP] Initialized stop thread lifecycle");
1129 }
1130
1131 // Now destroy it (moves INITIALIZED → DESTROYING)
1132 if (!lifecycle_destroy_once(&g_audio_stop_thread_lc)) {
1133 log_debug("[AUDIO_STOP] lifecycle_destroy_once() returned false - already destroying/destroyed");
1134 log_debug("[AUDIO_STOP] Audio stop already in progress or completed, skipping");
1135 return;
1136 }
1137 log_debug("[AUDIO_STOP] lifecycle_destroy_once() returned true, proceeding with audio stop");
1138
1139 // Signal audio worker thread to exit first (stop processing)
1140 log_debug("[AUDIO_STOP] Signaling audio worker thread to stop");
1141 atomic_store_bool(&g_audio_context.worker_should_stop, true);
1142 // Wake up the worker thread if it's waiting on condition variable
1143 mutex_lock(&g_audio_context.worker_mutex);
1144 cond_signal(&g_audio_context.worker_cond);
1145 mutex_unlock(&g_audio_context.worker_mutex);
1146 log_debug("[AUDIO_STOP] Worker thread stop signal sent");
1147
1148 // Signal audio sender thread to exit first.
1149 // This must happen before thread_pool_stop_all() is called, otherwise the sender
1150 // thread will be stuck in cond_wait() and thread_pool_stop_all() will hang forever.
1151 // The sender thread uses a condition variable to wait for packets - we must wake it up.
1152 // NOTE: Only do this if audio_sender_init() was actually called
1153 if (lifecycle_is_initialized(&g_audio_send_queue_lc)) {
1154 log_debug("[AUDIO_STOP] Signaling audio sender thread to exit");
1155 atomic_store_bool(&g_audio_sender_should_exit, true);
1156 mutex_lock(&g_audio_send_queue_mutex);
1157 cond_signal(&g_audio_send_queue_cond);
1158 mutex_unlock(&g_audio_send_queue_mutex);
1159 log_debug("[AUDIO_STOP] Signal sent to sender thread");
1160
1161 // Wait for audio sender thread to exit gracefully
1162 // This must complete before thread_pool_stop_all() to prevent deadlock
1163 // CRITICAL: Longer timeout to ensure thread completes any in-flight send operations
1164 // Network operations can block for 10-50ms, so we need to wait longer
1165 int wait_count = 0;
1166 log_debug("[AUDIO_STOP] Waiting for sender thread to exit, max 500ms");
1167 while (wait_count < 50 && !atomic_load_bool(&g_audio_sender_exited)) {
1168 platform_sleep_us(10 * US_PER_MS_INT); // 10ms * 50 = 500ms max wait
1169 wait_count++;
1170 }
1171
1172 if (!atomic_load_bool(&g_audio_sender_exited)) {
1173 log_warn("[AUDIO_STOP] Audio sender thread not responding after 500ms - will be joined by thread pool");
1174 } else {
1175 log_debug("[AUDIO_STOP] Sender thread exited successfully");
1176 }
1177 } else {
1178 log_debug("[AUDIO_STOP] Audio sender was never initialized (connection failed), skipping sender shutdown");
1179 }
1180
1181 log_debug("[AUDIO_STOP] Checking if capture thread was created: g_audio_capture_thread_created=%d",
1182 g_audio_capture_thread_created);
1183 if (!THREAD_IS_CREATED(g_audio_capture_thread_created)) {
1184 log_debug("[AUDIO_STOP] Capture thread was not created, committing lifecycle and returning");
1185 lifecycle_destroy_commit(&g_audio_stop_thread_lc);
1186 log_debug("[AUDIO_STOP] Lifecycle committed, returning");
1187 return;
1188 }
1189 log_debug("[AUDIO_STOP] Capture thread was created, waiting for it to exit");
1190
1191 // Note: We don't call signal_exit() here because that's for global shutdown only
1192 // The audio capture thread checks server_connection_is_active() to detect connection loss
1193
1194 // Wait for thread to exit gracefully
1195 // Threads check should_exit() every 1-5ms, so timeout can be much shorter than 500ms
1196 int wait_count2 = 0;
1197 log_debug("[AUDIO_STOP] Waiting for capture thread to exit, max 50ms");
1198 while (wait_count2 < 5 && !atomic_load_bool(&g_audio_capture_thread_exited)) {
1199 platform_sleep_us(10 * US_PER_MS_INT); // 10ms * 5 = 50ms max wait (threads respond in ~1-5ms)
1200 wait_count2++;
1201 }
1202
1203 if (!atomic_load_bool(&g_audio_capture_thread_exited)) {
1204 log_warn("[AUDIO_STOP] Audio capture thread not responding - will be joined by thread pool");
1205 } else {
1206 log_debug("[AUDIO_STOP] Capture thread exited successfully");
1207 }
1208
1209 // Thread will be joined by thread_pool_stop_all() in protocol_stop_connection()
1210 g_audio_capture_thread_created = false;
1211
1212 lifecycle_destroy_commit(&g_audio_stop_thread_lc);
1213
1214 log_debug("Audio capture thread stopped");
1215}

1236void audio_cleanup() {
1237 if (!GET_OPTION(audio_enabled)) {
1238 return;
1239 }
1240
1241 // Stop capture thread first (stops producing packets)
1242 audio_stop_thread();
1243
1244 // Stop async sender thread (drains queue and exits)
1245 audio_sender_cleanup();
1246
1247 // Terminate PortAudio FIRST to properly free device resources before cleanup
1248 // This must happen before audio_stop_duplex() and audio_destroy()
1249 audio_terminate_portaudio_final();
1250
1251 // Stop audio stream before destroying pipeline to prevent race condition.
1252 // PortAudio may invoke the callback one more time after we request stop.
1253 // We need to clear the pipeline pointer first so the callback can't access freed memory.
1254 if (g_audio_context.initialized) {
1255 audio_stop_duplex(&g_audio_context);
1256 }
1257
1258 // Clear the pipeline pointer from audio context BEFORE destroying pipeline
1259 // This prevents any lingering PortAudio callbacks from trying to access freed memory
1260 audio_set_pipeline(&g_audio_context, NULL);
1261
1262 // Sleep to allow CoreAudio threads to finish executing callbacks.
1263 // On macOS, CoreAudio's internal threads may continue running after Pa_StopStream() returns.
1264 // The duplex_callback may still be in-flight on other threads. Even after we set the pipeline
1265 // pointer to NULL, a CoreAudio thread may have already cached the pointer before the assignment.
1266 // This sleep ensures all in-flight callbacks have fully completed before we destroy the pipeline.
1267 // WORKAROUND: Skip sleep during shutdown (process is exiting anyway)
1268 // 500ms is sufficient on macOS for CoreAudio's internal thread pool to completely wind down.
1269 // platform_sleep_us(500 * US_PER_MS_INT); // 500ms - macOS CoreAudio needs time to shut down all threads
1270
1271 // Destroy audio pipeline (handles Opus, AEC, etc.)
1272 if (g_audio_pipeline) {
1273 client_audio_pipeline_destroy(g_audio_pipeline);
1274 g_audio_pipeline = NULL;
1275 log_debug("Audio pipeline destroyed");
1276 }
1277
1278 // Close WAV dumpers
1279 if (g_wav_capture_raw) {
1280 wav_writer_close(g_wav_capture_raw);
1281 g_wav_capture_raw = NULL;
1282 log_debug("Closed audio capture raw dump");

1285 wav_writer_close(g_wav_capture_processed);

1290 wav_writer_close(g_wav_playback_received);
1291 g_wav_playback_received = NULL;
1292 log_debug("Closed audio playback received dump");
1293 }
1294
1295 // Finally destroy the audio context
1296 if (g_audio_context.initialized) {
1297 audio_destroy(&g_audio_context);
1298 }
1299}
