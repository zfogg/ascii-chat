251static unsigned int get_reconnect_delay(unsigned int reconnect_attempt) {
252 // Use integer arithmetic for microsecond calculations
253 // Initial delay: 100,000 us (0.1 seconds)
254 // Additional delay per attempt: 200,000 us (0.2 seconds)
255 unsigned int delay_us = 100 * US_PER_MS_INT + (reconnect_attempt - 1) * 200 * US_PER_MS_INT;
256 if (delay_us > MAX_RECONNECT_DELAY)
257 delay_us = MAX_RECONNECT_DELAY;
258 return delay_us;
259}

276static int close_socket(socket_t socketfd) {

281 log_error("Failed to close socket: %s", network_error_string());

475 // Resolve server address using getaddrinfo() for IPv4/IPv6 support
476 // Special handling for localhost: ensure we try both IPv6 (::1) and IPv4 (127.0.0.1)
477 // Many systems only map "localhost" to 127.0.0.1 in /etc/hosts
478 bool is_localhost = (strcmp(address, "localhost") == 0 || is_localhost_ipv4(address) || is_localhost_ipv6(address));
479
480 struct addrinfo hints, *res = NULL, *addr_iter;
481 memset(&hints, 0, sizeof(hints));
482 hints.ai_family = AF_UNSPEC; // Allow IPv4 or IPv6
483 hints.ai_socktype = SOCK_STREAM;
484 if (is_localhost) {
485 hints.ai_flags = AI_NUMERICSERV; // Optimize for localhost
486 }
487
488 char port_str[16];
489 SAFE_SNPRINTF(port_str, sizeof(port_str), "%d", port);
490
491 // For localhost, try IPv6 loopback (::1) first, then fall back to DNS resolution
492 if (is_localhost) {
493 log_debug("Localhost detected - trying IPv6 loopback [::1]:%s first...", port_str);
494 hints.ai_family = AF_INET6;
495 hints.ai_flags = AI_NUMERICHOST | AI_NUMERICSERV;
496
497 int ipv6_result = getaddrinfo("::1", port_str, &hints, &res);
498 if (ipv6_result == 0 && res != NULL) {
499 // Try IPv6 loopback connection
500 g_sockfd = socket_create("ipv6", res->ai_family, res->ai_socktype, res->ai_protocol);
501 if (g_sockfd != INVALID_SOCKET_VALUE) {
502 log_debug("Trying IPv6 loopback connection to [::1]:%s...", port_str);
503 if (connect_with_timeout(g_sockfd, res->ai_addr, res->ai_addrlen, CONNECT_TIMEOUT)) {
504 log_debug("Connection successful using IPv6 loopback");
505 SAFE_STRNCPY(g_server_ip, "::1", sizeof(g_server_ip));
506 freeaddrinfo(res);
507 res = NULL; // Prevent double-free at connection_success label
508 goto connection_success;
509 }
510 if (socket_get_error(g_sockfd) != 0) {
511 log_debug("NETWORK_ERROR: %d", (int)socket_get_error(g_sockfd));
512 } else {
513 // log_debug("IPv6 loopback connection failed: %s", network_error_string());
514 }
515 close_socket(g_sockfd);
516 g_sockfd = INVALID_SOCKET_VALUE;
517 }
518 freeaddrinfo(res);
519 res = NULL;
520 }
521
522 // Check if user requested exit (Ctrl-C) before trying IPv4
523 if (should_exit()) {
524 log_debug("Exit requested during connection attempt");
525 return -1;
526 }
527
528 // IPv6 failed, try IPv4 loopback (127.0.0.1)
529 log_debug("IPv6 failed, trying IPv4 loopback 127.0.0.1:%s...", port_str);
530 hints.ai_family = AF_INET;
531
532 int ipv4_result = getaddrinfo("127.0.0.1", port_str, &hints, &res);
533 if (ipv4_result == 0 && res != NULL) {
534 g_sockfd = socket_create("ipv4", res->ai_family, res->ai_socktype, res->ai_protocol);
535 if (g_sockfd != INVALID_SOCKET_VALUE) {
536 log_debug("Trying IPv4 loopback connection to 127.0.0.1:%s...", port_str);
537 if (connect_with_timeout(g_sockfd, res->ai_addr, res->ai_addrlen, CONNECT_TIMEOUT)) {
538 log_debug("Connection successful using IPv4 loopback");
539 SAFE_STRNCPY(g_server_ip, "127.0.0.1", sizeof(g_server_ip));
540 freeaddrinfo(res);
541 res = NULL; // Prevent double-free at connection_success label
542 goto connection_success;
543 }
544 if (socket_get_error(g_sockfd) != 0) {
545 log_debug("NETWORK_ERROR: %d", (int)socket_get_error(g_sockfd));
546 } else {
547 // log_debug("IPv4 loopback connection failed: %s", network_error_string());
548 }
549 close_socket(g_sockfd);
550 g_sockfd = INVALID_SOCKET_VALUE;
551 }
552 freeaddrinfo(res);
553 res = NULL;
554 }
555
556 // Both IPv6 and IPv4 loopback failed for localhost
557 log_warn("Could not connect to localhost using either IPv6 or IPv4 loopback");
558 return -1;
559 }
560
561 // For non-localhost addresses, use standard resolution
562 log_debug("Resolving server address '%s' port %s...", address, port_str);
563 hints.ai_family = AF_UNSPEC;
564 hints.ai_flags = 0;
565 int getaddr_result = getaddrinfo(address, port_str, &hints, &res);
566 if (getaddr_result != 0) {
567 log_error("Failed to resolve server address '%s': %s", address, gai_strerror(getaddr_result));
568 return -1;
569 }
570
571 // Try each address returned by getaddrinfo() until one succeeds
572 // Prefer IPv6 over IPv4: try IPv6 addresses first, then fall back to IPv4
573 for (int address_family = AF_INET6; address_family >= AF_INET; address_family -= (AF_INET6 - AF_INET)) {
574 for (addr_iter = res; addr_iter != NULL; addr_iter = addr_iter->ai_next) {
575 // Skip addresses that don't match current pass (IPv6 first, then IPv4)
576 if (addr_iter->ai_family != address_family) {
577 continue;
578 }
579
580 // Create socket with appropriate address family
581 const char *socket_name =
582 (addr_iter->ai_family == AF_INET6) ? "client_socket_server_ipv6" : "client_socket_server_ipv4";
583 g_sockfd = socket_create(socket_name, addr_iter->ai_family, addr_iter->ai_socktype, addr_iter->ai_protocol);
584 if (g_sockfd == INVALID_SOCKET_VALUE) {
585 log_debug("Could not create socket for address family %d: %s", addr_iter->ai_family, network_error_string());
586 continue; // Try next address
587 }
588
589 // Log which address family we're trying
590 if (addr_iter->ai_family == AF_INET) {
591 log_debug("Trying IPv4 connection...");
592 } else if (addr_iter->ai_family == AF_INET6) {
593 log_debug("Trying IPv6 connection...");
594 }
595
596 // Attempt connection with timeout
597 if (connect_with_timeout(g_sockfd, addr_iter->ai_addr, addr_iter->ai_addrlen, CONNECT_TIMEOUT)) {
598 // Connection successful!
599 log_debug("Connection successful using %s", addr_iter->ai_family == AF_INET ? "IPv4"
600 : addr_iter->ai_family == AF_INET6 ? "IPv6"
601 : "unknown protocol");
602
603 // Extract server IP address for known_hosts
604 if (format_ip_address(addr_iter->ai_family, addr_iter->ai_addr, g_server_ip, sizeof(g_server_ip)) ==
605 ASCIICHAT_OK) {
606 log_debug("Resolved server IP: %s", g_server_ip);
607 } else {
608 log_warn("Failed to format server IP address");
609 }
610
611 goto connection_success; // Break out of both loops
612 }
613
614 // Connection failed - close socket and try next address
615 if (socket_get_error(g_sockfd) != 0) {
616 log_debug("NETWORK_ERROR: %d", (int)socket_get_error(g_sockfd));
617 } else {
618 // log_debug("Connection failed: %s", network_error_string());
619 }
620 close_socket(g_sockfd);
621 g_sockfd = INVALID_SOCKET_VALUE;
622 }
623 }
624
625connection_success:
626
627 if (res) {
628 freeaddrinfo(res);
629 }
630
631 // If we exhausted all addresses without success, fail
632 if (g_sockfd == INVALID_SOCKET_VALUE) {
633 log_warn("Could not connect to server %s:%d (tried all addresses)", address, port);
634 return -1;
635 }
636
637 // Connection successful - extract local port for client ID
638 struct sockaddr_storage local_addr = {0};
639 socklen_t addr_len = sizeof(local_addr);
640 if (getsockname(g_sockfd, (struct sockaddr *)&local_addr, &addr_len) == -1) {
641 log_error("Failed to get local socket address: %s", network_error_string());
642 close_socket(g_sockfd);
643 g_sockfd = INVALID_SOCKET_VALUE;
644 return -1;
645 }
646
647 // Extract port from either IPv4 or IPv6 address
648 int local_port = 0;
649 if (((struct sockaddr *)&local_addr)->sa_family == AF_INET) {
650 local_port = NET_TO_HOST_U16(((struct sockaddr_in *)&local_addr)->sin_port);
651 } else if (((struct sockaddr *)&local_addr)->sa_family == AF_INET6) {
652 local_port = NET_TO_HOST_U16(((struct sockaddr_in6 *)&local_addr)->sin6_port);
653 }
654 g_my_client_id = (uint32_t)local_port;
655
656 // Mark connection as active immediately after successful socket connection
657 atomic_store_bool(&g_connection_active, true);
658 atomic_store_bool(&g_connection_lost, false);
659 atomic_store_bool(&g_should_reconnect, false);
660
661 // Initialize crypto BEFORE starting protocol handshake
662 // Note: server IP is already set above in the connection loop
663 log_debug("CLIENT_CONNECT: Calling client_crypto_init()");
664 if (client_crypto_init() != 0) {

671
672 // Perform crypto handshake if encryption is enabled
673 log_debug("CLIENT_CONNECT: Calling client_crypto_handshake()");

675 if (handshake_result != 0) {
676 log_error("Crypto handshake failed");

812void server_connection_set_transport(acip_transport_t *transport) {
813 log_debug("[TRANSPORT_LIFECYCLE] server_connection_set_transport() called with transport=%p", (void *)transport);
814
815 // Clean up any existing transport
816 if (g_client_transport) {
817 log_warn("[TRANSPORT_LIFECYCLE] Replacing existing transport=%p with new transport=%p (potential premature "
818 "destruction?)",
819 (void *)g_client_transport, (void *)transport);
820 log_debug("[TRANSPORT_LIFECYCLE] OLD transport is_connected=%s before destruction",
821 acip_transport_is_connected(g_client_transport) ? "true" : "false");
822 acip_transport_destroy(g_client_transport);
823 log_debug("[TRANSPORT_LIFECYCLE] OLD transport destroyed");
824 }
825
826 log_debug("[TRANSPORT_LIFECYCLE] Setting g_client_transport to %p", (void *)transport);
827 g_client_transport = transport;
828
829 // Mark connection as active when transport is set
830 if (transport) {
831 log_debug("[TRANSPORT_LIFECYCLE] Transport is non-NULL, extracting socket...");

985static void server_connection_interrupt_websocket_recv(void) {
986 // Access the transport
987 acip_transport_t *transport = g_client_transport;
988 if (!transport || !transport->impl_data) {
989 return; // No transport
990 }
991
992 // Check if this is a WebSocket transport
993 if (transport->methods && transport->methods->get_type) {
994 acip_transport_type_t type = transport->methods->get_type(transport);
995 if (type == ACIP_TRANSPORT_WEBSOCKET) {
996 // For WebSocket, directly signal the recv condition variable to wake
997 // any blocked threads. This is async-signal-safe (doesn't require mutex).
998 // The waiting threads will re-acquire their mutexes and check is_connected.
999 websocket_transport_data_t *ws_data = (websocket_transport_data_t *)transport->impl_data;
1000
1001 // Signal recv condition - wakes threads waiting for fragments
1002 // This is safe to call without holding the recv_mutex
1003 cond_broadcast(&ws_data->recv_cond);
1004 cond_broadcast(&ws_data->state_cond);
1005 }

1018void server_connection_shutdown() {
1019 // NOTE: This function may be called from:
1020 // - Signal handlers on Unix (async-signal-safe context)
1021 // - SetConsoleCtrlHandler callback thread on Windows (separate thread context)
1022 // Only use atomic operations and simple system calls - NO mutex locks, NO malloc, NO logging.
1023
1024 atomic_store_bool(&g_connection_active, false);
1025 atomic_store_bool(&g_connection_lost, true);
1026
1027 if (g_sockfd != INVALID_SOCKET_VALUE) {
1028 // Only shutdown() the socket to interrupt blocking recv()/send() operations.
1029 // Do NOT close() here - on Windows, closing the socket while another thread
1030 // is using it is undefined behavior and can cause STATUS_STACK_BUFFER_OVERRUN.
1031 // The actual socket close happens in server_connection_close() which is called
1032 // from the main thread after worker threads have been joined.
1033 socket_shutdown(g_sockfd, SHUT_RDWR);
1034 }
1035
1036 // For WebSocket transport: interrupt blocking recv() calls
1037 // The recv() checks is_connected before waiting for data. Setting this to false
1038 // will cause any blocked recv() to return immediately with CONNECTION_CLOSED.
1039 // We use a function that's async-signal-safe (no mutexes) to do this.
1040 server_connection_interrupt_websocket_recv();
1041
1042 // DO NOT call log_set_terminal_output() here - it uses mutex which is NOT async-signal-safe.
1043 // The normal cleanup path in shutdown_client() will handle logging state.
1044}

1111asciichat_error_t threaded_send_packet(packet_type_t type, const void *data, size_t len) {
1112 // Lock mutex for entire send operation to prevent concurrent socket writes
1113 mutex_lock(&g_send_mutex);
1114
1115 // Check connection status and get transport reference
1116 if (!atomic_load_bool(&g_connection_active) || !g_client_transport) {
1117 log_debug("[TRANSPORT_LIFECYCLE] threaded_send_packet() check failed: active=%s, transport=%p",
1118 atomic_load_bool(&g_connection_active) ? "true" : "false", (void *)g_client_transport);
1119 mutex_unlock(&g_send_mutex);
1120 return SET_ERRNO(ERROR_NETWORK, "Connection not active or transport unavailable");
1121 }
1122
1123 // Get transport reference - transport has its own internal synchronization
1124 acip_transport_t *transport = g_client_transport;
1125 log_debug_every(LOG_RATE_SLOW, "[TRANSPORT_LIFECYCLE] threaded_send_packet() using transport=%p, is_connected=%s",
1126 (void *)transport, acip_transport_is_connected(transport) ? "true" : "false");
1127
1128 // Network I/O happens while holding mutex to prevent concurrent socket writes
1129 asciichat_error_t result = packet_send_via_transport(transport, type, data, len, 0);
1130
1131 // Unlock after send completes
1132 mutex_unlock(&g_send_mutex);
1133
1134 // If send failed due to network error, signal connection loss
1135 if (result != ASCIICHAT_OK) {
1136 log_debug("[TRANSPORT_LIFECYCLE] threaded_send_packet() send failed, calling server_connection_lost()");
1137 server_connection_lost();
1138 return result;
1139 }

1141 return ASCIICHAT_OK;
1142}

1200asciichat_error_t threaded_send_audio_opus(const uint8_t *opus_data, size_t opus_size, int sample_rate,
1201 int frame_duration) {
1202 // Get transport reference while holding mutex (brief lock)
1203 mutex_lock(&g_send_mutex);
1204
1205 // Check connection status and get transport reference
1206 if (!atomic_load_bool(&g_connection_active) || !g_client_transport) {
1207 mutex_unlock(&g_send_mutex);
1208 return SET_ERRNO(ERROR_NETWORK, "Connection not active or transport unavailable");
1209 }
1210
1211 // Get transport reference - transport has its own internal synchronization
1212 acip_transport_t *transport = g_client_transport;
1213 mutex_unlock(&g_send_mutex);
1214
1215 // Build Opus packet with header (outside mutex - no blocking I/O yet)
1216 size_t header_size = 16; // sample_rate (4), frame_duration (4), reserved (8)
1217 size_t total_size = header_size + opus_size;
1218 void *packet_data = buffer_pool_alloc(NULL, total_size);
1219 if (!packet_data) {
1220 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate buffer for Opus packet: %zu bytes", total_size);
1221 }
1222
1223 // Write header in network byte order
1224 uint8_t *buf = (uint8_t *)packet_data;
1225 uint32_t sr = HOST_TO_NET_U32((uint32_t)sample_rate);
1226 uint32_t fd = HOST_TO_NET_U32((uint32_t)frame_duration);
1227 memcpy(buf, &sr, 4);
1228 memcpy(buf + 4, &fd, 4);
1229 memset(buf + 8, 0, 8); // Reserved
1230
1231 // Copy Opus data
1232 memcpy(buf + header_size, opus_data, opus_size);
1233
1234 // Network I/O happens OUTSIDE the mutex to prevent deadlock on TCP buffer full
1235 asciichat_error_t result =
1236 packet_send_via_transport(transport, PACKET_TYPE_AUDIO_OPUS_BATCH, packet_data, total_size, 0);
1237
1238 // Clean up
1239 buffer_pool_free(NULL, packet_data, total_size);
1240
1241 // If send failed due to network error, signal connection loss
1242 if (result != ASCIICHAT_OK) {
1243 server_connection_lost();
1244 return result;
1245 }
1246
1247 return ASCIICHAT_OK;
