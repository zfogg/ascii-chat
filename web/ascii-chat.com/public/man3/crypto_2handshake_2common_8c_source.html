18asciichat_error_t crypto_handshake_init(const char *name, crypto_handshake_context_t *ctx, bool is_server) {
19 if (!name) {
20 return SET_ERRNO(ERROR_INVALID_PARAM, "Crypto context name is required");
21 }
22
23 if (!ctx) {
24 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters: ctx=%p", ctx);
25 }
26
27 // Register crypto context with debug naming system
28 NAMED_REGISTER_CRYPTO_CONTEXT(ctx, name, NULL);
29
30 // Zero out the context
31 memset(ctx, 0, sizeof(crypto_handshake_context_t));
32
33 // Initialize core crypto context
34 log_info("★★★ BEFORE crypto_init()");
35 debug_sync_print_state();
36 crypto_result_t result = crypto_init(&ctx->crypto_ctx);
37 log_info("★★★ AFTER crypto_init(), result=%d", result);
38 debug_sync_print_state();
39 if (result != CRYPTO_OK) {
40 return SET_ERRNO(ERROR_CRYPTO, "Failed to initialize crypto context: %s", crypto_result_to_string(result));
41 }
42
43 ctx->state = CRYPTO_HANDSHAKE_INIT;
44 ctx->is_server = is_server;
45 ctx->verify_server_key = false;
46 ctx->require_client_auth = false;
47 ctx->server_uses_client_auth = false; // Set to true only if authenticated packet received
48
49 // Load server keys if this is a server
50 if (is_server) {
51 log_dev("Server crypto handshake initialized (ephemeral keys)");
52 } else {
53 log_dev("Client crypto handshake initialized");
54 }
55
56 return ASCIICHAT_OK;

59// Set crypto parameters from crypto_parameters_packet_t
60asciichat_error_t crypto_handshake_set_parameters(crypto_handshake_context_t *ctx,
61 const crypto_parameters_packet_t *params) {
62 if (!ctx || !params) {
63 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters: ctx=%p, params=%p", ctx, params);
64 }
65
66 // Client receives network byte order and must convert
67 // Server uses host byte order and must NOT convert
68 if (ctx->is_server) {
69 // Server: values are already in host byte order
70 // Update crypto context with negotiated parameters directly
71 ctx->crypto_ctx.public_key_size = params->kex_public_key_size;
72 ctx->crypto_ctx.auth_public_key_size = params->auth_public_key_size;
73 ctx->crypto_ctx.shared_key_size = params->shared_secret_size;
74 ctx->crypto_ctx.signature_size = params->signature_size;
75 } else {
76 // Client: convert from network byte order
77 // Update crypto context with negotiated parameters directly
78 ctx->crypto_ctx.public_key_size = NET_TO_HOST_U16(params->kex_public_key_size);
79 ctx->crypto_ctx.auth_public_key_size = NET_TO_HOST_U16(params->auth_public_key_size);
80 ctx->crypto_ctx.shared_key_size = NET_TO_HOST_U16(params->shared_secret_size);
81 ctx->crypto_ctx.signature_size = NET_TO_HOST_U16(params->signature_size);
82 }
83 // Update crypto context with negotiated parameters directly
84 ctx->crypto_ctx.nonce_size = params->nonce_size;
85 ctx->crypto_ctx.mac_size = params->mac_size;
86 ctx->crypto_ctx.hmac_size = params->hmac_size;
87 ctx->crypto_ctx.auth_challenge_size =
88 AUTH_CHALLENGE_SIZE; // Auth challenge size is fixed for now, could be negotiated later
89 ctx->crypto_ctx.encryption_key_size =
90 (uint8_t)ctx->crypto_ctx.shared_key_size; // Use shared key size as encryption key size
91 ctx->crypto_ctx.private_key_size = ctx->crypto_ctx.public_key_size; // Same as public key for X25519
92 ctx->crypto_ctx.salt_size = ARGON2ID_SALT_SIZE; // Salt size doesn't change
93
94 log_debug("Crypto parameters set: kex_key=%u, auth_key=%u, sig=%u, "
95 "secret=%u, nonce=%u, mac=%u, hmac=%u",
96 ctx->crypto_ctx.public_key_size, ctx->crypto_ctx.auth_public_key_size, ctx->crypto_ctx.signature_size,
97 ctx->crypto_ctx.shared_key_size, ctx->crypto_ctx.nonce_size, ctx->crypto_ctx.mac_size,
98 ctx->crypto_ctx.hmac_size);
99
100 return ASCIICHAT_OK;
101}

103// Validate crypto packet size based on session parameters
104asciichat_error_t crypto_handshake_validate_packet_size(const crypto_handshake_context_t *ctx, uint16_t packet_type,
105 size_t packet_size) {
106 if (!ctx) {
107 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters: ctx=%p", ctx);
108 }
109
110 switch (packet_type) {
111 case PACKET_TYPE_CRYPTO_CAPABILITIES:
112 if (packet_size != sizeof(crypto_capabilities_packet_t)) {
113 // Don't return an error code, just set the errno and return the error code
114 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid crypto capabilities packet size: %zu (expected %zu)",
115 packet_size, sizeof(crypto_capabilities_packet_t));
116 }
117 break;
118
119 case PACKET_TYPE_CRYPTO_PARAMETERS:
120 if (packet_size != sizeof(crypto_parameters_packet_t)) {
121 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid crypto parameters packet size: %zu (expected %zu)", packet_size,
122 sizeof(crypto_parameters_packet_t));
123 }
124 break;
125
126 case PACKET_TYPE_CRYPTO_KEY_EXCHANGE_INIT:
127 // Server can send either:
128 // 1. Simple format: kex_public_key_size (when server has no identity key)
129 // 2. Authenticated format: kex_public_key_size + auth_public_key_size + signature_size
130 {
131 size_t simple_size = ctx->crypto_ctx.public_key_size;
132 size_t authenticated_size =
133 ctx->crypto_ctx.public_key_size + ctx->crypto_ctx.auth_public_key_size + ctx->crypto_ctx.signature_size;
134
135 if (packet_size != simple_size && packet_size != authenticated_size) {
136 return SET_ERRNO(ERROR_NETWORK_PROTOCOL,
137 "Invalid KEY_EXCHANGE_INIT size: %zu (expected %zu for simple or %zu for authenticated: "
138 "kex=%u + auth=%u + sig=%u)",
139 packet_size, simple_size, authenticated_size, ctx->crypto_ctx.public_key_size,
140 ctx->crypto_ctx.auth_public_key_size, ctx->crypto_ctx.signature_size);
141 }
142 }
143 break;
144
145 case PACKET_TYPE_CRYPTO_KEY_EXCHANGE_RESP:
146 // Client can send either:
147 // 1. Simple format: kex_public_key_size (when server has no identity key)
148 // 2. Authenticated format: kex_public_key_size + client_auth_key_size + client_sig_size + [gpg_key_id_len:1] +
149 // [gpg_key_id:0-16]
150 {
151 size_t simple_size = ctx->crypto_ctx.public_key_size;
152 // For authenticated format, use Ed25519 sizes since client has Ed25519 key
153 size_t ed25519_auth_size = ED25519_PUBLIC_KEY_SIZE; // Ed25519 public key is always 32 bytes
154 size_t ed25519_sig_size = ED25519_SIGNATURE_SIZE; // Ed25519 signature is always 64 bytes
155 size_t authenticated_min_size = ctx->crypto_ctx.public_key_size + ed25519_auth_size + ed25519_sig_size;
156 size_t authenticated_max_size = authenticated_min_size + 1 + 40; // +1 for length, +40 for max GPG key ID
157
158 if (packet_size != simple_size &&
159 (packet_size < authenticated_min_size || packet_size > authenticated_max_size)) {
160 return SET_ERRNO(ERROR_NETWORK_PROTOCOL,
161 "Invalid KEY_EXCHANGE_RESP size: %zu (expected %zu for simple or %zu-%zu for authenticated: "
162 "kex=%u + auth=%u + sig=%u + optional GPG key ID)",
163 packet_size, simple_size, authenticated_min_size, authenticated_max_size,
164 ctx->crypto_ctx.public_key_size, ed25519_auth_size, ed25519_sig_size);
165 }
166 }
167 break;
168
169 case PACKET_TYPE_CRYPTO_AUTH_CHALLENGE:
170 // Server sends: 1 byte auth_flags + auth_challenge_size byte nonce
171 {
172 size_t expected_size = AUTH_CHALLENGE_FLAGS_SIZE + ctx->crypto_ctx.auth_challenge_size;
173 if (packet_size != expected_size) {
174 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid AUTH_CHALLENGE size: %zu (expected %zu: flags=%d + nonce=%u)",
175 packet_size, expected_size, AUTH_CHALLENGE_FLAGS_SIZE, ctx->crypto_ctx.auth_challenge_size);
176 }
177 }
178 break;
179
180 case PACKET_TYPE_CRYPTO_AUTH_RESPONSE:
181 // Client sends: hmac_size + auth_challenge_size bytes client_nonce + [gpg_key_id_len:1] + [gpg_key_id:0-40]
182 {
183 size_t min_size = ctx->crypto_ctx.hmac_size + ctx->crypto_ctx.auth_challenge_size;
184 size_t max_size = min_size + 1 + 40; // +1 for length, +40 for max GPG key ID
185 if (packet_size < min_size || packet_size > max_size) {
186 return SET_ERRNO(ERROR_NETWORK_PROTOCOL,
187 "Invalid AUTH_RESPONSE size: %zu (expected %zu-%zu: hmac=%u + "
188 "nonce=%u + optional GPG key ID)",
189 packet_size, min_size, max_size, ctx->crypto_ctx.hmac_size,
190 ctx->crypto_ctx.auth_challenge_size);
191 }
192 }
193 break;
194
195 case PACKET_TYPE_CRYPTO_AUTH_FAILED:
196 // Variable size - just check reasonable limits
197 if (packet_size > MAX_AUTH_FAILED_PACKET_SIZE) {
198 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid AUTH_FAILED size: %zu (max %d)", packet_size,
199 MAX_AUTH_FAILED_PACKET_SIZE);
200 }
201 break;
202
203 case PACKET_TYPE_CRYPTO_SERVER_AUTH_RESP:
204 // Server sends: hmac_size bytes
205 if (packet_size != ctx->crypto_ctx.hmac_size) {
206 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid SERVER_AUTH_RESP size: %zu (expected %u)", packet_size,
207 ctx->crypto_ctx.hmac_size);
208 }
209 break;
210
211 case PACKET_TYPE_CRYPTO_HANDSHAKE_COMPLETE:
212 // Empty packet
213 if (packet_size != 0) {
214 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid HANDSHAKE_COMPLETE size: %zu (expected 0)", packet_size);
215 }
216 break;
217
218 case PACKET_TYPE_CRYPTO_NO_ENCRYPTION:
219 // Empty packet
220 if (packet_size != 0) {
221 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid NO_ENCRYPTION size: %zu (expected 0)", packet_size);
222 }
223 break;
224
225 case PACKET_TYPE_ENCRYPTED:
226 // Variable size - check reasonable limits
227 if (packet_size > MAX_ENCRYPTED_PACKET_SIZE) { // 64KB max for encrypted packets
228 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Invalid ENCRYPTED size: %zu (max %d)", packet_size,
229 MAX_ENCRYPTED_PACKET_SIZE);
230 }
231 break;
232
233 default:
234 return SET_ERRNO(ERROR_NETWORK_PROTOCOL, "Unknown crypto packet type: %u", packet_type);
235 }
236
237 return ASCIICHAT_OK;
238}

386 if (!ctx || !crypto_handshake_is_ready(ctx)) {

403 return SET_ERRNO(ERROR_NETWORK, "Failed to send REKEY_REQUEST packet");
404 }

406 log_debug("REKEY_REQUEST sent successfully, awaiting REKEY_RESPONSE");
407 return ASCIICHAT_OK;

442asciichat_error_t crypto_handshake_rekey_complete(crypto_handshake_context_t *ctx, socket_t socket) {
443 if (!ctx || !crypto_handshake_is_ready(ctx)) {
444 return SET_ERRNO(ERROR_INVALID_STATE, "Handshake not ready for rekeying: ctx=%p, ready=%d", ctx,
445 ctx ? crypto_handshake_is_ready(ctx) : 0);
446 }
447
448 if (!ctx->crypto_ctx.rekey_in_progress || !ctx->crypto_ctx.has_temp_key) {
449 return SET_ERRNO(ERROR_INVALID_STATE, "No rekey in progress or temp key missing");
450 }
451
452 // Encrypt empty payload with NEW key to prove possession
453 uint8_t plaintext[1] = {0}; // Minimal payload
454 uint8_t ciphertext[256]; // Sufficient for nonce + MAC + minimal payload
455 size_t ciphertext_len = 0;
456
457 // Temporarily swap keys to encrypt with NEW key
458 uint8_t old_shared_key[CRYPTO_SHARED_KEY_SIZE];
459 memcpy(old_shared_key, ctx->crypto_ctx.shared_key, CRYPTO_SHARED_KEY_SIZE);
460 memcpy(ctx->crypto_ctx.shared_key, ctx->crypto_ctx.temp_shared_key, CRYPTO_SHARED_KEY_SIZE);
461
462 // Encrypt with NEW key
463 crypto_result_t result =
464 crypto_encrypt(&ctx->crypto_ctx, plaintext, sizeof(plaintext), ciphertext, sizeof(ciphertext), &ciphertext_len);
465
466 // Restore old key immediately (commit will happen after successful send)
467 memcpy(ctx->crypto_ctx.shared_key, old_shared_key, CRYPTO_SHARED_KEY_SIZE);
468 sodium_memzero(old_shared_key, sizeof(old_shared_key));
469
470 if (result != CRYPTO_OK) {
471 crypto_rekey_abort(&ctx->crypto_ctx);
472 return SET_ERRNO(ERROR_CRYPTO, "Failed to encrypt REKEY_COMPLETE: %s", crypto_result_to_string(result));
473 }
474
475 // Send encrypted REKEY_COMPLETE
476 log_debug("Sending REKEY_COMPLETE (encrypted with NEW key, %zu bytes)", ciphertext_len);
477 int send_result = send_packet(socket, PACKET_TYPE_CRYPTO_REKEY_COMPLETE, ciphertext, ciphertext_len);
478 if (send_result != 0) {
479 crypto_rekey_abort(&ctx->crypto_ctx);
480 return SET_ERRNO(ERROR_NETWORK, "Failed to send REKEY_COMPLETE packet");
481 }
482
483 // Commit to new key (atomic switch)
484 result = crypto_rekey_commit(&ctx->crypto_ctx);
485 if (result != CRYPTO_OK) {
486 return SET_ERRNO(ERROR_CRYPTO, "Failed to commit rekey: %s", crypto_result_to_string(result));
487 }
488
489 log_debug("Session rekeying completed successfully (initiator side)");
490 return ASCIICHAT_OK;
491}

497asciichat_error_t crypto_handshake_process_rekey_request(crypto_handshake_context_t *ctx, const uint8_t *packet,
498 size_t packet_len) {
499 if (!ctx || !crypto_handshake_is_ready(ctx)) {
500 return SET_ERRNO(ERROR_INVALID_STATE, "Handshake not ready for rekeying: ctx=%p, ready=%d", ctx,
501 ctx ? crypto_handshake_is_ready(ctx) : 0);
502 }
503
504 // DDoS PROTECTION: Rate limit rekey requests
505 time_t now = time(NULL);
506 if (ctx->crypto_ctx.rekey_last_request_time > 0) {
507 time_t elapsed = now - ctx->crypto_ctx.rekey_last_request_time;
508 time_t min_request_interval_seconds = (time_t)(REKEY_MIN_REQUEST_INTERVAL / NS_PER_SEC_INT);
509 if (elapsed < min_request_interval_seconds) {
510 return SET_ERRNO(ERROR_CRYPTO,
511 "SECURITY: Rekey request rejected - too frequent (%ld sec since last, minimum %ld sec required)",
512 (long)elapsed, (long)min_request_interval_seconds);
513 }
514 }
515
516 // Update last request time
517 ctx->crypto_ctx.rekey_last_request_time = now;
518
519 // Validate packet size (should be 32 bytes for X25519 public key)
520 if (packet_len != CRYPTO_PUBLIC_KEY_SIZE) {
521 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid REKEY_REQUEST packet size: %zu (expected %d)", packet_len,
522 CRYPTO_PUBLIC_KEY_SIZE);
523 }
524
525 log_debug("Received REKEY_REQUEST with peer's new ephemeral public key (32 bytes)");
526
527 // Initialize our rekey process (generates our new ephemeral keypair)
528 crypto_result_t result = crypto_rekey_init(&ctx->crypto_ctx);
529 if (result != CRYPTO_OK) {
530 return SET_ERRNO(ERROR_CRYPTO, "Failed to initialize rekey: %s", crypto_result_to_string(result));
531 }
532
533 // Process peer's public key and compute new shared secret
534 result = crypto_rekey_process_request(&ctx->crypto_ctx, packet);
535 if (result != CRYPTO_OK) {
536 crypto_rekey_abort(&ctx->crypto_ctx);
537 return SET_ERRNO(ERROR_CRYPTO, "Failed to process REKEY_REQUEST: %s", crypto_result_to_string(result));
538 }
539
540 log_debug("REKEY_REQUEST processed successfully, new shared secret computed (responder side)");
541 return ASCIICHAT_OK;
542}

548asciichat_error_t crypto_handshake_process_rekey_response(crypto_handshake_context_t *ctx, const uint8_t *packet,
549 size_t packet_len) {
550 if (!ctx || !crypto_handshake_is_ready(ctx)) {
551 return SET_ERRNO(ERROR_INVALID_STATE, "Handshake not ready for rekeying: ctx=%p, ready=%d", ctx,
552 ctx ? crypto_handshake_is_ready(ctx) : 0);
553 }
554
555 // Validate packet size (should be 32 bytes for X25519 public key)
556 if (packet_len != CRYPTO_PUBLIC_KEY_SIZE) {
557 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid REKEY_RESPONSE packet size: %zu (expected %d)", packet_len,
558 CRYPTO_PUBLIC_KEY_SIZE);
559 }
560
561 if (!ctx->crypto_ctx.rekey_in_progress || !ctx->crypto_ctx.has_temp_key) {
562 return SET_ERRNO(ERROR_INVALID_STATE, "No rekey in progress or temp key missing");
563 }
564
565 log_debug("Received REKEY_RESPONSE with peer's new ephemeral public key (32 bytes)");
566
567 // Process peer's public key and compute new shared secret
568 crypto_result_t result = crypto_rekey_process_response(&ctx->crypto_ctx, packet);
569 if (result != CRYPTO_OK) {
570 crypto_rekey_abort(&ctx->crypto_ctx);
571 return SET_ERRNO(ERROR_CRYPTO, "Failed to process REKEY_RESPONSE: %s", crypto_result_to_string(result));
572 }
573
574 log_debug("REKEY_RESPONSE processed successfully, new shared secret computed (initiator side)");
575 return ASCIICHAT_OK;
576}

583asciichat_error_t crypto_handshake_process_rekey_complete(crypto_handshake_context_t *ctx, const uint8_t *packet,
584 size_t packet_len) {
585 if (!ctx || !crypto_handshake_is_ready(ctx)) {
586 return SET_ERRNO(ERROR_INVALID_STATE, "Handshake not ready for rekeying: ctx=%p, ready=%d", ctx,
587 ctx ? crypto_handshake_is_ready(ctx) : 0);
588 }
589
590 if (!ctx->crypto_ctx.rekey_in_progress || !ctx->crypto_ctx.has_temp_key) {
591 return SET_ERRNO(ERROR_INVALID_STATE, "No rekey in progress or temp key missing");
592 }
593
594 log_debug("Received REKEY_COMPLETE packet (%zu bytes), verifying with NEW key", packet_len);
595
596 // Temporarily swap keys to decrypt with NEW key
597 uint8_t old_shared_key[CRYPTO_SHARED_KEY_SIZE];
598 memcpy(old_shared_key, ctx->crypto_ctx.shared_key, CRYPTO_SHARED_KEY_SIZE);
599 memcpy(ctx->crypto_ctx.shared_key, ctx->crypto_ctx.temp_shared_key, CRYPTO_SHARED_KEY_SIZE);
600
601 // Attempt to decrypt with NEW key
602 uint8_t plaintext[256];
603 size_t plaintext_len = 0;
604 crypto_result_t result =
605 crypto_decrypt(&ctx->crypto_ctx, packet, packet_len, plaintext, sizeof(plaintext), &plaintext_len);
606
607 // Restore old key immediately
608 memcpy(ctx->crypto_ctx.shared_key, old_shared_key, CRYPTO_SHARED_KEY_SIZE);
609 sodium_memzero(old_shared_key, sizeof(old_shared_key));
610
611 if (result != CRYPTO_OK) {
612 crypto_rekey_abort(&ctx->crypto_ctx);
613 return SET_ERRNO(ERROR_CRYPTO, "REKEY_COMPLETE decryption failed (key mismatch): %s",
614 crypto_result_to_string(result));
615 }
616
617 log_debug("REKEY_COMPLETE verified successfully, committing to new key");
618
619 // Commit to new key (atomic switch)
620 result = crypto_rekey_commit(&ctx->crypto_ctx);
621 if (result != CRYPTO_OK) {
622 return SET_ERRNO(ERROR_CRYPTO, "Failed to commit rekey: %s", crypto_result_to_string(result));
623 }
624
625 log_debug("Session rekeying completed successfully (responder side)");
626 return ASCIICHAT_OK;
627}
