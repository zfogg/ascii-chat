31bool terminal_should_color_output(int fd) {
32 // Check color_mode setting - if explicitly set to NONE, disable colors
33 int color_mode = GET_OPTION(color_mode);
34 if (color_mode == COLOR_MODE_NONE) {
35 return false; // Color mode explicitly disabled
36 }
37
38 // Get current color setting (auto/true/false)
39 int color_setting = GET_OPTION(color);
40
41 // Priority 1: --color=true - Force colors ON (overrides everything)
42 if (color_setting == COLOR_SETTING_TRUE) {
43 return true; // ALWAYS colorize
44 }
45
46 // Priority 2: --color=false - Force colors OFF (overrides everything)
47 if (color_setting == COLOR_SETTING_FALSE) {
48 return false; // NEVER colorize
49 }
50
51 // Priority 3: --color=auto (default) - Smart detection
52 // Special case: Show colors for --help and --version even if not a TTY
53 // Check for help/version in global argv (set by main.c early)
54 if (g_argc > 1 && g_argv) {
55 for (int i = 1; i < g_argc; i++) {
56 if (strcmp(g_argv[i], "--help") == 0 || strcmp(g_argv[i], "-h") == 0 || strcmp(g_argv[i], "--version") == 0 ||
57 strcmp(g_argv[i], "-v") == 0 || strcmp(g_argv[i], "--show-capabilities") == 0) {
58 // For help/version/show-capabilities, use colors only if output is a TTY
59 // (respect piping convention: no colors unless --color=true)
60 int is_tty = platform_isatty(fd);
61 if (!is_tty) {
62 // Piping detected - fall through to check environment overrides
63 break;
64 }
65
66 // It's a TTY - show colors for help. Check environment overrides.
67 if (SAFE_GETENV("ASCII_CHAT_COLOR")) {
68 return true; // ASCII_CHAT_COLOR env var forces colors
69 }
70
71 // CLAUDECODE environment variable (LLM automation) - disable colors
72 if (SAFE_GETENV("CLAUDECODE")) {
73 return false; // NO colors in Claude Code environment
74 }
75
76 // For help output to TTY, show colors by default
77 return true;
78 }
79 }
80 }

83 if (SAFE_GETENV("ASCII_CHAT_COLOR")) {

206bool terminal_should_force_stderr(void) {
207 // If stdout is piped/redirected, force logs to stderr to avoid corruption
208 if (terminal_is_piped_output()) {
209 // Unless in TESTING environment where test framework may capture stdout
210 const char *testing = SAFE_GETENV("TESTING");
211 if (testing && strcmp(testing, "1") == 0) {
212 return false; // Allow stdout in test environments
213 }
214 return true; // Force stderr when piped
215 }
216 return false; // stdout is TTY, no need to force stderr

219bool terminal_can_prompt_user(void) {
220 // Must be fully interactive (stdin and stdout are TTYs)
221 if (!terminal_is_interactive()) {
222 return false;
223 }
224
225 // Must not be in snapshot mode (non-interactive capture)
226 if (GET_OPTION(snapshot_mode)) {
227 return false;
228 }
229
230 // Must not have automated prompt responses configured
231 const char *auto_response = SAFE_GETENV("ASCII_CHAT_QUESTION_PROMPT_RESPONSE");
232 if (auto_response && *auto_response != '\0') {
233 return false; // Automated responses configured, not interactive
234 }
235
236 // All checks passed, interactive prompts are appropriate
237 return true;
238}

240int terminal_choose_log_fd(log_level_t level) {
241 // When force_stderr is enabled (client mode), send ALL logs to stderr
242 // When terminal is NOT interactive (piped/redirected), send ALL logs to stderr
243 // to keep stdout clean for piped data (JSON, frames, etc.)
244 if (log_get_force_stderr() || !terminal_is_interactive()) {
245 return STDERR_FILENO;
246 }
247
248 // In interactive mode, route based on level:
249 // WARN and above (ERROR, FATAL) go to stderr
250 // Others (DEV, DEBUG, INFO) go to stdout
251 if (level >= LOG_WARN) {
252 return STDERR_FILENO;
253 }
254
255 return STDOUT_FILENO;
256}
