89static asciichat_error_t discovery_send_sdp(const uint8_t session_id[16], const uint8_t recipient_id[16],
90 const char *sdp_type, const char *sdp, void *user_data) {
91 (void)recipient_id; // Unused - direct connection, no need to route
92 (void)user_data; // Unused
93
94 ensure_mutex_initialized();
95 mutex_lock(&g_webrtc_mutex);
96
97 // Verify state
98 if (!g_tcp_transport) {
99 mutex_unlock(&g_webrtc_mutex);
100 SET_ERRNO(ERROR_INVALID_STATE, "TCP transport not set for direct signaling");
101 return ERROR_INVALID_STATE;
102 }
103
104 if (!g_session_context.is_set) {
105 mutex_unlock(&g_webrtc_mutex);
106 SET_ERRNO(ERROR_INVALID_STATE, "Session context not set for direct signaling");
107 return ERROR_INVALID_STATE;
108 }
109
110 // Verify inputs
111 if (!sdp) {
112 mutex_unlock(&g_webrtc_mutex);
113 SET_ERRNO(ERROR_INVALID_PARAM, "SDP is NULL");
114 return ERROR_INVALID_PARAM;
115 }
116
117 size_t sdp_len = strlen(sdp);
118 if (sdp_len > 4096) {
119 mutex_unlock(&g_webrtc_mutex);
120 SET_ERRNO(ERROR_INVALID_PARAM, "SDP too large (>4096 bytes)");
121 return ERROR_INVALID_PARAM;
122 }
123
124 // Construct packet
125 // Header: acip_webrtc_sdp_t (defined in network/acip/acds.h)
126 // Followed by: SDP string
127 uint8_t sdp_msg_buf[sizeof(acip_webrtc_sdp_t) + 4096];
128 acip_webrtc_sdp_t *sdp_msg = (acip_webrtc_sdp_t *)sdp_msg_buf;
129
130 memcpy(sdp_msg->session_id, session_id, 16);
131 memcpy(sdp_msg->sender_id, g_session_context.participant_id, 16);
132 memset(sdp_msg->recipient_id, 0, 16); // Broadcast to all
133 sdp_msg->sdp_type = (strcmp(sdp_type, "answer") == 0) ? 1 : 0; // 0=offer, 1=answer
134 sdp_msg->sdp_len = (uint16_t)sdp_len;
135 memcpy(sdp_msg_buf + sizeof(acip_webrtc_sdp_t), sdp, sdp_len);
136
137 // Send via direct TCP transport (not ACDS)
138 size_t msg_size = sizeof(acip_webrtc_sdp_t) + sdp_len;
139 asciichat_error_t result =
140 packet_send_via_transport(g_tcp_transport, PACKET_TYPE_ACIP_WEBRTC_SDP, sdp_msg_buf, msg_size, 0);
141
142 mutex_unlock(&g_webrtc_mutex);
143
144 if (result != ASCIICHAT_OK) {
145 log_error("Failed to send SDP via direct TCP: %d", result);
146 return result;
147 }
148
149 log_debug("Sent SDP %s directly to peer (TCP transport)", sdp_type);
150 return ASCIICHAT_OK;
151}

158static asciichat_error_t discovery_send_ice(const uint8_t session_id[16], const uint8_t recipient_id[16],
159 const char *candidate, const char *mid, void *user_data) {
160 (void)recipient_id; // Unused - direct connection, no need to route
161 (void)user_data; // Unused
162
163 ensure_mutex_initialized();
164 mutex_lock(&g_webrtc_mutex);
165
166 // Verify state
167 if (!g_tcp_transport) {
168 mutex_unlock(&g_webrtc_mutex);
169 SET_ERRNO(ERROR_INVALID_STATE, "TCP transport not set for direct signaling");
170 return ERROR_INVALID_STATE;
171 }
172
173 if (!g_session_context.is_set) {
174 mutex_unlock(&g_webrtc_mutex);
175 SET_ERRNO(ERROR_INVALID_STATE, "Session context not set for direct signaling");
176 return ERROR_INVALID_STATE;
177 }
178
179 // Verify inputs
180 if (!candidate || !mid) {
181 mutex_unlock(&g_webrtc_mutex);
182 SET_ERRNO(ERROR_INVALID_PARAM, "Candidate or mid is NULL");
183 return ERROR_INVALID_PARAM;
184 }
185
186 // Calculate payload length (candidate + null + mid + null)
187 size_t candidate_len = strlen(candidate);
188 size_t mid_len = strlen(mid);
189 size_t payload_len = candidate_len + 1 + mid_len + 1;
190
191 if (payload_len > 4096) {
192 mutex_unlock(&g_webrtc_mutex);
193 SET_ERRNO(ERROR_INVALID_PARAM, "ICE candidate + mid too large (>4096 bytes)");
194 return ERROR_INVALID_PARAM;
195 }
196
197 // Construct packet
198 // Header: acip_webrtc_ice_t (defined in network/acip/acds.h)
199 // Followed by: candidate string + null + mid string + null
200 uint8_t ice_msg_buf[sizeof(acip_webrtc_ice_t) + 4096];
201 acip_webrtc_ice_t *ice_msg = (acip_webrtc_ice_t *)ice_msg_buf;
202
203 // Fill header
204 memcpy(ice_msg->session_id, session_id, 16);
205 memcpy(ice_msg->sender_id, g_session_context.participant_id, 16);
206 memset(ice_msg->recipient_id, 0, 16); // Broadcast to all
207 ice_msg->candidate_len = HOST_TO_NET_U16((uint16_t)candidate_len);
208
209 // Copy candidate and mid after header
210 uint8_t *payload = ice_msg_buf + sizeof(acip_webrtc_ice_t);
211 memcpy(payload, candidate, candidate_len);
212 payload[candidate_len] = '\0';
213 memcpy(payload + candidate_len + 1, mid, mid_len);
214 payload[candidate_len + 1 + mid_len] = '\0';
215
216 log_debug("Sending ICE candidate directly to peer (candidate_len=%zu, mid=%s)", candidate_len, mid);
217
218 // Send via direct TCP transport (not ACDS)
219 size_t msg_size = sizeof(acip_webrtc_ice_t) + payload_len;
220 asciichat_error_t result =
221 packet_send_via_transport(g_tcp_transport, PACKET_TYPE_ACIP_WEBRTC_ICE, ice_msg_buf, msg_size, 0);
222
223 mutex_unlock(&g_webrtc_mutex);
224
225 if (result != ASCIICHAT_OK) {
226 log_error("Failed to send ICE candidate via direct TCP: %d", result);
227 return result;
228 }
229
230 log_debug("Sent ICE candidate directly to peer (TCP transport)");
231 return ASCIICHAT_OK;
232}

238webrtc_signaling_callbacks_t discovery_webrtc_get_direct_signaling_callbacks(acip_transport_t *tcp_transport,
239 const uint8_t session_id[16],
240 const uint8_t participant_id[16]) {
241 ensure_mutex_initialized();
242 mutex_lock(&g_webrtc_mutex);
243
244 g_tcp_transport = tcp_transport;
245
246 if (session_id && participant_id) {
247 memcpy(g_session_context.session_id, session_id, 16);
248 memcpy(g_session_context.participant_id, participant_id, 16);
249 g_session_context.is_set = true;
250 }
251
252 mutex_unlock(&g_webrtc_mutex);
253
254 webrtc_signaling_callbacks_t callbacks = {
255 .send_sdp = discovery_send_sdp,
256 .send_ice = discovery_send_ice,
257 .user_data = NULL, // Unused - we use global state
258 };
259
260 return callbacks;
