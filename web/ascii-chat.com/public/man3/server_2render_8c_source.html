935 // Accumulate all samples (including 0 or partial) until we have a full Opus frame
936 // This maintains continuous stream without silence padding
937 START_TIMER("accum_%s", client_id_snapshot);
938
939 int space_available = OPUS_FRAME_SAMPLES - opus_frame_accumulated;
940 int samples_to_copy = (samples_mixed <= space_available) ? samples_mixed : space_available;
941
942 // Only copy if we have samples, otherwise just wait for next frame
943 if (samples_to_copy > 0) {
944 SAFE_MEMCPY(opus_frame_buffer + opus_frame_accumulated,
945 (OPUS_FRAME_SAMPLES - opus_frame_accumulated) * sizeof(float), mix_buffer,
946 samples_to_copy * sizeof(float));
947 opus_frame_accumulated += samples_to_copy;
948 }
949
950 STOP_TIMER_AND_LOG_EVERY(dev, NS_PER_SEC_INT, 2 * NS_PER_MS_INT, "accum_%s", "Accumulate for client %s: took",
951 client_id_snapshot);
952
953 // Only encode and send when we have accumulated a full Opus frame
954 if (opus_frame_accumulated >= OPUS_FRAME_SAMPLES) {
955 // OPTIMIZATION: Don't check queue depth every iteration - it's expensive (requires lock)
956 // Only check periodically every 100 iterations (~0.6s at 172 fps)
957 // NOTE: backpressure_check_counter is now per-thread (not static), so each client thread has its own counter
958 bool apply_backpressure = false;
959
960 if (++backpressure_check_counter >= 100) {
961 backpressure_check_counter = 0;
962 size_t queue_depth = packet_queue_size(audio_queue_snapshot);
963 // Opus frames are produced at ~50 FPS (20ms each)
964 // Reduced threshold from 50 to 10 packets (~200ms buffer) to prevent audio gaps
965 // On localhost, queue shouldn't back up at all; aggressive thresholds cause frame skipping
966 apply_backpressure = (queue_depth > 10); // > 10 packets = ~200ms buffered at 50 FPS
967
968 if (apply_backpressure) {
969 log_warn_every(4500 * US_PER_MS_INT,
970 "Audio backpressure for client %s: queue depth %zu packets (%.1fms buffered)",
971 client_id_snapshot, queue_depth, (float)queue_depth / 50.0f * 1000.0f);
972 }
973 }
974
975 if (apply_backpressure) {
976 // Skip this packet to let the queue drain
977 // Reset accumulation buffer so fresh samples can be captured on next iteration.
978 // Without this reset, we'd loop forever with stale audio and no space for new samples
979 opus_frame_accumulated = 0;
980 // Reduced sleep from 5.8ms to 1ms to minimize audio gaps on localhost
981 platform_sleep_ns(1 * NS_PER_MS_INT);
982 continue;
983 }
984
985 // Encode accumulated Opus frame (960 samples = 20ms @ 48kHz)
986 uint8_t opus_buffer[1024]; // Max Opus frame size
987
988 START_TIMER("opus_encode_%s", client_id_snapshot);
989
990 int opus_size =
991 opus_codec_encode(opus_encoder, opus_frame_buffer, OPUS_FRAME_SAMPLES, opus_buffer, sizeof(opus_buffer));
992
993 STOP_TIMER_AND_LOG_EVERY(dev, NS_PER_SEC_INT, 10 * NS_PER_MS_INT, "opus_encode_%s",
994 "Opus encode for client %s: took", client_id_snapshot);
995
996 // DEBUG: Log mix buffer and encoding results to see audio levels being sent
997 {
998 float peak = 0.0f, rms = 0.0f;
999 for (int i = 0; i < OPUS_FRAME_SAMPLES; i++) {
1000 float abs_val = fabsf(opus_frame_buffer[i]);
1001 if (abs_val > peak)
1002 peak = abs_val;
1003 rms += opus_frame_buffer[i] * opus_frame_buffer[i];
1004 }
1005 rms = sqrtf(rms / OPUS_FRAME_SAMPLES);
1006 // NOTE: server_audio_frame_count is now per-thread (not static), so each client thread has its own counter
1007 server_audio_frame_count++;
1008 if (server_audio_frame_count <= 5 || server_audio_frame_count % 20 == 0) {
1009 // Log first 4 samples to verify they look like valid audio (not NaN/Inf/garbage)
1010 log_dev_every(4500 * US_PER_MS_INT,
1011 "Server audio frame #%d for client %u: samples_mixed=%d, Peak=%.6f, RMS=%.6f, opus_size=%d, "
1012 "first4=[%.4f,%.4f,%.4f,%.4f]",
1013 server_audio_frame_count, client_id_snapshot, samples_mixed, peak, rms, opus_size,
1014 opus_frame_buffer[0], opus_frame_buffer[1], opus_frame_buffer[2], opus_frame_buffer[3]);
1015 }
1016 }
1017
1018 // Always reset accumulation buffer after attempting to encode - we've consumed these samples
1019 // If we don't reset, new audio samples would be dropped while stale data sits in the buffer
1020 opus_frame_accumulated = 0;
1021
1022 if (opus_size <= 0) {
1023 log_error("Failed to encode audio to Opus for client %s: opus_size=%d", client_id_snapshot, opus_size);
1024 } else {
1025 // Wrap Opus data in proper batch packet format for client parsing:
1026 // Header: sample_rate (4B) + frame_duration (4B) + frame_count (4B)
1027 // Frame sizes: frame_count * 2B (one uint16 per frame with Opus data size)
1028 // Opus data: raw encoded audio
1029
1030 // Fixed values for Opus encoding (standard 48kHz, 960 samples per frame = 20ms)
1031 const uint32_t sample_rate = 48000;
1032 const uint32_t frame_duration = 20; // milliseconds (960 samples / 48kHz = 20ms)
1033 const uint32_t frame_count = 1; // one Opus frame per packet
1034 const uint16_t frame_size = (uint16_t)opus_size;
1035
1036 // Calculate total packet size: 16-byte header + frame_sizes + opus data
1037 const size_t header_size = 16; // 4 uint32s (sample_rate, frame_duration, frame_count, reserved)
1038 const size_t frame_sizes_size = (size_t)frame_count * sizeof(uint16_t);
1039 const size_t total_packet_size = header_size + frame_sizes_size + (size_t)opus_size;
1040
1041 // Allocate buffer for the complete packet
1042 uint8_t *packet_buffer = SAFE_MALLOC(total_packet_size, uint8_t *);
1043 if (!packet_buffer) {
1044 log_error("Failed to allocate Opus batch packet buffer for client %s (size=%zu)", client_id_snapshot,
1045 total_packet_size);
1046 } else {
1047 // Build header in network byte order
1048 uint8_t *ptr = packet_buffer;
1049
1050 // Write sample_rate (4 bytes, network byte order)
1051 uint32_t sr_net = htonl(sample_rate);
1052 memcpy(ptr, &sr_net, 4);
1053 ptr += 4;
1054
1055 // Write frame_duration (4 bytes, network byte order)
1056 uint32_t fd_net = htonl(frame_duration);
1057 memcpy(ptr, &fd_net, 4);
1058 ptr += 4;
1059
1060 // Write frame_count (4 bytes, network byte order)
1061 uint32_t fc_net = htonl(frame_count);
1062 memcpy(ptr, &fc_net, 4);
1063 ptr += 4;
1064
1065 // Write reserved field (4 bytes, all zeros)
1066 uint32_t reserved = 0;
1067 memcpy(ptr, &reserved, 4);
1068 ptr += 4;
1069
1070 // Write frame sizes (2 bytes per frame, network byte order)
1071 for (uint32_t i = 0; i < frame_count; i++) {
1072 uint16_t fs_net = htons(frame_size);
1073 memcpy(ptr, &fs_net, 2);
1074 ptr += 2;
1075 }
1076
1077 // Copy Opus data
1078 memcpy(ptr, opus_buffer, (size_t)opus_size);
1079
1080 // Queue the properly formatted packet
1081 START_TIMER("audio_queue_%s", client_id_snapshot);
1082
1083 int result = packet_queue_enqueue(audio_queue_snapshot, PACKET_TYPE_AUDIO_OPUS_BATCH, packet_buffer,
1084 total_packet_size, 0, true);
1085
1086 STOP_TIMER_AND_LOG_EVERY(dev, NS_PER_SEC_INT, 1 * NS_PER_MS_INT, "audio_queue_%s",
1087 "Audio queue for client %s: took", client_id_snapshot);
1088
1089 if (result < 0) {
1090 log_debug("Failed to queue Opus audio for client %s", client_id_snapshot);
1091 } else {
1092 // FPS tracking - audio packet successfully queued (handles lag detection and periodic reporting)
1093 fps_frame_ns(&audio_fps_tracker, time_get_ns(), "audio packet queued");
1094 }
1095
1096 // Free the allocated packet buffer after queuing (packet_queue_enqueue copies data)
1097 SAFE_FREE(packet_buffer);
1098 }
1099 }
1100 // NOTE: opus_frame_accumulated is already reset at line 928 after encode attempt
1101 }
1102
1103 // Audio mixing rate limiting using adaptive sleep system
1104 // Target: 10ms intervals (100 FPS) for 480 samples @ 48kHz
1105 // Use queue_depth=0 and target_depth=0 for constant-rate audio processing
1106 adaptive_sleep_do(&audio_sleep_state, 0, 0);
1107 }
1108
1109#ifdef DEBUG_THREADS
1110 log_debug("Audio render thread stopped for client %u", thread_client_id);
1111#endif
1112
1113 // Clean up Opus encoder
1114 if (opus_encoder) {
1115 opus_codec_destroy(opus_encoder);
1116 }
1117
1118 // Clean up thread-local error context before exit
1119 asciichat_errno_destroy();
1120
1121 return NULL;
1122}
