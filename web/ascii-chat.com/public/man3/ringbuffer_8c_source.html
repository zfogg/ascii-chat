19/* ============================================================================
20 * Ring Buffer Implementation
21 * ============================================================================
22 *
23 * THREAD SAFETY NOTE: This ring buffer is designed for single-producer,
24 * single-consumer (SPSC) use only. The atomic operations provide memory
25 * ordering guarantees but do NOT support concurrent writes from multiple
26 * producers. For multi-writer scenarios, external synchronization is required.
27 */

29ringbuffer_t *ringbuffer_create(size_t element_size, size_t capacity) {
30 if (element_size == 0 || capacity == 0) {
31 SET_ERRNO(ERROR_INVALID_PARAM, "Invalid ring buffer parameters: element_size=%zu, capacity=%zu", element_size,
32 capacity);
33 return NULL;
34 }
35
36 ringbuffer_t *rb;
37 rb = SAFE_CALLOC(1, sizeof(ringbuffer_t), ringbuffer_t *);
38
39 /* Round capacity up to power of 2 for optimization */
40 size_t actual_capacity = math_next_power_of_two(capacity);
41
42 rb->buffer = SAFE_CALLOC(actual_capacity, element_size, char *);
43
44 rb->element_size = element_size;
45 rb->capacity = actual_capacity;
46 rb->is_power_of_two = true;
47 rb->capacity_mask = actual_capacity - 1;
48 /* Atomic fields are zero-initialized by SAFE_CALLOC */
49
50 return rb;
51}

60bool ringbuffer_write(ringbuffer_t *rb, const void *data) {
61 if (!rb || !data)
62 return false;
63
64 size_t current_size = atomic_load_u64(&rb->size);
65 if (current_size >= rb->capacity) {
66 return false; /* Buffer full */
67 }
68
69 size_t head = atomic_load_u64(&rb->head);
70 size_t next_head = (head + 1) & rb->capacity_mask;
71
72 /* Copy data */
73 SAFE_MEMCPY(rb->buffer + (head * rb->element_size), rb->element_size, data, rb->element_size);
74
75 /* Update head and size atomically */
76 atomic_store_u64(&rb->head, next_head);
77 atomic_fetch_add_u64(&rb->size, 1);
78
79 return true;
80}

82bool ringbuffer_read(ringbuffer_t *rb, void *data) {
83 if (!rb || !data)
84 return false;
85
86 size_t current_size = atomic_load_u64(&rb->size);
87 if (current_size == 0) {
88 return false; /* Buffer empty */
89 }
90
91 size_t tail = atomic_load_u64(&rb->tail);
92 size_t next_tail = (tail + 1) & rb->capacity_mask;
93
94 /* Copy data */
95 SAFE_MEMCPY(data, rb->element_size, rb->buffer + (tail * rb->element_size), rb->element_size);
96
97 /* Update tail and size atomically */
98 atomic_store_u64(&rb->tail, next_tail);
99 atomic_fetch_sub_u64(&rb->size, 1);
100
101 return true;
102}

104bool ringbuffer_peek(ringbuffer_t *rb, void *data) {
105 if (!rb || !data)
106 return false;
107
108 size_t current_size = atomic_load_u64(&rb->size);
109 if (current_size == 0) {
110 return false; /* Buffer empty */
111 }
112
113 size_t tail = atomic_load_u64(&rb->tail);
114
115 /* Copy data without updating tail */
116 SAFE_MEMCPY(data, rb->element_size, rb->buffer + (tail * rb->element_size), rb->element_size);
117
118 return true;

151

158 SAFE_FREE(fb);

169 }

221bool framebuffer_write_frame(framebuffer_t *fb, const char *frame_data, size_t frame_size) {
222 if (!fb || !frame_data || frame_size == 0)
223 return false;
224
225 // Validate frame size to prevent overflow
226 if (frame_size > 10 * 1024 * 1024) { // 10MB max for ASCII frames
227 SET_ERRNO(ERROR_INVALID_PARAM, "Rejecting oversized frame: %zu bytes", frame_size);
228 return false;
229 }
230
231 // Thread-safe access to framebuffer (was missing, causing race conditions)
232 mutex_lock(&fb->mutex);
233
234 // Check if buffer is full - if so, we need to drop the oldest frame
235 if (ringbuffer_size(fb->rb) >= fb->rb->capacity) {
236 // Buffer is full, read and free the oldest frame before writing new one
237 frame_t old_frame;
238 if (ringbuffer_read(fb->rb, &old_frame)) {
239 if (IS_FRAME_VALID(&old_frame)) {
240 MARK_FRAME_FREED(&old_frame);
241 // Use buffer_pool_free since data was allocated with buffer_pool_alloc
242 buffer_pool_free(NULL, old_frame.data, old_frame.size);
243 } else if (!IS_MAGIC_VALID(old_frame.magic, FRAME_MAGIC)) {
244 SET_ERRNO(ERROR_INVALID_STATE, "CORRUPTION: Invalid old frame magic 0x%x when dropping", old_frame.magic);
245 }
246 }
247 }
248
249 // Allocate a copy of the frame data using buffer pool for better performance
250 char *frame_copy = (char *)buffer_pool_alloc(NULL, frame_size + 1);
251 if (!frame_copy) {
252 mutex_unlock(&fb->mutex);
253 SET_ERRNO(ERROR_MEMORY, "Failed to allocate %zu bytes from buffer pool for frame", frame_size + 1);
254 return false;
255 }
256
257 SAFE_MEMCPY(frame_copy, frame_size, frame_data, frame_size);
258 frame_copy[frame_size] = '\0'; // Ensure null termination
259
260 // Create a frame_t struct with the copy (store allocated size for proper cleanup)
261 frame_t frame = {.magic = MAGIC_FRAME_VALID, .size = frame_size + 1, .data = frame_copy};
262
263 bool result = ringbuffer_write(fb->rb, &frame);
264
265 if (!result) {
266 // If we still couldn't write to ringbuffer, return the buffer to pool
267 buffer_pool_free(NULL, frame_copy, frame_size + 1);
268 SET_ERRNO(ERROR_INVALID_STATE, "Failed to write frame to ringbuffer even after dropping oldest");
269 }
270
271 mutex_unlock(&fb->mutex);
272 return result;
273}

275bool framebuffer_read_frame(framebuffer_t *fb, frame_t *frame) {
276 if (!fb || !frame) {
277 return false;
278 }
279
280 // Initialize frame to safe values
281 frame->magic = 0;
282 frame->data = NULL;
283 frame->size = 0;
284
285 // Thread-safe access to framebuffer (fixes TOCTOU race condition)
286 mutex_lock(&fb->mutex);
287
288 bool result = ringbuffer_read(fb->rb, frame);
289
290 // Validate the frame we just read
291 if (result) {
292 if (!IS_MAGIC_VALID(frame->magic, FRAME_MAGIC)) {
293 SET_ERRNO(ERROR_INVALID_STATE, "CORRUPTION: Invalid frame magic 0x%x (expected 0x%x)", frame->magic, FRAME_MAGIC);
294 frame->data = NULL;
295 frame->size = 0;
296 mutex_unlock(&fb->mutex);
297 return false;
298 }
299
300 if (IS_FRAME_FREED(frame)) {
301 SET_ERRNO(ERROR_INVALID_STATE, "CORRUPTION: Reading already-freed frame!");
302 frame->data = NULL;
303 frame->size = 0;
304 mutex_unlock(&fb->mutex);
305 return false;
306 }

309 SET_ERRNO(ERROR_INVALID_STATE, "CORRUPTION: Frame size too large: %zu", frame->size);
310 SAFE_FREE(frame->data);
311 frame->data = NULL;
312 frame->size = 0;
313 mutex_unlock(&fb->mutex);
314 return false;

316 }

322void framebuffer_clear(framebuffer_t *fb) {
323 if (!fb || !fb->rb)
324 return;
325
326 // Thread-safe access to framebuffer (was missing, causing race conditions)
327 mutex_lock(&fb->mutex);
328
329 // Check the element size to determine frame type
330 if (fb->rb->element_size == sizeof(multi_source_frame_t)) {
331 // Multi-source frame buffer - read and free multi_source_frame_t
332 multi_source_frame_t multi_frame;
333 while (ringbuffer_read(fb->rb, &multi_frame)) {
334 if (IS_FRAME_VALID((frame_t *)&multi_frame)) {
335 MARK_FRAME_FREED((frame_t *)&multi_frame); // Mark as freed to detect use-after-free
336 buffer_pool_free(NULL, multi_frame.data, multi_frame.size);
337 } else if (!IS_MAGIC_VALID(multi_frame.magic, FRAME_MAGIC) && multi_frame.magic != 0) {
338 SET_ERRNO(ERROR_INVALID_STATE, "CORRUPTION: Invalid multi-source frame magic 0x%x during clear",
339 multi_frame.magic);
340 }
341 }
342 } else if (fb->rb->element_size == sizeof(frame_t)) {
343 // Single-source frame buffer - read and free frame_t
344 frame_t frame;
345 while (ringbuffer_read(fb->rb, &frame)) {
346 if (IS_FRAME_VALID(&frame)) {
347 MARK_FRAME_FREED(&frame); // Mark as freed to detect use-after-free
348 buffer_pool_free(NULL, frame.data, frame.size);
349 } else if (!IS_MAGIC_VALID(frame.magic, FRAME_MAGIC) && frame.magic != 0) {
350 SET_ERRNO(ERROR_INVALID_STATE, "CORRUPTION: Invalid frame magic 0x%x during clear", frame.magic);
351 }
352 }
353 } else {
354 SET_ERRNO(ERROR_INVALID_STATE, "Unknown frame buffer type with element size %zu", fb->rb->element_size);
355 }
356
357 // Clear the ringbuffer indices
358 ringbuffer_clear(fb->rb);
359
360 // Zero out the entire buffer to prevent any dangling pointers
361 if (fb->rb->buffer) {
362 // Check for integer overflow before multiplication
363 if (fb->rb->capacity > SIZE_MAX / fb->rb->element_size) {
364 SET_ERRNO(ERROR_INVALID_PARAM, "Buffer size would overflow: capacity=%zu, element_size=%zu", fb->rb->capacity,
365 fb->rb->element_size);
366 mutex_unlock(&fb->mutex);
367 return;
368 }
369 size_t buffer_size = fb->rb->capacity * fb->rb->element_size;
370 SAFE_MEMSET(fb->rb->buffer, buffer_size, 0, buffer_size);
371 }
372
373 mutex_unlock(&fb->mutex);
374}

378bool framebuffer_write_multi_frame(framebuffer_t *fb, const char *frame_data, size_t frame_size,
379 uint32_t source_client_id, uint32_t frame_sequence, uint32_t timestamp) {
380 if (!fb || !fb->rb || !frame_data || frame_size == 0) {
381 return false;
382 }
383
384 // Allocate memory for frame data using buffer pool for better performance
385 char *data_copy = (char *)buffer_pool_alloc(NULL, frame_size);
386 if (!data_copy) {
387 SET_ERRNO(ERROR_MEMORY, "Failed to allocate %zu bytes from buffer pool for multi-source frame", frame_size);
388 return false;
389 }
390
391 // Copy frame data
392 SAFE_MEMCPY(data_copy, frame_size, frame_data, frame_size);
393
394 // Convert client ID number to string for storage in frame metadata
395 char client_id_str[64];
396 snprintf(client_id_str, sizeof(client_id_str), "%u", source_client_id);
397
398 // Create multi-source frame
399 multi_source_frame_t multi_frame = {
400 .magic = MAGIC_FRAME_VALID,
401 .source_client_id = {0},
402 .frame_sequence = frame_sequence,
403 .timestamp = timestamp,
404 .size = frame_size,
405 .data = data_copy,
406 };
407 SAFE_MEMCPY(multi_frame.source_client_id, sizeof(multi_frame.source_client_id), client_id_str, strlen(client_id_str) + 1);
408
409 // Thread-safe access to framebuffer
410 mutex_lock(&fb->mutex);
411
412 // Try to write to ring buffer
413 bool success = ringbuffer_write(fb->rb, &multi_frame);
414 if (!success) {
415 // Buffer full, return the buffer to pool
416 buffer_pool_free(NULL, data_copy, frame_size);
417 log_debug("Frame buffer full, dropping multi-source frame from client %u", source_client_id);
418 }
419
420 mutex_unlock(&fb->mutex);
421 return success;
422}
