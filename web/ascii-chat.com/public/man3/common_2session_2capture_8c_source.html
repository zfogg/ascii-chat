114static void calculate_optimal_dimensions(ssize_t original_width, ssize_t original_height, ssize_t max_width,
115 ssize_t max_height, ssize_t *result_width, ssize_t *result_height) {
116 // Calculate original aspect ratio
117 float img_aspect = (float)original_width / (float)original_height;
118
119 // Check if image needs resizing
120 if (original_width <= max_width && original_height <= max_height) {
121 // Image is already within bounds - use as-is
122 *result_width = original_width;
123 *result_height = original_height;
124 return;
125 }
126
127 // Determine scaling factor based on which dimension is the limiting factor
128 if ((float)max_width / (float)max_height > img_aspect) {
129 // Max box is wider than image aspect - scale by height
130 *result_height = max_height;
131 *result_width = (ssize_t)(max_height * img_aspect);
132 } else {
133 // Max box is taller than image aspect - scale by width
134 *result_width = max_width;
135 *result_height = (ssize_t)(max_width / img_aspect);
136 }
137}

153session_capture_ctx_t *session_network_capture_create(uint32_t target_fps) {
154 // Network modes don't capture media - create minimal context for keyboard/audio only
155 session_capture_ctx_t *ctx = SAFE_CALLOC(1, sizeof(session_capture_ctx_t), session_capture_ctx_t *);
156 if (!ctx) {
157 return NULL;
158 }
159
160 // Set FPS (use provided value or default to 60)
161 ctx->target_fps = target_fps > 0 ? target_fps : 60;
162 ctx->resize_for_network = false;
163
164 // Audio and keyboard will be set up separately by callers
165 ctx->audio_enabled = false;
166 ctx->source = NULL;
167
168 // Initialize adaptive sleep for consistency
169 uint64_t baseline_sleep_ns = NS_PER_SEC_INT / ctx->target_fps;
170 adaptive_sleep_config_t sleep_config = {.baseline_sleep_ns = baseline_sleep_ns,
171 .min_speed_multiplier = 0.5,
172 .max_speed_multiplier = 2.0,
173 .speedup_rate = 0.1,
174 .slowdown_rate = 0.1};
175 adaptive_sleep_init(&ctx->sleep_state, &sleep_config);
176
177 // Initialize minimal FPS tracker (won't be used but keep structure consistent)
178 char *tracker_name = SAFE_MALLOC(32, char *);
179 if (!tracker_name) {
180 SAFE_FREE(ctx);
181 return NULL;

184 fps_init(&ctx->fps_tracker, 60, tracker_name);
185
186 ctx->start_time_ns = time_get_ns();
187 ctx->initialized = true;

189 log_debug("Created network capture context (no local media source)");

193session_capture_ctx_t *session_capture_create(const session_capture_config_t *config) {
194 // Auto-create config from command-line options if NULL
195 session_capture_config_t auto_config = {0};
196 if (!config) {
197 // Auto-initialization from GET_OPTION() only works when options have been parsed
198 // Verify by checking if we can access options (would SET_ERRNO if not)
199 const char *media_file = GET_OPTION(media_file);
200 bool media_from_stdin = GET_OPTION(media_from_stdin);
201
202 if (media_file[0] != '\0') {
203 // File or stdin streaming
204 auto_config.type = media_from_stdin ? MEDIA_SOURCE_STDIN : MEDIA_SOURCE_FILE;
205 auto_config.path = media_file;
206 auto_config.loop = GET_OPTION(media_loop) && !media_from_stdin;
207 } else if (GET_OPTION(test_pattern)) {
208 // Test pattern mode
209 auto_config.type = MEDIA_SOURCE_TEST;
210 auto_config.path = NULL;
211 } else {
212 // Webcam mode (default)
213 static char webcam_index_str[32];
214 unsigned int idx = GET_OPTION(webcam_index);
215 safe_snprintf(webcam_index_str, sizeof(webcam_index_str), "%u", idx);
216 auto_config.type = MEDIA_SOURCE_WEBCAM;
217 auto_config.path = webcam_index_str;
218 log_debug("Webcam mode: using device index %u", idx);
219 }
220
221 // Default settings suitable for local display
222 auto_config.target_fps = 60;
223 auto_config.resize_for_network = false;
224 config = &auto_config;
225 }
226
227 // Check if we should exit before starting initialization
228 if (config->should_exit_callback && config->should_exit_callback(config->callback_data)) {
229 return NULL;
230 }
231
232 // Allocate context
233 session_capture_ctx_t *ctx = SAFE_CALLOC(1, sizeof(session_capture_ctx_t), session_capture_ctx_t *);
234
235 // Store configuration
236 ctx->target_fps = config->target_fps > 0 ? config->target_fps : 60;
237 ctx->resize_for_network = config->resize_for_network;
238
239 // Store audio configuration
240 ctx->audio_enabled = config->enable_audio;
241 ctx->audio_fallback_enabled = config->audio_fallback_to_mic;
242 ctx->mic_audio_ctx = config->mic_audio_ctx;
243 ctx->using_file_audio = false;
244 ctx->file_has_audio = false;
245
246 // Create media source from config type and path
247 ctx->source = media_source_create(config->type, config->path);
248
249 if (!ctx->source) {
250 // Preserve existing error if set, otherwise set generic error
251 asciichat_error_t existing_error = GET_ERRNO();
252 if (existing_error == ASCIICHAT_OK) {
253 SET_ERRNO(ERROR_MEDIA_INIT, "Failed to create media source");
254 }
255 SAFE_FREE(ctx);
256 return NULL;
257 }
258
259 // Set up exit callback on media source for graceful shutdown during I/O
260 // This allows Ctrl+C to abort blocking FFmpeg calls like YouTube HTTP requests
261 if (config->should_exit_callback && ctx->source) {
262 media_source_set_exit_callback(ctx->source, config->should_exit_callback, config->callback_data);
263 }
264
265 // Detect if media source has audio
266 if (ctx->audio_enabled && ctx->source) {
267 ctx->file_has_audio = media_source_has_audio(ctx->source);
268 if (ctx->file_has_audio) {
269 ctx->using_file_audio = true;
270 log_info("Audio capture enabled: using file audio");
271 } else if (ctx->audio_fallback_enabled && ctx->mic_audio_ctx) {
272 ctx->using_file_audio = false;
273 log_info("Audio capture enabled: file has no audio, using microphone fallback");
274 } else {
275 ctx->audio_enabled = false;
276 log_debug("Audio capture disabled: no file audio and no fallback configured");
277 }
278 }
279
280 // Enable loop if requested (only for file sources)
281 if (config->loop && config->type == MEDIA_SOURCE_FILE) {
282 media_source_set_loop(ctx->source, true);
283 }
284
285 // Set initial pause state if requested (only for file sources)
286 // Note: We don't pause yet - we'll pause AFTER reading the first frame in the render loop
287 // This is because paused sources return NULL from read_video()
288 if (config->type == MEDIA_SOURCE_FILE && GET_OPTION(pause)) {
289 ctx->should_pause_after_first_frame = true;
290 log_debug("Will pause after first frame (--pause flag)");
291 }
292
293 // Perform initial seek if requested
294 if (config->initial_seek_timestamp > 0.0) {
295 log_debug("Seeking to %.2f seconds", config->initial_seek_timestamp);
296 asciichat_error_t seek_err = media_source_seek(ctx->source, config->initial_seek_timestamp);
297 if (seek_err != ASCIICHAT_OK) {
298 log_warn("Failed to seek to %.2f seconds: %d", config->initial_seek_timestamp, seek_err);
299 // Continue anyway - seeking is best-effort
300 } else {
301 log_debug("Successfully seeked to %.2f seconds", config->initial_seek_timestamp);
302 // Codec buffers are flushed during seek, so next frame read will use correct position
303
304 // Reset timing state after seek to prevent FPS calculation confusion
305 // (frame_count and elapsed_time must match the new playback position)
306 ctx->frame_count = 0;
307 ctx->start_time_ns = time_get_ns();
308
309 // For snapshot mode with immediate snapshot (delay=0), we need to wait for prefetch thread to
310 // decode the seeked frame. This is a workaround for a timing issue where the prefetch thread
311 // needs time to decode the frame at the new position after seeking.
312 // The sleep duration must be long enough for the prefetch thread to decode one frame.
313 float snapshot_delay = GET_OPTION(snapshot_delay);
314 if (GET_OPTION(snapshot_mode) && snapshot_delay == 0.0f) {
315 // HTTP streams are slow - need 1+ second for first frame decode after seek
316 // Local files are faster - 100-200ms is usually sufficient
317 // Use 1 second as a safe default to handle both cases
318 log_debug("Waiting for prefetch thread after seek (snapshot_delay=0, HTTP streams need ~1 second)");
319 platform_sleep_us(US_PER_SEC_INT); // 1 second - ensures prefetch thread has delivered seeked frame
320 }
321 }
322 }
323
324 // Initialize adaptive sleep for frame rate limiting
325 // Calculate baseline sleep time in nanoseconds from target FPS
326 uint64_t baseline_sleep_ns = NS_PER_SEC_INT / ctx->target_fps;
327 adaptive_sleep_config_t sleep_config = {
328 .baseline_sleep_ns = baseline_sleep_ns,
329 .min_speed_multiplier = 0.5, // Allow slowing down to 50% of baseline
330 .max_speed_multiplier = 2.0, // Allow speeding up to 200% of baseline
331 .speedup_rate = 0.1, // Adapt by 10% per frame if possible
332 .slowdown_rate = 0.1 // Adapt by 10% per frame if possible
333 };
334 adaptive_sleep_init(&ctx->sleep_state, &sleep_config);
335
336 // Initialize FPS tracker
337 // Note: Must allocate tracker_name on heap since fps_init stores a pointer to it
338 char *tracker_name = SAFE_MALLOC(32, char *);
339 if (!tracker_name) {
340 media_source_destroy(ctx->source);
341 SAFE_FREE(ctx);
342 return NULL;
343 }
344 safe_snprintf(tracker_name, 32, "CAPTURE_%u", ctx->target_fps);
345 fps_init(&ctx->fps_tracker, (int)ctx->target_fps, tracker_name);
346
347 // Record start time for FPS calculation (nanoseconds)
348 ctx->start_time_ns = time_get_ns();
349
350 ctx->initialized = true;
351 return ctx;
352}

379image_t *session_capture_read_frame(session_capture_ctx_t *ctx) {
380 if (!ctx || !ctx->initialized || !ctx->source) {
381 return NULL;
382 }
383
384 static uint64_t last_frame_time_ns = 0;
385 uint64_t frame_request_time_ns = time_get_ns();
386
387 image_t *frame = media_source_read_video(ctx->source);
388
389 if (frame) {
390 uint64_t frame_available_time_ns = time_get_ns();
391
392 // Track frame for FPS reporting
393 fps_frame_ns(&ctx->fps_tracker, frame_available_time_ns, "frame captured");
394 ctx->frame_count++;
395
396 // Log frame-to-frame timing
397 if (last_frame_time_ns > 0) {
398 uint64_t time_since_last_frame_ns = time_elapsed_ns(last_frame_time_ns, frame_request_time_ns);
399 uint64_t time_to_get_frame_ns = time_elapsed_ns(frame_request_time_ns, frame_available_time_ns);
400 double since_last_ms = (double)time_since_last_frame_ns / NS_PER_MS;
401 double to_get_ms = (double)time_to_get_frame_ns / NS_PER_MS;
402
403 if (ctx->frame_count % 30 == 0) {
404 log_dev_every(3 * US_PER_SEC_INT, "FRAME_TIMING[%lu]: since_last=%.1f ms, to_get=%.1f ms", ctx->frame_count,
405 since_last_ms, to_get_ms);
406 }
407 }
408 last_frame_time_ns = frame_available_time_ns;
409
410 // Handle --pause flag: pause after first frame is read
411 if (ctx->should_pause_after_first_frame && !ctx->paused_after_first_frame) {
412 media_source_pause(ctx->source);
413 ctx->paused_after_first_frame = true;
414 log_info("Paused (--pause flag)");
415 }
416 }
417
418 return frame;
419}

421image_t *session_capture_process_for_transmission(session_capture_ctx_t *ctx, image_t *frame) {
422 if (!ctx || !frame) {
423 SET_ERRNO(ERROR_INVALID_PARAM, "session_capture_process_for_transmission: NULL parameter");
424 return NULL;
425 }
426
427 // If resize is not enabled, just create a copy
428 if (!ctx->resize_for_network) {
429 image_t *copy = image_new(frame->w, frame->h);
430 if (!copy) {
431 SET_ERRNO(ERROR_MEMORY, "Failed to allocate image copy");
432 return NULL;
433 }
434 memcpy(copy->pixels, frame->pixels, (size_t)frame->w * (size_t)frame->h * sizeof(rgb_pixel_t));
435 return copy;
436 }
437
438 // Calculate optimal dimensions for network transmission
439 ssize_t resized_width, resized_height;
440 calculate_optimal_dimensions(frame->w, frame->h, SESSION_MAX_FRAME_WIDTH, SESSION_MAX_FRAME_HEIGHT, &resized_width,
441 &resized_height);
442
443 // Check if resizing is actually needed
444 if (frame->w == resized_width && frame->h == resized_height) {
445 // No resizing needed - create a copy
446 image_t *copy = image_new(frame->w, frame->h);
447 if (!copy) {
448 SET_ERRNO(ERROR_MEMORY, "Failed to allocate image copy");
449 return NULL;
450 }
451 memcpy(copy->pixels, frame->pixels, (size_t)frame->w * (size_t)frame->h * sizeof(rgb_pixel_t));
452 return copy;
453 }
454
455 // Create new image for resized frame
456 image_t *resized = image_new(resized_width, resized_height);
457 if (!resized) {
458 SET_ERRNO(ERROR_MEMORY, "Failed to allocate resized image buffer");
459 return NULL;
460 }
461
462 // Perform resizing operation
463 image_resize(frame, resized);
464
465 return resized;
466}

489double session_capture_get_current_fps(session_capture_ctx_t *ctx) {
490 if (!ctx || !ctx->initialized || ctx->frame_count == 0) {
491 return 0.0;
492 }
493
494 // Calculate elapsed time in nanoseconds then convert to seconds
495 uint64_t elapsed_ns = time_elapsed_ns(ctx->start_time_ns, time_get_ns());
496 double elapsed_sec = time_ns_to_s(elapsed_ns);
497
498 if (elapsed_sec <= 0.0) {
499 return 0.0;
500 }
501
502 return (double)ctx->frame_count / elapsed_sec;
503}
