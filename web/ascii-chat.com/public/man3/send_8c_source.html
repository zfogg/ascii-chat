41asciichat_error_t packet_send_via_transport(acip_transport_t *transport, packet_type_t type, const void *payload,
42 size_t payload_len, uint32_t client_id) {
43 if (!transport) {
44 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid transport");
45 }
46
47// Sanity check: catch suspiciously large packets (likely a bug)
48#define MAX_REASONABLE_PACKET_SIZE (25 * 1024 * 1024) // 25 MB
49 if (payload_len > MAX_REASONABLE_PACKET_SIZE) {
50 log_error("★ PACKET_SIZE_VALIDATION_FAILURE: Attempting to send %zu bytes (%zu MB) with type=%d (0x%04x), exceeds "
51 "max %d bytes (25 MB) - likely a bug in caller's length calculation",
52 payload_len, payload_len / (1024 * 1024), type, type, MAX_REASONABLE_PACKET_SIZE);
53 return SET_ERRNO(ERROR_INVALID_PARAM, "Packet payload too large: %zu bytes (max 25MB)", payload_len);
54 }
55
56 log_info("★ PACKET_SEND_VIA_TRANSPORT START: type=%d (0x%04x), payload_len=%zu bytes, client_id=%u, transport=%p",
57 type, type, payload_len, client_id, (void *)transport);
58
59 // Build packet header
60 packet_header_t header;
61 header.magic = HOST_TO_NET_U64(PACKET_MAGIC);
62 header.type = HOST_TO_NET_U16(type);
63 header.length = HOST_TO_NET_U32((uint32_t)payload_len);
64 header.client_id = client_id;
65
66 // Calculate CRC32 if we have payload
67 if (payload && payload_len > 0) {
68 header.crc32 = HOST_TO_NET_U32(asciichat_crc32((const uint8_t *)payload, payload_len));
69 log_debug("★ PACKET_SEND: CRC32=0x%08x calculated for %zu byte payload", header.crc32, payload_len);
70 } else {
71 header.crc32 = 0;
72 log_debug("★ PACKET_SEND: No payload, CRC32=0");
73 }
74
75 log_debug("★ PACKET_SEND: magic=0x%016llx, type=%d, length=%u bytes, client_id=%u", header.magic, type, header.length,
76 header.client_id);
77
78 // Calculate total packet size
79 size_t total_size = sizeof(header) + payload_len;
80 log_debug("★ PACKET_SEND: Header=%zu bytes + Payload=%zu bytes = Total=%zu bytes", sizeof(header), payload_len,
81 total_size);
82
83 // Allocate buffer for complete packet
84 uint8_t *packet = SAFE_MALLOC(total_size, uint8_t *);
85 if (!packet) {
86 log_error("★ PACKET_SEND: Memory allocation FAILED for %zu byte packet buffer", total_size);
87 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate packet buffer");
88 }
89
90 log_debug("★ PACKET_SEND: Packet buffer allocated at %p", (void *)packet);
91
92 // Build complete packet: header + payload
93 memcpy(packet, &header, sizeof(header));
94 if (payload && payload_len > 0) {
95 memcpy(packet + sizeof(header), payload, payload_len);
96 log_debug("★ PACKET_SEND: Payload copied to buffer offset %zu", sizeof(header));
97 }
98
99 // Send via transport (transport handles encryption if crypto_ctx present)
100 // Transport does NOT take ownership - it only copies data to its own buffers
101 log_info("★ PACKET_SEND: Calling acip_transport_send with %zu total bytes", total_size);
102 asciichat_error_t result = acip_transport_send(transport, packet, total_size);
103
104 if (result == ASCIICHAT_OK) {
105 log_info("★ PACKET_SEND_VIA_TRANSPORT COMPLETE: SUCCESS - sent %zu bytes (type=%d, client_id=%u)", total_size, type,
106 client_id);
107 } else {
108 log_error("★ PACKET_SEND_VIA_TRANSPORT FAILED: acip_transport_send returned error %d (%s) when sending %zu bytes "
109 "type=%d to client_id=%u",
110 result, asciichat_error_string(result), total_size, type, client_id);
111 }
112
113 // Always free the packet buffer
114 SAFE_FREE(packet);
115 return result;
116}

126asciichat_error_t acip_send_audio_batch(acip_transport_t *transport, const float *samples, uint32_t num_samples,
127 uint32_t batch_count) {
128 if (!transport || !samples || num_samples == 0) {
129 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters");
130 }
131
132 // Build batch header
133 audio_batch_packet_t header;
134 header.batch_count = batch_count;
135 header.total_samples = (uint32_t)num_samples;
136 header.sample_rate = (uint32_t)AUDIO_SAMPLE_RATE;
137 header.channels = 1;
138
139 // Calculate total size (header + float samples)
140 size_t samples_size = num_samples * sizeof(float);
141 size_t total_size = sizeof(header) + samples_size;
142
143 uint8_t *buffer = buffer_pool_alloc(NULL, total_size);
144 if (!buffer) {
145 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate buffer");
146 }
147
148 // Copy header and samples
149 memcpy(buffer, &header, sizeof(header));
150 memcpy(buffer + sizeof(header), samples, samples_size);
151
152 asciichat_error_t result = packet_send_via_transport(transport, PACKET_TYPE_AUDIO_BATCH, buffer, total_size, 0);
153
154 buffer_pool_free(NULL, buffer, total_size);
155 return result;
156}

169asciichat_error_t acip_send_audio_opus_batch(acip_transport_t *transport, const void *opus_data, size_t opus_len,
170 const uint16_t *frame_sizes, uint32_t frame_count, uint32_t sample_rate,
171 uint32_t frame_duration) {
172 if (!transport || !opus_data || !frame_sizes || frame_count == 0) {
173 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters");
174 }
175
176 // Build batch header (16 bytes)
177 uint8_t header[16];
178 *(uint32_t *)(header + 0) = HOST_TO_NET_U32(sample_rate);
179 *(uint32_t *)(header + 4) = HOST_TO_NET_U32(frame_duration);
180 *(uint32_t *)(header + 8) = HOST_TO_NET_U32(frame_count);
181 *(uint32_t *)(header + 12) = 0; // reserved
182
183 // Calculate sizes array size
184 size_t sizes_len = frame_count * sizeof(uint16_t);
185 size_t total_size = sizeof(header) + sizes_len + opus_len;
186
187 uint8_t *buffer = buffer_pool_alloc(NULL, total_size);
188 if (!buffer) {
189 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate buffer");
190 }
191
192 // Build packet: header + sizes + opus_data
193 memcpy(buffer, header, sizeof(header));
194
195 // Convert frame sizes to network byte order before copying
196 // IMPORTANT: Server expects network byte order and will apply NET_TO_HOST_U16()
197 uint16_t *sizes_buf = (uint16_t *)(buffer + sizeof(header));
198 for (uint32_t i = 0; i < frame_count; i++) {
199 sizes_buf[i] = HOST_TO_NET_U16(frame_sizes[i]);
200 }
201
202 memcpy(buffer + sizeof(header) + sizes_len, opus_data, opus_len);
203
204 log_info("★ OPUS_BATCH_SEND_DEBUG: header=16, frame_count=%u, sizes_len=%zu, opus_len=%zu, TOTAL=%zu", frame_count,
205 sizes_len, opus_len, total_size);
206
207 asciichat_error_t result = packet_send_via_transport(transport, PACKET_TYPE_AUDIO_OPUS_BATCH, buffer, total_size, 0);
208
209 buffer_pool_free(NULL, buffer, total_size);
210 return result;
211}

247asciichat_error_t acip_send_error(acip_transport_t *transport, uint32_t error_code, const char *message) {
248 if (!transport) {
249 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid transport");
250 }
251
252 // Calculate message length (max MAX_ERROR_MESSAGE_LENGTH)
253 size_t msg_len = 0;
254 if (message) {
255 msg_len = strlen(message);
256 if (msg_len > MAX_ERROR_MESSAGE_LENGTH) {
257 msg_len = MAX_ERROR_MESSAGE_LENGTH;
258 }
259 }
260
261 error_packet_t header;
262 header.error_code = HOST_TO_NET_U32(error_code);
263 header.message_length = HOST_TO_NET_U32((uint32_t)msg_len);
264
265 // Allocate buffer for header + message
266 size_t total_size = sizeof(header) + msg_len;
267 uint8_t *buffer = buffer_pool_alloc(NULL, total_size);
268 if (!buffer) {
269 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate buffer");
270 }
271
272 // Build packet
273 memcpy(buffer, &header, sizeof(header));
274 if (msg_len > 0) {
275 memcpy(buffer + sizeof(header), message, msg_len);
276 }
277
278 asciichat_error_t result = packet_send_via_transport(transport, PACKET_TYPE_ERROR_MESSAGE, buffer, total_size, 0);
279
280 buffer_pool_free(NULL, buffer, total_size);
281 return result;
282}
