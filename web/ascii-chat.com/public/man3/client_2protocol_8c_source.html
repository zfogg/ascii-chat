344static void handle_ascii_frame_packet(const void *data, size_t len) {
345 fflush(stdout);
346 static int frame_count = 0;
347 frame_count++;
348 if (frame_count == 1) {
349 }
350
351 if (should_exit()) {
352 return;
353 }
354
355 if (!data || len < sizeof(ascii_frame_packet_t)) {
356 disconnect_server_for_bad_data("ASCII_FRAME payload too small: %zu (min %zu)", len, sizeof(ascii_frame_packet_t));
357 return;

360 // FPS tracking for received ASCII frames using reusable tracker utility
361 static fps_t fps_tracker = {0};
362 static bool fps_tracker_initialized = false;
363
364 // Initialize FPS tracker on first frame
365 if (!fps_tracker_initialized) {
366 int fps = GET_OPTION(fps);
367 int expected_fps = fps > 0 ? ((fps > 144) ? 144 : fps) : DEFAULT_MAX_FPS;
368 fps_init(&fps_tracker, expected_fps, "ASCII_RX");
369 fps_tracker_initialized = true;
370 }
371
372 // Track this frame and detect lag
373 fps_frame_ns(&fps_tracker, time_get_ns(), "ASCII frame");
374
375 // Extract header from the packet
376 ascii_frame_packet_t header;
377 memcpy(&header, data, sizeof(ascii_frame_packet_t));
378
379 // Convert from network byte order
380 header.width = NET_TO_HOST_U32(header.width);
381 header.height = NET_TO_HOST_U32(header.height);
382 header.original_size = NET_TO_HOST_U32(header.original_size);
383 header.compressed_size = NET_TO_HOST_U32(header.compressed_size);
384 header.checksum = NET_TO_HOST_U32(header.checksum);
385 header.flags = NET_TO_HOST_U32(header.flags);
386
387 // Get the frame data (starts after the header)
388 const char *frame_data_ptr = (const char *)data + sizeof(ascii_frame_packet_t);
389 size_t frame_data_len = len - sizeof(ascii_frame_packet_t);
390
391 // Decode frame data (handles both compressed and uncompressed)
392 bool is_compressed = (header.flags & FRAME_FLAG_IS_COMPRESSED) && header.compressed_size > 0;
393 char *frame_data =
394 decode_frame_data(frame_data_ptr, frame_data_len, is_compressed, header.original_size, header.compressed_size);
395 if (!frame_data) {
396 return; // Error already logged by decode_frame_data
397 }
398
399 // Verify checksum
400 uint32_t actual_crc = asciichat_crc32(frame_data, header.original_size);
401 if (actual_crc != header.checksum) {
402 log_error("Frame checksum mismatch: got 0x%x, expected 0x%x (size=%u, first_bytes=%02x%02x%02x%02x)", actual_crc,
403 header.checksum, header.original_size, (unsigned char)frame_data[0], (unsigned char)frame_data[1],
404 (unsigned char)frame_data[2], (unsigned char)frame_data[3]);
405
406 // DEBUG: Try software CRC32 to compare
407 uint32_t sw_crc = asciichat_crc32_sw(frame_data, header.original_size);
408 log_error("Software CRC32: 0x%x (matches: %s)", sw_crc, (sw_crc == header.checksum) ? "YES" : "NO");
409
410 SAFE_FREE(frame_data);
411 return;
412 }
413
414 // Track frame dimension changes
415 static uint32_t last_width = 0;
416 static uint32_t last_height = 0;
417
418 if (header.width > 0 && header.height > 0) {
419 if (header.width != last_width || header.height != last_height) {
420 last_width = header.width;
421 last_height = header.height;
422 }
423 }
424
425 // Handle snapshot mode timing
426 bool take_snapshot = false;
427 if (GET_OPTION(snapshot_mode)) {
428 static uint64_t first_frame_time_ns = 0;
429 static bool first_frame_recorded = false;
430 static int snapshot_frame_count = 0;
431
432 snapshot_frame_count++;
433 log_debug("Snapshot frame %d received", snapshot_frame_count);
434
435 if (!first_frame_recorded) {
436 first_frame_time_ns = time_get_ns();
437 g_snapshot_first_frame_time_ns = first_frame_time_ns; // Also store globally for access from data_reception_thread
438 log_info("[SNAPSHOT_INIT] SET g_snapshot_first_frame_time_ns=%llu first_frame_time_ns=%llu",
439 (unsigned long long)g_snapshot_first_frame_time_ns, (unsigned long long)first_frame_time_ns);
440 first_frame_recorded = true;
441
442 if (GET_OPTION(snapshot_delay) == 0) {
443 log_debug("Snapshot captured immediately (delay=0)!");
444 take_snapshot = true;
445 signal_exit();
446 } else {
447 log_debug("Snapshot mode: first frame received, waiting %.2f seconds for webcam warmup...",
448 GET_OPTION(snapshot_delay));
449 }
450 } else {
451 uint64_t current_time_ns = time_get_ns();
452 double elapsed = time_ns_to_s(time_elapsed_ns(first_frame_time_ns, current_time_ns));
453 double snapshot_delay = GET_OPTION(snapshot_delay);
454
455 // Log every 0.5 seconds to debug timing
456 static uint64_t last_debug_log_ns = 0;
457 if (current_time_ns - last_debug_log_ns > 500 * 1000 * 1000) { // 500ms
458 log_info("SNAPSHOT_DELAY_CHECK: elapsed=%.3f target=%.3f first_frame_ns=%llu current_ns=%llu",
459 elapsed, snapshot_delay, (unsigned long long)first_frame_time_ns, (unsigned long long)current_time_ns);
460 last_debug_log_ns = current_time_ns;
461 }
462
463 if (elapsed >= snapshot_delay) {
464 char duration_str[32];
465 time_pretty((uint64_t)(elapsed * 1e9), -1, duration_str, sizeof(duration_str));
466 log_info("🎬 SNAPSHOT TIMEOUT FIRED: Snapshot captured after %s! (target=%.3f) AT handle_ascii_frame_packet",
467 duration_str, snapshot_delay);
468 take_snapshot = true;
469 signal_exit();
470 }
471 }
472 }
473
474 // Check if we need to clear console before rendering this frame
475 // IMPORTANT: We track if this is the first frame to ensure proper initialization
476 static bool first_frame_rendered = false;
477
478 if (!first_frame_rendered) {
479 // Always clear display and disable logging before rendering the first frame
480 // This ensures clean ASCII display regardless of packet arrival order
481 log_debug("First frame - clearing display and disabling terminal logging");
482 log_set_terminal_output(false);
483 display_full_reset();
484 first_frame_rendered = true;
485 g_server_state_initialized = true; // Mark as initialized
486 g_should_clear_before_next_frame = false; // Clear any pending clear request
487 log_debug("CLIENT_DISPLAY: Display cleared, ready for ASCII frames");
488 } else if (g_should_clear_before_next_frame) {
489 // Subsequent clear request from server (e.g., after client list changes)
490 log_debug("CLIENT_DISPLAY: Clearing display for layout change");
491 log_set_terminal_output(false);
492 display_full_reset();
493 g_should_clear_before_next_frame = false;
494 }
495
496 // Safety check before rendering
497 if (!frame_data || header.original_size == 0) {
498 log_error("Invalid frame data for rendering: frame_data=%p, size=%u", frame_data, header.original_size);
499 if (frame_data) {
500 SAFE_FREE(frame_data);
501 }
502 return;
503 }
504
505 // Client-side FPS limiting for rendering (display)
506 // Server may send at 144fps for high-refresh displays, but this client renders at its requested FPS
507 static uint64_t last_render_time_ns = 0;
508
509 // Don't limit frame rate in snapshot mode - always render the final frame
510 if (!take_snapshot) {
511 // Get the client's desired FPS (what we told the server we can display)
512 int fps = GET_OPTION(fps);
513 int client_display_fps = fps > 0 ? fps : DEFAULT_MAX_FPS;
514 uint64_t render_interval_us = US_PER_SEC_INT / (uint64_t)client_display_fps;
515
516 uint64_t render_time_ns = time_get_ns();
517 uint64_t render_elapsed_us = 0;
518
519 if (last_render_time_ns != 0) {
520 render_elapsed_us = time_ns_to_us(time_elapsed_ns(last_render_time_ns, render_time_ns));
521 }
522
523 // Skip rendering if not enough time has passed (frame rate limiting)
524 if (last_render_time_ns != 0) {
525 if (render_elapsed_us > 0 && render_elapsed_us < render_interval_us) {
526 // Drop this frame to maintain display FPS limit
527 SAFE_FREE(frame_data);
528 return;
529 }
530 }
531
532 // Update last render time
533 last_render_time_ns = render_time_ns;
534 }
535
536 // Increment global frame counter BEFORE rendering (to track unique frames received)
537 int total_frames = atomic_fetch_add_u64(&g_frames_rendered, 1) + 1;
538
539 // DEBUG: Periodically log frame stats on client side
540 static int client_frame_counter = 0;
541 client_frame_counter++;
542 if (client_frame_counter % 60 == 1) {
543 // Count lines and check for issues
544 int line_count = 0;
545 size_t frame_len = strlen(frame_data);
546 for (size_t i = 0; i < frame_len; i++) {
547 if (frame_data[i] == '\n')
548 line_count++;
549 }
550 log_info("🎬 CLIENT_FRAME: #%d received - %zu bytes, %d newlines, %ux%u", total_frames, frame_len, line_count,
551 header.width, header.height);
552 }
553
554 // Track frames actually reaching display (after frame rate limiting)
555 static int frames_to_display = 0;
556 frames_to_display++;
557 if (frames_to_display % 10 == 1) {
558 log_info("📺 FRAME_TO_DISPLAY: #%d (received: %d)", frames_to_display, total_frames);
559 }
560
561 // Render ASCII art frame (display_render_frame will apply effects like --matrix)
562 log_debug("🎬 CALLING_DISPLAY_RENDER: frame_data=%p len=%zu, calling display_render_frame()", frame_data,
563 strlen(frame_data));
564 display_render_frame(frame_data);
565 log_debug("🎬 DISPLAY_RENDER_RETURNED: frame was rendered");
566
567 SAFE_FREE(frame_data);
568}

696static void handle_audio_opus_batch_packet(const void *data, size_t len) {
697 // Validate parameters
698 if (!data || len == 0) {
699 SET_ERRNO(ERROR_INVALID_PARAM, "Invalid opus batch packet: data=%p, len=%zu", data, len);
700 return;
701 }
702
703 // DEBUG: Log first 12 bytes of received Opus batch header
704 if (len >= 12) {
705 const uint8_t *bytes = (const uint8_t *)data;
706 log_info("OPUS_BATCH_RECV_DEBUG: first_bytes=[%02x %02x %02x %02x][%02x %02x %02x %02x][%02x %02x %02x %02x]",
707 bytes[0], bytes[1], bytes[2], bytes[3], bytes[4], bytes[5], bytes[6], bytes[7], bytes[8], bytes[9],
708 bytes[10], bytes[11]);
709
710 // Parse header manually to check values
711 uint32_t sr, fd, fc;
712 memcpy(&sr, bytes, 4);
713 memcpy(&fd, bytes + 4, 4);
714 memcpy(&fc, bytes + 8, 4);
715 log_info("OPUS_BATCH_RECV_PARSED: sample_rate(net)=0x%08x frame_duration(net)=0x%08x frame_count(net)=0x%08x", sr,
716 fd, fc);
717 log_info("OPUS_BATCH_RECV_PARSED: sample_rate=%u frame_duration=%u frame_count=%u", NET_TO_HOST_U32(sr),
718 NET_TO_HOST_U32(fd), NET_TO_HOST_U32(fc));
719 }
720
721 if (!GET_OPTION(audio_enabled)) {
722 log_warn_every(NS_PER_MS_INT, "Received opus batch packet but audio is disabled");
723 return;
724 }
725
726 // Parse batch header using av_receive_audio_opus_batch() for consistency
727 const uint8_t *opus_data = NULL;
728 size_t opus_size = 0;
729 const uint16_t *frame_sizes = NULL;
730 int sample_rate = 0;
731 int frame_duration = 0;
732 int frame_count = 0;
733
734 asciichat_error_t result = packet_parse_opus_batch(data, len, &opus_data, &opus_size, &frame_sizes, &sample_rate,
735 &frame_duration, &frame_count);
736
737 if (result != ASCIICHAT_OK) {
738 log_warn("Failed to parse Opus batch packet");
739 return;
740 }
741
742 if (frame_count <= 0 || frame_count > 256 || opus_size == 0) {
743 log_warn("Invalid Opus batch: frame_count=%d, opus_size=%zu", frame_count, opus_size);
744 return;
745 }
746
747 // Calculate samples per frame
748 int samples_per_frame = (sample_rate * frame_duration) / 1000;
749 if (samples_per_frame <= 0 || samples_per_frame > 2880) {
750 log_warn("Invalid Opus frame parameters: samples_per_frame=%d", samples_per_frame);
751 return;
752 }
753
754 // Allocate buffer for all decoded samples
755 size_t max_decoded_samples = (size_t)samples_per_frame * (size_t)frame_count;
756 float *all_samples = SAFE_MALLOC(max_decoded_samples * sizeof(float), float *);
757 if (!all_samples) {
758 SET_ERRNO(ERROR_MEMORY, "Failed to allocate memory for Opus batch decoding");
759 return;
760 }
761
762 // Decode each Opus frame using frame_sizes array
763 int total_decoded_samples = 0;
764 size_t opus_offset = 0;
765
766 for (int i = 0; i < frame_count; i++) {
767 // Get frame size (convert from network byte order)
768 size_t frame_size = (size_t)NET_TO_HOST_U16(frame_sizes[i]);
769
770 if (opus_offset + frame_size > opus_size) {
771 log_warn("Opus batch truncated at frame %d (offset=%zu, frame_size=%zu, total=%zu)", i, opus_offset, frame_size,
772 opus_size);
773 break;
774 }
775
776 // Decode frame through audio pipeline
777 float *frame_buffer = all_samples + total_decoded_samples;
778 int remaining_space = (int)(max_decoded_samples - (size_t)total_decoded_samples);
779 int decoded = audio_decode_opus(opus_data + opus_offset, frame_size, frame_buffer, remaining_space);
780
781 if (decoded <= 0) {
782 log_warn("Failed to decode Opus frame %d in batch, decoded=%d", i, decoded);
783 break;
784 }
785
786 total_decoded_samples += decoded;
787 opus_offset += frame_size;
788 }
789
790 if (total_decoded_samples > 0) {
791 // Track received packet for analysis
792 if (GET_OPTION(audio_analysis_enabled)) {
793 audio_analysis_track_received_packet(len);
794 }
795
796 // Process decoded audio through audio subsystem
797 audio_process_received_samples(all_samples, total_decoded_samples);
798
799 log_debug_every(LOG_RATE_DEFAULT, "Processed Opus batch: %d decoded samples from %d frames", total_decoded_samples,
800 frame_count);
801 }
802
803 // Clean up
804 SAFE_FREE(all_samples);
805}

877 log_debug("Active client count changed from %u to %u - will clear console before next frame", g_last_active_count,

879 g_should_clear_before_next_frame = true;
880 }
881 } else {
882 // First state packet received
883 g_server_state_initialized = true;
884 // Clear terminal before the very first frame
885 g_should_clear_before_next_frame = true;
886 }
887
888 g_last_active_count = active_count;
889}

991static void *data_reception_thread_func(void *arg) {
992 (void)arg;
993
994 log_warn("[FRAME_RECV_LOOP] 🔄 THREAD_STARTED: Data reception thread active, callbacks initialized");
995
996#ifdef DEBUG_THREADS
997 log_debug("[FRAME_RECV_LOOP] Thread lifecycle tracking enabled");
998#endif
999
1000 int packet_count = 0;
1001 while (!should_exit() && server_connection_is_active()) {
1002 // Main loop: receive and process packets while connection is active
1003 // When connection becomes inactive or shutdown is requested, thread exits cleanly
1004
1005 // Debug: periodically check loop condition
1006 if (packet_count % 50 == 0) {
1007 bool exit_flag = should_exit();
1008 bool conn_active = server_connection_is_active();
1009 if (packet_count > 0) {
1010 log_debug("[FRAME_RECV_LOOP] DEBUG: packet_count=%d, should_exit=%d, server_connection_is_active=%d",
1011 packet_count, exit_flag, conn_active);
1012 }
1013 }
1014
1015 // Receive and dispatch packet using ACIP transport API
1016 // This combines packet reception, decryption, parsing, handler dispatch, and cleanup
1017 acip_transport_t *transport = server_connection_get_transport();
1018 if (packet_count < 3) {
1019 }
1020 if (!transport) {
1021 log_error("[FRAME_RECV_LOOP] ❌ NO_TRANSPORT: connection lost, transport not available");
1022 server_connection_lost();
1023 break;
1024 }
1025
1026 log_debug("[FRAME_RECV_LOOP] 📥 RECV_WAITING: awaiting packet #%d from server (transport ready)", packet_count + 1);
1027
1028 if (packet_count == 0) {
1029 }
1030 // Use short 16ms timeout to allow rendering at 60 FPS
1031 // Instead of blocking indefinitely waiting for a packet,
1032 // we timeout quickly so main thread can continue rendering
1033 asciichat_error_t acip_result = acip_client_receive_and_dispatch(transport, &g_acip_client_callbacks);
1034 if (packet_count == 0) {
1035 }
1036
1037 if (acip_result == ASCIICHAT_OK) {
1038 packet_count++;
1039 log_debug("[FRAME_RECV_LOOP] ✅ PACKET_%d_DISPATCHED: callbacks processed successfully", packet_count);
1040 } else if (acip_result == ERROR_NETWORK_TIMEOUT) {
1041 // Network timeouts are expected - socket may have no data available
1042 // Continue looping to allow snapshot timer to expire
1043 log_debug("[FRAME_RECV_LOOP] ⏱️ TIMEOUT: No data available, retrying (packets received: %d)", packet_count);
1044 continue;
1045 } else {
1046 // Handle receive/dispatch errors - ALWAYS exit on network errors
1047 // NOTE: Log to file even if terminal logging is disabled (important for debugging)
1048 log_warn("[FRAME_RECV_LOOP] ⚠️ RECV_ERROR_BEFORE_SNAPSHOT_CHECK: acip_result=%d: %s (after %d packets received)", acip_result,
1049 asciichat_error_string(acip_result), packet_count);
1050
1051 // Before breaking due to error, check if snapshot_delay timer has fired in snapshot mode
1052 bool should_break_now = true;
1053 if (GET_OPTION(snapshot_mode) && g_snapshot_first_frame_time_ns > 0) {
1054 uint64_t current_time_ns = time_get_ns();
1055 double elapsed = time_ns_to_s(time_elapsed_ns(g_snapshot_first_frame_time_ns, current_time_ns));
1056 double snapshot_delay = GET_OPTION(snapshot_delay);
1057
1058 if (elapsed >= snapshot_delay) {
1059 char duration_str[32];
1060 time_pretty((uint64_t)(elapsed * 1e9), -1, duration_str, sizeof(duration_str));
1061 log_info("🎬 SNAPSHOT TIMEOUT FIRED (ON_ERROR): Snapshot captured after %s! (target=%.3f)",
1062 duration_str, snapshot_delay);
1063 signal_exit();
1064 should_break_now = false; // Don't disconnect, let normal shutdown handle it
1065 }
1066 }
1067
1068 if (should_break_now) {
1069 // Network errors (ERROR_NETWORK, ERROR_NETWORK_PROTOCOL, etc) always disconnect
1070 if (acip_result == ERROR_NETWORK || acip_result == ERROR_NETWORK_PROTOCOL) {
1071 log_warn("[FRAME_RECV_LOOP] ⚠️ NETWORK_ERROR: Server disconnected after %d packets, exiting loop",
1072 packet_count);
1073 server_connection_lost();
1074 break;
1075 }
1076
1077 // Check errno context for additional error details
1078 asciichat_error_context_t err_ctx;
1079 if (HAS_ERRNO(&err_ctx)) {
1080 if (err_ctx.code == ERROR_CRYPTO) {
1081 // Security violation - exit immediately
1082 log_error("[FRAME_RECV_LOOP] ❌ SECURITY_VIOLATION: Server crypto policy violated - EXITING");
1083 log_error("SECURITY: This is a critical security violation - exiting immediately");
1084 exit(1);
1085 }
1086 }
1087
1088 // Other errors - still disconnect to prevent infinite loop
1089 log_error("[FRAME_RECV_LOOP] ❌ RECV_FAILED: packet #%d failed after %d packets, disconnecting", packet_count + 1, packet_count);
1090 server_connection_lost();
1091 break;
1092 }
1093 }
1094 }
1095
1096 // Log why the loop exited (DEBUG to avoid affecting timing)
1097 bool final_exit_flag = should_exit();
1098 bool final_conn_active = server_connection_is_active();
1099 log_debug("[FRAME_RECV_LOOP] ⏹️ LOOP_EXIT: should_exit=%d server_connection_is_active=%d (packets_received=%d)",
1100 final_exit_flag, final_conn_active, packet_count);
1101
1102 // Before exiting due to connection loss, check if snapshot_delay timer has fired
1103 // If it has, take the snapshot and exit cleanly instead of abandoning the snapshot
1104 if (GET_OPTION(snapshot_mode)) {
1105 log_debug("[FRAME_RECV_LOOP] ⏰ SNAPSHOT_CHECK_ON_EXIT: snapshot_mode enabled, g_snapshot_first_frame_time_ns=%llu",
1106 (unsigned long long)g_snapshot_first_frame_time_ns);
1107
1108 if (g_snapshot_first_frame_time_ns > 0) {
1109 uint64_t current_time_ns = time_get_ns();
1110 double elapsed = time_ns_to_s(time_elapsed_ns(g_snapshot_first_frame_time_ns, current_time_ns));
1111 double snapshot_delay = GET_OPTION(snapshot_delay);
1112
1113 log_debug("[FRAME_RECV_LOOP] ⏰ SNAPSHOT_CHECK_DETAILS: elapsed=%.3f target=%.3f", elapsed, snapshot_delay);
1114
1115 if (elapsed >= snapshot_delay) {
1116 char duration_str[32];
1117 time_pretty((uint64_t)(elapsed * 1e9), -1, duration_str, sizeof(duration_str));
1118 log_info("🎬 SNAPSHOT TIMEOUT FIRED (ON_CONNECTION_LOSS): Snapshot captured after %s! (target=%.3f)",
1119 duration_str, snapshot_delay);
1120 signal_exit();
1121 }
1122 }
1123 }
1124
1125 log_warn("[FRAME_RECV_LOOP] 🔴 THREAD_EXITING: received %d packets total, connection inactive or shutdown requested",
1126 packet_count);
1127
1128#ifdef DEBUG_THREADS
1129 log_debug("[FRAME_RECV_LOOP] Thread lifecycle tracking - exit");
1130#endif
1131
1132 atomic_store_bool(&g_data_thread_exited, true);
1133
1134 // Clean up thread-local error context before exit
1135 asciichat_errno_destroy();
1136
1137 log_warn("[FRAME_RECV_LOOP] ✅ THREAD_CLEANUP: error context destroyed, thread terminating");
1138 return NULL;
1139}

1253void protocol_stop_connection() {
1254 // Log final frame count before shutting down (always log for diagnostic purposes)
1255 int final_frame_count = atomic_load_u64(&g_frames_rendered);
1256 log_info("📊 CLIENT SESSION STATS: %d unique frames rendered during connection", final_frame_count);
1257
1258 log_debug("[PROTOCOL_STOP] 1. Starting protocol_stop_connection");
1259
1260 // In snapshot mode, data reception thread was never started, but capture thread may still be running
1261 // Always stop the capture thread to prevent use-after-free when transport is destroyed
1262 if (GET_OPTION(snapshot_mode)) {
1263 log_debug("[PROTOCOL_STOP] Snapshot mode: flushing H.265 encoder before stopping capture thread");
1264 // Flush the encoder to output any buffered frames before exiting
1265 if (g_h265_encoder) {
1266 uint8_t flush_buf[1024];
1267 size_t flush_size = sizeof(flush_buf);
1268 asciichat_error_t flush_result = h265_encoder_flush(g_h265_encoder, flush_buf, &flush_size);
1269 if (flush_result == ASCIICHAT_OK && flush_size > 0) {
1270 log_info("[PROTOCOL_STOP] Flushed H.265 encoder: got %zu bytes of buffered frames", flush_size);
1271 // Send the flushed frame if we got one
1272 acip_transport_t *transport = server_connection_get_transport();
1273 if (transport && server_connection_is_active()) {
1274 uint32_t width = 320, height = 240; // TODO: track actual dimensions
1275 log_debug("[PROTOCOL_STOP] Sending flushed frame: %ux%u, %zu bytes", width, height, flush_size);
1276 }
1277 }
1278 }
1279 log_debug("[PROTOCOL_STOP] Snapshot mode: stopping capture thread before returning");
1280 capture_stop_thread();
1281 return;
1282 }
1283
1284 // Don't call signal_exit() here - that's for global shutdown only!
1285 // We just want to stop threads for this connection, not exit the entire client
1286
1287 // Shutdown the socket FIRST to interrupt any blocking network operations in worker threads
1288 // This must happen BEFORE audio_stop_thread() so audio sender thread unblocks from network send
1289 log_debug("[PROTOCOL_STOP] 2. About to call server_connection_shutdown");
1290 server_connection_shutdown();
1291 log_debug("[PROTOCOL_STOP] 3. server_connection_shutdown() returned");
1292
1293 // Signal audio sender thread to exit
1294 // Must happen after socket shutdown so any blocked network calls fail
1295 // Audio sender is created in ALL modes except snapshot mode
1296 log_debug("[PROTOCOL_STOP] 4. About to call audio_stop_thread");
1297 audio_stop_thread();
1298 log_debug("[PROTOCOL_STOP] 5. audio_stop_thread() returned");
1299
1300 // Early return if data thread was never created (e.g., mirror mode)
1301 // In mirror mode, we only need to stop the audio sender (done above)
1302 if (!g_data_thread_created) {
1303 log_debug("[PROTOCOL_STOP] 6. Data thread not created, returning");
1304 return;
1305 }
1306
1307 // Stop keepalive/ping thread - it checks connection status and will exit
1308 log_debug("[PROTOCOL_STOP] 7. About to call keepalive_stop_thread");
1309 keepalive_stop_thread();
1310 log_debug("[PROTOCOL_STOP] 8. keepalive_stop_thread() returned");
1311
1312 // Stop webcam capture thread
1313 log_debug("[PROTOCOL_STOP] 9. About to call capture_stop_thread");
1314 capture_stop_thread();
1315 log_debug("[PROTOCOL_STOP] 10. capture_stop_thread() returned");
1316
1317 // Wait for data reception thread to exit gracefully
1318 log_debug("[PROTOCOL_STOP] 11. Waiting for data thread to exit");
1319 // Thread checks should_exit() every read cycle (typically <1-5ms), so timeout can be much shorter
1320 int wait_count = 0;
1321 while (wait_count < 5 && !atomic_load_bool(&g_data_thread_exited)) {
1322 platform_sleep_us(10 * US_PER_MS_INT); // 10ms * 5 = 50ms max wait
1323 wait_count++;
1324 }
1325
1326 if (!atomic_load_bool(&g_data_thread_exited)) {
1327 log_warn("Data thread not responding after 50ms - will be joined by thread pool");
1328 }
1329 log_debug("[PROTOCOL_STOP] 12. Data thread wait complete");
1330
1331 // Join all threads in the client worker pool (in stop_id order)
1332 // This handles the data reception thread and (eventually) all other worker threads
1333 log_debug("[PROTOCOL_STOP] 13. About to call thread_pool_stop_all");
1334 if (g_client_worker_pool) {
1335 asciichat_error_t result = thread_pool_stop_all(g_client_worker_pool);
1336 if (result != ASCIICHAT_OK) {
1337 log_error("Failed to stop client worker threads");
1338 LOG_ERRNO_IF_SET("Thread pool stop failed");
1339 }
1340 }
1341 log_debug("[PROTOCOL_STOP] 14. thread_pool_stop_all() returned");
1342
1343 g_data_thread_created = false;
1344
1345#ifdef DEBUG_THREADS
1346 log_debug("Data reception thread stopped and joined by thread pool");
1347#endif
1348 log_debug("[PROTOCOL_STOP] 15. protocol_stop_connection complete");
1349}

1371static void acip_on_ascii_frame(const ascii_frame_packet_t *header, const void *frame_data, size_t data_len,
1372 void *ctx) {
1373 (void)ctx;
1374
1375 log_info("[FRAME_RECV_CALLBACK] 🎬 FRAME_RECEIVED: width=%u, height=%u, data_len=%zu bytes, flags=0x%x",
1376 header->width, header->height, data_len, header->flags);
1377
1378 // Reconstruct full packet for existing handler (header + data)
1379 // IMPORTANT: header is already in HOST byte order from ACIP layer,
1380 // but handle_ascii_frame_packet() expects NETWORK byte order and does conversion.
1381 // So we need to convert back to network order before passing.
1382 size_t total_len = sizeof(*header) + data_len;
1383 log_debug("[FRAME_RECV_CALLBACK] 📦 FRAME_SIZE: header=%zu + data=%zu = total=%zu", sizeof(*header), data_len,
1384 total_len);
1385
1386 uint8_t *packet = SAFE_MALLOC(total_len, uint8_t *);
1387 if (!packet) {
1388 log_error("[FRAME_RECV_CALLBACK] ❌ ALLOC_FAILED: cannot allocate %zu bytes for frame", total_len);
1389 return;
1390 }
1391
1392 log_debug("[FRAME_RECV_CALLBACK] ✅ BUFFER_ALLOCATED: packet=%p, capacity=%zu", (void *)packet, total_len);
1393
1394 // Convert header fields back to network byte order for handle_ascii_frame_packet()
1395 ascii_frame_packet_t net_header = *header;
1396 net_header.width = HOST_TO_NET_U32(header->width);
1397 net_header.height = HOST_TO_NET_U32(header->height);
1398 net_header.original_size = HOST_TO_NET_U32(header->original_size);
1399 net_header.compressed_size = HOST_TO_NET_U32(header->compressed_size);
1400 net_header.checksum = HOST_TO_NET_U32(header->checksum);
1401 net_header.flags = HOST_TO_NET_U32(header->flags);
1402
1403 log_debug("[FRAME_RECV_CALLBACK] 🔄 BYTE_ORDER_CONVERSION: converting header to network byte order");
1404
1405 memcpy(packet, &net_header, sizeof(net_header));
1406 memcpy(packet + sizeof(net_header), frame_data, data_len);
1407
1408 log_info("[FRAME_RECV_CALLBACK] 📥 FRAME_DISPATCH: calling handle_ascii_frame_packet() with %zu bytes", total_len);
1409 handle_ascii_frame_packet(packet, total_len);
1410 log_info("[FRAME_RECV_CALLBACK] ✅ FRAME_DISPATCH_COMPLETE: frame processing finished");
1411
1412 SAFE_FREE(packet);
1413 log_debug("[FRAME_RECV_CALLBACK] 🗑️ BUFFER_FREED: frame callback cleanup complete");
1414}
