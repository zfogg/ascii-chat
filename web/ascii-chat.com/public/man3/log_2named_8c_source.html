85static const char *find_generic_type_prefix(const char *start, const char *p, const char **out_type,
86 size_t *out_type_len) {
87 if (p == start) {
88 *out_type = NULL;
89 *out_type_len = 0;
90 return p;
91 }
92
93 const char *word_start = p - 1;
94 /* Skip back over whitespace, '=', ':' */
95 while (word_start > start &&
96 (*word_start == ' ' || *word_start == '\t' || *word_start == '=' || *word_start == ':')) {
97 word_start--;
98 }
99
100 /* Check if we have a digit (we're in a decimal number) */
101 if (!isdigit(*word_start)) {
102 /* Not a digit, so look for type keywords before us */
103 const char *check = word_start;
104 while (check > start && (isalnum(*check) || *check == '_')) {
105 check--;
106 }
107 check++;
108
109 const char *type_start = check;
110 size_t type_len = word_start - type_start + 1;
111
112 /* Check for known type keywords */
113 static const struct {
114 const char *type;
115 size_t len;
116 } known_types[] = {
117 {"socket", 6}, {"client", 6}, {"connection", 10}, {"fd", 2}, {"file", 4}, {"descriptor", 11},
118 {"thread", 6}, {"decoder", 7}, {"encoder", 7}, {"context", 7}, {"handler", 7},
119 };
120
121 for (size_t i = 0; i < sizeof(known_types) / sizeof(known_types[0]); i++) {
122 if (type_len == known_types[i].len && strncmp(type_start, known_types[i].type, type_len) == 0) {
123 *out_type = type_start;
124 *out_type_len = type_len;
125 return type_start;
126 }
127 }
128 }
129
130 *out_type = NULL;
131 *out_type_len = 0;
132 return p;
133}

144static const char *find_fd_prefix_start(const char *start, const char *p) {

147

149 /* Skip back over whitespace, '=', ':' */

152 word_start--;
153 }
154
155 /* Check if we have a digit (we're in a decimal number) */
156 if (!isdigit(*word_start)) {
157 /* Not a digit, so look for fd/file/descriptor/socket/sockfd keywords before us */
158 const char *check = word_start;
159 while (check > start && (isalnum(*check) || *check == '_')) {
160 check--;
161 }
162 check++; /* Move back to start of word */
163

188static const char *find_packet_type_prefix_start(const char *start, const char *p) {
189 if (p == start)
190 return p;
191
192 const char *word_start = p - 1;
193 /* Skip back over whitespace, '=', ':', '(' */
194 while (word_start > start && (*word_start == ' ' || *word_start == '\t' || *word_start == '=' || *word_start == ':' ||
195 *word_start == '(')) {
196 word_start--;
197 }
198
199 /* Check if we have a digit (we're in a decimal number) */
200 if (!isdigit(*word_start)) {
201 /* Not a digit, so look for packet type keywords before us */
202 const char *check = word_start;
203 while (check > start && (isalnum(*check) || *check == '_')) {
204 check--;
205 }
206 check++; /* Move back to start of word */
207
208 size_t word_len = word_start - check + 1;
209 /* Check for "type" keyword (part of "packet type") */
210 if (word_len >= 4 && strncasecmp(check, "type", 4) == 0) {
211 /* Look for "packet" before "type" */
212 const char *packet_check = check - 1;
213 while (packet_check > start && (*packet_check == ' ' || *packet_check == '\t')) {
214 packet_check--;
215 }
216 if (packet_check > start) {
217 const char *packet_start = packet_check;
218 while (packet_start > start && (isalnum(*packet_start) || *packet_start == '_')) {
219 packet_start--;
220 }
221 packet_start++;
222 size_t packet_len = packet_check - packet_start + 1;
223 if (packet_len == 6 && strncasecmp(packet_start, "packet", 6) == 0) {
224 return packet_start; /* Return start of "packet" in "packet type" */
225 }
226 }
227 }
228 /* Check for "packet" keyword alone */
229 if (word_len >= 6 && strncasecmp(check, "packet", 6) == 0) {
230 return check; /* Return start of "packet" */
231 }
232 }
233
234 return p; /* No prefix found, return original position */
235}

409static bool try_format_hex_address(const char *message, const char *hex_start, char *output,
410 size_t output_size, size_t *out_pos, const char **p) {
411 const char *hex_ptr = hex_start + 2; /* Skip "0x" */
412
413 /* Parse hex digits */
414 uintptr_t address = 0;
415 int hex_count = 0;
416 while (*hex_ptr && isxdigit(*hex_ptr) && hex_count < 16) {
417 address = (address << 4) | (unsigned int)(*hex_ptr > '9' ? (*hex_ptr | 0x20) - 'a' + 10 : *hex_ptr - '0');
418 hex_ptr++;
419 hex_count++;
420 }
421
422 if (hex_count == 0) {
423 return false; /* No valid hex digits */
424 }
425
426 /* Check if already inside parentheses (part of previous formatting) */
427 bool is_already_in_parens = (hex_start > message && *(hex_start - 1) == '(');
428 if (is_already_in_parens) {
429 return false;
430 }
431
432 /* Check if address is registered */
433 const char *name = named_get(address);
434 const char *type = named_get_type(address);
435 if (!name || !type) {
436 return false;
437 }
438
439 /* Format the output */
440 const char *fmt_spec = named_get_format_spec(address);
441 if (!fmt_spec) {
442 fmt_spec = "0x%tx";
443 }
444
445 char id_buffer[128];
446 int id_written = snprintf(id_buffer, sizeof(id_buffer), fmt_spec, (ptrdiff_t)address);
447 if (id_written <= 0 || id_written >= (int)sizeof(id_buffer)) {
448 return false;
449 }
450
451 char temp_output[512];
452 int temp_written = snprintf(temp_output, sizeof(temp_output), "%s/%s (%s)", type, name, id_buffer);
453 if (temp_written <= 0 || (size_t)temp_written >= sizeof(temp_output)) {
454 return false;
455 }
456
457 if (*out_pos + temp_written >= output_size - 1) {
458 return false; /* Buffer overflow */
459 }
460
461 memcpy(output + *out_pos, temp_output, temp_written);
462 *out_pos += temp_written;
463 *p = hex_ptr;
464 return true;
465}

467int log_named_format_message(const char *message, char *output, size_t output_size) {
468 if (!message || !output || output_size == 0) {
469 return -1;
470 }
471
472 size_t out_pos = 0;
473 const char *p = message;
474 bool any_transformed = false;
475
476 while (*p && out_pos < output_size - 1) {
477 /* Look for "0x" prefix indicating a hex address */
478 if (*p == '0' && *(p + 1) == 'x' && isxdigit(*(p + 2))) {
479 const char *hex_start = p;
480 if (try_format_hex_address(message, hex_start, output, output_size, &out_pos, &p)) {
481 any_transformed = true;
482 continue;
483 }
484
485 /* Not registered or formatting failed - copy original hex address */
486 const char *hex_end = hex_start + 2; /* Start after "0x" */
487 while (*hex_end && isxdigit(*hex_end)) {
488 hex_end++;
489 }
490 size_t hex_len = hex_end - hex_start;
491 if (hex_len < output_size - out_pos) {
492 memcpy(output + out_pos, hex_start, hex_len);
493 out_pos += hex_len;
494 p = hex_end;
495 } else {
496 /* Buffer overflow - truncate and stop */
497 break;
498 }
499 } else if (isdigit(*p)) {
500 /* Look for decimal integers that might be FDs or packet types */
501 const char *int_start = p;
502 int fd_value = 0;
503 int digit_count = 0;
504
505 /* Parse decimal digits */
506 while (*p && isdigit(*p) && digit_count < 6) { /* FDs rarely exceed 999999 */
507 fd_value = fd_value * 10 + (*p - '0');
508 p++;
509 digit_count++;
510 }
511
512 /* Skip if this integer is part of already-formatted output like "(fd=20)" or "(type/name (...))" */
513 bool is_already_formatted = false;
514 if (int_start - message >= 5) {
515 /* Check for "(fd=", "(pkt_type=", "(socket=", "(sockfd=", or "(type/" patterns to prevent re-formatting.
516 * Named objects are formatted as: type/name (0xaddress) or type/name (key=value)
517 * We need to detect if we're already inside a formatted output.
518 */
519 const char *check = int_start - 1;
520 while (check > message && isspace(*check))
521 check--;
522
523 // Only access *check if pointer is within bounds
524 if (check >= message && *check == '=') {
525 /* Found "=", now check what prefix it has for patterns like "(fd=20)" or "(socket=20)" or "(pkt_type=123)" */
526 const char *eq_pos = check;
527 check--;
528 while (check > message && (isalnum(*check) || *check == '_')) {
529 check--;
530 }
531 check++;
532
533 size_t prefix_len = eq_pos - check;
534 if ((prefix_len == 2 && strncmp(check, "fd", 2) == 0) ||
535 (prefix_len == 6 && strncmp(check, "socket", 6) == 0) ||
536 (prefix_len == 6 && strncmp(check, "sockfd", 6) == 0) ||
537 (prefix_len == 8 && strncmp(check, "pkt_type", 8) == 0)) {
538 is_already_formatted = true;
539 }
540 } else if (check >= message && *check == '/') {
541 /* Found "/", now check if this is a type/name pattern.
542 * Format is "(type/name (...)" where type is word characters (thread, mutex, socket, etc.)
543 * Must verify "/" is preceded by word characters AND "(" to ensure it's a formatted type.
544 */
545 const char *slash_pos = check;
546 check--;
547 while (check > message && (isalnum(*check) || *check == '_')) {
548 check--;
549 }
550 check++;
551
552 /* Check that the "/" was preceded by word characters (indicating a type name) */
553 size_t type_len = slash_pos - check;
554 if (type_len > 0 && (isalpha(*check) || *check == '_')) {
555 /* Verify this is a registered type/name pattern by checking for "(" before the type
556 * AND checking if the type actually exists in the registry */
557 const char *type_start = check;
558 const char *before_type = type_start - 1;
559 while (before_type > message && isspace(*before_type)) {
560 before_type--;
561 }
562
563 /* Only mark as formatted if:
564 * 1. Preceded by "(" - ensures pattern looks like "(type/name ...)"
565 * 2. Type exists in registry - prevents false positives on random text */
566 if (before_type >= message && *before_type == '(' && is_type_in_registry(type_start, type_len)) {
567 is_already_formatted = true;
568 }
569 }
570 }
571 }
572
573 /* Check if this is a registered packet type */
574 if (!is_already_formatted && digit_count > 0 && has_packet_type_prefix(message, int_start)) {
575 const char *name = named_get_packet_type(fd_value);
576 if (name) {
577 const char *prefix_start = find_packet_type_prefix_start(message, int_start);
578 size_t prefix_len = int_start - prefix_start;
579 size_t saved_pos = out_pos;
580
581 if (out_pos >= prefix_len) {
582 out_pos -= prefix_len;
583 }
584
585 if (write_formatted_packet_type(fd_value, name, output, output_size, &out_pos)) {
586 any_transformed = true;
587 continue;
588 }
589
590 out_pos = saved_pos; /* Restore on failure */
591 }
592 }
593
594
595 /* Check if this is a registered file descriptor */
596 if (!is_already_formatted && digit_count > 0 && has_fd_prefix(message, int_start)) {
597 const char *name = named_get_fd(fd_value);
598 if (name) {
599 const char *prefix_start = find_fd_prefix_start(message, int_start);
600 size_t prefix_len = int_start - prefix_start;
601 size_t saved_pos = out_pos;
602
603 if (out_pos >= prefix_len) {
604 out_pos -= prefix_len;
605 }
606
607 if (write_formatted_fd(fd_value, name, output, output_size, &out_pos)) {
608 any_transformed = true;
609 continue;
610 }
611
612 out_pos = saved_pos; /* Restore on failure */
613 }
614 }
615
616 /* Check if this integer has a generic type prefix (socket, client, connection, etc.)
617 * DISABLED: This feature has a backtracking bug that corrupts output when applied after
618 * other formatting operations. Re-enable only after fixing the backtracking logic to
619 * properly track correspondence between input and output positions.
620 * See commit b86fed2a8 which introduced the bug.
621 */
622 if (false && !is_already_formatted && digit_count > 0) {
623 const char *type_name = NULL;
624 size_t type_len = 0;
625 const char *prefix_start = find_generic_type_prefix(message, int_start, &type_name, &type_len);
626
627 if (type_name && prefix_start != int_start) {
628 /* Skip if already formatted with a name (e.g., "socket/listener.0" already has a /) */
629 bool skip_format = false;
630 /* Check if this looks like it's already been formatted by looking for / in the pattern */
631 const char *check_ptr = prefix_start;
632 while (check_ptr < int_start && *check_ptr && *check_ptr != '/') {
633 check_ptr++;
634 }
635 if (check_ptr < int_start && *check_ptr == '/') {
636 /* Already formatted (contains /), skip */
637 skip_format = true;
638 }
639
640 if (!skip_format) {
641 /* We found a type prefix like "socket", "client", etc.
642 * Try to look up a registered name for this type+id */
643
644 /* Normalize type name aliases: "sockfd" → "socket", "descriptor" → "fd" */
645 const char *lookup_type = type_name;
646 size_t lookup_len = type_len;
647 if (type_len == 6 && strncmp(type_name, "sockfd", 6) == 0) {
648 lookup_type = "socket";
649 lookup_len = 6;
650 } else if (type_len == 10 && strncmp(type_name, "descriptor", 10) == 0) {
651 lookup_type = "fd";
652 lookup_len = 2;
653 }
654
655 const char *name = named_get_by_type_and_id(lookup_type, lookup_len, fd_value);
656
657 char temp_output[512];
658 char type_str[32];
659 char id_buffer[64];
660
661 /* Safely copy type name */
662 if (type_len < sizeof(type_str)) {
663 memcpy(type_str, type_name, type_len);
664 type_str[type_len] = '\0';
665
666 int id_written = snprintf(id_buffer, sizeof(id_buffer), "%d", fd_value);
667 if (id_written > 0 && id_written < (int)sizeof(id_buffer)) {
668 int temp_written = 0;
669 if (name) {
670 /* Format with name: type/name (type=value) */
671 temp_written =
672 snprintf(temp_output, sizeof(temp_output), "%s/%s (%s=%s)", type_str, name, type_str, id_buffer);
673
674 if (temp_written > 0 && (size_t)temp_written < sizeof(temp_output)) {
675 /* Backtrack to remove the prefix we already copied */
676 size_t prefix_len = int_start - prefix_start;
677 if (out_pos >= prefix_len) {
678 out_pos -= prefix_len;
679 }
680
681 int copy_len = temp_written;
682 if (out_pos + copy_len < output_size - 1) {
683 memcpy(output + out_pos, temp_output, copy_len);
684 out_pos += copy_len;
685 any_transformed = true;
686 continue;
687 }
688 }
689 }
690 }
691 }
692 }
693 }
694 }
695
696 /* Not a registered FD or formatting failed - copy original number */
697 if (!copy_unformatted_decimal(int_start, p, output, output_size, &out_pos)) {
698 break;
699 }
700 } else {
701 /* Check if this is the start of "type=DIGIT" pattern */
702 if (*p == 't' && p + 4 < message + strlen(message) && is_type_equals_pattern(p)) {
703 if (!type_equals_inside_parens(p, message)) {
704 const char *digit_end = p + 5;
705 int pkt_value = parse_decimal_digits(digit_end, &digit_end);
706
707 if (pkt_value > 0) {
708 const char *name = named_get_packet_type(pkt_value);
709 if (name && write_formatted_type_equals(pkt_value, name, output, output_size, &out_pos)) {
710 p = digit_end;
711 any_transformed = true;
712 continue;
713 }
714 }
715 }
716 }
717
718 /* Regular character - copy it */
719 copy_char_to_output(output, &out_pos, *p++);
720 }
721 }
722
723 /* Null terminate */
724 if (out_pos < output_size) {
725 output[out_pos] = '\0';
726 } else {
727 output[output_size - 1] = '\0';
728 }
729
730 return any_transformed ? (int)out_pos : -1;
731}

748 static _Thread_local char work_buffer[NAMED_FORMAT_BUFFER_SIZE];
749

756
757 if (result <= 0) {
758 /* No transformation was made, we've reached a fixpoint */
759 return current == message ? message : format_buffer;
760 }
761
762 /* Check if the message changed */
763 if (strcmp(current, format_buffer) == 0) {
764 /* No actual change, return the current version */
765 return format_buffer;
766 }
767
768 /* Message changed, prepare for next iteration */
769 if (current == message) {
770 /* First iteration, use format_buffer as input for next iteration */
771 current = format_buffer;
772 } else {
773 /* Subsequent iterations, swap buffers */
774 strcpy(work_buffer, format_buffer);
775 current = work_buffer;
776 }
777 }
778
779 /* Max iterations reached, return last formatted version */
780 return format_buffer;
781}

796const char *colorize_named_string(const char *name_str) {
797#define COLORIZE_BUFFERS 4
798#define COLORIZE_BUFFER_SIZE 512
799 static char buffers[COLORIZE_BUFFERS][COLORIZE_BUFFER_SIZE];
800 static int buffer_idx = 0;
801
802 if (!name_str || name_str[0] == '\0') {
803 return name_str;
804 }
805
806 char *current_buf = buffers[buffer_idx];
807 buffer_idx = (buffer_idx + 1) % COLORIZE_BUFFERS;
808
809 /* Find the slash separator (type/name) */
810 const char *slash = strchr(name_str, '/');
811 if (!slash) {
812 /* No slash found - just return the string as-is */
813 return name_str;
814 }
815
816 /* Find opening paren for address */
817 const char *paren = strchr(name_str, '(');
818
819 /* Extract type (before slash) */
820 size_t type_len = slash - name_str;
821 char type_buf[256];
822 strncpy(type_buf, name_str, type_len < sizeof(type_buf) - 1 ? type_len : sizeof(type_buf) - 1);
823 type_buf[type_len] = '\0';
824
825 /* Extract name (between slash and paren, or end of string) */
826 const char *name_start = slash + 1;
827 size_t name_len;
828 if (paren && paren > slash) {
829 /* Skip spaces before paren */
830 const char *name_end = paren - 1;
831 while (name_end > name_start && *name_end == ' ') {
832 name_end--;
833 }
834 name_len = name_end - name_start + 1;
835 } else {
836 name_len = strlen(name_start);
837 }
838
839 char name_buf[256];
840 strncpy(name_buf, name_start, name_len < sizeof(name_buf) - 1 ? name_len : sizeof(name_buf) - 1);
841 name_buf[name_len] = '\0';
842
843 /* Apply colors */
844 const char *type_colored = colored_string(LOG_COLOR_WARN, type_buf);
845 const char *name_colored = colored_string(LOG_COLOR_DEV, name_buf);
846
847 /* Format: type(yellow)/name(blue) and add address part if present */
848 if (paren) {
849 /* Extract address (0x...) from parentheses */
850 const char *addr_start = paren + 1;
851 const char *addr_end = strchr(addr_start, ')');
852 if (addr_end) {
853 size_t addr_len = addr_end - addr_start;
854 char addr_buf[128];
855 strncpy(addr_buf, addr_start, addr_len < sizeof(addr_buf) - 1 ? addr_len : sizeof(addr_buf) - 1);
856 addr_buf[addr_len] = '\0';
857
858 const char *addr_colored = colored_string(LOG_COLOR_GREY, addr_buf);
859 safe_snprintf(current_buf, COLORIZE_BUFFER_SIZE, "%s/%s (%s)", type_colored, name_colored, addr_colored);
860 } else {
861 safe_snprintf(current_buf, COLORIZE_BUFFER_SIZE, "%s/%s %s", type_colored, name_colored, paren);
862 }
863 } else {
864 safe_snprintf(current_buf, COLORIZE_BUFFER_SIZE, "%s/%s", type_colored, name_colored);
865 }
866
867 return current_buf;
868#undef COLORIZE_BUFFERS
869#undef COLORIZE_BUFFER_SIZE
870}
