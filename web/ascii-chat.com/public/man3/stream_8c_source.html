221static int collect_video_sources(image_source_t *sources, int max_sources) {
222 int source_count = 0;
223
224 // Check for shutdown before acquiring locks to prevent lock corruption
225 if (atomic_load_bool(&g_should_exit)) {
226 return 0;
227 }
228
229 // Collect client info snapshots WITHOUT holding rwlock
230 typedef struct {
231 char client_id[MAX_CLIENT_ID_LEN];
232 bool is_active;
233 bool is_sending_video;
234 video_frame_buffer_t *video_buffer;
235 } client_snapshot_t;
236
237 client_snapshot_t client_snapshots[MAX_CLIENTS];
238 int snapshot_count = 0;
239
240 // NO LOCK: All fields are atomic or stable pointers
241 for (int i = 0; i < MAX_CLIENTS; i++) {
242 client_info_t *client = &g_client_manager.clients[i];
243
244 if (client->client_id[0] == '\0') {
245 continue; // Skip uninitialized clients
246 }
247
248 // Snapshot all needed client state (all atomic reads or stable pointers)
249 SAFE_STRNCPY(client_snapshots[snapshot_count].client_id, client->client_id,
250 sizeof(client_snapshots[snapshot_count].client_id) - 1);
251 client_snapshots[snapshot_count].is_active = atomic_load_bool(&client->active);
252 client_snapshots[snapshot_count].is_sending_video = atomic_load_bool(&client->is_sending_video);
253 client_snapshots[snapshot_count].video_buffer = client->incoming_video_buffer; // Stable pointer
254 snapshot_count++;
255 }
256
257 // Process frames (expensive operations)
258 log_dev_every(5 * NS_PER_MS_INT, "collect_video_sources: Processing %d snapshots", snapshot_count);
259 for (int i = 0; i < snapshot_count && source_count < max_sources; i++) {
260 client_snapshot_t *snap = &client_snapshots[i];
261
262 if (!snap->is_active) {
263 log_dev_every(5 * NS_PER_MS_INT, "collect_video_sources: Skipping inactive client %u", snap->client_id);
264 continue;
265 }
266
267 log_dev_every(5 * NS_PER_MS_INT, "collect_video_sources: Client %s: is_sending_video=%d", snap->client_id,
268 snap->is_sending_video);
269
270 SAFE_STRNCPY(sources[source_count].client_id, snap->client_id, sizeof(sources[source_count].client_id) - 1);
271 sources[source_count].image = NULL; // Will be set if video is available
272 sources[source_count].has_video = false;
273
274 // Declare these outside the if block so they're accessible later
275 multi_source_frame_t current_frame = {0};
276 bool got_new_frame = false;
277
278 // Always try to get the last available video frame for consistent ASCII generation
279 // The double buffer ensures we always have the last valid frame
280 if (snap->is_sending_video && snap->video_buffer) {
281 // Get the latest frame (always available from double buffer)
282 const video_frame_t *frame = video_frame_get_latest(snap->video_buffer);
283
284 if (!frame) {
285 continue; // Skip to next snapshot
286 }
287
288 // Try to access frame fields ONE AT A TIME to pinpoint the hang
289 void *frame_data_ptr = frame->data;
290
291 size_t frame_size_val = frame->size;
292
293 // Compute hash of incoming frame to verify it's changing
294 uint32_t incoming_hash = 0;
295 if (frame_data_ptr && frame_size_val > 0) {
296 for (size_t i = 0; i < frame_size_val && i < 1000; i++) {
297 uint8_t byte = ((unsigned char *)frame_data_ptr)[i];
298 incoming_hash = (uint32_t)((uint64_t)incoming_hash * 31 + byte);
299 }
300 }
301
302 // DIAGNOSTIC: Track incoming frame changes from buffer
303 static uint32_t last_buffer_hash = 0;
304 if (incoming_hash != last_buffer_hash) {
305 log_info("BUFFER_FRAME CHANGE: Client %u got NEW frame from buffer: hash=0x%08x (prev=0x%08x) size=%zu",
306 snap->client_id, incoming_hash, last_buffer_hash, frame_size_val);
307 last_buffer_hash = incoming_hash;
308 } else {
309 log_dev_every(25000, "BUFFER_FRAME DUPLICATE: Client %u frame hash=0x%08x size=%zu (no change)",
310 snap->client_id, incoming_hash, frame_size_val);
311 }
312
313 // DETAILED BUFFER INSPECTION: Extract and log frame dimensions + first pixels
314 if (frame_data_ptr && frame_size_val >= 8) {
315 uint32_t width_net, height_net;
316 memcpy(&width_net, frame_data_ptr, sizeof(uint32_t));
317 memcpy(&height_net, (char *)frame_data_ptr + sizeof(uint32_t), sizeof(uint32_t));
318 uint32_t width = NET_TO_HOST_U32(width_net);
319 uint32_t height = NET_TO_HOST_U32(height_net);
320
321 // Extract first 3 RGB pixels to inspect actual pixel data
322 uint8_t *pixel_ptr = (uint8_t *)frame_data_ptr + 8;
323 uint32_t first_pixel_rgb = 0;
324 if (frame_size_val >= 11) {
325 first_pixel_rgb = ((uint32_t)pixel_ptr[0] << 16) | ((uint32_t)pixel_ptr[1] << 8) | (uint32_t)pixel_ptr[2];
326 }
327
328 log_info("BUFFER_INSPECT: Client %u dims=%ux%u pixel_data_size=%zu first_pixel_rgb=0x%06x data_hash=0x%08x",
329 snap->client_id, width, height, frame_size_val - 8, first_pixel_rgb, incoming_hash);
330 }
331
332 log_debug_every(5 * NS_PER_MS_INT, "Video mixer: client %u incoming frame hash=0x%08x size=%zu", snap->client_id,
333 incoming_hash, frame_size_val);
334
335 if (frame_data_ptr && frame_size_val > 0 && frame_size_val >= (sizeof(uint32_t) * 2 + 3)) {
336 // PARSE AND VALIDATE DIMENSIONS BEFORE COPYING
337 // Don't trust frame->size - calculate correct size from dimensions
338 uint32_t peek_width = NET_TO_HOST_U32(read_u32_unaligned(frame_data_ptr));
339 uint32_t peek_height = NET_TO_HOST_U32(read_u32_unaligned(frame_data_ptr + sizeof(uint32_t)));
340
341 // Reject obviously corrupted dimensions
342 if (peek_width == 0 || peek_height == 0 || peek_width > 4096 || peek_height > 2160) {
343 log_debug("Per-client %u: rejected dimensions %ux%u as corrupted", snap->client_id, peek_width, peek_height);
344 continue;
345 }
346
347 // Validate dimensions
348 if (image_validate_dimensions((size_t)peek_width, (size_t)peek_height) != ASCIICHAT_OK) {
349 continue;
350 }
351
352 // Calculate CORRECT frame size based on dimensions (don't trust frame->size)
353 size_t correct_frame_size = sizeof(uint32_t) * 2;
354 {
355 size_t rgb_size = 0;
356 if (image_calc_rgb_size((size_t)peek_width, (size_t)peek_height, &rgb_size) != ASCIICHAT_OK) {
357 log_debug("Per-client: rgb_size calc failed for %ux%u", peek_width, peek_height);
358 continue;
359 }
360 correct_frame_size += rgb_size;
361 }
362
363 log_debug_every(NS_PER_MS_INT, "Per-client: frame dimensions=%ux%u, frame_size=%zu, correct_size=%zu",
364 peek_width, peek_height, frame_size_val, correct_frame_size);
365
366 // Verify frame is at least large enough for the correct size
367 if (frame_size_val < correct_frame_size) {
368 log_debug("Per-client: frame too small: got %zu, need %zu", frame_size_val, correct_frame_size);
369 continue;
370 }
371
372 // We have frame data - copy ONLY the correct amount based on dimensions
373 // Use SAFE_MALLOC (not buffer pool - image_new_from_pool uses pool and causes overlap)
374 current_frame.data = SAFE_MALLOC(correct_frame_size, void *);
375
376 if (current_frame.data) {
377 memcpy(current_frame.data, frame->data, correct_frame_size);
378 current_frame.size = correct_frame_size;
379 SAFE_STRNCPY(current_frame.source_client_id, snap->client_id, sizeof(current_frame.source_client_id) - 1);
380 current_frame.timestamp = (uint32_t)(frame->capture_timestamp_ns / NS_PER_SEC_INT);
381 got_new_frame = true;
382 }
383 } else {
384 }
385 }
386
387 multi_source_frame_t *frame_to_use = got_new_frame ? &current_frame : NULL;
388
389 if (frame_to_use && frame_to_use->data && frame_to_use->size > sizeof(uint32_t) * 2) {
390 // Parse the image data
391 // Format: [width:4][height:4][rgb_data:w*h*3]
392 // Use unaligned read helpers - frame data may not be aligned
393 uint32_t img_width = NET_TO_HOST_U32(read_u32_unaligned(frame_to_use->data));
394 uint32_t img_height = NET_TO_HOST_U32(read_u32_unaligned(frame_to_use->data + sizeof(uint32_t)));
395
396 // Debug logging to understand the data
397 if (img_width == 0xBEBEBEBE || img_height == 0xBEBEBEBE) {
398 SET_ERRNO(ERROR_INVALID_STATE, "UNINITIALIZED MEMORY DETECTED! First 16 bytes of frame data:");
399 uint8_t *bytes = (uint8_t *)frame_to_use->data;
400 SET_ERRNO(ERROR_INVALID_STATE,
401 " %02X %02X %02X %02X %02X %02X %02X %02X %02X %02X %02X %02X %02X %02X %02X %02X", bytes[0],
402 bytes[1], bytes[2], bytes[3], bytes[4], bytes[5], bytes[6], bytes[7], bytes[8], bytes[9], bytes[10],
403 bytes[11], bytes[12], bytes[13], bytes[14], bytes[15]);
404 }
405
406 // Validate dimensions using image utility function
407 if (image_validate_dimensions((size_t)img_width, (size_t)img_height) != ASCIICHAT_OK) {
408 SET_ERRNO(ERROR_INVALID_STATE,
409 "Per-client: Invalid image dimensions from client %u: %ux%u (data may be corrupted)", snap->client_id,
410 img_width, img_height);
411 source_count++;
412 continue;
413 }
414
415 // Calculate expected frame size with overflow checking
416 size_t expected_size = sizeof(uint32_t) * 2;
417 {
418 size_t rgb_size = 0;
419 if (image_calc_rgb_size((size_t)img_width, (size_t)img_height, &rgb_size) != ASCIICHAT_OK) {
420 SET_ERRNO(ERROR_INVALID_STATE, "Per-client: RGB size calculation failed for client %u: %ux%u",
421 snap->client_id, img_width, img_height);
422 source_count++;
423 continue;
424 }
425 expected_size += rgb_size;
426 }
427 if (frame_to_use->size != expected_size) {
428 SET_ERRNO(ERROR_INVALID_STATE,
429 "Per-client: Frame size mismatch from client %u: got %zu, expected %zu for %ux%u image",
430 snap->client_id, frame_to_use->size, expected_size, img_width, img_height);
431 source_count++;
432 continue;
433 }
434
435 // Extract pixel data pointer
436 rgb_pixel_t *pixels = (rgb_pixel_t *)(frame_to_use->data + (sizeof(uint32_t) * 2));
437
438 // Create image from buffer pool for consistent video pipeline management
439 image_t *img = image_new_from_pool(img_width, img_height);
440 if (!img) {
441 log_error("Per-client: image_new_from_pool failed for %ux%u", img_width, img_height);
442 continue;
443 }
444 memcpy(img->pixels, pixels, (size_t)img_width * (size_t)img_height * sizeof(rgb_pixel_t));
445 sources[source_count].image = img;
446 sources[source_count].has_video = true;
447
448 // Free temporary frame buffer - image has its own pixel data now
449 if (got_new_frame) {
450 SAFE_FREE(current_frame.data);
451 }
452 } else {
453 // frame_to_use check failed - clean up allocated frame data
454 if (got_new_frame && current_frame.data) {
455 SAFE_FREE(current_frame.data);
456 }
457 }
458
459 // Increment source count for this active client (with or without video)
460 source_count++;
461 }
462
463 return source_count;
464}

476static image_t *create_single_source_composite(image_source_t *sources, int source_count,
477 const char *target_client_id __attribute__((unused)),
478 unsigned short width __attribute__((unused)),
479 unsigned short height __attribute__((unused))) {

482 for (int i = 0; i < source_count; i++) {

484 single_source = sources[i].image;

486 }

488

523static void calculate_optimal_grid_layout(image_source_t *sources, int source_count, int sources_with_video,
524 int terminal_width, int terminal_height, int *out_cols, int *out_rows) {
525 // Special cases
526 if (sources_with_video == 0) {
527 *out_cols = 0;
528 *out_rows = 0;
529 return;
530 }
531
532 if (sources_with_video == 1) {
533 *out_cols = 1;
534 *out_rows = 1;
535 return;
536 }
537
538 // ASCII character aspect ratio: characters are ~2x taller than wide
539 // So we need to adjust terminal dimensions to visual space
540 const float CHAR_ASPECT = 2.0f; // Character height / width ratio
541
542 // Calculate average aspect ratio of all video sources
543 float avg_aspect = 0.0f;
544 int aspect_count = 0;
545 for (int i = 0; i < source_count; i++) {
546 if (sources[i].has_video && sources[i].image) {
547 float aspect = (float)sources[i].image->w / (float)sources[i].image->h;
548 avg_aspect += aspect;
549 aspect_count++;
550 }
551 }
552 if (aspect_count > 0) {
553 avg_aspect /= aspect_count;
554 } else {
555 avg_aspect = 1.6f; // Default aspect ratio
556 }
557
558 // Try all reasonable grid configurations
559 int best_cols = 1;
560 int best_rows = sources_with_video;
561 float best_utilization = 0.0f;
562
563 // Try grid configurations from 1x1 up to reasonable limits
564 for (int cols = 1; cols <= sources_with_video; cols++) {
565 int rows = (sources_with_video + cols - 1) / cols; // Ceiling division
566
567 // Skip configurations with too many empty cells (more than one row worth)
568 int total_cells = cols * rows;
569 int empty_cells = total_cells - sources_with_video;
570 if (empty_cells > cols) {
571 continue;
572 }
573
574 // Calculate cell dimensions for this configuration
575 int cell_width = terminal_width / cols;
576 int cell_height = terminal_height / rows;
577
578 // Skip if cells would be too small
579 if (cell_width < 20 || cell_height < 10) {
580 continue;
581 }
582
583 // Calculate total area utilized by all videos in this configuration
584 // For each video, calculate how much space it would use in a cell
585 // IMPORTANT: Account for character aspect ratio (chars are 2x taller than wide)
586 float total_area_used = 0.0f;
587 int cell_area = cell_width * cell_height;
588
589 for (int i = 0; i < sources_with_video; i++) {
590 // Use average aspect ratio for calculation
591 float video_aspect = avg_aspect;
592
593 // Calculate VISUAL cell aspect ratio (accounting for character shape)
594 // A cell that is cell_width chars wide and cell_height chars tall
595 // has visual aspect = cell_width / (cell_height * CHAR_ASPECT)
596 float cell_visual_aspect = (float)cell_width / ((float)cell_height * CHAR_ASPECT);
597
598 // Calculate fitted dimensions while preserving aspect ratio
599 int fitted_width, fitted_height;
600
601 if (video_aspect > cell_visual_aspect) {
602 // Video is wider than cell - fit to width
603 fitted_width = cell_width;
604 // Visual height needed: cell_width / video_aspect
605 // Character height: visual_height / CHAR_ASPECT
606 fitted_height = (int)((cell_width / video_aspect) / CHAR_ASPECT);
607 } else {
608 // Video is taller than cell - fit to height
609 fitted_height = cell_height;
610 // Visual width needed: cell_height * CHAR_ASPECT * video_aspect
611 fitted_width = (int)(cell_height * CHAR_ASPECT * video_aspect);
612 }
613
614 // Clamp to cell bounds
615 if (fitted_width > cell_width) {
616 fitted_width = cell_width;
617 }
618 if (fitted_height > cell_height) {
619 fitted_height = cell_height;
620 }
621
622 // Add area used by this video
623 total_area_used += fitted_width * fitted_height;
624 }
625
626 // Calculate utilization percentage
627 float total_available_area = (float)(cell_area * sources_with_video);
628 float utilization = total_area_used / total_available_area;
629
630 float test_cell_visual_aspect = (float)cell_width / ((float)cell_height * CHAR_ASPECT);
631 log_dev_every(LOG_RATE_NORMAL, " Testing %dx%d: cell=%dx%d (visual aspect %.2f), utilization=%.1f%%", cols, rows,
632 cell_width, cell_height, test_cell_visual_aspect, utilization * 100.0f);
633
634 // Prefer configurations with better utilization
635 if (utilization > best_utilization) {
636 best_utilization = utilization;
637 best_cols = cols;
638 best_rows = rows;
639 }
640 }
641
642 *out_cols = best_cols;
643 *out_rows = best_rows;
644
645 float terminal_visual_aspect = (float)terminal_width / ((float)terminal_height * CHAR_ASPECT);
646 log_dev_every(LOG_RATE_NORMAL,
647 "Grid layout: %d clients -> %dx%d grid (%.1f%% utilization) | terminal=%dx%d (char aspect %.2f, VISUAL "
648 "aspect %.2f), video aspect: %.2f",
649 sources_with_video, best_cols, best_rows, best_utilization * 100.0f, terminal_width, terminal_height,
650 (float)terminal_width / (float)terminal_height, terminal_visual_aspect, avg_aspect);
651}

664static image_t *create_multi_source_composite(image_source_t *sources, int source_count, int sources_with_video,
665 const char *target_client_id, unsigned short width,
666 unsigned short height) {
667 (void)target_client_id; // Unused - composite is same for all clients now
668
669 // Calculate optimal grid layout using space-maximizing algorithm
670 int grid_cols, grid_rows;
671 calculate_optimal_grid_layout(sources, source_count, sources_with_video, width, height, &grid_cols, &grid_rows);
672
673 // Calculate composite dimensions in PIXELS for half-block mode
674 // Terminal dimensions are in CHARACTERS, need to convert to pixels:
675 // - Width: 1 char = 1 horizontal pixel
676 // - Height: 1 char = 2 vertical pixels (half-block = 2 pixels per char)
677 const int PIXELS_PER_CHAR_HEIGHT = 2;
678 int composite_width_px = width; // chars = pixels horizontally
679 int composite_height_px = height * PIXELS_PER_CHAR_HEIGHT; // chars * 2 = pixels vertically
680
681 // Create composite with final dimensions - no recreation needed
682 image_t *composite = image_new_from_pool(composite_width_px, composite_height_px);
683 image_clear(composite);
684
685 // Place each source in the grid
686 int video_source_index = 0; // Track only sources with video
687 for (int i = 0; i < source_count && video_source_index < 9; i++) { // Max 9 sources in 3x3 grid
688 if (!sources[i].image)
689 continue;
690
691 int row = video_source_index / grid_cols;
692 int col = video_source_index % grid_cols;
693 video_source_index++;
694
695 // Use actual composite dimensions for cell calculations
696 // Composite is now in PIXELS (already converted from terminal characters)
697 int cell_width_px = composite->w / grid_cols;
698 int cell_height_px = composite->h / grid_rows;
699
700 // Calculate aspect ratios
701 float src_aspect = (float)sources[i].image->w / (float)sources[i].image->h;
702 float cell_visual_aspect = (float)cell_width_px / (float)cell_height_px;
703
704 int target_width_px, target_height_px;
705
706 // CONTAIN strategy: Fill one dimension completely, let other scale down (never overflow)
707 // Compare aspects to decide which dimension to fill
708 if (src_aspect > cell_visual_aspect) {
709 // Video is WIDER than cell → fill WIDTH (height will be smaller)
710 target_width_px = cell_width_px;
711 target_height_px = (int)((cell_width_px / src_aspect) + 0.5f);
712 } else {
713 // Video is TALLER than cell → fill HEIGHT (width will be smaller)
714 target_height_px = cell_height_px;
715 target_width_px = (int)((cell_height_px * src_aspect) + 0.5f);
716 }
717
718 log_dev_every(LOG_RATE_NORMAL, "Cell %d: %dx%d px, video %.1f, cell %.2f → target %dx%d px (fill %s)",
719 video_source_index - 1, cell_width_px, cell_height_px, src_aspect, cell_visual_aspect,
720 target_width_px, target_height_px, (src_aspect > cell_visual_aspect) ? "WIDTH" : "HEIGHT");
721
722 // Create resized image with standard allocation
723 image_t *resized = image_new_from_pool(target_width_px, target_height_px);
724 image_resize(sources[i].image, resized);
725
726 // Calculate cell position in pixels (after any composite recreation)
727 int cell_x_offset_px = col * cell_width_px;
728 int cell_y_offset_px = row * cell_height_px;
729
730 // Grid centering strategy:
731 // - Multi-client: Apply padding ONLY to edge cells to center the grid as a whole
732 // (left column gets left padding, right column gets right padding)
733 // (top row gets top padding, bottom row gets bottom padding)
734 // This keeps cells edge-to-edge while centering the entire grid
735 // - Single client: Center the image within the cell
736 int x_padding_px, y_padding_px;
737
738 // Center images within their cells for all layouts
739 // This prevents gaps/stripes between clients
740 x_padding_px = (cell_width_px - target_width_px) / 2;
741 y_padding_px = (cell_height_px - target_height_px) / 2;
742
743 // Define cell boundaries for clipping (prevents bleeding into adjacent cells)
744 int cell_x_min = cell_x_offset_px;
745 int cell_x_max = cell_x_offset_px + cell_width_px - 1;
746 int cell_y_min = cell_y_offset_px;
747 int cell_y_max = cell_y_offset_px + cell_height_px - 1;
748
749 // Copy resized image to composite with cell boundary clipping
750 // This allows images to fill cells completely while preventing overlap
751
752 for (int y = 0; y < resized->h; y++) {
753 for (int x = 0; x < resized->w; x++) {
754 // Calculate destination position
755 int dst_x = cell_x_offset_px + x_padding_px + x;
756 int dst_y = cell_y_offset_px + y_padding_px + y;
757
758 // Clip to cell boundaries (prevents bleeding into adjacent cells)
759 if (dst_x < cell_x_min || dst_x > cell_x_max || dst_y < cell_y_min || dst_y > cell_y_max) {
760 continue; // Skip pixels outside cell boundaries
761 }
762
763 // Additional composite boundary check
764 if (dst_x < 0 || dst_x >= composite->w || dst_y < 0 || dst_y >= composite->h) {
765 continue; // Skip pixels outside composite
766 }
767
768 // Copy pixel
769 int src_idx = (y * resized->w) + x;
770 int dst_idx = (dst_y * composite->w) + dst_x;
771 composite->pixels[dst_idx] = resized->pixels[src_idx];
772 }
773 }
774
775 image_destroy_to_pool(resized);
776 }
777
778 return composite;
779}

790static char *convert_composite_to_ascii(image_t *composite, const char *target_client_id, unsigned short width,
791 unsigned short height) {
792 // LOCK OPTIMIZATION: Don't call find_client_by_id() - it would acquire rwlock unnecessarily
793 // Instead, the render thread already has snapshot of client state, so we just need palette data
794 // which is stable after initialization

797 // since palette is initialized once and never changes
798 client_info_t *render_client = NULL;

801 for (int i = 0; i < MAX_CLIENTS; i++) {

811 return NULL;

815 // Terminal caps are set once during handshake and never change, so this is safe
816 bool has_terminal_caps_snapshot = render_client->has_terminal_caps;
817 if (!has_terminal_caps_snapshot) {
818 SET_ERRNO(ERROR_INVALID_STATE, "Per-client %u: Terminal capabilities not received", target_client_id);
819 return NULL;

826 return NULL;

830 // Palette data is stable after initialization, so no locking needed
831 const int h = caps_snapshot.render_mode == RENDER_MODE_HALF_BLOCK ? height * 2 : height;
832
833 // DEBUG: Log dimensions being used for ASCII conversion
834 log_dev_every(LOG_RATE_SLOW, "convert_composite_to_ascii: composite=%dx%d, terminal=%dx%d, h=%d (mode=%d)",
835 composite->w, composite->h, width, height, h, caps_snapshot.render_mode);

838 // with proper CHAR_ASPECT correction. The composite may have been pre-fitted in pixel space,
839 // but ascii_convert will handle terminal character aspect ratio properly when aspect_ratio=true.
840 uint64_t convert_start_ns = time_get_ns();
841 char *ascii_frame = ascii_convert_with_capabilities(composite, width, h, &caps_snapshot, true, false,
842 render_client->client_palette_chars);
843 uint64_t convert_end_ns = time_get_ns();

845
846 if (convert_duration_ns > 5 * NS_PER_MS_INT) { // Log if > 5ms
847 char duration_str[32];
848 time_pretty((uint64_t)((double)convert_duration_ns), -1, duration_str, sizeof(duration_str));
849 log_warn("SLOW_ASCII_CONVERT: Client %u took %s to convert %dx%d image to ASCII", target_client_id, duration_str,
850 composite->w, composite->h);
851 }
852
853 return ascii_frame;

956// Compute hash of all active video sources for cache invalidation
957// Uses hardware-accelerated CRC32 for ultra-fast hashing
958char *create_mixed_ascii_frame_for_client(const char *target_client_id, unsigned short width, unsigned short height,
959 bool wants_stretch, size_t *out_size, bool *out_grid_changed,
960 int *out_sources_count) {
961 (void)wants_stretch; // Unused - we always handle aspect ratio ourselves
962
963 // Register stream atomics with named debug registry
964 static bool stream_atomics_registered = false;
965 if (!stream_atomics_registered) {
966 NAMED_REGISTER_ATOMIC(&g_previous_active_video_count, "server_video_source_count_for_layout_detection", NULL);
967 stream_atomics_registered = true;
968 }
969
970 uint64_t frame_gen_start_ns = time_get_ns();
971
972 // Initialize output parameters
973 if (out_grid_changed) {
974 *out_grid_changed = false;
975 }
976 if (out_sources_count) {
977 *out_sources_count = 0;
978 }
979
980 if (!out_size || width == 0 || height == 0) {
981 SET_ERRNO(ERROR_INVALID_PARAM,
982 "Invalid parameters for create_mixed_ascii_frame_for_client: width=%u, height=%u, out_size=%p", width,
983 height, out_size);
984 return NULL;
985 }
986
987 // Collect all active clients and their image sources
988 image_source_t sources[MAX_CLIENTS];
989 uint64_t collect_start_ns = time_get_ns();
990 int source_count = collect_video_sources(sources, MAX_CLIENTS);
991 uint64_t collect_end_ns = time_get_ns();
992
993 // Count sources that actually have video data
994 int sources_with_video = 0;
995 for (int i = 0; i < source_count; i++) {
996 if (sources[i].has_video && sources[i].image) {
997 sources_with_video++;
998 }
999 }
1000
1001 static uint64_t last_detailed_log = 0;
1002 uint64_t now_ns = collect_end_ns;
1003 if (now_ns - last_detailed_log > 333 * NS_PER_MS_INT) { // Log every 333ms (3x per second)
1004 last_detailed_log = now_ns;
1005 log_info("FRAME_GEN_START: target_client=%u sources=%d collect=%.1fms", target_client_id, sources_with_video,
1006 (collect_end_ns - collect_start_ns) / NS_PER_MS);
1007 }
1008
1009 // Return the source count for debugging/tracking
1010 if (out_sources_count) {
1011 *out_sources_count = sources_with_video;
1012 }
1013
1014 // GRID LAYOUT CHANGE DETECTION:
1015 // Check if the number of active video sources has changed
1016 // NOTE: We only UPDATE the count and SIGNAL the change via out parameter
1017 // Broadcasting CLEAR_CONSOLE must happen AFTER the new frames are written to buffers
1018 // to prevent race condition where CLEAR arrives before new frame is ready
1019 uint64_t previous_count = atomic_load_u64(&g_previous_active_video_count);
1020 if (sources_with_video != (int)previous_count) {
1021 // Use compare-and-swap to ensure only ONE thread detects the change
1022 if (atomic_cas_u64(&g_previous_active_video_count, &previous_count, (uint64_t)sources_with_video)) {
1023 log_dev_every(
1024 LOG_RATE_DEFAULT,
1025 "Grid layout changing: %d -> %d active video sources - caller will broadcast clear AFTER buffering frame",
1026 previous_count, sources_with_video);
1027 if (out_grid_changed) {
1028 *out_grid_changed = true; // Signal to caller
1029 }
1030 }
1031 }
1032
1033 // No active video sources - don't generate placeholder frames
1034 image_t *composite = NULL;
1035
1036 if (sources_with_video == 0) {
1037 *out_size = 0;
1038 // No active video sources for client - this isn't an error.
1039 // Return NULL to indicate no frame should be sent
1040 return NULL;
1041 }
1042
1043 START_TIMER("composite_create");
1044 if (sources_with_video == 1) {
1045 // Single source handling - create composite and convert to ASCII
1046 // Note: create_single_source_composite returns a reference to sources[i].image
1047 // which could be modified by other threads. Make a copy to prevent concurrent
1048 // modification during ascii_convert_with_capabilities.
1049 image_t *single_source = create_single_source_composite(sources, source_count, target_client_id, width, height);
1050 if (single_source) {
1051 composite = image_new_copy(single_source);
1052 if (!composite) {
1053 SET_ERRNO(ERROR_MEMORY, "Failed to copy single source composite");
1054 *out_size = 0;
1055 return NULL;
1056 }
1057 }
1058 } else {
1059 // Multiple sources - create grid layout
1060 composite =
1061 create_multi_source_composite(sources, source_count, sources_with_video, target_client_id, width, height);
1062 }
1063 STOP_TIMER_AND_LOG(dev, 500000, "composite_create", "Composite creation (single/multi)");
1064
1065 char *out = NULL;
1066
1067 if (!composite) {
1068 SET_ERRNO(ERROR_INVALID_STATE, "Per-client %s: Failed to create composite image", target_client_id);
1069 *out_size = 0;
1070 out = NULL;
1071 }
1072
1073 // Convert composite to ASCII using client capabilities
1074 // Pass terminal dimensions so the frame can be padded to full width
1075 START_TIMER("ascii_convert");
1076 char *ascii_frame = convert_composite_to_ascii(composite, target_client_id, width, height);
1077 STOP_TIMER_AND_LOG(dev, 500000, "ascii_convert", "ASCII conversion (composite to ASCII)");
1078
1079 if (ascii_frame) {
1080 // The frame should have been null-terminated by the padding functions.
1081 // Use strlen() which is optimized and reliable
1082 size_t ascii_len = strlen(ascii_frame);
1083
1084 // Safety check: don't accept unreasonably large frames (10MB limit)
1085 if (ascii_len > 10 * 1024 * 1024) {
1086 log_error("Frame size exceeds 10MB safety limit (possible buffer overflow)");
1087 SET_ERRNO(ERROR_INVALID_PARAM, "Frame size exceeds 10MB");
1088 return NULL;
1089 }
1090
1091 // Ensure frame ends with a reset sequence to avoid garbage at terminal
1092 // This prevents color codes from leaking into uninitialized memory
1093 const char reset_seq[] = "\033[0m";
1094 const size_t reset_len = 4;
1095
1096 if (ascii_len >= reset_len) {
1097 // Check if frame already ends with reset
1098 const char *frame_end = ascii_frame + ascii_len - reset_len;
1099 if (strncmp(frame_end, reset_seq, reset_len) == 0) {
1100 // Frame properly ends with reset, use full length
1101 *out_size = ascii_len;
1102 } else {
1103 // Frame doesn't end with reset - this is the REAL bug!
1104 // Truncate to the last occurrence of reset sequence
1105 const char *last_reset = NULL;
1106 for (const char *p = ascii_frame + ascii_len - reset_len; p >= ascii_frame; p--) {
1107 if (strncmp(p, reset_seq, reset_len) == 0) {
1108 last_reset = p;
1109 break;
1110 }
1111 }
1112
1113 if (last_reset) {
1114 // Include the reset sequence and truncate after it
1115 *out_size = (size_t)(last_reset - ascii_frame) + reset_len;
1116 ascii_frame[*out_size] = '\0'; // Ensure null termination
1117 log_warn("Frame was missing reset at end (had garbage), truncated from %zu to %zu bytes", ascii_len,
1118 *out_size);
1119 } else {
1120 // No reset found, use full length as fallback
1121 *out_size = ascii_len;
1122 log_warn("Frame has no reset sequences, sending full %zu bytes", ascii_len);
1123 }
1124 }
1125 } else {
1126 // Frame too short to have a reset, use as-is
1127 *out_size = ascii_len;
1128 }
1129
1130 log_dev_every(LOG_RATE_SLOW, "create_mixed_ascii_frame_for_client: Final frame size=%zu bytes for client %u",
1131 *out_size, target_client_id);
1132
1133 // Debug: Log the last 50 bytes of the frame to see what's really there
1134 if (*out_size >= 50) {
1135 char hex_buf[300] = {0};
1136 size_t hex_len = 0;
1137 const uint8_t *last_bytes = (const uint8_t *)ascii_frame + (*out_size - 50);
1138 for (int i = 0; i < 50 && hex_len < sizeof(hex_buf) - 5; i++) {
1139 hex_len += snprintf(hex_buf + hex_len, sizeof(hex_buf) - hex_len, "%02X ", last_bytes[i]);
1140 }
1141 log_dev_every(4500 * US_PER_MS_INT, "FRAME_LAST_50_BYTES (hex): %s", hex_buf);
1142
1143 // Also log as ASCII for readability
1144 char ascii_buf[100] = {0};
1145 for (int i = 0; i < 50 && i < (int)sizeof(ascii_buf) - 1; i++) {
1146 if (last_bytes[i] >= 32 && last_bytes[i] < 127) {
1147 ascii_buf[i] = (char)last_bytes[i];
1148 } else if (last_bytes[i] == '\n') {
1149 ascii_buf[i] = 'N';
1150 } else if (last_bytes[i] == '\0') {
1151 ascii_buf[i] = '0';
1152 } else {
1153 ascii_buf[i] = '.';
1154 }
1155 }
1156 log_dev_every(4500 * US_PER_MS_INT, "FRAME_LAST_50_ASCII: %s", ascii_buf);
1157 }
1158
1159 out = ascii_frame;
1160 } else {
1161 SET_ERRNO(ERROR_TERMINAL, "Per-client %u: Failed to convert image to ASCII", target_client_id);
1162 *out_size = 0;
1163 }
1164
1165 START_TIMER("image_cleanup");
1166 if (composite) {
1167 // For single source, composite is a malloc-allocated copy, not from pool
1168 // Check alloc method to determine correct destroy function
1169 if (composite->alloc_method == IMAGE_ALLOC_POOL) {
1170 image_destroy_to_pool(composite);
1171 } else {
1172 image_destroy(composite);
1173 }
1174 }
1175 for (int i = 0; i < source_count; i++) {
1176 if (sources[i].image) {
1177 image_destroy_to_pool(sources[i].image);
1178 }
1179 }
1180 STOP_TIMER_AND_LOG(dev, 500000, "image_cleanup", "Image cleanup (composite + sources)");
1181
1182 uint64_t frame_gen_end_ns = time_get_ns();
1183 uint64_t frame_gen_duration_ns = frame_gen_end_ns - frame_gen_start_ns;
1184 if (frame_gen_duration_ns > 10 * NS_PER_MS_INT) { // Log if > 10ms
1185 char duration_str[32];
1186 time_pretty((uint64_t)((double)frame_gen_duration_ns), -1, duration_str, sizeof(duration_str));
1187 log_warn("SLOW_FRAME_GENERATION: Client %u full frame gen took %s", target_client_id, duration_str);
1188 }
1189
1190 return out;
1191}
