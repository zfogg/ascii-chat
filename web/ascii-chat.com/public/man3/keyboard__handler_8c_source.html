54static int next_color_mode(int current) {
55 // Cycle: TERM_COLOR_NONE → TERM_COLOR_16 → TERM_COLOR_256 → TERM_COLOR_TRUECOLOR → TERM_COLOR_NONE
56 switch (current) {
57 case 0: // TERM_COLOR_NONE
58 return 1; // TERM_COLOR_16
59 case 1: // TERM_COLOR_16
60 return 2; // TERM_COLOR_256
61 case 2: // TERM_COLOR_256
62 return 3; // TERM_COLOR_TRUECOLOR
63 default:
64 return 0; // Back to TERM_COLOR_NONE
65 }
66}

71static int next_render_mode(int current) {
72 // 3 render modes: FOREGROUND (0), BACKGROUND (1), HALF_BLOCK (2)
73 return (current + 1) % 3;

87void session_handle_keyboard_input(session_capture_ctx_t *capture, session_display_ctx_t *display, keyboard_key_t key) {
88 // Debug: log all key codes to help identify unknown keys
89 if (key != KEY_NONE) {
90 log_info("KEYBOARD INPUT: code=%d (0x%02x) char='%c'", key, key, (key >= 32 && key < 127) ? key : '?');
91 if (key == 'x' || key == 'X') {
92 log_info("!!! X KEY DETECTED AT HANDLER ENTRY !!!");
93 }
94 }
95
96 switch ((int)key) {
97 // ===== HELP SCREEN TOGGLE =====
98 case KEY_QUESTION: {
99 log_info("✓✓✓ USER PRESSED ? KEY - HELP SCREEN TOGGLE ✓✓✓");
100 if (display) {
101 log_info("Toggling help screen (display=%p)", (void *)display);
102 keyboard_help_toggle(display);
103 // Render help screen immediately so user sees it
104 keyboard_help_render(display);
105 log_info("✓ Help screen toggle complete");
106 } else {
107 log_error("ERROR: Cannot toggle help - display context is NULL");
108 }
109 break;
110 }
111
112 // ===== HELP SCREEN CLOSE / QUIT =====
113 case KEY_ESCAPE: {
114 if (display && keyboard_help_is_active(display)) {
115 // Close help screen if it's active
116 keyboard_help_toggle(display);
117 terminal_clear_screen();
118 } else {
119 // If help screen is not active, quit the app (like Ctrl-C)
120 // The signal handler will gracefully shutdown all modes (client, server, mirror, etc.)
121 raise(SIGINT);
122 }
123 break;
124 }
125
126 // ===== SEEK CONTROLS (file sources only) =====
127 case KEY_LEFT: { // Seek backward 30 seconds
128 if (capture) {
129 media_source_t *source = (media_source_t *)session_capture_get_media_source(capture);
130 if (source && media_source_get_type(source) == MEDIA_SOURCE_FILE) {
131 double current_pos = media_source_get_position(source);
132 if (current_pos >= 0.0) {
133 double new_pos = current_pos - 30.0;
134 if (new_pos < 0.0) {
135 new_pos = 0.0;
136 }
137 asciichat_error_t err = media_source_seek(source, new_pos);
138 if (err == ASCIICHAT_OK) {
139 log_info("Seeked backward to %.1f seconds", new_pos);
140 }
141 void *audio_ctx = session_capture_get_audio_context(capture);
142 if (audio_ctx) {
143 audio_flush_playback_buffers((audio_context_t *)audio_ctx);
144 }
145 }
146 }
147 }
148 break;
149 }
150
151 case KEY_RIGHT: { // Seek forward 30 seconds
152 if (capture) {
153 media_source_t *source = (media_source_t *)session_capture_get_media_source(capture);
154 if (source && media_source_get_type(source) == MEDIA_SOURCE_FILE) {
155 double current_pos = media_source_get_position(source);
156 double duration = media_source_get_duration(source);
157 if (current_pos >= 0.0) {
158 double new_pos = current_pos + 30.0;
159 // Clamp to duration if known
160 if (duration > 0.0 && new_pos > duration) {
161 new_pos = duration;
162 }
163 asciichat_error_t err = media_source_seek(source, new_pos);
164 if (err == ASCIICHAT_OK) {
165 log_info("Seeked forward to %.1f seconds", new_pos);
166 }
167 void *audio_ctx = session_capture_get_audio_context(capture);
168 if (audio_ctx) {
169 audio_flush_playback_buffers((audio_context_t *)audio_ctx);
170 }
171 }
172 }
173 }
174 break;
175 }
176
177 // ===== VOLUME CONTROLS =====
178 case KEY_DOWN: { // Decrease volume 10%
179 double current_volume = GET_OPTION(speakers_volume);
180 double new_volume = clamp_volume(current_volume - 0.1);
181 options_set_double("speakers_volume", new_volume);
182 double verify_volume = GET_OPTION(speakers_volume);
183 log_info("Volume DOWN: %.0f%% → %.0f%% (verified: %.0f%%)", current_volume * 100.0, new_volume * 100.0,
184 verify_volume * 100.0);
185 break;
186 }
187
188 case KEY_UP: { // Increase volume 10%
189 double current_volume = GET_OPTION(speakers_volume);
190 double new_volume = clamp_volume(current_volume + 0.1);
191 options_set_double("speakers_volume", new_volume);
192 double verify_volume = GET_OPTION(speakers_volume);
193 log_info("Volume UP: %.0f%% → %.0f%% (verified: %.0f%%)", current_volume * 100.0, new_volume * 100.0,
194 verify_volume * 100.0);
195 break;
196 }
197
198 // ===== PLAY/PAUSE CONTROL =====
199 case KEY_SPACE: { // Toggle play/pause
200 if (capture) {
201 media_source_t *source = (media_source_t *)session_capture_get_media_source(capture);
202 if (source && media_source_get_type(source) == MEDIA_SOURCE_FILE) {
203 media_source_toggle_pause(source);
204 if (media_source_is_paused(source)) {
205 log_info("Paused");
206 } else {
207 log_info("Playing");
208 }
209 }
210 }
211 break;
212 }
213
214 // ===== COLOR MODE CONTROL =====
215 case KEY_C:
216 case 'C': {
217 int current_mode = (int)GET_OPTION(color_mode);
218 int next_mode = next_color_mode(current_mode);
219 options_set_int("color_mode", next_mode);
220
221 const char *mode_names[] = {"Mono", "16-color", "256-color", "Truecolor"};
222 if (next_mode >= 0 && next_mode <= 3) {
223 log_info("Color mode: %s", mode_names[next_mode]);
224 }
225 break;
226 }
227
228 // ===== MUTE CONTROL =====
229 case KEY_M:
230 case 'M': {
231 double current_volume = GET_OPTION(speakers_volume);
232 log_debug("Mute toggle: current_volume=%.2f, g_mute_saved_volume=%.2f, threshold=0.01", current_volume,
233 g_mute_saved_volume);
234
235 if (current_volume > 0.01) { // If not already muted
236 // Save current volume and mute
237 g_mute_saved_volume = current_volume;
238 options_set_double("speakers_volume", 0.0);
239 double verify = GET_OPTION(speakers_volume);
240 log_info("Muted: saved %.0f%%, set to 0%% (verified: %.2f)", g_mute_saved_volume * 100.0, verify);
241 } else {
242 // Restore previous volume
243 double restore_volume = g_mute_saved_volume > 0.0 ? g_mute_saved_volume : 0.5;
244 options_set_double("speakers_volume", restore_volume);
245 double verify = GET_OPTION(speakers_volume);
246 log_info("Unmuted: restored %.0f%% (verified: %.2f)", restore_volume * 100.0, verify);
247 }
248 break;
249 }
250
251 // ===== RENDER MODE CONTROL =====
252 case KEY_R:
253 case 'R': {
254 int current_mode = (int)GET_OPTION(render_mode);
255 int next_mode = next_render_mode(current_mode);
256 options_set_int("render_mode", next_mode);
257
258 const char *mode_names[] = {"Foreground", "Background", "Half-block"};
259 if (next_mode >= 0 && next_mode <= 2) {
260 log_info("Render mode: %s", mode_names[next_mode]);
261 }
262 break;
263 }
264
265 // ===== COLOR FILTER CONTROL =====
266 case KEY_F:
267 case 'F': {
268 int current_filter = (int)GET_OPTION(color_filter);
269 int next_filter = next_color_filter(current_filter);
270 options_set_int("color_filter", next_filter);
271
272 const char *filter_names[] = {"None", "Black", "White", "Green", "Magenta", "Fuchsia", "Orange",
273 "Teal", "Cyan", "Pink", "Red", "Yellow", "Rainbow"};
274 if (next_filter >= 0 && next_filter < (int)COLOR_FILTER_COUNT) {
275 log_info("Color filter: %s", filter_names[next_filter]);
276 }
277 break;
278 }
279
280 // ===== HORIZONTAL FLIP CONTROL =====
281 case 'X':
282 case 'x': {
283 log_info("✓ X KEY PRESSED - handling horizontal flip");
284 bool current_flip_x = (bool)GET_OPTION(flip_x);
285 options_set_bool("flip_x", !current_flip_x);
286 log_info("Horizontal flip: %s (was: %s)", !current_flip_x ? "enabled" : "disabled",
287 current_flip_x ? "enabled" : "disabled");
288 break;
289 }
290
291 // ===== VERTICAL FLIP CONTROL =====
292 case 'Y':
293 case 'y': {
294 bool current_flip_y = (bool)GET_OPTION(flip_y);
295 options_set_bool("flip_y", !current_flip_y);
296 log_info("Vertical flip: %s", !current_flip_y ? "enabled" : "disabled");
297 break;
298 }
299
300 // ===== MATRIX RAIN EFFECT CONTROL =====
301 case KEY_0: {
302 bool current_matrix = (bool)GET_OPTION(matrix_rain);
303 options_set_bool("matrix_rain", !current_matrix);
304 log_info("Matrix rain effect: %s", !current_matrix ? "enabled" : "disabled");
305 break;
306 }
307
308 // ===== FPS COUNTER TOGGLE =====
309 case '-': {
310 bool current = (bool)GET_OPTION(fps_counter);
311 options_set_bool("fps_counter", !current);
312 log_info("FPS counter: %s", !current ? "enabled" : "disabled");
313 break;
314 }
315
316 // ===== LOCK DEBUG (debug builds only) =====
317#ifndef NDEBUG
318 case KEY_BACKTICK: {
319 debug_sync_trigger_print();
320 log_debug("Lock state dump triggered via backtick key");
321 break;
322 }
323#endif
324
325 default:
326 // Unknown key - silently ignore
327 break;
328 }
