59bool validate_shell_safe(const char *str, const char *allowed_chars) {
60 if (!str) {
61 return false;
62 }
63
64 // Shell metacharacters that are dangerous
65 // ; & | $ ` \ " ' < > ( ) [ ] { } * ? ! ~ # @ space tab newline
66 const char *dangerous = ";|$`\\\"'<>()[]{}*?!~#@ \t\n\r";
67
68 // First pass: validate UTF-8 encoding
69 if (!utf8_is_valid(str)) {
70 return false;
71 }
72
73 // Second pass: validate each character for shell safety
74 // Convert to codepoints and check only ASCII range for dangerous characters
75 // (non-ASCII characters are allowed by default)
76 const uint8_t *p = (const uint8_t *)str;
77 while (*p) {
78 uint32_t codepoint;
79 int decode_len = utf8_decode(p, &codepoint);
80 if (decode_len < 0) {
81 return false; // Should not happen after utf8_is_valid check
82 }
83
84 // Only validate ASCII range where shell metacharacters exist
85 // Non-ASCII characters (codepoint > 127) are allowed
86 if (codepoint <= 127) {
87 unsigned char c = (unsigned char)codepoint;
88
89 // Allow ASCII alphanumeric characters
90 if (isalnum(c)) {
91 p += decode_len;
92 continue;
93 }
94
95 // Check if character is in allowed_chars list
96 if (allowed_chars) {
97 bool is_allowed = false;
98 for (size_t j = 0; allowed_chars[j] != '\0'; j++) {
99 if (c == (unsigned char)allowed_chars[j]) {
100 is_allowed = true;
101 break;
102 }
103 }
104 if (is_allowed) {
105 p += decode_len;
106 continue;
107 }
108 }
109
110 // Check if character is dangerous
111 bool is_dangerous = false;
112 for (size_t j = 0; dangerous[j] != '\0'; j++) {
113 if (c == (unsigned char)dangerous[j]) {
114 is_dangerous = true;
115 break;
116 }
117 }
118 if (is_dangerous) {
119 return false;
120 }
121 }
122
123 p += decode_len;
124 }
125
126 return true;
127}

130 if (!str || !out_buffer || out_buffer_size == 0) {

271const char *colored_string(log_color_t color, const char *text) {
272#define COLORED_BUFFERS 4
273#define COLORED_BUFFER_SIZE 256
274 static char buffers[COLORED_BUFFERS][COLORED_BUFFER_SIZE];
275 static int buffer_idx = 0;
276
277 if (!text) {
278 return "";
279 }
280
281 // Check if we should use colors
282 // Use global flag that persists even after options cleanup
283 bool use_colors = false;
284
285 // Priority 1: If --color was explicitly passed, force colors
286 if (g_color_flag_passed && g_color_flag_value) {
287 use_colors = true;
288 }
289 // Priority 2: If --color NOT explicitly passed, use terminal detection
290 else if (!g_color_flag_passed) {
291 use_colors = terminal_should_color_output(STDOUT_FILENO);
292 }
293 // Priority 3: If --color=false was explicitly passed, disable colors
294 // (use_colors stays false)
295
296 if (!use_colors) {
297 // No colors, just return the text directly
298 return text;
299 }
300
301 // Use rotating buffer to handle multiple calls in same fprintf
302 char *current_buf = buffers[buffer_idx];
303 buffer_idx = (buffer_idx + 1) % COLORED_BUFFERS;
304
305 const char *color_code = log_level_color(color);
306 const char *reset_code = log_level_color(LOG_COLOR_RESET);
307
308 // Ensure we never pass NULL to snprintf (handle uninitialized color system)
309 if (!color_code)
310 color_code = "";
311 if (!reset_code)
312 reset_code = "";
313
314 // Format into rotating static buffer: color_code + text + reset_code
315 safe_snprintf(current_buf, COLORED_BUFFER_SIZE, "%s%s%s", color_code, text, reset_code);
316 return current_buf;
317}

319void truncate_utf8_with_ellipsis(const char *input, char *output, size_t output_size, int max_width) {
320 if (!input || !output || output_size < 2) {
321 if (output && output_size > 0)
322 output[0] = '\0';
323 return;
324 }
325
326 // Copy input to output first
327 int input_len = strlen(input);
328 if (input_len >= (int)output_size) {
329 input_len = output_size - 1;
330 }
331 memcpy(output, input, input_len);
332 output[input_len] = '\0';
333
334 // Get display width of current string
335 int current_width = utf8_display_width(output);
336
337 // If it fits, we're done
338 if (current_width <= max_width) {
339 return;
340 }
341
342 // Need to truncate - reserve space for "..." (3 chars, 3 width)
343 int available_width = max_width - 3;
344 if (available_width <= 0) {
345 // No room for ellipsis, just show "..."
346 output[0] = '.';
347 output[1] = '.';
348 output[2] = '.';
349 output[3] = '\0';
350 return;
351 }
352
353 // Truncate byte by byte until we fit in available_width
354 for (int i = input_len - 1; i > 0; i--) {
355 output[i] = '\0';
356 current_width = utf8_display_width(output);
357 if (current_width <= available_width) {
358 // Found the right length - add ellipsis
359 strcat(output, "...");
360 return;
361 }
362 }
363
364 // Fallback: just "..."
365 output[0] = '.';
366 output[1] = '.';
367 output[2] = '.';
368 output[3] = '\0';
369}
