119static bool discovery_participant_render_should_exit(void *user_data) {
120 discovery_session_t *discovery = (discovery_session_t *)user_data;
121
122 // Check global exit flag first
123 if (should_exit()) {
124 return true;
125 }
126
127 // Process discovery session events (keep NAT negotiation responsive)
128 // 50ms timeout for responsiveness
129 asciichat_error_t result = discovery_session_process(discovery, 50 * NS_PER_MS_INT);
130 if (result != ASCIICHAT_OK && result != ERROR_NETWORK_TIMEOUT) {
131 log_error("Discovery session process failed: %d", result);
132 signal_exit();
133 return true;
134 }
135
136 // Exit if session is no longer active
137 if (!discovery_session_is_active(discovery)) {
138 // Still negotiating - continue loop
139 return false;
140 }
141
142 return false;
143}

189static asciichat_error_t discovery_run(session_capture_ctx_t *capture, session_display_ctx_t *display,
190 void *user_data) {
191 (void)user_data; // Unused
192
193 asciichat_error_t result = ASCIICHAT_OK;
194
195 // Wait for session to become active (host negotiation complete)
196 // This processes ACDS events until we have a determined role
197 while (!should_exit()) {
198 result = discovery_session_process(g_discovery, 50 * NS_PER_MS_INT);
199 if (result != ASCIICHAT_OK && result != ERROR_NETWORK_TIMEOUT) {
200 log_error("Discovery session process failed: %d", result);
201 return result;
202 }
203
204 if (discovery_session_is_active(g_discovery)) {
205 break; // Session is active, role is determined
206 }
207 }
208
209 if (should_exit()) {
210 log_info("Shutdown requested during discovery negotiation");
211 return ASCIICHAT_OK;
212 }
213
214 // Session is active - run based on our role
215 bool we_are_host = discovery_session_is_host(g_discovery);
216
217 if (we_are_host) {
218 // HOST ROLE: Capture own media, manage participants, mix and broadcast
219 log_info("Hosting session - capturing and broadcasting");
220
221 // Create session host for managing participants
222 session_host_config_t host_config = {
223 .port = (int)GET_OPTION(port),
224 .ipv4_address = NULL, // Bind to any
225 .ipv6_address = NULL,
226 .max_clients = 32,
227 .encryption_enabled = false, // TODO: Add crypto support
228 .key_path = NULL,
229 .password = NULL,
230 .callbacks = {0},
231 .user_data = NULL,
232 };
233
234 session_host_t *host = session_host_create(&host_config);
235 if (!host) {
236 log_fatal("Failed to create session host");
237 return ERROR_MEMORY;
238 }
239
240 // Start host (accept connections, render threads)
241 result = session_host_start(host);
242 if (result != ASCIICHAT_OK) {
243 log_error("Failed to start session host: %d", result);
244 session_host_destroy(host);
245 return result;
246 }
247
248 // Add memory participant for host's own media
249 uint32_t host_participant_id = session_host_add_memory_participant(host);
250 if (host_participant_id == 0) {
251 log_error("Failed to add memory participant for host");
252 session_host_stop(host);
253 session_host_destroy(host);
254 return ERROR_INVALID_STATE;
255 }
256
257 log_info("Host participating with ID %u", host_participant_id);
258
259 // Main loop: capture own media and keep discovery responsive
260 while (!should_exit() && discovery_session_is_active(g_discovery)) {
261 // Capture frame from webcam (reuse existing capture context)
262 image_t *frame = session_capture_read_frame(capture);
263 if (frame) {
264 // Inject host's frame into mixer
265 session_host_inject_frame(host, host_participant_id, frame);
266 }
267
268 // Keep discovery session responsive (NAT negotiations, migrations)
269 result = discovery_session_process(g_discovery, 10 * NS_PER_MS_INT);
270 if (result != ASCIICHAT_OK && result != ERROR_NETWORK_TIMEOUT) {
271 log_error("Discovery session process failed: %d", result);
272 break;
273 }
274
275 // Frame rate limiting (60 FPS)
276 session_capture_sleep_for_fps(capture);
277 }
278
279 // Cleanup
280 session_host_stop(host);
281 session_host_destroy(host);
282
283 if (should_exit()) {
284 return ASCIICHAT_OK;
285 }
286
287 // If we reach here, discovery session is no longer active (role change)
288 // Return error to trigger potential reconnection retry
289 if (!discovery_session_is_active(g_discovery)) {
290 log_info("Session ended or role changed");
291 return ASCIICHAT_OK;
292 }
293 } else {
294 // PARTICIPANT ROLE: Capture local media and display host's frames
295 log_info("Participant in session - displaying host's frames");
296
297 // Use the unified render loop that handles capture, ASCII conversion, and display
298 result = session_render_loop(capture, display, discovery_participant_render_should_exit,
299 NULL, // No custom capture callback
300 NULL, // No custom sleep callback
301 discovery_keyboard_handler, // Keyboard handler for interactive controls
302 g_discovery); // Pass discovery session as user_data
303
304 if (result != ASCIICHAT_OK) {
305 log_error("Render loop failed with error code: %d", result);
306 }
307 }
308
309 return result;
310}

323int discovery_main(void) {

325
326 // Discovery-specific setup: initialize ACDS connection and session negotiation
327 // Note: Shared setup (keepawake, splash, terminal, capture, display, audio)
328 // is handled by session_client_like_run()
329
330 // Get session string from options (command-line argument)
331 const char *session_string = GET_OPTION(session_string);
332 bool is_initiator = (session_string == NULL || session_string[0] == '\0');
333
334 int port_int = GET_OPTION(port);
335
336 log_debug("Discovery: is_initiator=%d, port=%d", is_initiator, port_int);
337
338 // Create discovery session configuration
339 discovery_config_t discovery_config = {
340 .acds_address = GET_OPTION(discovery_server),
341 .acds_port = (uint16_t)GET_OPTION(discovery_port),
342 .session_string = is_initiator ? NULL : session_string,
343 .local_port = (uint16_t)port_int,
344 .on_state_change = on_discovery_state_change,
345 .on_session_ready = on_session_ready,
346 .on_error = on_discovery_error,
347 .callback_user_data = NULL,
348 .should_exit_callback = discovery_capture_should_exit_adapter,
349 .exit_callback_data = NULL,
350 };
351
352 // Create discovery session (stored in global for discovery_run() to access)
353 g_discovery = discovery_session_create(&discovery_config);
354 if (!g_discovery) {
355 log_fatal("Failed to create discovery session");
356 return ERROR_MEMORY;
357 }
358
359 // Start discovery session (connects to ACDS, creates/joins, initiates NAT negotiation)
360 log_debug("Discovery: starting discovery session");
361 asciichat_error_t result = discovery_session_start(g_discovery);
362 if (result != ASCIICHAT_OK) {
363 log_fatal("Failed to start discovery session: %d", result);
364 discovery_session_destroy(g_discovery);
365 g_discovery = NULL;
366 return result;

407 asciichat_error_t session_result = session_client_like_run(&config);
408 log_debug("Discovery: session_client_like_run() returned %d", session_result);
409
410 // Cleanup discovery session
411 log_debug("Discovery: cleaning up");
412
413 if (g_discovery) {
414 discovery_session_stop(g_discovery);
415 discovery_session_destroy(g_discovery);
416 g_discovery = NULL;
417 }
418
419 return (session_result == ASCIICHAT_OK) ? 0 : (int)session_result;
