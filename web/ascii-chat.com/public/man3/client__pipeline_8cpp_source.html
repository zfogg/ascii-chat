125 // User reported clipping with +6dB makeup gain

156client_audio_pipeline_t *client_audio_pipeline_create(const client_audio_pipeline_config_t *config) {
157 client_audio_pipeline_t *p = SAFE_CALLOC(1, sizeof(client_audio_pipeline_t), client_audio_pipeline_t *);
158 if (!p) {
159 log_error("Failed to allocate client audio pipeline");
160 return NULL;
161 }
162
163 // Use default config if none provided
164 if (config) {
165 p->config = *config;
166 } else {
167 p->config = client_audio_pipeline_default_config();
168 }
169
170 p->flags = p->config.flags;
171 p->frame_size = p->config.sample_rate * p->config.frame_size_ns / 1000;
172
173 // No mutex needed - full-duplex means single callback thread handles all AEC3
174
175 // Initialize Opus encoder/decoder first (no exceptions)
176 int opus_error = 0;
177 p->encoder = opus_encoder_create(p->config.sample_rate, 1, OPUS_APPLICATION_VOIP, &opus_error);
178 if (!p->encoder || opus_error != OPUS_OK) {
179 log_error("Failed to create Opus encoder: %d", opus_error);
180 goto error;
181 }
182 opus_encoder_ctl(p->encoder, OPUS_SET_BITRATE(p->config.opus_bitrate));
183
184 // Disable DTX (Discontinuous Transmission) to prevent "beeps".
185 // DTX stops sending frames during silence, causing audible clicks/beeps when audio resumes.
186 // opus_encoder_ctl(p->encoder, OPUS_SET_DTX(0));
187
188 // Create Opus decoder
189 p->decoder = opus_decoder_create(p->config.sample_rate, 1, &opus_error);
190 if (!p->decoder || opus_error != OPUS_OK) {
191 log_error("Failed to create Opus decoder: %d", opus_error);
192 goto error;
193 }
194
195 // Create WebRTC AEC3 Echo Cancellation
196 // AEC3 provides production-grade acoustic echo cancellation with:
197 // - Automatic network delay estimation (0-500ms)
198 // - Adaptive filtering to actual echo path
199 // - Residual echo suppression via spectral subtraction
200 // - Jitter buffer handling via side information
201 if (p->flags.echo_cancel) {
202 // Configure AEC3 for better low-frequency (bass) echo cancellation
203 webrtc::EchoCanceller3Config aec3_config;
204
205 // Increase filter length for bass frequencies (default 13 blocks = ~17ms)
206 // Bass at 80Hz has 12.5ms period, so we need at least 50+ blocks (~67ms)
207 // to properly model the echo path for low frequencies
208 aec3_config.filter.main.length_blocks = 50; // ~67ms (was 13)
209 aec3_config.filter.shadow.length_blocks = 50; // ~67ms (was 13)
210 aec3_config.filter.main_initial.length_blocks = 25; // ~33ms (was 12)
211 aec3_config.filter.shadow_initial.length_blocks = 25; // ~33ms (was 12)
212
213 // More aggressive low-frequency suppression thresholds
214 // Lower values = more aggressive echo suppression
215 aec3_config.echo_audibility.audibility_threshold_lf = 5; // (was 10)
216
217 // Create AEC3 using the factory
218 auto factory = webrtc::EchoCanceller3Factory(aec3_config);
219
220 std::unique_ptr<webrtc::EchoControl> echo_control = factory.Create(static_cast<int>(p->config.sample_rate), // 48kHz
221 1, // num_render_channels (speaker output)
222 1 // num_capture_channels (microphone input)
223 );
224
225 if (!echo_control) {
226 log_warn("Failed to create WebRTC AEC3 instance - echo cancellation unavailable");
227 p->echo_canceller = NULL;
228 } else {
229 // Successfully created AEC3 - wrap in our C++ wrapper for C compatibility
230 auto wrapper = new WebRTCAec3Wrapper();
231 wrapper->aec3 = std::move(echo_control);
232 wrapper->config = aec3_config;
233 p->echo_canceller = wrapper;
234
235 log_info("✓ WebRTC AEC3 initialized (67ms filter for bass, adaptive delay)");
236
237 // Create persistent AudioBuffer instances for AEC3
238 p->aec3_render_buffer = new webrtc::AudioBuffer(48000, 1, 48000, 1, 48000, 1);
239 p->aec3_capture_buffer = new webrtc::AudioBuffer(48000, 1, 48000, 1, 48000, 1);
240
241 auto *render_buf = static_cast<webrtc::AudioBuffer *>(p->aec3_render_buffer);
242 auto *capture_buf = static_cast<webrtc::AudioBuffer *>(p->aec3_capture_buffer);
243
244 // Zero-initialize channel data
245 float *const *render_ch = render_buf->channels();
246 float *const *capture_ch = capture_buf->channels();
247 if (render_ch && render_ch[0]) {
248 memset(render_ch[0], 0, 480 * sizeof(float)); // 10ms at 48kHz
249 }
250 if (capture_ch && capture_ch[0]) {
251 memset(capture_ch[0], 0, 480 * sizeof(float));
252 }
253
254 // Prime filterbank state with dummy processing cycle
255 render_buf->SplitIntoFrequencyBands();
256 render_buf->MergeFrequencyBands();
257 capture_buf->SplitIntoFrequencyBands();
258 capture_buf->MergeFrequencyBands();
259
260 log_info(" - AudioBuffer filterbank state initialized");
261
262 // Warm up AEC3 with 10 silent frames to initialize internal state
263 for (int warmup = 0; warmup < 10; warmup++) {
264 memset(render_ch[0], 0, 480 * sizeof(float));
265 memset(capture_ch[0], 0, 480 * sizeof(float));
266
267 render_buf->SplitIntoFrequencyBands();
268 wrapper->aec3->AnalyzeRender(render_buf);
269 render_buf->MergeFrequencyBands();
270
271 wrapper->aec3->AnalyzeCapture(capture_buf);
272 capture_buf->SplitIntoFrequencyBands();
273 wrapper->aec3->SetAudioBufferDelay(0);
274 wrapper->aec3->ProcessCapture(capture_buf, false);
275 capture_buf->MergeFrequencyBands();
276 }
277 log_info(" - AEC3 warmed up with 10 silent frames");
278 log_info(" - Persistent AudioBuffer instances created");
279 }
280 }
281
282 // Initialize debug WAV writers for AEC3 analysis (if echo_cancel enabled)
283 p->debug_wav_aec3_in = NULL;
284 p->debug_wav_aec3_out = NULL;
285 if (p->flags.echo_cancel) {
286 // Open WAV files to capture AEC3 input and output
287 p->debug_wav_aec3_in = wav_writer_open("/tmp/aec3_input.wav", 48000, 1);
288 p->debug_wav_aec3_out = wav_writer_open("/tmp/aec3_output.wav", 48000, 1);
289 if (p->debug_wav_aec3_in) {
290 log_info("Debug: Recording AEC3 input to /tmp/aec3_input.wav");
291 }
292 if (p->debug_wav_aec3_out) {
293 log_info("Debug: Recording AEC3 output to /tmp/aec3_output.wav");
294 }
295
296 log_info("✓ AEC3 echo cancellation enabled (full-duplex mode, no ring buffer delay)");
297 }
298
299 // Initialize audio processing components (compressor, noise gate, filters)
300 // These are applied in the capture path after AEC3 and before Opus encoding
301 {
302 float sample_rate = (float)p->config.sample_rate;
303
304 // Initialize compressor with config values
305 compressor_init(&p->compressor, sample_rate);
306 compressor_set_params(&p->compressor, p->config.comp_threshold_db, p->config.comp_ratio, p->config.comp_attack_ns,
307 p->config.comp_release_ns, p->config.comp_makeup_db);
308 log_info("✓ Capture compressor: threshold=%.1fdB, ratio=%.1f:1, makeup=+%.1fdB", p->config.comp_threshold_db,
309 p->config.comp_ratio, p->config.comp_makeup_db);
310
311 // Initialize noise gate with config values
312 noise_gate_init(&p->noise_gate, sample_rate);
313 noise_gate_set_params(&p->noise_gate, p->config.gate_threshold, p->config.gate_attack_ns, p->config.gate_release_ns,
314 p->config.gate_hysteresis);
315 log_info("✓ Capture noise gate: threshold=%.4f (%.1fdB)", p->config.gate_threshold,
316 20.0f * log10f(p->config.gate_threshold + 1e-10f));
317
318 // Initialize PLAYBACK noise gate - cuts quiet received audio before speakers
319 // Very low threshold - only cut actual silence, not quiet voice audio
320 // The server sends audio with RMS=0.01-0.02, so threshold must be below that
321 noise_gate_init(&p->playback_noise_gate, sample_rate);
322 noise_gate_set_params(&p->playback_noise_gate,
323 0.002f, // -54dB threshold - only cut near-silence
324 1.0f, // 1ms attack - fast open
325 50.0f, // 50ms release - smooth close
326 0.4f); // Hysteresis
327 log_info("✓ Playback noise gate: threshold=0.002 (-54dB)");
328
329 // Initialize highpass filter (removes low-frequency rumble)
330 highpass_filter_init(&p->highpass, p->config.highpass_hz, sample_rate);
331 log_info("✓ Capture highpass filter: %.1f Hz", p->config.highpass_hz);
332
333 // Initialize lowpass filter (removes high-frequency hiss)
334 lowpass_filter_init(&p->lowpass, p->config.lowpass_hz, sample_rate);
335 log_info("✓ Capture lowpass filter: %.1f Hz", p->config.lowpass_hz);
336 }
337
338 p->initialized = true;
339
340 // Initialize startup fade-in to prevent initial microphone click
341 // 200ms at 48kHz = 9600 samples - gradual ramp from silence to full volume
342 // Longer fade-in (200ms vs 50ms) gives much smoother transition without audible pop
343 p->capture_fadein_remaining = (p->config.sample_rate * 200) / 1000; // 200ms worth of samples
344 log_info("✓ Capture fade-in: %d samples (200ms)", p->capture_fadein_remaining);
345
346 log_info("Audio pipeline created: %dHz, %dms frames, %dkbps Opus", p->config.sample_rate, p->config.frame_size_ns,
347 p->config.opus_bitrate / 1000);
348
349 (void)NAMED_REGISTER_CLIENT_AUDIO_PIPELINE(p, "audio_pipeline", NULL);
350
351 return p;
352
353error:
354 if (p->encoder)
355 opus_encoder_destroy(p->encoder);
356 if (p->decoder)
357 opus_decoder_destroy(p->decoder);
358 if (p->echo_canceller) {
359 delete static_cast<WebRTCAec3Wrapper *>(p->echo_canceller);
360 }
361 SAFE_FREE(p);
362 return NULL;
363}

365void client_audio_pipeline_destroy(client_audio_pipeline_t *pipeline) {
366 if (!pipeline)
367 return;
368
369 // Clean up WebRTC AEC3 AudioBuffer instances
370 if (pipeline->aec3_render_buffer) {
371 delete static_cast<webrtc::AudioBuffer *>(pipeline->aec3_render_buffer);
372 pipeline->aec3_render_buffer = NULL;
373 }
374 if (pipeline->aec3_capture_buffer) {
375 delete static_cast<webrtc::AudioBuffer *>(pipeline->aec3_capture_buffer);
376 pipeline->aec3_capture_buffer = NULL;
377 }
378
379 // Clean up WebRTC AEC3
380 if (pipeline->echo_canceller) {
381 delete static_cast<WebRTCAec3Wrapper *>(pipeline->echo_canceller);
382 pipeline->echo_canceller = NULL;
383 }
384
385 // Clean up Opus
386 if (pipeline->encoder) {
387 opus_encoder_destroy(pipeline->encoder);
388 pipeline->encoder = NULL;
389 }
390 if (pipeline->decoder) {
391 opus_decoder_destroy(pipeline->decoder);
392 pipeline->decoder = NULL;
393 }
394
395 // Clean up debug WAV writers
396 if (pipeline->debug_wav_aec3_in) {
397 wav_writer_close((wav_writer_t *)pipeline->debug_wav_aec3_in);
398 pipeline->debug_wav_aec3_in = NULL;
399 }
400 if (pipeline->debug_wav_aec3_out) {
401 wav_writer_close((wav_writer_t *)pipeline->debug_wav_aec3_out);
402 pipeline->debug_wav_aec3_out = NULL;
403 }
404
405 NAMED_UNREGISTER(pipeline);
406
407 SAFE_FREE(pipeline);
408}

459int client_audio_pipeline_playback(client_audio_pipeline_t *pipeline, const uint8_t *opus_in, int opus_len,
460 float *output, int num_samples) {
461 if (!pipeline || !opus_in || !output) {
462 return -1;
463 }
464
465 // No mutex needed - Opus decoder is only used from this thread
466
467 // Decode Opus
468 int decoded_samples = opus_decode_float(pipeline->decoder, opus_in, opus_len, output, num_samples, 0);
469
470 if (decoded_samples < 0) {
471 log_error("Opus decoding failed: %d", decoded_samples);
472 return -1;
473 }
474
475 // Apply playback noise gate - cut quiet background audio before it reaches speakers
476 if (decoded_samples > 0) {
477 noise_gate_process_buffer(&pipeline->playback_noise_gate, output, decoded_samples);
478 }
479
480 // NOTE: Render signal is queued to AEC3 in output_callback() when audio plays,
481 // not here. The capture thread drains the queue and processes AEC3.
482
483 return decoded_samples;
484}

516void client_audio_pipeline_process_duplex(client_audio_pipeline_t *pipeline, const float *render_samples,
517 int render_count, const float *capture_samples, int capture_count,
518 float *processed_output) {
519 if (!pipeline || !processed_output)
520 return;
521
522 // Copy capture samples to output buffer for processing
523 if (capture_samples && capture_count > 0) {
524 memcpy(processed_output, capture_samples, capture_count * sizeof(float));
525 } else {
526 memset(processed_output, 0, capture_count * sizeof(float));
527 return;
528 }
529
530 // Check for AEC3 bypass
531 static int bypass_aec3 = -1;
532 if (bypass_aec3 == -1) {
533 const char *env = platform_getenv("BYPASS_AEC3");
534 bypass_aec3 = (env && (strcmp(env, "1") == 0 || strcmp(env, "true") == 0)) ? 1 : 0;
535 if (bypass_aec3) {
536 log_warn("AEC3 BYPASSED (full-duplex mode) via BYPASS_AEC3=1");
537 }
538 }
539
540 // Debug WAV recording
541 if (pipeline->debug_wav_aec3_in) {
542 wav_writer_write((wav_writer_t *)pipeline->debug_wav_aec3_in, capture_samples, capture_count);
543 }
544
545 // Apply startup fade-in using smoothstep curve
546 if (pipeline->capture_fadein_remaining > 0) {
547 const int total_fadein_samples = (pipeline->config.sample_rate * 200) / 1000;
548 for (int i = 0; i < capture_count && pipeline->capture_fadein_remaining > 0; i++) {
549 float progress = 1.0f - ((float)pipeline->capture_fadein_remaining / (float)total_fadein_samples);
550 float gain = smoothstep(progress);
551 processed_output[i] *= gain;
552 pipeline->capture_fadein_remaining--;
553 }
554 }
555
556 // WebRTC AEC3 processing - INLINE, no ring buffer, no mutex
557 if (!bypass_aec3 && pipeline->flags.echo_cancel && pipeline->echo_canceller) {
558 auto wrapper = static_cast<WebRTCAec3Wrapper *>(pipeline->echo_canceller);
559 if (wrapper && wrapper->aec3) {
560 const int webrtc_frame_size = 480; // 10ms at 48kHz
561
562 auto *render_buf = static_cast<webrtc::AudioBuffer *>(pipeline->aec3_render_buffer);
563 auto *capture_buf = static_cast<webrtc::AudioBuffer *>(pipeline->aec3_capture_buffer);
564
565 if (render_buf && capture_buf) {
566 float *const *render_channels = render_buf->channels();
567 float *const *capture_channels = capture_buf->channels();
568
569 if (render_channels && render_channels[0] && capture_channels && capture_channels[0]) {
570 // Verify render_samples is valid before accessing
571 if (!render_samples && render_count > 0) {
572 log_warn_every(1000000, "AEC3: render_samples is NULL but render_count=%d", render_count);
573 return;
574 }
575
576 // Process in 10ms chunks (AEC3 requirement)
577 int render_offset = 0;
578 int capture_offset = 0;
579
580 while (capture_offset < capture_count || render_offset < render_count) {
581 // STEP 1: Feed render signal (what's playing to speakers)
582 // In full-duplex, this is THE EXACT audio being played RIGHT NOW
583 if (render_samples && render_offset < render_count) {
584 int render_chunk = (render_offset + webrtc_frame_size <= render_count) ? webrtc_frame_size
585 : (render_count - render_offset);
586 if (render_chunk == webrtc_frame_size) {
587 // Scale float [-1,1] to WebRTC int16-range [-32768, 32767]
588 copy_buffer_with_gain(&render_samples[render_offset], render_channels[0], webrtc_frame_size, 32768.0f);
589 render_buf->SplitIntoFrequencyBands();
590 wrapper->aec3->AnalyzeRender(render_buf);
591 render_buf->MergeFrequencyBands();
592 g_render_frames_fed.fetch_add(1, std::memory_order_relaxed);
593 }
594 render_offset += render_chunk;
595 }
596
597 // STEP 2: Process capture (microphone input)
598 if (capture_offset < capture_count) {
599 int capture_chunk = (capture_offset + webrtc_frame_size <= capture_count)
600 ? webrtc_frame_size
601 : (capture_count - capture_offset);
602 if (capture_chunk == webrtc_frame_size) {
603 // Scale float [-1,1] to WebRTC int16-range [-32768, 32767]
604 copy_buffer_with_gain(&processed_output[capture_offset], capture_channels[0], webrtc_frame_size,
605 32768.0f);
606
607 // AEC3 sequence: AnalyzeCapture, split, ProcessCapture, merge
608 wrapper->aec3->AnalyzeCapture(capture_buf);
609 capture_buf->SplitIntoFrequencyBands();
610
611 // NOTE: SetAudioBufferDelay() is just an initial hint when use_external_delay_estimator=false
612 // (default). AEC3's internal delay estimator will find the actual delay (~144ms in practice). We don't
613 // call it here - let AEC3 estimate delay automatically.
614
615 wrapper->aec3->ProcessCapture(capture_buf, false);
616 capture_buf->MergeFrequencyBands();
617
618 // Scale back to float range and apply soft clip to prevent distortion
619 // Use gentle soft_clip (threshold=0.6, steepness=2.5) to leave headroom for compressor
620 for (int j = 0; j < webrtc_frame_size; j++) {
621 float sample = capture_channels[0][j] / 32768.0f;
622 processed_output[capture_offset + j] = soft_clip(sample, 0.6f, 2.5f);
623 }
624
625 // Log AEC3 metrics periodically
626 static int duplex_log_count = 0;
627 if (++duplex_log_count % 100 == 1) {
628 webrtc::EchoControl::Metrics metrics = wrapper->aec3->GetMetrics();
629 log_info("AEC3 DUPLEX: ERL=%.1f ERLE=%.1f delay=%dms", metrics.echo_return_loss,
630 metrics.echo_return_loss_enhancement, metrics.delay_ms);
631 audio_analysis_set_aec3_metrics(metrics.echo_return_loss, metrics.echo_return_loss_enhancement,
632 metrics.delay_ms);
633 }
634 }
635 capture_offset += capture_chunk;
636 }
637 }
638 }
639 }
640 }
641 }
642
643 // Debug WAV recording (after AEC3)
644 if (pipeline->debug_wav_aec3_out) {
645 wav_writer_write((wav_writer_t *)pipeline->debug_wav_aec3_out, processed_output, capture_count);
646 }
647
648 // Apply manual AGC (simple pre-gain to boost quiet microphones)
649 // TODO: Replace with proper WebRTC AGC module for adaptive gain control
650 static int agc_call_count = 0;
651 agc_call_count++;
652 if (agc_call_count <= 3 || agc_call_count % 100 == 0) {
653 log_info("AGC check #%d: flags.agc=%d, agc_max_gain=%.1f", agc_call_count, pipeline->flags.agc,
654 pipeline->config.agc_max_gain);
655 }
656
657 if (pipeline->flags.agc) {
658 // Convert dB to linear gain: linear = 10^(dB/20)
659 const float agc_pregain = powf(10.0f, pipeline->config.agc_max_gain / 20.0f);
660 for (int i = 0; i < capture_count; i++) {
661 processed_output[i] *= agc_pregain;
662 }
663 if (agc_call_count <= 3 || agc_call_count % 100 == 0) {
664 log_info("AGC: Applied %.1f dB pre-gain (%.2fx multiplier)", pipeline->config.agc_max_gain, agc_pregain);
665 }
666 }
667
668 // Apply capture processing chain: filters, noise gate, compressor
669 if (pipeline->flags.highpass) {
670 highpass_filter_process_buffer(&pipeline->highpass, processed_output, capture_count);
671 }
672 if (pipeline->flags.lowpass) {
673 lowpass_filter_process_buffer(&pipeline->lowpass, processed_output, capture_count);
674 }
675 if (pipeline->flags.noise_gate) {
676 noise_gate_process_buffer(&pipeline->noise_gate, processed_output, capture_count);
677 }
678 if (pipeline->flags.compressor) {
679 for (int i = 0; i < capture_count; i++) {
680 float gain = compressor_process_sample(&pipeline->compressor, processed_output[i]);
681 processed_output[i] *= gain;
682 }
683 // Apply soft clipping after compressor - threshold=0.7 gives 3dB headroom
684 soft_clip_buffer(processed_output, capture_count, 0.7f, 3.0f);
685 }
686}
