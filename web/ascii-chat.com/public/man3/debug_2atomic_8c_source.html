51int debug_atomic_format_timing(const atomic_t *atomic, char *buffer, size_t size) {
52 if (!atomic || !buffer || size == 0) return 0;
53
54 // If never accessed, return empty (so caller can skip output)
55 if (atomic->last_load_time_ns == 0 && atomic->last_store_time_ns == 0) {
56 return 0;
57 }
58
59 int offset = 0;
60 uint64_t now_ns = time_get_ns();
61
62 char load_str[64] = "";
63 char store_str[64] = "";
64 char count_str[128] = "";
65
66 if (atomic->last_load_time_ns > 0 && atomic->last_load_time_ns <= now_ns) {
67 char elapsed_str[64];
68 time_pretty(now_ns - atomic->last_load_time_ns, -1, elapsed_str, sizeof(elapsed_str));
69 snprintf(load_str, sizeof(load_str), "load=%s", elapsed_str);
70 }
71
72 if (atomic->last_store_time_ns > 0 && atomic->last_store_time_ns <= now_ns) {
73 char elapsed_str[64];
74 time_pretty(now_ns - atomic->last_store_time_ns, -1, elapsed_str, sizeof(elapsed_str));
75 snprintf(store_str, sizeof(store_str), "store=%s", elapsed_str);
76 }
77
78 if (atomic->load_count > 0 || atomic->store_count > 0 || atomic->cas_count > 0 || atomic->fetch_count > 0) {
79 snprintf(count_str, sizeof(count_str), "[ops: load=%lu store=%lu cas=%lu/%lu fetch=%lu]",
80 atomic->load_count, atomic->store_count, atomic->cas_success_count, atomic->cas_count, atomic->fetch_count);
81 }
82
83 offset += snprintf(buffer + offset, size - offset, "%s %s %s", load_str, store_str, count_str);
84 return offset;
85}
