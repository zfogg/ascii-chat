74asciichat_error_t named_init(void) {
75 // Idempotent: if already initialized, return success
76 if (lifecycle_is_initialized(&g_named_registry.lifecycle)) {
77 return ASCIICHAT_OK;
78 }
79
80 if (!lifecycle_init(&g_named_registry.lifecycle, "named_registry")) {
81 return ASCIICHAT_OK;

83
84 if (rwlock_init(&g_named_registry.entries_lock, "named_registry_lock") != 0) {
85 lifecycle_shutdown(&g_named_registry.lifecycle);
86 return ASCIICHAT_OK; // Continue even if rwlock init fails
87 }
88
89 lifecycle_init_commit(&g_named_registry.lifecycle);
90 return ASCIICHAT_OK;
91}

93void named_destroy(void) {
94 // Always attempt to shut down the lifecycle, but always cleanup entries
95 lifecycle_shutdown(&g_named_registry.lifecycle);
96
97 rwlock_wrlock(&g_named_registry.entries_lock);
98
99 // Find the registry's own rwlock entry (allocated during named_init bootstrap)
100 // This entry will be unregistered by rwlock_destroy, so we must not free it here
101 uintptr_t registry_lock_key = (uintptr_t)(const void *)&g_named_registry.entries_lock;
102 named_entry_t *registry_lock_entry = NULL;
103 HASH_FIND(hh, g_named_registry.entries, &registry_lock_key, sizeof(uintptr_t), registry_lock_entry);
104
105 for (named_entry_t *e = g_named_registry.entries; e != NULL;) {
106 named_entry_t *next = e->hh.next;
107
108 // Skip the registry lock entry - it will be cleaned up by rwlock_destroy
109 // which calls NAMED_UNREGISTER after we release the lock
110 if (e == registry_lock_entry) {
111 e = next;
112 continue;
113 }
114
115 // Free all other entries
116 free(e->name);
117 if (e->type) free(e->type);
118 if (e->format_spec) free(e->format_spec);
119 if (e->file) free(e->file);
120 if (e->func) free(e->func);
121 free(e);
122 e = next;
123 }
124
125 rwlock_wrunlock(&g_named_registry.entries_lock);
126
127 // rwlock_destroy will call NAMED_UNREGISTER on the lock entry, which will:
128 // 1. Acquire the lock (now safe since we released it)
129 // 2. Find and remove the lock entry from the hash
130 // 3. Free the lock entry's fields
131 rwlock_destroy(&g_named_registry.entries_lock);
132}

168const char *named_register(uintptr_t key, const char *base_name, const char *type, const char *format_spec,
169 const char *file, int line, const char *func, uintptr_t parent_key) {
170 if (!base_name || !type || !format_spec) {
171 return "?";
172 }
173
174 if (!lifecycle_is_initialized(&g_named_registry.lifecycle)) {
175 return base_name;
176 }
177
178 // If parent_key provided, look up parent name and create hierarchical name
179 char final_base_name[256];
180 if (parent_key != 0) {
181 rwlock_rdlock(&g_named_registry.entries_lock);
182 const char *parent_name = named_lookup_name_unlocked(parent_key);
183 rwlock_rdunlock(&g_named_registry.entries_lock);
184
185 if (parent_name) {
186 snprintf(final_base_name, sizeof(final_base_name), "%s#%s", parent_name, base_name);
187 } else {
188 // Parent not found, fall back to base name
189 snprintf(final_base_name, sizeof(final_base_name), "%s", base_name);
190 }
191 } else {
192 snprintf(final_base_name, sizeof(final_base_name), "%s", base_name);
193 }
194
195 // Generate unique name using atomic counters (only for non-hierarchical names)
196 char name_buffer[256];
197 if (parent_key == 0) {
198 uint64_t counter = 0;
199 if (strcmp(type, "mutex") == 0) {
200 counter = __sync_fetch_and_add(&mutex_counter, 1);
201 } else if (strcmp(type, "rwlock") == 0) {
202 counter = __sync_fetch_and_add(&rwlock_counter, 1);
203 } else if (strcmp(type, "cond") == 0) {
204 counter = __sync_fetch_and_add(&cond_counter, 1);
205 } else if (strcmp(type, "atomic") == 0) {
206 counter = __sync_fetch_and_add(&atomic_counter, 1);
207 }
208 snprintf(name_buffer, sizeof(name_buffer), "%s.%"PRIu64, final_base_name, counter);
209 } else {
210 // Hierarchical names don't get counters - parent already has one
211 snprintf(name_buffer, sizeof(name_buffer), "%s", final_base_name);
212 }
213
214 // Allocate entry BEFORE acquiring lock (avoid long critical section)
215 named_entry_t *entry = malloc(sizeof(named_entry_t));
216 if (!entry) return base_name;
217
218 entry->key = key;
219 entry->name = entry_strdup(name_buffer);
220 named_entry_init(entry, type, format_spec, file, line, func);
221
222 // Only hold lock for the hash table operation
223 rwlock_wrlock(&g_named_registry.entries_lock);
224 HASH_ADD(hh, g_named_registry.entries, key, sizeof(uintptr_t), entry);
225 rwlock_wrunlock(&g_named_registry.entries_lock);
226
227 return entry->name;
228}

230const char *named_register_fmt(uintptr_t key, const char *type, const char *format_spec, const char *file, int line,
231 const char *func, const char *fmt, ...) {
232 if (!type || !format_spec || !fmt) {
233 return "?";
234 }

236 if (!lifecycle_is_initialized(&g_named_registry.lifecycle)) {
237 return fmt;

240 // Format the name BEFORE acquiring lock
241 va_list args;
242 va_start(args, fmt);
243 char *full_name = NULL;
244 int ret = vasprintf(&full_name, fmt, args);
245 va_end(args);
246

249 }
250

259 entry->key = key;

336const char *named_describe(uintptr_t key, const char *type_hint) {
337 if (!type_hint) type_hint = "object";
338
339 static _Thread_local char buffer[DESCRIBE_BUFFER_SIZE];
340
341 if (!lifecycle_is_initialized(&g_named_registry.lifecycle)) {
342 snprintf(buffer, sizeof(buffer), "%s (0x%tx)", type_hint, (ptrdiff_t)key);
343 return buffer;
344 }
345
346 rwlock_rdlock(&g_named_registry.entries_lock);
347 named_entry_t *entry = NULL;
348 HASH_FIND(hh, g_named_registry.entries, &key, sizeof(uintptr_t), entry);
349 rwlock_rdunlock(&g_named_registry.entries_lock);
350
351 if (entry) {
352 const char *type = entry->type ? entry->type : type_hint;
353 const char *name = entry->name ? entry->name : "unknown";
354 if (entry->file && entry->func && entry->line > 0) {
355 snprintf(buffer, sizeof(buffer), "%s/%s (0x%tx) @ %s:%d:%s()", type, name, (ptrdiff_t)key,
356 entry->file, entry->line, entry->func);
357 } else {
358 snprintf(buffer, sizeof(buffer), "%s/%s (0x%tx)", type, name, (ptrdiff_t)key);
359 }
360 } else {
361 snprintf(buffer, sizeof(buffer), "%s (0x%tx)", type_hint ? type_hint : "unknown", (ptrdiff_t)key);
362 }
363
364 return buffer;
365}

377void named_registry_for_each(named_iter_callback_t callback, void *user_data) {
378 if (!callback || !lifecycle_is_initialized(&g_named_registry.lifecycle)) {
379 return;
380 }
381
382 // Snapshot entries while holding read lock
383 named_iter_entry_t entries[256];
384 int count = 0;
385
386 rwlock_rdlock(&g_named_registry.entries_lock);
387 for (named_entry_t *e = g_named_registry.entries; e != NULL && count < 256; e = e->hh.next) {
388 entries[count].key = e->key;
389 strncpy(entries[count].name, e->name ? e->name : "?", MAX_NAME_LEN - 1);
390 entries[count].name[MAX_NAME_LEN - 1] = '\0';
391 count++;
392 }
393 rwlock_rdunlock(&g_named_registry.entries_lock);
394
395 // Call callback outside the lock
396 for (int i = 0; i < count; i++) {
397 callback(entries[i].key, entries[i].name, user_data);
398 }
399}
