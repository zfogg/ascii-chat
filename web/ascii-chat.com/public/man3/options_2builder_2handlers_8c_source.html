63static bool is_set_bool(const void *field, const option_descriptor_t *desc) {
64 // Safely read the bool value (may be uninitialized)
65 unsigned char value_byte = 0;
66 memcpy(&value_byte, field, 1);
67 bool value = (value_byte != 0);
68
69 bool default_val = false;
70 if (desc && desc->default_value) {
71 unsigned char default_byte = 0;
72 memcpy(&default_byte, desc->default_value, 1);
73 default_val = (default_byte != 0);
74 }
75 return value != default_val;
76}

120static void apply_env_bool(void *field, const char *env_value, const option_descriptor_t *desc) {
121 // Safely read the current bool value (may be uninitialized)
122 unsigned char current_byte = 0;
123 memcpy(&current_byte, field, 1);
124 bool current_value = (current_byte != 0);
125
126 // Safely read the default value
127 bool default_val = false;
128 if (desc && desc->default_value) {
129 unsigned char default_byte = 0;
130 memcpy(&default_byte, desc->default_value, 1);
131 default_val = (default_byte != 0);
132 }
133
134 if (current_value != default_val) {
135 return; // Already set, skip env var
136 }
137
138 bool value = false;
139 if (env_value) {
140 value = (strcmp(env_value, "1") == 0 || strcmp(env_value, "true") == 0 || strcmp(env_value, "yes") == 0 ||
141 strcmp(env_value, "on") == 0);
142 } else if (desc && desc->default_value) {
143 unsigned char default_byte = 0;
144 memcpy(&default_byte, desc->default_value, 1);
145 value = (default_byte != 0);
146 }
147
148 unsigned char value_byte = value ? 1 : 0;
149 memcpy(field, &value_byte, 1);
150}

152static void apply_env_int(void *field, const char *env_value, const option_descriptor_t *desc) {
153 if (env_value) {
154 char *endptr;
155 long parsed = strtol(env_value, &endptr, 10);
156 if (*endptr == '\0' && parsed >= INT_MIN && parsed <= INT_MAX) {
157 int value = (int)parsed;
158 memcpy(field, &value, sizeof(int));
159 }
160 return;
161 }
162 // No env var — apply descriptor default only if field hasn't been changed from descriptor default
163 int current_value = 0;
164 memcpy(&current_value, field, sizeof(int));
165 int default_val = desc->default_value ? *(const int *)desc->default_value : 0;
166 if (current_value != default_val)
167 return;
168 if (desc->default_value) {
169 int value = *(const int *)desc->default_value;
170 memcpy(field, &value, sizeof(int));
171 }
172}

242// --- apply_cli handlers ---
243// Helper function to parse boolean value from string
244// Accepts: "true", "1", "yes", "on" → true
245// "false", "0", "no", "off" → false
246// Returns ERROR_USAGE if value is invalid
247static asciichat_error_t parse_bool_value(const char *value_str, bool *out_value, const option_descriptor_t *desc) {
248 if (!value_str || value_str[0] == '\0') {
249 return SET_ERRNO(ERROR_USAGE, "Option --%s requires a value (true/false, yes/no, 1/0, on/off)",
250 desc ? desc->long_name : "unknown");
251 }
252
253 // Check for true values
254 if (strcasecmp(value_str, "true") == 0 || strcasecmp(value_str, "yes") == 0 || strcasecmp(value_str, "1") == 0 ||
255 strcasecmp(value_str, "on") == 0) {
256 *out_value = true;
257 return ASCIICHAT_OK;
258 }
259
260 // Check for false values
261 if (strcasecmp(value_str, "false") == 0 || strcasecmp(value_str, "no") == 0 || strcasecmp(value_str, "0") == 0 ||
262 strcasecmp(value_str, "off") == 0) {
263 *out_value = false;
264 return ASCIICHAT_OK;
265 }
266
267 // Invalid value
268 return SET_ERRNO(ERROR_USAGE, "Invalid boolean value for --%s: '%s' (use: true/false, yes/no, 1/0, on/off)",
269 desc ? desc->long_name : "unknown", value_str);
270}

272static asciichat_error_t apply_cli_bool(void *field, const char *opt_value, const option_descriptor_t *desc) {
273 bool new_value;
274
275 if (opt_value == NULL) {
276 // No value provided (flag without =value), toggle the current value
277 unsigned char current_byte = 0;
278 memcpy(&current_byte, field, 1);
279 bool current_value = (current_byte != 0);
280 new_value = !current_value;
281 } else {
282 // Value provided (--option=value), parse it
283 asciichat_error_t parse_result = parse_bool_value(opt_value, &new_value, desc);
284 if (parse_result != ASCIICHAT_OK) {
285 return parse_result;
286 }
287 }
288
289 // Set the new value
290 unsigned char new_value_byte = new_value ? 1 : 0;
291 memcpy(field, &new_value_byte, 1);
292
293 return ASCIICHAT_OK;
294}

296static asciichat_error_t apply_cli_int(void *field, const char *opt_value, const option_descriptor_t *desc) {
297 // Reject empty strings
298 if (!opt_value || opt_value[0] == '\0') {
299 return SET_ERRNO(ERROR_USAGE, "Option --%s requires a numeric value", desc ? desc->long_name : "unknown");
300 }
301
302 char *endptr;
303 long value = strtol(opt_value, &endptr, 10);
304 if (*endptr != '\0' || value < INT_MIN || value > INT_MAX) {
305 return SET_ERRNO(ERROR_USAGE, "Option --%s: '%s' is not a valid integer", desc ? desc->long_name : "unknown",
306 opt_value);
307 }
308 int int_value = (int)value;
309
310 // Check numeric range constraints if defined in descriptor's metadata
311 if (desc && desc->metadata.numeric_range.max != 0) {
312 if (int_value < desc->metadata.numeric_range.min || int_value > desc->metadata.numeric_range.max) {
313 return SET_ERRNO(ERROR_USAGE, "Value %d out of range [%d-%d]", int_value, desc->metadata.numeric_range.min,
314 desc->metadata.numeric_range.max);
315 }
316 }
317
318 memcpy(field, &int_value, sizeof(int));
319
320 return ASCIICHAT_OK;
321}

338static asciichat_error_t apply_cli_double(void *field, const char *opt_value, const option_descriptor_t *desc) {
339 (void)desc;
340 // Handle optional arguments - if no value provided, keep default
341 if (!opt_value || !opt_value[0]) {
342 return ASCIICHAT_OK;
343 }
344
345 char *endptr;
346 double value = strtod(opt_value, &endptr);
347 // Check: endptr must have advanced past the input, and must be at the string end
348 if (endptr == opt_value || *endptr != '\0') {
349 return ERROR_USAGE;
350 }
351 memcpy(field, &value, sizeof(double));

353 return ASCIICHAT_OK;
