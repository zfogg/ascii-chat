133static bool symbolizer_path_is_executable(const char *path) {
134#ifdef _WIN32
135 if (!path || path[0] == '\0') {
136 return false;
137 }
138
139 DWORD attrs = GetFileAttributesA(path);
140 if (attrs == INVALID_FILE_ATTRIBUTES) {
141 return false;
142 }
143 if (attrs & FILE_ATTRIBUTE_DIRECTORY) {
144 return false;
145 }
146
147 HANDLE handle = CreateFileA(path, GENERIC_READ, FILE_SHARE_READ | FILE_SHARE_WRITE, NULL, OPEN_EXISTING,
148 FILE_ATTRIBUTE_NORMAL, NULL);
149 if (handle == INVALID_HANDLE_VALUE) {
150 return false;
151 }
152
153 CloseHandle(handle);
154 return true;
155#else
156 return (path && path[0] != '\0' && access(path, X_OK) == 0);
157#endif
158}

160static const char *get_llvm_symbolizer_command(void) {
161 // Check if already initialized
162 if (lifecycle_is_initialized(&g_llvm_symbolizer_lc)) {
163 goto check_available;
164 }
165
166 // Try to win the initialization race
167 if (!lifecycle_init(&g_llvm_symbolizer_lc, "llvm_symbolizer")) {
168 // Lost race - wait for winner to complete
169 while (!lifecycle_is_initialized(&g_llvm_symbolizer_lc)) {
170 // Spin-wait
171 }
172 goto check_available;
173 }
174
175 // Winner: do the initialization
176 {
177 const char *env_path = SAFE_GETENV("LLVM_SYMBOLIZER_PATH");
178 bool available = false;
179
180 g_llvm_symbolizer_cmd[0] = '\0';
181
182 if (env_path && env_path[0] != '\0') {
183 if (symbolizer_path_is_executable(env_path)) {
184 SAFE_STRNCPY(g_llvm_symbolizer_cmd, env_path, sizeof(g_llvm_symbolizer_cmd));
185 available = true;
186 log_dev("Using llvm-symbolizer from LLVM_SYMBOLIZER_PATH: %s", env_path);
187 } else {
188 log_warn("LLVM_SYMBOLIZER_PATH is set but not executable: %s", env_path);
189 }
190 }
191
192 if (!available && platform_is_binary_in_path(LLVM_SYMBOLIZER_BIN)) {
193 available = true;
194 g_llvm_symbolizer_cmd[0] = '\0'; // Use binary name from PATH
195 log_dev("Found %s in PATH", LLVM_SYMBOLIZER_BIN);
196 }
197
198 atomic_store_bool(&g_llvm_symbolizer_available, available);
199 }
200
201check_available:
202
203 if (!atomic_load_bool(&g_llvm_symbolizer_available)) {
204 return NULL;
205 }
206
207 if (g_llvm_symbolizer_cmd[0] != '\0') {
208 return g_llvm_symbolizer_cmd;
209 }
210
211 return LLVM_SYMBOLIZER_BIN;
212}

268static symbolizer_type_t detect_symbolizer(void) {
269 // Prefer llvm-symbolizer on all platforms (including macOS)
270 // We handle ASLR ourselves using dyld APIs on macOS
271 const char *llvm_symbolizer = get_llvm_symbolizer_command();
272 if (llvm_symbolizer) {
273 log_dev("Using llvm-symbolizer for symbol resolution");
274 return SYMBOLIZER_LLVM;
275 }
276
277 const char *addr2line_cmd = get_addr2line_command();
278 if (addr2line_cmd) {
279 log_debug("Using addr2line command: %s", addr2line_cmd);
280 return SYMBOLIZER_ADDR2LINE;
281 }
282
283 log_warn("No symbolizer found in PATH (tried %s, %s) - using native backend", LLVM_SYMBOLIZER_BIN, ADDR2LINE_BIN);
284 return SYMBOLIZER_NONE;
285}

291asciichat_error_t symbol_cache_init(void) {
292 if (!lifecycle_init(&g_symbol_cache_lc, "symbol_cache")) {
293 return 0; // Already initialized
294 }
295
296 // Detect which symbolizer is available (once at init)
297 bool expected = false;
298 if (atomic_cas_bool(&g_symbolizer_detected, &expected, true)) {
299 g_symbolizer_type = detect_symbolizer();
300 }
301
302 // Initialize rwlock for thread safety (uthash requires external locking)
303 if (rwlock_init(&g_symbol_cache_lock, "symbol_cache") != 0) {
304 lifecycle_init_abort(&g_symbol_cache_lc);
305 return SET_ERRNO(ERROR_THREAD, "Failed to initialize symbol cache rwlock");
306 }
307
308 // Initialize uthash head to NULL (required)
309 g_symbol_cache = NULL;
310
311 atomic_store_u64(&g_cache_hits, 0);
312 atomic_store_u64(&g_cache_misses, 0);
313
314 log_dev("Symbol cache initialized");
315 return 0;
316}

318void symbol_cache_destroy(void) {
319 if (!lifecycle_shutdown(&g_symbol_cache_lc)) {
320 return;
321 }
322
323 // Mark as uninitialized FIRST to prevent new inserts during cleanup
324
325 // Acquire write lock to prevent any concurrent operations
326 rwlock_wrlock(&g_symbol_cache_lock);
327
328 // Count entries before freeing for debugging
329 size_t entry_count = HASH_COUNT(g_symbol_cache);
330
331 // Free all symbol entries using HASH_ITER
332 symbol_entry_t *entry = NULL, *tmp = NULL;
333 size_t freed_count = 0;
334 HASH_ITER(hh, g_symbol_cache, entry, tmp) {
335 if (entry) {
336 HASH_DEL(g_symbol_cache, entry);
337 if (entry->symbol) {
338 free(entry->symbol);
339 }
340 free(entry);
341 freed_count++;
342 }
343 }
344
345 // Release lock and destroy rwlock
346 rwlock_wrunlock(&g_symbol_cache_lock);
347 rwlock_destroy(&g_symbol_cache_lock);
348
349 g_symbol_cache = NULL;
350
351 log_dev("Symbol cache cleaned up: %zu entries counted, %zu entries freed (hits=%llu, misses=%llu)", entry_count,
352 freed_count, (unsigned long long)atomic_load_u64(&g_cache_hits),
353 (unsigned long long)atomic_load_u64(&g_cache_misses));
354}

356const char *symbol_cache_lookup(void *addr) {
357 if (!lifecycle_is_initialized(&g_symbol_cache_lc) || !addr) {
358 return NULL;
359 }
360
361 rwlock_rdlock(&g_symbol_cache_lock);
362
363 symbol_entry_t *entry = NULL;
364 HASH_FIND_PTR(g_symbol_cache, &addr, entry);
365
366 const char *result = NULL;
367 if (entry) {
368 // Copy string while holding lock to prevent use-after-free
369 result = platform_strdup(entry->symbol);
370 atomic_fetch_add_u64(&g_cache_hits, 1);
371 } else {
372 atomic_fetch_add_u64(&g_cache_misses, 1);
373 }
374
375 rwlock_rdunlock(&g_symbol_cache_lock);
376 return result;
377}

379bool symbol_cache_insert(void *addr, const char *symbol) {
380 if (!lifecycle_is_initialized(&g_symbol_cache_lc) || !addr || !symbol) {
381 return false;
382 }
383
384 // Acquire write lock to make the entire operation atomic
385 rwlock_wrlock(&g_symbol_cache_lock);
386
387 // Double-check cache is still initialized after acquiring lock
388 // (cleanup might have marked it uninitialized between our check and lock acquisition)
389 if (!lifecycle_is_initialized(&g_symbol_cache_lc)) {
390 rwlock_wrunlock(&g_symbol_cache_lock);
391 return false;
392 }
393
394 // Check if entry already exists
395 symbol_entry_t *existing = NULL;
396 HASH_FIND_PTR(g_symbol_cache, &addr, existing);
397
398 if (existing) {
399 // Entry exists - update symbol if different
400 if (existing->symbol && strcmp(existing->symbol, symbol) != 0) {
401 // Free old symbol and allocate new one
402 free(existing->symbol);
403 existing->symbol = platform_strdup(symbol);
404 if (!existing->symbol) {
405 rwlock_wrunlock(&g_symbol_cache_lock);
406 return false;
407 }
408 }
409 rwlock_wrunlock(&g_symbol_cache_lock);
410 return true;
411 }
412
413 // Create new entry
414 symbol_entry_t *entry = (symbol_entry_t *)malloc(sizeof(symbol_entry_t));
415 if (!entry) {
416 rwlock_wrunlock(&g_symbol_cache_lock);
417 return false;
418 }
419
420 entry->addr = addr;
421 entry->symbol = platform_strdup(symbol);
422 if (!entry->symbol) {
423 free(entry);
424 rwlock_wrunlock(&g_symbol_cache_lock);
425 return false;
426 }
427
428 // Add to hash table
429 HASH_ADD_PTR(g_symbol_cache, addr, entry);
430
431 // Release lock
432 rwlock_wrunlock(&g_symbol_cache_lock);
433
434 return true;
435}

476static char *parse_llvm_symbolizer_result(FILE *fp, void *addr) {
477 char func_name[512] = "??";
478 char file_location[512] = "??:0";
479 char blank_line[8];
480
481 // Read function name line
482 if (fgets(func_name, sizeof(func_name), fp) == NULL) {
483 return NULL;
484 }
485 func_name[strcspn(func_name, "\n")] = '\0';
486
487 // Read file location line
488 if (fgets(file_location, sizeof(file_location), fp) == NULL) {
489 return NULL;
490 }
491 file_location[strcspn(file_location, "\n")] = '\0';
492
493 // Read blank separator line (and discard it)
494 if (fgets(blank_line, sizeof(blank_line), fp) == NULL) {
495 // End of output - not an error
496 }
497
498 // Remove column number (last :N) from file_location
499 char *last_colon = strrchr(file_location, ':');
500 if (last_colon) {
501 *last_colon = '\0';
502 }
503
504 // Extract relative path and COPY IT IMMEDIATELY
505 // IMPORTANT: extract_project_relative_path uses a static buffer internally.
506 // SAFE_MALLOC's debug memory tracking also calls extract_project_relative_path,
507 // which would overwrite the static buffer. So we must copy before allocating.
508 const char *rel_path_tmp = extract_project_relative_path(file_location);
509 char rel_path[512];
510 SAFE_STRNCPY(rel_path, rel_path_tmp, sizeof(rel_path));
511
512 // Allocate result buffer (AFTER copying rel_path to local storage)
513 char *result = SAFE_MALLOC(1024, char *);
514 if (!result) {
515 return NULL;
516 }
517
518 // Format symbol
519 bool has_func = (strcmp(func_name, "??") != 0 && strlen(func_name) > 0);
520 bool has_file =
521 (strcmp(file_location, "??:0") != 0 && strcmp(file_location, "??:?") != 0 && strcmp(file_location, "??") != 0);
522
523 // Remove () from function name if present (llvm-symbolizer includes them)
524 char clean_func[512];
525 SAFE_STRNCPY(clean_func, func_name, sizeof(clean_func));
526 char *paren = strstr(clean_func, "()");
527 if (paren) {
528 *paren = '\0';
529 }
530
531 if (!has_func && !has_file) {
532 SAFE_SNPRINTF(result, 1024, "%p", addr);
533 } else if (has_func && has_file) {
534 SAFE_SNPRINTF(result, 1024, "%s() (%s)", clean_func, rel_path);
535 } else if (has_func) {
536 SAFE_SNPRINTF(result, 1024, "%s()", clean_func);
537 } else {
538 SAFE_SNPRINTF(result, 1024, "%s (unknown function)", rel_path);
539 }
540
541 return result;
542}

553static char **run_llvm_symbolizer_batch(void *const *buffer, int size) {
554 if (size <= 0 || !buffer) {
555 return NULL;
556 }
557
558 const char *symbolizer_cmd = get_llvm_symbolizer_command();
559 if (!symbolizer_cmd) {
560 log_debug("llvm-symbolizer not available - skipping symbolization");
561 return NULL;
562 }
563
564 // Allocate result array
565 char **result = SAFE_CALLOC((size_t)(size + 1), sizeof(char *), char **);
566 if (!result) {
567 return NULL;
568 }
569
570 // ============================================================================
571 // Unified platform-agnostic implementation using get_binary_file_address_offsets()
572 // ============================================================================
573
574 // Group addresses by binary using platform-independent interface
575 // Works identically on Linux, macOS, and Windows
576 typedef struct {
577 char binary_path[PLATFORM_MAX_PATH_LENGTH];
578 uintptr_t file_offsets[64]; // Max addresses per binary
579 int original_indices[64];
580 int count;
581 } binary_group_t;
582
583 binary_group_t groups[8]; // Support up to 8 different binaries
584 int num_groups = 0;
585
586 // First pass: Call get_binary_file_address_offsets() for each address
587 // (same function works on all platforms - zero #ifdefs needed!)
588 for (int i = 0; i < size; i++) {
589 platform_binary_match_t match;
590 if (get_binary_file_address_offsets(buffer[i], &match, 1) == 0) {
591 // Could not find binary - use raw address as fallback
592 result[i] = SAFE_MALLOC(32, char *);
593 if (result[i]) {
594 SAFE_SNPRINTF(result[i], 32, "%p", buffer[i]);
595 }
596 continue;
597 }
598
599 // Find or create group for this binary
600 int group_idx = -1;
601 for (int g = 0; g < num_groups; g++) {
602 if (strcmp(groups[g].binary_path, match.path) == 0) {
603 group_idx = g;
604 break;
605 }
606 }
607
608 if (group_idx < 0 && num_groups < 8) {
609 group_idx = num_groups++;
610 SAFE_STRNCPY(groups[group_idx].binary_path, match.path, sizeof(groups[group_idx].binary_path));
611 groups[group_idx].count = 0;
612 }
613
614 if (group_idx >= 0 && groups[group_idx].count < 64) {
615 int idx = groups[group_idx].count++;
616 groups[group_idx].file_offsets[idx] = match.file_offset;
617 groups[group_idx].original_indices[idx] = i;
618 }
619 }
620
621 // Second pass: batch symbolize each group
622 for (int g = 0; g < num_groups; g++) {
623 if (groups[g].count == 0) {
624 continue;
625 }
626
627 // Escape binary path for shell
628 char escaped_binary_path[PLATFORM_MAX_PATH_LENGTH * 2];
629 if (!escape_path_for_shell(groups[g].binary_path, escaped_binary_path, sizeof(escaped_binary_path))) {
630 // On error, fill with fallback addresses
631 for (int j = 0; j < groups[g].count; j++) {
632 int orig_idx = groups[g].original_indices[j];
633 if (!result[orig_idx]) {
634 result[orig_idx] = SAFE_MALLOC(32, char *);
635 if (result[orig_idx]) {
636 SAFE_SNPRINTF(result[orig_idx], 32, "%p", buffer[orig_idx]);
637 }
638 }
639 }
640 continue;
641 }
642
643 // Build command with all addresses for this binary
644 char cmd[8192];
645 int offset = safe_snprintf(cmd, sizeof(cmd), "%s --demangle --output-style=LLVM --relativenames -e %s ",
646 symbolizer_cmd, escaped_binary_path);
647
648 for (int j = 0; j < groups[g].count && offset < (int)sizeof(cmd) - 32; j++) {
649 int n =
650 safe_snprintf(cmd + offset, sizeof(cmd) - (size_t)offset, "0x%lx ", (unsigned long)groups[g].file_offsets[j]);
651 if (n > 0) {
652 offset += n;
653 }
654 }
655
656 // Suppress stderr
657 strncat(cmd, "2>/dev/null", sizeof(cmd) - strlen(cmd) - 1);
658
659 // Note: No LOG_IO here because output is read and consumed by this function
660 FILE *fp = NULL;
661 if (platform_popen("llvm-symbolizer", cmd, "r", &fp) != ASCIICHAT_OK || !fp) {
662 // popen() failed - fill with fallback addresses instead of leaving NULL
663 for (int j = 0; j < groups[g].count; j++) {
664 int orig_idx = groups[g].original_indices[j];
665 if (!result[orig_idx]) {
666 result[orig_idx] = SAFE_MALLOC(32, char *);
667 if (result[orig_idx]) {
668 SAFE_SNPRINTF(result[orig_idx], 32, "%p", buffer[orig_idx]);
669 }
670 }
671 }
672 continue;
673 }
674
675 // Extract binary name (basename) for display
676 const char *binary_name = strrchr(groups[g].binary_path, '/');
677 if (!binary_name) {
678 binary_name = strrchr(groups[g].binary_path, '\\');
679 }
680 binary_name = binary_name ? binary_name + 1 : groups[g].binary_path;
681
682 // Parse results in order
683 for (int j = 0; j < groups[g].count; j++) {
684 int orig_idx = groups[g].original_indices[j];
685 char *parsed = parse_llvm_symbolizer_result(fp, buffer[orig_idx]);
686
687 if (parsed) {
688 // Prepend binary name in brackets
689 char *with_binary = SAFE_MALLOC(1024, char *);
690 if (with_binary) {
691 SAFE_SNPRINTF(with_binary, 1024, "[%s] %s", binary_name, parsed);
692 SAFE_FREE(parsed);
693 result[orig_idx] = with_binary;
694 } else {
695 result[orig_idx] = parsed;
696 }
697 } else {
698 // Fallback: show raw address with binary name
699 result[orig_idx] = SAFE_MALLOC(256, char *);
700 if (result[orig_idx]) {
701 SAFE_SNPRINTF(result[orig_idx], 256, "[%s] %p", binary_name, buffer[orig_idx]);
702 }
703 }
704 }
705
706 platform_pclose(&fp);
707 }
708
709 return result;
710}

718static char **run_addr2line_batch(void *const *buffer, int size) {
719 if (size <= 0 || !buffer) {
720 log_error("Invalid parameters: buffer=%p, size=%d", (void *)buffer, size);
721 return NULL;
722 }
723
724 const char *addr2line_cmd = get_addr2line_command();
725 if (!addr2line_cmd) {
726 log_debug("addr2line not available - skipping symbolization");
727 return NULL;
728 }
729
730 // Get executable path
731 char exe_path[PLATFORM_MAX_PATH_LENGTH];
732 if (!platform_get_executable_path(exe_path, sizeof(exe_path))) {
733 return NULL;
734 }
735
736 // SECURITY: Validate executable path to prevent command injection
737 // Paths from system APIs should be safe, but validate to be thorough
738 if (!validate_shell_safe(exe_path, ".-/\\:")) {
739 log_error("Invalid executable path - contains unsafe characters: %s", exe_path);
740 return NULL;
741 }
742
743 // Escape exe_path for shell (auto-detects platform and quoting needs)
744 char escaped_exe_path_buf[PLATFORM_MAX_PATH_LENGTH * 2];
745 if (!escape_path_for_shell(exe_path, escaped_exe_path_buf, sizeof(escaped_exe_path_buf))) {
746 log_error("Failed to escape executable path for shell command");
747 return NULL;
748 }
749 const char *escaped_exe_path = escaped_exe_path_buf;
750
751 if (!validate_shell_safe(addr2line_cmd, ".-/\\:_")) {
752 log_warn("addr2line path contains unsafe characters: %s", addr2line_cmd);
753 return NULL;
754 }
755
756 // Escape addr2line command for shell (auto-detects platform and quoting needs)
757 char escaped_addr2line_buf[PLATFORM_MAX_PATH_LENGTH * 2];
758 if (!escape_path_for_shell(addr2line_cmd, escaped_addr2line_buf, sizeof(escaped_addr2line_buf))) {
759 log_error("Failed to escape addr2line path for shell command");
760 return NULL;
761 }
762 const char *escaped_addr2line_cmd = escaped_addr2line_buf;
763
764 // Build addr2line command
765 char cmd[4096];
766 int offset = safe_snprintf(cmd, sizeof(cmd), "%s -e %s -f -C -i ", escaped_addr2line_cmd, escaped_exe_path);
767 if (offset <= 0 || offset >= (int)sizeof(cmd)) {
768 log_error("Failed to build addr2line command");
769 return NULL;
770 }
771
772 // Use explicit hex format with 0x prefix since Windows %p doesn't include it
773 for (int i = 0; i < size; i++) {
774 int n = safe_snprintf(cmd + offset, sizeof(cmd) - (size_t)offset, "0x%llx ", (unsigned long long)buffer[i]);
775 if (n <= 0 || offset + n >= (int)sizeof(cmd)) {
776 log_error("Failed to build addr2line command");
777 break;
778 }
779 offset += n;
780 }
781
782 // Allocate result array BEFORE executing addr2line (to avoid leak on popen failure)
783 char **result = SAFE_CALLOC((size_t)(size + 1), sizeof(char *), char **);
784 if (!result) {
785 return NULL;
786 }
787
788 // Execute addr2line
789 // Note: No LOG_IO here because output is read and consumed by this function
790 FILE *fp = NULL;
791 if (platform_popen("addr2line", cmd, "r", &fp) != ASCIICHAT_OK || !fp) {
792 log_error("Failed to execute addr2line command");
793 // Fill result array with fallback values instead of leaving NULL entries
794 for (int i = 0; i < size; i++) {
795 result[i] = SAFE_MALLOC(32, char *);
796 if (result[i]) {
797 SAFE_SNPRINTF(result[i], 32, "%p", buffer[i]);
798 }
799 }
800 result[size] = NULL;
801 return result;
802 }
803
804 // Parse output
805 for (int i = 0; i < size; i++) {
806 char func_name[256];
807 char file_line[512];
808
809 if (fgets(func_name, sizeof(func_name), fp) == NULL) {
810 break;
811 }
812 if (fgets(file_line, sizeof(file_line), fp) == NULL) {
813 break;
814 }
815
816 // Remove newlines
817 func_name[strcspn(func_name, "\n")] = '\0';
818 file_line[strcspn(file_line, "\n")] = '\0';
819
820 // Extract relative path and COPY IT IMMEDIATELY
821 // IMPORTANT: extract_project_relative_path uses a static buffer internally.
822 // SAFE_MALLOC's debug memory tracking also calls extract_project_relative_path,
823 // which would overwrite the static buffer. So we must copy before allocating.
824 const char *rel_path_tmp = extract_project_relative_path(file_line);
825 char rel_path[512];
826 SAFE_STRNCPY(rel_path, rel_path_tmp, sizeof(rel_path));
827
828 // Allocate result buffer (AFTER copying rel_path to local storage)
829 result[i] = SAFE_MALLOC(1024, char *);
830
831 // Format symbol
832 bool has_func = (strcmp(func_name, "??") != 0);
833 bool has_file = (strcmp(file_line, "??:0") != 0 && strcmp(file_line, "??:?") != 0);
834
835 if (!has_func && !has_file) {
836 // Complete unknown - show raw address
837 SAFE_SNPRINTF(result[i], 1024, "%p", buffer[i]);
838 } else if (has_func && has_file) {
839 // Best case - both function and file:line known
840 if (strstr(rel_path, ":") != NULL) {
841 SAFE_SNPRINTF(result[i], 1024, "%s in %s()", rel_path, func_name);
842 } else {
843 SAFE_SNPRINTF(result[i], 1024, "%s() at %s", func_name, rel_path);
844 }
845 } else if (has_func) {
846 // Function known but file unknown (common for library functions)
847 SAFE_SNPRINTF(result[i], 1024, "%s() at %p", func_name, buffer[i]);
848 } else {
849 // File known but function unknown (rare)
850 SAFE_SNPRINTF(result[i], 1024, "%s (unknown function)", rel_path);
851 }
852 }
853
854 platform_pclose(&fp);
855 return result;
856}

858char **symbol_cache_resolve_batch(void *const *buffer, int size) {
859 if (size <= 0 || !buffer) {
860 log_error("Invalid parameters: buffer=%p, size=%d", (void *)buffer, size);
861 return NULL;
862 }
863
864 // DO NOT auto-initialize here - causes circular dependency during lock_debug_init()
865 // The cache must be initialized explicitly by platform_init() before use
866 if (!lifecycle_is_initialized(&g_symbol_cache_lc)) {
867 // Cache not initialized - fall back to uncached resolution
868 // This happens during early initialization before platform_init() completes
869 char **result = run_llvm_symbolizer_batch(buffer, size);
870 if (!result) {
871 result = run_addr2line_batch(buffer, size);
872 }
873 return result;
874 }
875
876 // Allocate result array (size + 1 for NULL terminator)
877 // CALLOC zeros the memory, so result[size] is already NULL
878 char **result = SAFE_CALLOC((size_t)(size + 1), sizeof(char *), char **);
879 if (!result) {
880 return NULL;
881 }
882
883 // Ensure NULL terminator is explicitly set (CALLOC already did this, but be explicit)
884 result[size] = NULL;
885
886 // First pass: check cache for all addresses
887 int uncached_count = 0;
888 void *uncached_addrs[size];
889 int uncached_indices[size];
890
891 for (int i = 0; i < size; i++) {
892 const char *cached = symbol_cache_lookup(buffer[i]);
893 if (cached) {
894 // Cache hit - symbol_cache_lookup already duplicated the string
895 result[i] = (char *)cached;
896 } else {
897 // Cache miss - track for batch resolution
898 uncached_addrs[uncached_count] = buffer[i];
899 uncached_indices[uncached_count] = i;
900 uncached_count++;
901 }
902 }
903
904 // Second pass: resolve uncached addresses with selected symbolizer
905 if (uncached_count > 0) {
906 char **resolved = NULL;
907
908 // Use the detected symbolizer type
909 switch (g_symbolizer_type) {
910 case SYMBOLIZER_LLVM:
911 resolved = run_llvm_symbolizer_batch(uncached_addrs, uncached_count);
912 break;
913 case SYMBOLIZER_ADDR2LINE:
914 resolved = run_addr2line_batch(uncached_addrs, uncached_count);
915 break;
916 case SYMBOLIZER_NONE:
917 default:
918 // No symbolizer available - will fall through to raw address handling
919 resolved = NULL;
920 break;
921 }
922
923 if (resolved) {
924 for (int i = 0; i < uncached_count; i++) {
925 int orig_idx = uncached_indices[i];
926 if (resolved[i]) {
927 result[orig_idx] = platform_strdup(resolved[i]);
928 // If strdup failed, use sentinel string instead of NULL
929 if (!result[orig_idx]) {
930 log_error("Failed to duplicate string for result[%d]", orig_idx);
931 result[orig_idx] = platform_strdup(NULL_SENTINEL);
932 }
933 // Only insert into cache if strdup succeeded (and it's not the sentinel)
934 if (result[orig_idx] && strcmp(result[orig_idx], NULL_SENTINEL) != 0) {
935 if (!symbol_cache_insert(uncached_addrs[i], resolved[i])) {
936 log_error("Failed to insert symbol into cache for result[%d]", orig_idx);
937 }
938 }
939 SAFE_FREE(resolved[i]);
940 } else {
941 // resolved[i] is NULL - use sentinel string
942 if (!result[orig_idx]) {
943 result[orig_idx] = platform_strdup(NULL_SENTINEL);
944 }
945 }
946 if (!result[orig_idx]) {
947 log_error("Failed to allocate memory for result[%d]", orig_idx);
948 }
949 }
950
951 SAFE_FREE(resolved);
952
953 } else {
954 // addr2line failed - fill uncached entries with raw addresses or sentinel
955 for (int i = 0; i < uncached_count; i++) {
956 int orig_idx = uncached_indices[i];
957 if (!result[orig_idx]) {
958 result[orig_idx] = SAFE_MALLOC(32, char *);
959 if (result[orig_idx]) {
960 SAFE_SNPRINTF(result[orig_idx], 32, "%p", uncached_addrs[i]);
961 } else {
962 result[orig_idx] = platform_strdup(NULL_SENTINEL);
963 }
964 }
965 if (!result[orig_idx]) {
966 log_error("Failed to allocate memory for result[%d]", orig_idx);
967 }
968 }
969 }
970 }
971
972 return result;
973}

975void symbol_cache_free_symbols(char **symbols) {
976 if (!symbols) {
977 return;
978 }
979
980 // The array is NULL-terminated (allocated with size+1, with result[size] = NULL)
981 // The terminator is a SINGLE NULL at index 'size'
982 // Failed allocations use the NULL_SENTINEL string "[NULL]" instead of NULL,
983 // so there are no NULL entries in the middle - only at the terminator
984 // This makes iteration safe: we can iterate until we find the first NULL (the terminator)
985
986 // Iterate through entries, freeing all non-NULL entries until we hit the NULL terminator
987 for (int i = 0; i < 64; i++) { // Reasonable limit to prevent infinite loop
988 if (symbols[i] == NULL) {
989 // Found NULL - this is the terminator, stop here
990 break;
991 }
992
993 // Found a non-NULL entry - check if it's the sentinel string
994 // Both regular strings and sentinel strings are allocated (with strdup),
995 // so we free them all
996 SAFE_FREE(symbols[i]);
997 symbols[i] = NULL; // Clear pointer after freeing
998 }
999
1000 // Free the array itself
1001 SAFE_FREE(symbols);
1002}
