295char *client_get_public_key_hex(void) {
296 if (g_crypto_handshake_ctx.state == CRYPTO_HANDSHAKE_DISABLED) {
297 WASM_ERROR("No crypto context (call client_generate_keypair first)");
298 return NULL;
299 }
300
301 // Get public key from handshake context
302 const uint8_t *pubkey = g_crypto_handshake_ctx.crypto_ctx.public_key;
303 if (!pubkey) {
304 WASM_ERROR("Public key not available in crypto context");
305 return NULL;
306 }
307
308 // Allocate buffer for hex string (32 bytes = 64 hex chars + null terminator)
309 char *hex_buffer = SAFE_MALLOC(65, char *);
310 if (!hex_buffer) {
311 WASM_ERROR("Failed to allocate hex buffer");
312 return NULL;
313 }
314
315 // Convert public key to hex
316 for (int i = 0; i < 32; i++) {
317 snprintf(hex_buffer + (i * 2), 3, "%02x", pubkey[i]);
318 }
319 hex_buffer[64] = '\0';
320
321 return hex_buffer;

331EMSCRIPTEN_KEEPALIVE
332int client_handle_key_exchange_init(const uint8_t *packet, size_t packet_len) {
333 WASM_LOG("=== client_handle_key_exchange_init CALLED ===");
334 WASM_LOG_INT(" packet_len", (int)packet_len);
335 WASM_LOG_INT(" g_crypto_handshake_ctx.state BEFORE", g_crypto_handshake_ctx.state);
336
337 // Safety check: if handshake context is not in INIT state, reinitialize it
338 // This handles cases where previous handshakes weren't properly cleaned up
339 if (g_crypto_handshake_ctx.state != CRYPTO_HANDSHAKE_INIT) {
340 WASM_LOG("Handshake context not in INIT state, reinitializing...");
341 WASM_LOG_INT(" Previous state", g_crypto_handshake_ctx.state);
342
343 // Destroy and reset
344 if (g_crypto_handshake_ctx.state != CRYPTO_HANDSHAKE_DISABLED) {
345 crypto_handshake_destroy(&g_crypto_handshake_ctx);
346 }
347 memset(&g_crypto_handshake_ctx, 0, sizeof(g_crypto_handshake_ctx));
348
349 // Reinitialize to INIT state
350 asciichat_error_t init_result = crypto_handshake_init("wasm-client", &g_crypto_handshake_ctx, false);
351 if (init_result != ASCIICHAT_OK) {
352 WASM_ERROR("Failed to reinitialize crypto handshake context");
353 WASM_LOG_INT(" init result", init_result);
354 return -1;
355 }
356 WASM_LOG("Crypto handshake context reinitialized");
357 }
358
359 if (!packet || packet_len == 0) {
360 WASM_ERROR("Invalid packet data");
361 return -1;
362 }
363
364 // Extract packet type and payload
365 if (packet_len < sizeof(packet_header_t)) {
366 WASM_ERROR("Packet too small for header");
367 return -1;
368 }
369
370 const packet_header_t *header = (const packet_header_t *)packet;
371 packet_type_t packet_type = ntohs(header->type);
372 const uint8_t *payload_src = packet + sizeof(packet_header_t);
373 size_t payload_len = packet_len - sizeof(packet_header_t);
374
375 WASM_LOG_INT(" packet_type", packet_type);
376 WASM_LOG_INT(" payload_len", (int)payload_len);
377
378 // Allocate payload copy from buffer pool (crypto function takes ownership and frees it)
379 // The raw packet pointer from JS cannot be passed directly because the crypto
380 // handshake function calls buffer_pool_free() on the payload when done.
381 uint8_t *payload = NULL;
382 if (payload_len > 0) {
383 payload = buffer_pool_alloc(NULL, payload_len);
384 if (!payload) {
385 WASM_ERROR("Failed to allocate payload buffer");
386 g_connection_state = CONNECTION_STATE_ERROR;
387 return -1;
388 }
389 memcpy(payload, payload_src, payload_len);
390 }
391
392 // Process key exchange using transport-abstracted handshake
393 WASM_LOG("Calling crypto_handshake_client_key_exchange...");
394
395 asciichat_error_t result = crypto_handshake_client_key_exchange(&g_crypto_handshake_ctx, &g_wasm_transport,
396 packet_type, payload, payload_len);
397
398 WASM_LOG_INT(" handshake result", result);
399 WASM_LOG_INT(" g_crypto_handshake_ctx.state AFTER", g_crypto_handshake_ctx.state);
400
401 if (result != ASCIICHAT_OK) {
402 WASM_ERROR("Failed to process KEY_EXCHANGE_INIT");
403 g_connection_state = CONNECTION_STATE_ERROR;
404 return -1;
405 }
406
407 g_connection_state = CONNECTION_STATE_HANDSHAKE;
408 WASM_LOG("=== KEY_EXCHANGE_INIT processed successfully ===");
409 return 0;
410}

601EMSCRIPTEN_KEEPALIVE
602char *client_parse_packet(const uint8_t *raw_packet, size_t packet_len) {
603 if (!raw_packet || packet_len < sizeof(packet_header_t)) {
604 WASM_ERROR("Invalid packet data");
605 return NULL;
606 }
607
608 // Parse packet header directly from bytes
609 const packet_header_t *header = (const packet_header_t *)raw_packet;
610
611 // Validate magic number (convert from network byte order)
612 uint64_t magic = NET_TO_HOST_U64(header->magic);
613 if (magic != PACKET_MAGIC) {
614 WASM_ERROR("Invalid packet magic number");
615 return NULL;
616 }
617
618 // Convert header fields from network byte order
619 uint16_t type = NET_TO_HOST_U16(header->type);
620 uint32_t length = NET_TO_HOST_U32(header->length);
621 uint32_t client_id = NET_TO_HOST_U32(header->client_id);
622 uint32_t crc32 = NET_TO_HOST_U32(header->crc32);
623
624 // Build JSON response with packet metadata
625 char *json = SAFE_MALLOC(1024, char *);
626 if (!json) {
627 WASM_ERROR("Failed to allocate JSON buffer");
628 return NULL;
629 }
630
631 snprintf(json, 1024, "{\"type\":%u,\"length\":%u,\"client_id\":%u,\"crc32\":%u}", type, length, client_id, crc32);
632
633 return json;
634}

646EMSCRIPTEN_KEEPALIVE
647int client_serialize_packet(uint16_t packet_type, const uint8_t *payload, size_t payload_len, uint32_t client_id,
648 uint8_t *output, size_t *out_len) {
649 if (!output || !out_len) {
650 WASM_ERROR("Invalid output parameters");
651 return -1;
652 }
653
654 // Calculate CRC32 of payload (use software version for WASM)
655 uint32_t crc = payload && payload_len > 0 ? asciichat_crc32_sw(payload, payload_len) : 0;
656
657 // Build packet header (convert to network byte order)
658 packet_header_t header = {.magic = HOST_TO_NET_U64(PACKET_MAGIC),
659 .type = HOST_TO_NET_U16(packet_type),
660 .length = HOST_TO_NET_U32((uint32_t)payload_len),
661 .crc32 = HOST_TO_NET_U32(crc),
662 .client_id = HOST_TO_NET_U32(client_id)};
663
664 // Copy header to output
665 memcpy(output, &header, sizeof(packet_header_t));
666
667 // Copy payload to output
668 if (payload && payload_len > 0) {
669 memcpy(output + sizeof(packet_header_t), payload, payload_len);
670 }
671
672 *out_len = sizeof(packet_header_t) + payload_len;
673 return 0;
674}
