38static void format_volume_bar(double volume, char *bar_output, size_t bar_output_size) {
39 if (!bar_output || bar_output_size < 25) {
40 return;
41 }
42
43 // Clamp volume to [0.0, 1.0]
44 if (volume < 0.0) {
45 volume = 0.0;
46 }
47 if (volume > 1.0) {
48 volume = 1.0;
49 }
50
51 // Calculate filled blocks (10 blocks total for 10% granularity)
52 int filled = (int)(volume * 10.0);
53 int empty = 10 - filled;
54
55 // Build bar: "[====== ] 80%"
56 // Use simple ASCII characters to avoid UTF-8 encoding issues
57 snprintf(bar_output, bar_output_size, "[%.*s%.*s] %d%%", filled, "==========", empty, " ",
58 (int)(volume * 100.0));
59}

119static void build_help_line(char *output, size_t output_size, const char *content, int max_width) {
120 if (!output || output_size < 256 || !content || max_width < 6) {
121 return;
122 }
123
124 // Available width: max_width - 3 (left border "║ ") - 1 (right border "║")
125 int content_available = max_width - 4;
126 if (content_available < 1)
127 content_available = 1;
128
129 // Truncate content if needed (with ellipsis indicator)
130 char truncated[256];
131 truncate_utf8_with_ellipsis(content, truncated, sizeof(truncated), content_available);
132
133 int content_width = utf8_display_width(truncated);
134 int padding = content_available - content_width;
135 if (padding < 0)
136 padding = 0;
137
138 // Build the line
139 char *pos = output;
140 int remaining = output_size;
141
142 // Left border and spacing
143 int n = snprintf(pos, remaining, "║ %s", truncated);
144 if (n > 0) {
145 pos += n;
146 remaining -= n;
147 }
148
149 // Padding spaces
150 for (int i = 0; i < padding && remaining > 1; i++) {
151 *pos++ = ' ';
152 remaining--;
153 }
154
155 // Right border
156 if (remaining > 3) {
157 snprintf(pos, remaining, "║");
158 }
159}

175static void build_settings_line(char *output, size_t output_size, const char *label, const char *value, int max_width,
176 int label_width) {
177 if (!output || output_size < 256 || !label || !value || max_width < 20) {
178 return;
179 }
180
181 // Align all values to start at the same column by padding labels to fixed width
182 const int MAX_LABEL_WIDTH = label_width;
183
184 int actual_label_width = utf8_display_width(label);
185
186 // Calculate label padding to align all values vertically
187 int label_padding = MAX_LABEL_WIDTH - actual_label_width;
188 if (label_padding < 0) {
189 label_padding = 0;
190 }
191
192 // Fixed prefix: "║ " (3) + label (padded to MAX_LABEL_WIDTH) + " : " (3) = 6 + MAX_LABEL_WIDTH columns
193 int fixed_prefix = 1 + 2 + MAX_LABEL_WIDTH + 3;
194 int right_border = 1;
195
196 // Available space for value + final padding
197 int available = max_width - fixed_prefix - right_border;
198 if (available < 4)
199 available = 4; // Minimum space for truncated value
200
201 // Truncate value if needed (with ellipsis indicator)
202 char truncated_value[256];
203 truncate_utf8_with_ellipsis(value, truncated_value, sizeof(truncated_value), available);
204
205 int value_width = utf8_display_width(truncated_value);
206 int padding = available - value_width;
207 if (padding < 0)
208 padding = 0;
209
210 // Build the line: "║ <label><label_pad>: <value><padding>║"
211 char *pos = output;
212 int remaining = output_size;
213
214 // Left border, spacing, and label
215 int n = snprintf(pos, remaining, "║ %s", label);
216 if (n > 0) {
217 pos += n;
218 remaining -= n;
219 }
220
221 // Add label padding spaces to align values
222 for (int i = 0; i < label_padding && remaining > 1; i++) {
223 *pos++ = ' ';
224 remaining--;
225 }
226
227 // Colon with spacing (one space before and after)
228 n = snprintf(pos, remaining, " : ");
229 if (n > 0) {
230 pos += n;
231 remaining -= n;
232 }
233
234 // Value (already truncated)
235 n = snprintf(pos, remaining, "%s", truncated_value);
236 if (n > 0) {
237 pos += n;
238 remaining -= n;
239 }
240
241 // Final padding spaces
242 for (int i = 0; i < padding && remaining > 1; i++) {
243 *pos++ = ' ';
244 remaining--;
245 }
246
247 // Right border
248 if (remaining > 3) {
249 snprintf(pos, remaining, "║");
250 }
251}
