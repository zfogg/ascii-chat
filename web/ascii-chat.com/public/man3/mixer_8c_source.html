42// Compressor implementation
43void compressor_init(compressor_t *comp, float sample_rate) {
44 comp->sample_rate = sample_rate;
45 comp->envelope = 0.0f;
46 comp->gain_lin = 1.0f;
47
48 // Set default parameters with 0dB makeup gain (unity)
49 // Soft clip at 0.7 provides 3dB headroom to prevent hard clipping
50 // Ducking and crowd scaling naturally reduce volume, no compensation needed
51 compressor_set_params(comp, -10.0f, 4.0f, 10 * NS_PER_MS_INT, 100 * NS_PER_MS_INT, 0.0f);
52}

54void compressor_set_params(compressor_t *comp, float threshold_dB, float ratio, uint64_t attack_ns, uint64_t release_ns,
55 float makeup_dB) {
56 comp->threshold_dB = threshold_dB;
57 comp->ratio = ratio;
58 comp->attack_ns = attack_ns;
59 comp->release_ns = release_ns;
60 comp->makeup_dB = makeup_dB;
61 comp->knee_dB = 2.0f; // Fixed soft knee
62
63 // Calculate time constants
64 float attack_tau = (float)attack_ns / NS_PER_SEC;
65 float release_tau = (float)release_ns / NS_PER_SEC;
66 comp->attack_coeff = expf(-1.0f / (attack_tau * comp->sample_rate + 1e-12f));
67 comp->release_coeff = expf(-1.0f / (release_tau * comp->sample_rate + 1e-12f));
68}

70static float compressor_gain_reduction_db(const compressor_t *comp, float level_dB) {
71 float over = level_dB - comp->threshold_dB;
72 float knee = comp->knee_dB;
73
74 if (knee > 0.0f) {
75 if (over <= -knee * 0.5f)
76 return 0.0f;
77 if (over >= knee * 0.5f)
78 return (1.0f / comp->ratio - 1.0f) * over;
79 float x = over + knee * 0.5f;
80 return (1.0f / comp->ratio - 1.0f) * (x * x) / (2.0f * knee);
81 }
82 if (over <= 0.0f) {
83 return 0.0f;
84 }
85 return (1.0f / comp->ratio - 1.0f) * over;
86}

88float compressor_process_sample(compressor_t *comp, float sidechain) {
89 float x = fabsf(sidechain);
90
91 // Update envelope with attack/release
92 if (x > comp->envelope)
93 comp->envelope = comp->attack_coeff * comp->envelope + (1.0f - comp->attack_coeff) * x;
94 else
95 comp->envelope = comp->release_coeff * comp->envelope + (1.0f - comp->release_coeff) * x;
96
97 // Calculate gain reduction
98 float level_dB = linear_to_db(comp->envelope);
99 float gr_dB = compressor_gain_reduction_db(comp, level_dB);
100 float target_lin = db_to_linear(gr_dB + comp->makeup_dB);
101
102 // Smooth gain changes
103 if (target_lin < comp->gain_lin)
104 comp->gain_lin = comp->attack_coeff * comp->gain_lin + (1.0f - comp->attack_coeff) * target_lin;
105 else
106 comp->gain_lin = comp->release_coeff * comp->gain_lin + (1.0f - comp->release_coeff) * target_lin;
107
108 return comp->gain_lin;
109}

111// Ducking implementation
112int ducking_init(ducking_t *duck, int num_sources, float sample_rate) {
113 if (!duck) {
114 return SET_ERRNO(ERROR_INVALID_PARAM, "ducking_init: duck is NULL");
115 }
116
117 // Set default parameters
118 // threshold_dB: sources below this aren't considered "speaking"
119 // leader_margin_dB: sources within this margin of loudest are all "leaders"
120 // atten_dB: how much to attenuate non-leaders (was -12dB, too aggressive)
121 duck->threshold_dB = -45.0f; // More lenient threshold
122 duck->leader_margin_dB = 6.0f; // Wider margin = more sources treated as leaders
123 duck->atten_dB = -6.0f; // Only -6dB attenuation (was -12dB)
124 duck->attack_ns = 10 * NS_PER_MS_INT; // Slower attack (10ms)
125 duck->release_ns = 200 * NS_PER_MS_INT; // Slower release (200ms)
126 duck->envelope = NULL;
127 duck->gain = NULL;
128
129 // Calculate time constants
130 float attack_tau = (float)duck->attack_ns / NS_PER_SEC;
131 float release_tau = (float)duck->release_ns / NS_PER_SEC;
132 duck->attack_coeff = expf(-1.0f / (attack_tau * sample_rate + 1e-12f));
133 duck->release_coeff = expf(-1.0f / (release_tau * sample_rate + 1e-12f));
134
135 // Allocate arrays with overflow checking
136 size_t envelope_size = 0;
137 if (checked_size_mul((size_t)num_sources, sizeof(float), &envelope_size) != ASCIICHAT_OK) {
138 return SET_ERRNO(ERROR_BUFFER_OVERFLOW, "Ducking envelope array size overflow: %d sources", num_sources);
139 }
140 duck->envelope = SAFE_MALLOC(envelope_size, float *);
141 if (!duck->envelope) {
142 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate ducking envelope array");
143 }
144
145 size_t gain_size = 0;
146 if (checked_size_mul((size_t)num_sources, sizeof(float), &gain_size) != ASCIICHAT_OK) {
147 SAFE_FREE(duck->envelope);
148 duck->envelope = NULL;
149 return SET_ERRNO(ERROR_BUFFER_OVERFLOW, "Ducking gain array size overflow: %d sources", num_sources);
150 }
151 duck->gain = SAFE_MALLOC(gain_size, float *);
152 if (!duck->gain) {
153 SAFE_FREE(duck->envelope);
154 duck->envelope = NULL;
155 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate ducking gain array");
156 }
157
158 // Initialize
159 SAFE_MEMSET(duck->envelope, (size_t)num_sources * sizeof(float), 0, (size_t)num_sources * sizeof(float));
160 for (int i = 0; i < num_sources; i++) {
161 duck->gain[i] = 1.0f;
162 }
163
164 return ASCIICHAT_OK;
165}

185void ducking_process_frame(ducking_t *duck, float *envelopes, float *gains, int num_sources) {
186 // Find the loudest source
187 float max_dB = -120.0f;
188 float env_dB[MIXER_MAX_SOURCES];
189
190 for (int i = 0; i < num_sources; i++) {
191 env_dB[i] = linear_to_db(envelopes[i]);
192 if (env_dB[i] > max_dB)
193 max_dB = env_dB[i];
194 }
195
196 // Calculate ducking gain for each source
197 float leader_cut = db_to_linear(duck->atten_dB);
198
199 for (int i = 0; i < num_sources; i++) {
200 bool is_speaking = env_dB[i] > duck->threshold_dB;
201 bool is_leader = is_speaking && (env_dB[i] >= max_dB - duck->leader_margin_dB);
202
203 float target;
204 if (is_speaking && !is_leader) {
205 target = leader_cut;
206 } else {
207 target = 1.0f;
208 }
209
210 // Smooth gain transitions
211 if (target < gains[i])
212 gains[i] = duck->attack_coeff * gains[i] + (1.0f - duck->attack_coeff) * target;
213 else
214 gains[i] = duck->release_coeff * gains[i] + (1.0f - duck->release_coeff) * target;
215 }
216}

221 if (max_sources <= 0 || max_sources > MIXER_MAX_SOURCES) {

223 return NULL;
224 }
225

227 SET_ERRNO(ERROR_INVALID_PARAM, "Invalid sample_rate: %d (must be 1-192000)", sample_rate);
228 return NULL;
229 }

234 SET_ERRNO(ERROR_MEMORY, "Failed to allocate mixer structure");
235 return NULL;
236 }
237
238 mixer->num_sources = 0;
239 mixer->max_sources = max_sources;
240 mixer->sample_rate = sample_rate;
241
242 // Allocate source management arrays with overflow checking
243 size_t buffers_size = 0;
244 if (checked_size_mul((size_t)max_sources, sizeof(audio_ring_buffer_t *), &buffers_size) != ASCIICHAT_OK) {
245 SET_ERRNO(ERROR_BUFFER_OVERFLOW, "Mixer source buffers array overflow: %d sources", max_sources);

258 SAFE_FREE(mixer->source_buffers);

263 if (!mixer->source_ids) {
264 SAFE_FREE(mixer->source_buffers);
265 SAFE_FREE(mixer);
266 return NULL;
267 }
268
269 size_t active_size = 0;
270 if (checked_size_mul((size_t)max_sources, sizeof(bool), &active_size) != ASCIICHAT_OK) {
271 SET_ERRNO(ERROR_BUFFER_OVERFLOW, "Mixer source active array overflow: %d sources", max_sources);
272 SAFE_FREE(mixer->source_buffers);
273 SAFE_FREE(mixer->source_ids);
274 SAFE_FREE(mixer);
275 return NULL;
276 }
277 mixer->source_active = SAFE_MALLOC(active_size, bool *);
278 if (!mixer->source_active) {
279 SAFE_FREE(mixer->source_buffers);
280 SAFE_FREE(mixer->source_ids);
281 SAFE_FREE(mixer);
282 return NULL;
283 }
284
285 // Initialize arrays
286 SAFE_MEMSET((void *)mixer->source_buffers, buffers_size, 0, buffers_size);
287 SAFE_MEMSET(mixer->source_ids, ids_size, 0, ids_size);
288 SAFE_MEMSET(mixer->source_active, active_size, 0, active_size);
289
290 // OPTIMIZATION 1: Initialize bitset optimization structures
291 mixer->active_sources_mask = 0ULL; // No sources active initially
292 for (int i = 0; i < 256; i++) {
293 mixer->source_id_to_index[i] = MIXER_HASH_INVALID;
294 }
295
296 // Allocate mix buffer BEFORE rwlock_init so cleanup path is correct
297 size_t mix_buffer_size = 0;
298 if (checked_size_mul(MIXER_FRAME_SIZE, sizeof(float), &mix_buffer_size) != ASCIICHAT_OK) {
299 SET_ERRNO(ERROR_BUFFER_OVERFLOW, "Mixer mix buffer size overflow: %zu samples", (size_t)MIXER_FRAME_SIZE);
300 SAFE_FREE(mixer->source_buffers);
301 SAFE_FREE(mixer->source_ids);
302 SAFE_FREE(mixer->source_active);
303 SAFE_FREE(mixer);
304 return NULL;
305 }
306 mixer->mix_buffer = SAFE_MALLOC(mix_buffer_size, float *);
307 if (!mixer->mix_buffer) {
308 SET_ERRNO(ERROR_MEMORY, "Failed to allocate mix buffer");
309 SAFE_FREE(mixer->source_buffers);
310 SAFE_FREE(mixer->source_ids);
311 SAFE_FREE(mixer->source_active);
312 SAFE_FREE(mixer);
313 return NULL;
314 }
315
316 // OPTIMIZATION 2: Initialize reader-writer lock
317 if (rwlock_init(&mixer->source_lock, "audio_mixer_sources") != 0) {
318 SET_ERRNO(ERROR_THREAD, "Failed to initialize mixer source lock");
319 SAFE_FREE(mixer->source_buffers);
320 SAFE_FREE(mixer->source_ids);
321 SAFE_FREE(mixer->source_active);

329 mixer->base_gain = 1.0f; // Unity gain - let compressor handle loudness, avoid clipping

332 if (ducking_init(&mixer->ducking, max_sources, (float)sample_rate) != ASCIICHAT_OK) {
333 rwlock_destroy(&mixer->source_lock);
334 SAFE_FREE(mixer->source_buffers);
335 SAFE_FREE(mixer->source_ids);
336 SAFE_FREE(mixer->source_active);
337 SAFE_FREE(mixer->mix_buffer);
338 SAFE_FREE(mixer);

340 }

342
343 log_debug("Audio mixer created: max_sources=%d, sample_rate=%d", max_sources, sample_rate);
344
345 // Register mixer and its sync primitives for debugging
346 NAMED_REGISTER_MIXER(mixer, "audio_mixer", NULL);
347 NAMED_REGISTER_RWLOCK(&mixer->source_lock, "source_lock", (uintptr_t)(const void *)(mixer));

349 return mixer;
350}

378int mixer_add_source(mixer_t *mixer, const char *client_id, audio_ring_buffer_t *buffer) {
379 if (!mixer || !buffer)
380 return -1;
381
382 // OPTIMIZATION 2: Acquire write lock for source modification
383 rwlock_wrlock(&mixer->source_lock);
384
385 // Find an empty slot
386 int slot = -1;
387 for (int i = 0; i < mixer->max_sources; i++) {
388 if (mixer->source_ids[i] == NULL) {
389 slot = i;
390 break;
391 }
392 }
393
394 if (slot == -1) {
395 rwlock_wrunlock(&mixer->source_lock);
396 log_warn("Mixer: No available slots for client %s", client_id);
397 return -1;
398 }
399
400 mixer->source_buffers[slot] = buffer;
401 // Duplicate the client_id string - we own this memory now
402 mixer->source_ids[slot] = SAFE_MALLOC(strlen(client_id) + 1, char *);
403 if (!mixer->source_ids[slot]) {
404 log_warn("Mixer: Failed to allocate memory for client_id %s", client_id);
405 mixer->source_buffers[slot] = NULL;
406 rwlock_wrunlock(&mixer->source_lock);
407 return -1;
408 }
409 strcpy((char *)mixer->source_ids[slot], client_id);
410 mixer->source_active[slot] = true;
411 mixer->num_sources++;
412
413 // OPTIMIZATION 1: Update bitset optimization structures
414 mixer->active_sources_mask |= (1ULL << slot); // Set bit for this slot
415 // Hash table optimization disabled for string-based client_ids (linear search is acceptable for small mixer sizes)
416
417 rwlock_wrunlock(&mixer->source_lock);
418
419 log_info("Mixer: Added source for client %s at slot %d", client_id, slot);
420 return slot;
421}

423void mixer_remove_source(mixer_t *mixer, const char *client_id) {
424 if (!mixer)
425 return;
426
427 // OPTIMIZATION 2: Acquire write lock for source modification
428 rwlock_wrlock(&mixer->source_lock);
429
430 for (int i = 0; i < mixer->max_sources; i++) {

483int mixer_process(mixer_t *mixer, float *output, int num_samples) {
484 if (!mixer || !output || num_samples <= 0)
485 return -1;
486
487 // THREAD SAFETY: Acquire read lock to protect against concurrent source add/remove
488 // This prevents race conditions where source_buffers[i] could be set to NULL while we read it
489 rwlock_rdlock(&mixer->source_lock);
490
491 // Clear output buffer
492 SAFE_MEMSET(output, num_samples * sizeof(float), 0, num_samples * sizeof(float));
493
494 // Count active sources
495 int active_count = 0;
496 for (int i = 0; i < mixer->max_sources; i++) {
497 if (mixer->source_ids[i] != 0 && mixer->source_active[i] && mixer->source_buffers[i]) {
498 active_count++;
499 }
500 }
501
502 if (active_count == 0) {
503 // No active sources, output silence
504 return 0;
505 }
506
507 // Process in frames for efficiency
508 for (int frame_start = 0; frame_start < num_samples; frame_start += MIXER_FRAME_SIZE) {
509 int frame_size = (frame_start + MIXER_FRAME_SIZE > num_samples) ? (num_samples - frame_start) : MIXER_FRAME_SIZE;
510
511 // Clear mix buffer
512 SAFE_MEMSET(mixer->mix_buffer, frame_size * sizeof(float), 0, frame_size * sizeof(float));
513
514 // Temporary buffers for source audio
515 float source_samples[MIXER_MAX_SOURCES][MIXER_FRAME_SIZE];
516 int source_count = 0;
517 int source_map[MIXER_MAX_SOURCES]; // Maps source index to slot
518
519 // Read from each active source
520 for (int i = 0; i < mixer->max_sources; i++) {
521 if (mixer->source_ids[i] != 0 && mixer->source_active[i] && mixer->source_buffers[i]) {
522 // Read samples from this source's ring buffer
523 size_t samples_read_size =
524 audio_ring_buffer_read(mixer->source_buffers[i], source_samples[source_count], frame_size);
525 int samples_read = (int)samples_read_size;
526
527 // Accept partial frames - pad with silence if needed
528 // This prevents audio dropouts when ring buffers are temporarily under-filled
529 if (samples_read > 0) {
530 // Pad remaining samples with silence if we got a partial frame
531 if (samples_read < frame_size) {
532 SAFE_MEMSET(&source_samples[source_count][samples_read], (frame_size - samples_read) * sizeof(float), 0,
533 (frame_size - samples_read) * sizeof(float));
534 }
535
536 source_map[source_count] = i;
537 source_count++;
538
539 if (source_count >= MIXER_MAX_SOURCES)
540 break;
541 }
542 }
543 }
544
545 // OPTIMIZATION: Batch envelope calculation per-frame instead of per-sample
546 // Calculate peak amplitude for each source over the entire frame
547 int speaking_count = 0;
548
549 for (int i = 0; i < source_count; i++) {
550 int slot = source_map[i];
551 float peak = 0.0f;
552
553 // Find peak amplitude in frame (much faster than per-sample envelope)
554 for (int s = 0; s < frame_size; s++) {
555 float abs_sample = fabsf(source_samples[i][s]);
556 if (abs_sample > peak)
557 peak = abs_sample;
558 }
559
560 // Update envelope using frame peak (one update per frame instead of per-sample)
561 if (peak > mixer->ducking.envelope[slot]) {
562 mixer->ducking.envelope[slot] =
563 mixer->ducking.attack_coeff * mixer->ducking.envelope[slot] + (1.0f - mixer->ducking.attack_coeff) * peak;
564 } else {
565 mixer->ducking.envelope[slot] =
566 mixer->ducking.release_coeff * mixer->ducking.envelope[slot] + (1.0f - mixer->ducking.release_coeff) * peak;
567 }
568
569 // Count speaking sources
570 if (mixer->ducking.envelope[slot] > db_to_linear(-60.0f))
571 speaking_count++;
572 }
573
574 // Apply ducking ONCE per frame (not per-sample)
575 ducking_process_frame(&mixer->ducking, mixer->ducking.envelope, mixer->ducking.gain, mixer->max_sources);
576
577 // Calculate crowd scaling ONCE per frame
578 float crowd_gain = (speaking_count > 0) ? (1.0f / powf((float)speaking_count, mixer->crowd_alpha)) : 1.0f;
579 float pre_bus = mixer->base_gain * crowd_gain;
580
581 // Pre-calculate combined gains for each source (ducking * pre_bus)
582 float combined_gains[MIXER_MAX_SOURCES];
583 for (int i = 0; i < source_count; i++) {
584 int slot = source_map[i];
585 combined_gains[i] = mixer->ducking.gain[slot] * pre_bus;
586 }
587
588 // OPTIMIZATION: Fast mixing loop - simple multiply-add with pre-calculated gains
589 // NO per-sample compressor to avoid expensive log/pow calls (480x per iteration)
590 for (int s = 0; s < frame_size; s++) {
591 float mix = 0.0f;
592 for (int i = 0; i < source_count; i++) {
593 mix += source_samples[i][s] * combined_gains[i];
594 }
595
596 // Store in mix buffer for frame-level compression below
597 mixer->mix_buffer[s] = mix;
598 }
599
600 // OPTIMIZATION: Apply compression ONCE per frame instead of per-sample
601 // This reduces expensive log10f/powf calls from 480x to 1x per iteration
602 // Calculate frame peak for compressor sidechain
603 float frame_peak = 0.0f;
604 for (int s = 0; s < frame_size; s++) {
605 float abs_val = fabsf(mixer->mix_buffer[s]);
606 if (abs_val > frame_peak)
607 frame_peak = abs_val;
608 }
609
610 // Process compressor with frame peak (1 call instead of 480)
611 float comp_gain = compressor_process_sample(&mixer->compressor, frame_peak);
612
613 // Apply compression gain and soft clipping to all samples
614 for (int s = 0; s < frame_size; s++) {
615 float mix = mixer->mix_buffer[s] * comp_gain;
616
617 // Soft clip at 0.7 to provide 3dB headroom and prevent hard clipping
618 // With +6dB makeup gain, peaks can exceed 1.0, so we need headroom
619 // threshold=0.7 maps asymptotically toward 1.0, preventing harsh distortion
620 output[frame_start + s] = soft_clip(mix, 0.7f, 3.0f);
621 }
622 }
623
624 rwlock_rdunlock(&mixer->source_lock);
625 return num_samples;
626}

628int mixer_process_excluding_source(mixer_t *mixer, float *output, int num_samples, uint32_t exclude_client_id) {
629 if (!mixer || !output || num_samples <= 0)
630 return -1;
631
632 // Only use timing in debug builds - snprintf + hashtable ops are expensive in hot path
633#ifndef NDEBUG
634 START_TIMER("mixer_total");
635#endif
636
637 // THREAD SAFETY: Acquire read lock to protect against concurrent source add/remove
638 // This prevents race conditions where source_buffers[i] could be set to NULL while we read it
639 rwlock_rdlock(&mixer->source_lock);
640
641 // Clear output buffer
642 SAFE_MEMSET(output, num_samples * sizeof(float), 0, num_samples * sizeof(float));
643
644 // OPTIMIZATION 1: O(1) exclusion using bitset and hash table
645 uint8_t exclude_index = mixer_hash_get_slot(mixer, exclude_client_id);
646 uint64_t active_mask = mixer->active_sources_mask;
647
648 // Validate exclude_index before using in bitshift
649 // - exclude_index == MIXER_HASH_INVALID means "not found" (sentinel value from initialization)
650 // - exclude_index >= MIXER_MAX_SOURCES would cause undefined behavior in bitshift
651 bool valid_exclude = (exclude_index < MIXER_MAX_SOURCES && exclude_index != MIXER_HASH_INVALID);
652
653 if (valid_exclude) {
654 // Apply exclusion to prevent echo feedback
655 uint64_t mask_without_excluded = active_mask & ~(1ULL << exclude_index);
656 active_mask = mask_without_excluded;
657
658 // DIAGNOSTIC: Log which client was excluded and which remain active
659 log_dev_every(
660 4500 * US_PER_MS_INT,
661 "MIXER EXCLUSION: exclude_client=%u, exclude_index=%u, active_mask_before=0x%llx, active_mask_after=0x%llx",
662 exclude_client_id, exclude_index, (unsigned long long)(active_mask | (1ULL << exclude_index)),
663 (unsigned long long)active_mask);
664 } else {
665 // DIAGNOSTIC: Failed to exclude - log why
666 log_dev_every(4500 * US_PER_MS_INT,
667 "MIXER EXCLUSION FAILED: exclude_client=%u, exclude_index=%u (valid=%d), lookup_id=%u",
668 exclude_client_id, exclude_index, valid_exclude,
669 (exclude_index < MIXER_MAX_SOURCES && exclude_index != 0xFF) ? mixer->source_ids[exclude_index] : 0);
670 }
671
672 // Fast check: any sources to mix?
673 if (active_mask == 0) {
674 rwlock_rdunlock(&mixer->source_lock);
675#ifndef NDEBUG
676 STOP_TIMER("mixer_total");
677#endif
678 return 0; // No sources to mix (excluding the specified client), output silence
679 }
680
681 // Process in frames for efficiency
682 for (int frame_start = 0; frame_start < num_samples; frame_start += MIXER_FRAME_SIZE) {
683 int frame_size = (frame_start + MIXER_FRAME_SIZE > num_samples) ? (num_samples - frame_start) : MIXER_FRAME_SIZE;
684
685 uint64_t read_start_ns = time_get_ns();
686
687 // Clear mix buffer
688 SAFE_MEMSET(mixer->mix_buffer, frame_size * sizeof(float), 0, frame_size * sizeof(float));
689
690 // Temporary buffers for source audio
691 float source_samples[MIXER_MAX_SOURCES][MIXER_FRAME_SIZE];
692 int source_count = 0;
693 int source_map[MIXER_MAX_SOURCES]; // Maps source index to slot
694
695 // OPTIMIZATION 1: Iterate only over active sources using bitset
696 uint64_t current_mask = active_mask;
697 while (current_mask && source_count < MIXER_MAX_SOURCES) {
698 int i = find_first_set_bit(current_mask); // Portable: find first set bit
699 current_mask &= current_mask - 1; // Clear lowest set bit
700
701 // Verify source is valid (defensive programming)
702 if (i < mixer->max_sources && mixer->source_ids[i] != 0 && mixer->source_buffers[i]) {
703 // Read samples from this source's ring buffer
704 size_t samples_read_size =
705 audio_ring_buffer_read(mixer->source_buffers[i], source_samples[source_count], frame_size);
706 int samples_read = (int)samples_read_size;
707
708 // Accept partial frames - pad with silence if needed
709 // This prevents audio dropouts when ring buffers are temporarily under-filled
710 if (samples_read > 0) {
711 // Pad remaining samples with silence if we got a partial frame
712 if (samples_read < frame_size) {
713 SAFE_MEMSET(&source_samples[source_count][samples_read], (frame_size - samples_read) * sizeof(float), 0,
714 (frame_size - samples_read) * sizeof(float));
715 }
716
717 // DIAGNOSTIC: Calculate RMS of this source's audio
718 float source_rms = 0.0f;
719 if (frame_start == 0) { // Only log for first frame to reduce spam
720 float sum_squares = 0.0f;
721 for (int s = 0; s < samples_read; s++) {
722 sum_squares += source_samples[source_count][s] * source_samples[source_count][s];
723 }
724 source_rms = sqrtf(sum_squares / (float)samples_read);
725 log_info_every(NS_PER_MS_INT, "MIXER SOURCE READ: client_id=%u, slot=%d, samples_read=%d, RMS=%.6f",
726 mixer->source_ids[i], i, samples_read, source_rms);
727 }
728
729 source_map[source_count] = i;
730 source_count++;
731 }
732 }
733 }
734
735 uint64_t read_end_ns = time_get_ns();
736 uint64_t read_time_ns = time_elapsed_ns(read_start_ns, read_end_ns);
737
738 if (read_time_ns > 10 * NS_PER_MS_INT) {
739 char read_time_str[32];
740 time_pretty(read_time_ns, -1, read_time_str, sizeof(read_time_str));
741 log_warn_every(LOG_RATE_DEFAULT, "Mixer: Slow source reading took %s for %d sources", read_time_str,
742 source_count);
743 }
744
745 // OPTIMIZATION: Batch envelope calculation per-frame instead of per-sample
746 // Calculate peak amplitude for each source over the entire frame
747 int speaking_count = 0;
748
749 for (int i = 0; i < source_count; i++) {
750 int slot = source_map[i];
751 float peak = 0.0f;
752
753 // Find peak amplitude in frame (much faster than per-sample envelope)
754 for (int s = 0; s < frame_size; s++) {
755 float abs_sample = fabsf(source_samples[i][s]);
756 if (abs_sample > peak)
757 peak = abs_sample;
758 }
759
760 // Update envelope using frame peak (one update per frame instead of per-sample)
761 if (peak > mixer->ducking.envelope[slot]) {
762 mixer->ducking.envelope[slot] =
763 mixer->ducking.attack_coeff * mixer->ducking.envelope[slot] + (1.0f - mixer->ducking.attack_coeff) * peak;
764 } else {
765 mixer->ducking.envelope[slot] =
766 mixer->ducking.release_coeff * mixer->ducking.envelope[slot] + (1.0f - mixer->ducking.release_coeff) * peak;
767 }
768
769 // Count speaking sources
770 if (mixer->ducking.envelope[slot] > db_to_linear(-60.0f))
771 speaking_count++;
772 }
773
774 // Apply ducking ONCE per frame (not per-sample)
775 ducking_process_frame(&mixer->ducking, mixer->ducking.envelope, mixer->ducking.gain, mixer->max_sources);
776
777 // Calculate crowd scaling ONCE per frame
778 float crowd_gain = (speaking_count > 0) ? (1.0f / powf((float)speaking_count, mixer->crowd_alpha)) : 1.0f;
779 float pre_bus = mixer->base_gain * crowd_gain;
780
781 // Pre-calculate combined gains for each source (ducking * pre_bus)
782 float combined_gains[MIXER_MAX_SOURCES];
783 for (int i = 0; i < source_count; i++) {
784 int slot = source_map[i];
785 combined_gains[i] = mixer->ducking.gain[slot] * pre_bus;
786 }
787
788 // OPTIMIZATION: Fast mixing loop - simple multiply-add with pre-calculated gains
789 // NO per-sample compressor to avoid expensive log/pow calls (480x per iteration)
790 for (int s = 0; s < frame_size; s++) {
791 float mix = 0.0f;
792 for (int i = 0; i < source_count; i++) {
793 mix += source_samples[i][s] * combined_gains[i];
794 }
795
796 // Store in mix buffer for frame-level compression below
797 mixer->mix_buffer[s] = mix;
798 }
799
800 // OPTIMIZATION: Apply compression ONCE per frame instead of per-sample
801 // This reduces expensive log10f/powf calls from 480x to 1x per iteration
802 // Calculate frame peak for compressor sidechain
803 float frame_peak = 0.0f;
804 for (int s = 0; s < frame_size; s++) {
805 float abs_val = fabsf(mixer->mix_buffer[s]);
806 if (abs_val > frame_peak)
807 frame_peak = abs_val;
808 }
809
810 // Process compressor with frame peak (1 call instead of 480)
811 float comp_gain = compressor_process_sample(&mixer->compressor, frame_peak);
812
813 // Apply compression gain and soft clipping to all samples
814 for (int s = 0; s < frame_size; s++) {
815 float mix = mixer->mix_buffer[s] * comp_gain;
816
817 // Soft clip at 0.7 to provide 3dB headroom and prevent hard clipping
818 // With +6dB makeup gain, peaks can exceed 1.0, so we need headroom
819 // threshold=0.7 maps asymptotically toward 1.0, preventing harsh distortion
820 output[frame_start + s] = soft_clip(mix, 0.7f, 3.0f);
821 }
822 }
823
824 rwlock_rdunlock(&mixer->source_lock);
825
826#ifndef NDEBUG
827 STOP_TIMER_AND_LOG_EVERY(warn, NS_PER_SEC_INT, 2 * NS_PER_MS_INT, "mixer_total", "Mixer took");
828#endif
829
830 return num_samples;
831}

839 if (!gate)
840 return;

842 gate->sample_rate = sample_rate;
843 gate->envelope = 0.0f;
844 gate->gate_open = false;
845
846 // Default parameters
847 // Slower attack (10ms) prevents clicking artifacts during gate transitions
848 // threshold=0.01, attack=10ms, release=50ms, hysteresis=0.9
849 noise_gate_set_params(gate, 0.01f, 10 * NS_PER_MS_INT, 50 * NS_PER_MS_INT, 0.9f);

851
852void noise_gate_set_params(noise_gate_t *gate, float threshold, uint64_t attack_ns, uint64_t release_ns,
853 float hysteresis) {
854 if (!gate)
855 return;
856
857 gate->threshold = threshold;
858 gate->attack_ns = attack_ns;
859 gate->release_ns = release_ns;
860 gate->hysteresis = hysteresis;
861
862 // Calculate coefficients for envelope follower
863 // Using exponential moving average: coeff = 1 - exp(-1 / (time_s * sample_rate))
864 float attack_s = (float)attack_ns / NS_PER_SEC;
865 float release_s = (float)release_ns / NS_PER_SEC;
866 gate->attack_coeff = 1.0f - expf(-1.0f / (attack_s * gate->sample_rate + 1e-12f));
867 gate->release_coeff = 1.0f - expf(-1.0f / (release_s * gate->sample_rate + 1e-12f));
868}
869
870float noise_gate_process_sample(noise_gate_t *gate, float input, float peak_amplitude) {
871 if (!gate)
872 return input;
873
874 // Determine target state with hysteresis
875 float target;
876 if (gate->gate_open) {
877 // Gate is open - use lower threshold (hysteresis) to close
878 target = (peak_amplitude > gate->threshold * gate->hysteresis) ? 1.0f : 0.0f;
879 } else {
880 // Gate is closed - use normal threshold to open
881 target = (peak_amplitude > gate->threshold) ? 1.0f : 0.0f;
882 }
883
884 // Update gate state
885 gate->gate_open = (target > 0.5f);
886
887 // Update envelope with appropriate coefficient
888 float coeff = (target > gate->envelope) ? gate->attack_coeff : gate->release_coeff;
889 gate->envelope += coeff * (target - gate->envelope);
890
891 // Apply gate
892 return input * gate->envelope;
893}

946 if (!filter)

973void lowpass_filter_init(lowpass_filter_t *filter, float cutoff_hz, float sample_rate) {
974 if (!filter)
975 return;
976
977 filter->cutoff_hz = cutoff_hz;
978 filter->sample_rate = sample_rate;
979

981 // alpha = dt / (RC + dt) where RC = 1 / (2 * pi * fc)
982 float dt = 1.0f / sample_rate;
983 float rc = 1.0f / (2.0f * (float)M_PI * cutoff_hz);
984 filter->alpha = dt / (rc + dt);
985
986 lowpass_filter_reset(filter);
987}

1001 float output = filter->alpha * input + (1.0f - filter->alpha) * filter->prev_output;
1002
1003 filter->prev_output = output;
1004
1005 return output;

1022float soft_clip(float sample, float threshold, float steepness) {
1023 if (sample > threshold) {
1024 // Soft clip positive values using tanh curve
1025 // Maps samples above threshold asymptotically toward 1.0
1026 return threshold + (1.0f - threshold) * tanhf((sample - threshold) * steepness);
1027 }
1028 if (sample < -threshold) {
1029 // Soft clip negative values symmetrically
1030 return -threshold + (-1.0f + threshold) * tanhf((sample + threshold) * steepness);
1031 }
1032 return sample;
1033}
