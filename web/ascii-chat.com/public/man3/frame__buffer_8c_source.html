66void frame_buffer_append(frame_buffer_t *buf, const char *str, size_t len) {
67 if (!buf || !str || len == 0) {
68 return;
69 }
70
71 // Grow buffer if needed
72 if (buf->len + len > buf->capacity) {
73 size_t new_capacity = buf->capacity * 2;
74 while (new_capacity < buf->len + len) {
75 new_capacity *= 2;
76 }
77
78 char *new_data = SAFE_MALLOC(new_capacity, char *);
79 if (!new_data) {
80 return; // Failed to grow, silently ignore
81 }
82
83 if (buf->len > 0 && buf->data) {
84 memcpy(new_data, buf->data, buf->len);
85 }
86
87 SAFE_FREE(buf->data);
88 buf->data = new_data;
89 buf->capacity = new_capacity;
90 }
91
92 // Append the data
93 memcpy(buf->data + buf->len, str, len);
94 buf->len += len;

97void frame_buffer_printf(frame_buffer_t *buf, const char *fmt, ...) {
98 if (!buf || !fmt) {
99 return;
100 }
101
102 // Ensure we have space for a formatted string
103 // Start with a reasonable buffer size
104 size_t remaining = buf->capacity - buf->len;
105 if (remaining < 256) {
106 // Grow buffer to have at least 512 bytes of space
107 size_t new_capacity = buf->capacity + 512;
108 char *new_data = SAFE_MALLOC(new_capacity, char *);
109 if (!new_data) {
110 return;
111 }
112
113 if (buf->len > 0 && buf->data) {
114 memcpy(new_data, buf->data, buf->len);
115 }
116
117 SAFE_FREE(buf->data);
118 buf->data = new_data;
119 buf->capacity = new_capacity;
120 remaining = buf->capacity - buf->len;
121 }
122
123 // Format the string into the buffer
124 va_list args;
125 va_start(args, fmt);
126 int written = vsnprintf(buf->data + buf->len, remaining, fmt, args);
127 va_end(args);
128
129 if (written > 0 && written < (int)remaining) {
130 buf->len += (size_t)written;
131 }
132 // If vsnprintf failed or truncated, we silently ignore (buffer may be full)
133}
