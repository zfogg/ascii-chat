48wav_writer_t *wav_writer_open(const char *filepath, int sample_rate, int channels) {
49 if (!filepath || sample_rate <= 0 || channels <= 0 || channels > 2) {
50 return NULL;
51 }
52
53 wav_writer_t *writer = SAFE_MALLOC(sizeof(wav_writer_t), wav_writer_t *);
54 if (!writer) {
55 return NULL;
56 }
57
58 writer->file = platform_fopen("file_stream", filepath, "wb");
59 if (!writer->file) {
60 SAFE_FREE(writer);
61 return NULL;
62 }
63
64 writer->samples_written = 0;
65 writer->sample_rate = sample_rate;
66 writer->channels = channels;
67
68 // Write WAV header (we'll update sizes when closing)
69 wav_riff_header_t riff = {.riff = {'R', 'I', 'F', 'F'},
70 .file_size = 0, // Will update on close
71 .wave = {'W', 'A', 'V', 'E'}};
72
73 wav_fmt_chunk_t fmt = {.fmt = {'f', 'm', 't', ' '},
74 .chunk_size = 16,
75 .audio_format = 3, // IEEE float
76 .num_channels = (uint16_t)channels,
77 .sample_rate = (uint32_t)sample_rate,
78 .byte_rate = (uint32_t)(sample_rate * channels * sizeof(float)),
79 .block_align = (uint16_t)(channels * sizeof(float)),
80 .bits_per_sample = 32};
81
82 wav_data_header_t data = {
83 .data = {'d', 'a', 't', 'a'},
84 .data_size = 0 // Will update on close
85 };
86
87 fwrite(&riff, sizeof(riff), 1, writer->file);
88 fwrite(&fmt, sizeof(fmt), 1, writer->file);
89 fwrite(&data, sizeof(data), 1, writer->file);

91 return writer;

94int wav_writer_write(wav_writer_t *writer, const float *samples, int num_samples) {
95 if (!writer || !writer->file || !samples || num_samples <= 0) {
96 return -1;
97 }
98
99 size_t written = fwrite(samples, sizeof(float), num_samples, writer->file);
100 if (written != (size_t)num_samples) {
101 return -1;
102 }
103
104 writer->samples_written += num_samples;
105
106 // Flush immediately to ensure data is written to disk (critical for real-time analysis)
107 fflush(writer->file);
108
109 return 0;
110}

112void wav_writer_close(wav_writer_t *writer) {
113 if (!writer) {
114 return;
115 }
116
117 if (writer->file) {
118 // Update file size and data size in header
119 uint32_t data_size = (uint32_t)(writer->samples_written * sizeof(float));
120 uint32_t file_size = data_size + 36; // 36 = header size without RIFF chunk
121
122 // Seek to file size field (offset 4)
123 fseek(writer->file, 4, SEEK_SET);
124 fwrite(&file_size, sizeof(uint32_t), 1, writer->file);
125
126 // Seek to data size field (offset 40)
127 fseek(writer->file, 40, SEEK_SET);
128 fwrite(&data_size, sizeof(uint32_t), 1, writer->file);
129
130 // Flush all buffered data to ensure headers are written before closing
131 fflush(writer->file);
132 fclose(writer->file);
133 }
134
135 SAFE_FREE(writer);
136}
