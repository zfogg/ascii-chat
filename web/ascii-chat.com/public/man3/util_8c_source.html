49asciichat_error_t platform_localtime(const time_t *timer, struct tm *result) {
50 if (!timer || !result) {
51 return ERROR_INVALID_PARAM;
52 }
53 // Use thread-safe localtime_r for both WASM and native builds
54 // IMPORTANT: localtime() is NOT thread-safe and causes deadlocks in threaded WASM builds
