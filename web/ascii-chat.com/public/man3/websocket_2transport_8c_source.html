122static void *websocket_service_thread(void *arg) {
123 websocket_transport_data_t *ws_data = (websocket_transport_data_t *)arg;
124
125 log_info(">>> SERVICE_THREAD_START: owns_context=%d, wsi=%p, context=%p, send_queue=%p", ws_data->owns_context,
126 (void *)ws_data->wsi, (void *)ws_data->context, (void *)ws_data->send_queue);
127
128 int loop_count = 0;
129 int total_messages_sent = 0;
130 uint64_t last_service_call = time_get_ns();
131
132 while (ws_data->service_running) {
133 loop_count++;
134
135 // CRITICAL: Check if we're destroying FIRST before doing anything else
136 // This prevents accessing invalid pointers or contexts
137 if (atomic_load_bool(&ws_data->is_destroying) || !ws_data->context) {
138 if (loop_count <= 10) {
139 log_info("[LOOP %d] Destroying flag set or context NULL, exiting service thread", loop_count);
140 }
141 break;
142 }
143
144 // Periodically drain pending-free queue to free buffers deferred from compression
145 // Do this at the start of each loop to ensure buffers are freed after compression completes
146 if (loop_count % 10 == 0) {
147 drain_pending_free_queue(ws_data);
148 }
149
150 // ALWAYS log first 10 loops to debug why queue checks aren't working
151 if (loop_count <= 10) {
152 log_info("[LOOP %d] owns_context=%d, wsi=%p, service_running=%d", loop_count, ws_data->owns_context,
153 (void *)ws_data->wsi, ws_data->service_running);
154 }
155
156 // Check if THIS transport (client or server) has data queued to send
157 // Note: we only check for CLIENT transports here (owns_context=true)
158 // because SERVER transports are already handled by the SERVER_WRITEABLE callback.
159 // Clients need explicit triggering via lws_callback_on_writable().
160 if (ws_data->owns_context && ws_data->wsi) {
161 if (loop_count <= 10) {
162 log_info("[LOOP %d] CLIENT condition TRUE - checking queue", loop_count);
163 }
164
165 // Wait for connection to be established before sending
166 // The CLIENT_ESTABLISHED callback sets is_connected = true
167 // If we try to send before handshake completes, lws_write() fails
168 mutex_lock(&ws_data->state_mutex);
169 bool connected = ws_data->is_connected;
170 mutex_unlock(&ws_data->state_mutex);
171
172 if (!connected) {
173 if (loop_count <= 10) {
174 log_info("[LOOP %d] CLIENT not connected yet, skipping queue drain", loop_count);
175 }
176
177 // Check if connection attempt failed (CONNECTION_ERROR was called)
178 // If so, break from service loop to avoid indefinite retry
179 mutex_lock(&ws_data->state_mutex);
180 bool failed = ws_data->connection_failed;
181 mutex_unlock(&ws_data->state_mutex);
182
183 if (failed) {
184 log_info("[LOOP %d] Connection attempt failed, exiting service thread", loop_count);
185 break;
186 }
187
188 // Sleep briefly to avoid busy-waiting
189 platform_sleep_us(10000); // 10ms
190 } else {
191 // This is a CLIENT transport - request WRITEABLE callback to send queued messages
192 // lws_write() must be called from within LWS callbacks, not from external threads
193 int messages_sent = 0;
194
195 mutex_lock(&ws_data->send_mutex);
196 bool has_data = !ringbuffer_is_empty(ws_data->send_queue);
197
198 if (loop_count <= 10) {
199 log_info("[LOOP %d] Queue check: has_data=%d", loop_count, has_data);
200 }
201
202 if (has_data) {
203 log_info(">>> SERVICE_THREAD: CLIENT queue has data, requesting CLIENT_WRITEABLE callback");
204 // Request CLIENT_WRITEABLE callback instead of calling lws_write() directly
205 // lws_write() MUST be called from within a callback context, not from external threads
206 lws_callback_on_writable(ws_data->wsi);
207 messages_sent++;
208 total_messages_sent++;
209 }
210 mutex_unlock(&ws_data->send_mutex);
211
212 if (messages_sent > 0) {
213 log_info(">>> SERVICE_BATCH: requested %d writable callbacks (total: %d)", messages_sent,
214 total_messages_sent);
215 }
216 } // End of if (connected) block
217 } else {
218 if (loop_count <= 10) {
219 log_info("[LOOP %d] CLIENT condition FALSE (owns_context=%d, wsi=%p)", loop_count, ws_data->owns_context,
220 (void *)ws_data->wsi);
221 }
222 }
223
224 // Service libwebsockets (processes network events, triggers callbacks)
225 // Call frequently (1ms) to handle both incoming AND outgoing data responsively
226 uint64_t now_ns = time_get_ns();
227 uint64_t time_since_last_service = now_ns - last_service_call;
228 int result = 0;
229
230 // Call lws_service() every 1ms for low-latency frame transmission and reception
231 if (time_since_last_service >= 1000000ULL) { // 1ms
232 last_service_call = now_ns;
233 uint64_t service_start_ns = now_ns;
234
235 bool is_destroying = atomic_load_bool(&ws_data->is_destroying);
236 if (!ws_data->context || is_destroying) {
237 // Context is invalid or being destroyed, don't call lws_service
238 if (loop_count <= 50) {
239 log_info("[LOOP %d] Skipping lws_service: context=%p, destroying=%d", loop_count, (void *)ws_data->context,
240 (int)is_destroying);
241 }
242 } else {
243 // Call lws_service with valid context
244 // Use 1ms timeout for responsive frame transmission and reception
245 result = lws_service(ws_data->context, 1);
246 }
247 uint64_t service_end_ns = time_get_ns();
248
249 if (loop_count <= 50) {
250 char service_duration_str[32];
251 time_pretty(service_end_ns - service_start_ns, -1, service_duration_str, sizeof(service_duration_str));
252 log_info("[LOOP %d] lws_service() returned %d, duration %s", loop_count, result, service_duration_str);
253 }
254
255 if (result < 0) {
256 log_fatal("🔴 lws_service error: %d at loop %d", result, loop_count);
257 break;
258 }
259 } else {
260 // Sleep briefly between lws_service() calls to avoid busy-waiting
261 platform_sleep_us(100); // 0.1ms
262 }
263
264 // Check if connection is still alive
265 if (loop_count <= 50 && ws_data->is_connected) {
266 log_info("[LOOP %d] After lws_service: is_connected=true, wsi=%p", loop_count, (void *)ws_data->wsi);
267 } else if (loop_count <= 50) {
268 log_warn("[LOOP %d] After lws_service: is_connected=false, wsi=%p", loop_count, (void *)ws_data->wsi);
269 }
270 }
271
272 // log_debug("WebSocket service thread exiting"); // Disabled: causes AddressSanitizer heap-use-after-free crash
273 return NULL;
274}

286static int websocket_callback(struct lws *wsi, enum lws_callback_reasons reason, void *user, void *in, size_t len) {
287 websocket_transport_data_t *ws_data = (websocket_transport_data_t *)user;
288
289 // Early exit during shutdown to prevent crashes in libwebsockets callbacks
290 if (ws_data && atomic_load_bool(&ws_data->is_destroying)) {
291 return 0;
292 }
293
294 switch (reason) {
295 case LWS_CALLBACK_CLIENT_ESTABLISHED: {
296 uint64_t now_ns = time_get_ns();
297 log_fatal("🟢🟢🟢 WebSocket CLIENT_ESTABLISHED! wsi=%p, ws_data=%p, timestamp=%llu", (void *)wsi, (void *)ws_data,
298 (unsigned long long)now_ns);
299 if (ws_data) {
300 mutex_lock(&ws_data->state_mutex);
301 log_fatal(" [ESTABLISHED] Setting is_connected=true (was %d)", ws_data->is_connected);
302 ws_data->is_connected = true;
303 cond_signal(&ws_data->state_cond);
304 mutex_unlock(&ws_data->state_mutex);
305 log_fatal(" [ESTABLISHED] State updated, wsi=%p ready for send/recv", (void *)wsi);
306 }
307 break;
308 }
309
310 case LWS_CALLBACK_CLIENT_RECEIVE: {
311 // Received data from server - may be fragmented for large messages
312 uint64_t now_ns = time_get_ns();
313 if (!ws_data || !in || len == 0) {
314 log_debug("CLIENT_RECEIVE: ws_data=%p, in=%p, len=%zu - skipping", (void *)ws_data, in, len);
315 break;
316 }
317
318 // Properly detect first/final bits from WebSocket frame
319 // lws_remaining_packet_payload() returns 0 when we're at the final frame
320 bool is_first = 1; // First frame of a fragmented message (safe to assume - we get callbacks per frame)
321 bool is_final = (lws_remaining_packet_payload(wsi) == 0); // Final frame when no more payload remains
322
323 log_info("🟡 LWS_CALLBACK_CLIENT_RECEIVE: %zu bytes (first=%d, final=%d), wsi=%p, timestamp=%llu", len, is_first,
324 is_final, (void *)wsi, (unsigned long long)now_ns);
325
326 // Queue this fragment immediately with first/final flags.
327 // Per LWS design, each fragment is processed individually by the callback.
328 // We must NOT manually reassemble fragments - that breaks LWS's internal state machine.
329 // Instead, queue each fragment with metadata, and let the receiver decide on reassembly.
330
331 websocket_recv_msg_t msg;
332 msg.data = buffer_pool_alloc(NULL, len);
333 if (!msg.data) {
334 log_error("Failed to allocate buffer for fragment (%zu bytes)", len);
335 break;
336 }
337
338 memcpy(msg.data, in, len);
339 msg.len = len;
340 msg.first = is_first;
341 msg.final = is_final;
342
343 mutex_lock(&ws_data->recv_mutex);
344 bool success = ringbuffer_write(ws_data->recv_queue, &msg);
345 if (!success) {
346 // Queue is full - drop the fragment and log warning
347 log_warn("WebSocket receive queue full - dropping fragment (len=%zu, first=%d, final=%d)", len, is_first,
348 is_final);
349 buffer_pool_free(NULL, msg.data, msg.len);
350 mutex_unlock(&ws_data->recv_mutex);
351 break;
352 }
353
354 // Signal waiting recv() call that a fragment is available
355 cond_signal(&ws_data->recv_cond);
356 mutex_unlock(&ws_data->recv_mutex);
357 break;
358 }
359
360 case LWS_CALLBACK_CLIENT_CLOSED:
361 case LWS_CALLBACK_CLOSED: {
362 uint64_t now_ns = time_get_ns();
363 log_fatal("🔴🔴🔴 WebSocket connection CLOSED! reason=%d, wsi=%p, ws_data=%p, is_connected=%d, timestamp=%llu",
364 reason, (void *)wsi, (void *)ws_data, ws_data ? ws_data->is_connected : -1, (unsigned long long)now_ns);
365 if (ws_data) {
366 mutex_lock(&ws_data->state_mutex);
367 log_fatal(" [CLOSE] Setting is_connected=false (was %d)", ws_data->is_connected);
368 ws_data->is_connected = false;
369 mutex_unlock(&ws_data->state_mutex);
370
371 // Wake any blocking recv() calls
372 cond_broadcast(&ws_data->recv_cond);
373 }
374 break;
375 }
376
377 case LWS_CALLBACK_CLIENT_CONNECTION_ERROR: {
378 uint64_t now_ns = time_get_ns();
379 log_fatal("🔴🔴🔴 WebSocket CONNECTION ERROR! reason=%d, error=%s, wsi=%p, ws_data=%p, timestamp=%llu", reason,
380 in ? (const char *)in : "unknown", (void *)wsi, (void *)ws_data, (unsigned long long)now_ns);
381 if (ws_data) {
382 mutex_lock(&ws_data->state_mutex);
383 ws_data->is_connected = false;
384 ws_data->connection_failed = true; // Signal service thread to exit
385 cond_signal(&ws_data->state_cond); // Wake anyone waiting on connection
386 mutex_unlock(&ws_data->state_mutex);
387
388 // Wake any blocking recv() calls
389 cond_broadcast(&ws_data->recv_cond);
390 }
391 break;
392 }
393
394 case LWS_CALLBACK_CLIENT_WRITEABLE: {
395 // Socket is writable - process queued messages with FRAGMENTATION
396 // CRITICAL FIX: Fragment large messages into ~4KB chunks to avoid internal buffering
397 // libwebsockets #464: Sending messages > rx_buffer_size causes ultra-slow buffering
398 uint64_t now_ns = time_get_ns();
399 log_info("🟡 LWS_CALLBACK_CLIENT_WRITEABLE FIRED for wsi=%p, ws_data=%p, is_connected=%d, timestamp=%llu",
400 (void *)wsi, (void *)ws_data, ws_data ? ws_data->is_connected : -1, (unsigned long long)now_ns);
401 if (!ws_data) {
402 log_warn(" [CLIENT_WRITEABLE] ws_data is NULL, breaking");
403 break;
404 }
405
406 // CRITICAL: Don't try to write if not fully connected
407 // libwebsockets can hit assertion if we write before connection establishment completes
408 mutex_lock(&ws_data->state_mutex);
409 bool connected = ws_data->is_connected;
410 mutex_unlock(&ws_data->state_mutex);
411
412 if (!connected) {
413 log_debug(" [CLIENT_WRITEABLE] Skipping write - not connected yet");
414 break;
415 }
416
417 websocket_recv_msg_t msg;
418 int message_count = 0;
419
420 mutex_lock(&ws_data->send_mutex);
421
422 // Process all messages in queue until pipe is choked or queue is empty
423 // This ensures remainders are sent before new messages
424 while (ringbuffer_peek(ws_data->send_queue, &msg)) {
425 // Check pipe state BEFORE dequeuing
426 if (lws_send_pipe_choked(ws_data->wsi)) {
427 log_debug("Pipe choked, stopping message processing");
428 break;
429 }
430 ringbuffer_read(ws_data->send_queue, &msg);
431
432 log_debug("WebSocket CLIENT_WRITEABLE: sending queued %zu bytes in fragments (msg %d)", msg.len,
433 message_count + 1);
434
435 // CRITICAL: Let libwebsockets handle fragmentation internally
436 // We send the entire message at once with LWS_WRITE_BINARY
437 // libwebsockets automatically fragments based on its MTU and manages FIN bits
438 log_debug("WebSocket sending %zu bytes (first=%d) with LWS_WRITE_BINARY", msg.len, msg.first);
439
440 int result = lws_write(ws_data->wsi, msg.data + LWS_PRE, msg.len, LWS_WRITE_BINARY);
441
442 if (result < 0) {
443 // Error - pipe is full or other error
444 log_debug("WebSocket write returned %d (pipe choked), re-queueing", result);
445 ringbuffer_write(ws_data->send_queue, &msg);
446 break; // Stop processing, wait for next writeable callback
447 } else {
448 // Success - lws_write handles ALL fragmentation internally
449 // The entire message is queued for sending (fragmented internally by libwebsockets)
450 deferred_buffer_free(ws_data, msg.data, LWS_PRE + msg.len);
451 message_count++;
452 log_debug("Message queued for send, will be fragmented by libwebsockets");
453 }
454 } // End while loop - continues until queue empty or pipe choked
455
456 mutex_unlock(&ws_data->send_mutex);
457
458 // Request another callback if more messages are queued
459 // Must request even if pipe was choked so we drain queue once TCP buffer drains
460 mutex_lock(&ws_data->send_mutex);
461 bool has_more = ringbuffer_peek(ws_data->send_queue, &msg);
462 mutex_unlock(&ws_data->send_mutex);
463
464 if (has_more) {
465 lws_callback_on_writable(ws_data->wsi);
466 }
467 break;
468 }
469
470 default:
471 break;
472 }
473
474 return 0;
475}

481static asciichat_error_t websocket_send(acip_transport_t *transport, const void *data, size_t len) {
482 START_TIMER("ws_send_total");
483 websocket_transport_data_t *ws_data = (websocket_transport_data_t *)transport->impl_data;
484
485 // For server-side transports (owns_context=false), the connection is already established
486 // For client-side transports (owns_context=true), wait for connection to be established
487 // before sending, instead of failing immediately. This allows sends to work even if
488 // they're called before the service thread has fully established the connection.
489 if (ws_data->owns_context) {
490 // Wait for connection with same timeout as recv()
491 const uint64_t CONNECT_WAIT_TIMEOUT_NS = 100 * 1000000ULL; // 100ms per wait iteration
492 uint64_t wait_start_ns = time_get_ns();
493
494 mutex_lock(&ws_data->state_mutex);
495 while (!ws_data->is_connected && !ws_data->connection_failed) {
496 uint64_t elapsed_ns = time_get_ns() - wait_start_ns;
497 if (elapsed_ns > 30 * 1000000000ULL) { // 30 second total timeout
498 log_error("[WEBSOCKET_SEND] Connection timeout after 30 seconds, cannot send");
499 mutex_unlock(&ws_data->state_mutex);
500 return SET_ERRNO(ERROR_NETWORK, "WebSocket connection timeout");
501 }
502 // Wait for connection with timeout
503 cond_timedwait(&ws_data->state_cond, &ws_data->state_mutex, CONNECT_WAIT_TIMEOUT_NS);
504 }
505 bool connected = ws_data->is_connected;
506 bool connection_failed = ws_data->connection_failed;
507 mutex_unlock(&ws_data->state_mutex);
508
509 if (connection_failed && !connected) {
510 log_error("[WEBSOCKET_SEND] Connection failed during establishment");
511 return SET_ERRNO(ERROR_NETWORK, "WebSocket connection failed");
512 }
513
514 log_dev_every(1000000, "websocket_send (client): is_connected=%d, wsi=%p, send_len=%zu", connected,
515 (void *)ws_data->wsi, len);
516 } else {
517 log_info("[WEBSOCKET_SEND_SERVER] ★★★ Server transport send: wsi=%p, len=%zu (bypassing is_connected check)",
518 (void *)ws_data->wsi, len);
519 }
520
521 // Check if encryption is needed (matching tcp_send logic)
522 const void *send_data = data;
523 size_t send_len = len;
524 uint8_t *encrypted_packet = NULL;
525 size_t encrypted_packet_size = 0;
526
527 if (len >= sizeof(packet_header_t) && transport->crypto_ctx && crypto_is_ready(transport->crypto_ctx)) {
528 const packet_header_t *header = (const packet_header_t *)data;
529 uint16_t packet_type = NET_TO_HOST_U16(header->type);
530
531 if (!packet_is_handshake_type((packet_type_t)packet_type)) {
532 // Encrypt the entire packet (header + payload)
533 size_t ciphertext_size = len + CRYPTO_NONCE_SIZE + CRYPTO_MAC_SIZE;
534 // Use SAFE_MALLOC (not buffer pool - encrypted_packet also uses pool and causes overlap)
535 uint8_t *ciphertext = SAFE_MALLOC(ciphertext_size, uint8_t *);
536 if (!ciphertext) {
537 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate ciphertext buffer for WebSocket");
538 }
539
540 size_t ciphertext_len;
541 START_TIMER("ws_encrypt_packet");
542 crypto_result_t result =
543 crypto_encrypt(transport->crypto_ctx, data, len, ciphertext, ciphertext_size, &ciphertext_len);
544 double _ws_encrypt_elapsed = STOP_TIMER("ws_encrypt_packet");
545 if (_ws_encrypt_elapsed >= 0.0) {
546 char _ws_encrypt_duration[32];
547 time_pretty((uint64_t)_ws_encrypt_elapsed, -1, _ws_encrypt_duration, sizeof(_ws_encrypt_duration));
548 log_info("[WS] encrypt %zu bytes in %s", len, _ws_encrypt_duration);
549 }
550 if (result != CRYPTO_OK) {
551 SAFE_FREE(ciphertext);
552 return SET_ERRNO(ERROR_CRYPTO, "Failed to encrypt WebSocket packet: %s", crypto_result_to_string(result));
553 }
554
555 // Build PACKET_TYPE_ENCRYPTED wrapper: header + ciphertext
556 size_t total_encrypted_size = sizeof(packet_header_t) + ciphertext_len;
557 encrypted_packet = buffer_pool_alloc(NULL, total_encrypted_size);
558 if (!encrypted_packet) {
559 SAFE_FREE(ciphertext);
560 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate encrypted packet buffer");
561 }
562
563 packet_header_t encrypted_header;
564 encrypted_header.magic = HOST_TO_NET_U64(PACKET_MAGIC);
565 encrypted_header.type = HOST_TO_NET_U16(PACKET_TYPE_ENCRYPTED);
566 encrypted_header.length = HOST_TO_NET_U32((uint32_t)ciphertext_len);
567 encrypted_header.crc32 = HOST_TO_NET_U32(asciichat_crc32(ciphertext, ciphertext_len));
568 encrypted_header.client_id = 0;
569
570 memcpy(encrypted_packet, &encrypted_header, sizeof(encrypted_header));
571 memcpy(encrypted_packet + sizeof(encrypted_header), ciphertext, ciphertext_len);
572 SAFE_FREE(ciphertext);
573
574 send_data = encrypted_packet;
575 send_len = total_encrypted_size;
576 encrypted_packet_size = total_encrypted_size;
577
578 log_dev_every(1000000, "WebSocket: encrypted packet (original type %d as PACKET_TYPE_ENCRYPTED, %zu bytes)",
579 packet_type, send_len);
580 }
581 }
582
583 // libwebsockets requires LWS_PRE bytes before the payload for protocol headers
584 // Allocate a temporary buffer for this send to avoid thread-safety issues
585 // Each send() call gets its own buffer, preventing race conditions with concurrent sends
586 size_t required_size = LWS_PRE + send_len;
587 uint8_t *send_buffer = SAFE_MALLOC(required_size, uint8_t *);
588 if (!send_buffer) {
589 if (encrypted_packet)
590 buffer_pool_free(NULL, encrypted_packet, encrypted_packet_size);
591 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate WebSocket send buffer");
592 }
593
594 // Copy data after LWS_PRE offset
595 memcpy(send_buffer + LWS_PRE, send_data, send_len);
596
597 // Server-side transports cannot call lws_write() directly
598 // They must queue data and send from LWS_CALLBACK_SERVER_WRITEABLE
599 if (!ws_data->owns_context) {
600 // Queue the data for server-side sending
601 // IMPORTANT: Allocate with LWS_PRE padding because lws_write() needs to write
602 // the WebSocket frame header backwards into the LWS_PRE region
603 // Use buffer_pool instead of SAFE_MALLOC to avoid use-after-free with
604 // permessage-deflate compression (same issue as client-side sends)
605 websocket_recv_msg_t msg;
606 size_t buffer_size = LWS_PRE + send_len;
607 msg.data = buffer_pool_alloc(NULL, buffer_size);
608 if (!msg.data) {
609 SAFE_FREE(send_buffer);
610 if (encrypted_packet)
611 buffer_pool_free(NULL, encrypted_packet, send_len);
612 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate send queue buffer");
613 }
614 // Copy data AFTER the LWS_PRE region
615 memcpy(msg.data + LWS_PRE, send_data, send_len);
616 msg.len = send_len;
617 msg.first = 1;
618 msg.final = 1;
619
620 mutex_lock(&ws_data->send_mutex);
621 bool success = ringbuffer_write(ws_data->send_queue, &msg);
622
623 if (!success) {
624 mutex_unlock(&ws_data->send_mutex);
625 log_error("WebSocket server send queue FULL - cannot queue %zu bytes (queue size=%d)", send_len,
626 WEBSOCKET_SEND_QUEUE_SIZE);
627 buffer_pool_free(NULL, msg.data, buffer_size);
628 SAFE_FREE(send_buffer);
629 if (encrypted_packet)
630 buffer_pool_free(NULL, encrypted_packet, encrypted_packet_size);
631 return SET_ERRNO(ERROR_NETWORK, "Send queue full (cannot queue %zu bytes)", send_len);
632 }
633 mutex_unlock(&ws_data->send_mutex);
634
635 log_debug(">>> SERVER FRAME QUEUED: %zu bytes for wsi=%p", send_len, (void *)ws_data->wsi);
636
637 // Notify libwebsockets that there's data to send - triggers LWS_CALLBACK_SERVER_WRITEABLE
638 // This is essential! Without this, the queued frames never get transmitted.
639 START_TIMER("ws_callback");
640 lws_callback_on_writable(ws_data->wsi);
641 STOP_TIMER_AND_LOG(info, 0, "ws_callback", "[WEBSOCKET] lws_callback_on_writable");
642 log_debug(">>> REQUESTED SERVER_WRITEABLE CALLBACK for wsi=%p", (void *)ws_data->wsi);
643
644 SAFE_FREE(send_buffer);
645 if (encrypted_packet)
646 buffer_pool_free(NULL, encrypted_packet, encrypted_packet_size);
647 double _ws_send_server_elapsed = STOP_TIMER("ws_send_total");
648 if (_ws_send_server_elapsed >= 0.0) {
649 char _ws_send_server_duration[32];
650 time_pretty((uint64_t)_ws_send_server_elapsed, -1, _ws_send_server_duration, sizeof(_ws_send_server_duration));
651 log_info("[WS_TOTAL] send_server %zu bytes in %s", len, _ws_send_server_duration);
652 }
653 return ASCIICHAT_OK;
654 }
655
656 // Client-side: queue entire message for service thread to send
657 // Avoid fragmentation race conditions by sending atomic messages from service thread
658 // libwebsockets handles automatic fragmentation internally when needed
659 // Allocate with LWS_PRE padding because lws_write() needs to write
660 // the WebSocket frame header backwards into the LWS_PRE region
661 // Use buffer_pool instead of SAFE_MALLOC to avoid use-after-free with permessage-deflate
662 // compression. libwebsockets may buffer this data asynchronously after lws_write() returns,
663 // so we can't free it immediately. buffer_pool_free() safely returns it to the pool instead
664 // of deallocating, preventing the use-after-free race condition with the compression layer.
665 websocket_recv_msg_t msg;
666 size_t buffer_size = LWS_PRE + send_len;
667 msg.data = buffer_pool_alloc(NULL, buffer_size);
668 if (!msg.data) {
669 SAFE_FREE(send_buffer);
670 if (encrypted_packet)
671 buffer_pool_free(NULL, encrypted_packet, encrypted_packet_size);
672 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate client send queue buffer");
673 }
674 // Copy data AFTER the LWS_PRE region
675 memcpy(msg.data + LWS_PRE, send_data, send_len);
676 msg.len = send_len;
677 msg.first = 1;
678 msg.final = 1;
679
680 mutex_lock(&ws_data->send_mutex);
681 bool success = ringbuffer_write(ws_data->send_queue, &msg);
682
683 if (!success) {
684 mutex_unlock(&ws_data->send_mutex);
685 log_error("WebSocket client send queue FULL - cannot queue %zu bytes (queue size=%d)", send_len,
686 WEBSOCKET_SEND_QUEUE_SIZE);
687 buffer_pool_free(NULL, msg.data, buffer_size);
688 SAFE_FREE(send_buffer);
689 if (encrypted_packet)
690 buffer_pool_free(NULL, encrypted_packet, encrypted_packet_size);
691 return SET_ERRNO(ERROR_NETWORK, "Client send queue full (cannot queue %zu bytes)", send_len);
692 }
693 mutex_unlock(&ws_data->send_mutex);
694
695 log_debug(">>> QUEUED CLIENT MESSAGE: %zu bytes queued at %p for service thread (wsi=%p)", send_len,
696 (void *)ws_data->wsi, (void *)ws_data->wsi);
697
698 log_dev_every(1000000, "WebSocket client: queued %zu bytes for service thread to send", send_len);
699 SAFE_FREE(send_buffer);
700 if (encrypted_packet)
701 buffer_pool_free(NULL, encrypted_packet, encrypted_packet_size);
702 double _ws_send_client_elapsed = STOP_TIMER("ws_send_total");
703 if (_ws_send_client_elapsed >= 0.0) {
704 char _ws_send_client_duration[32];
705 time_pretty((uint64_t)_ws_send_client_elapsed, -1, _ws_send_client_duration, sizeof(_ws_send_client_duration));
706 log_info("[WS_TOTAL] send_client %zu bytes in %s", len, _ws_send_client_duration);
707 }
708 return ASCIICHAT_OK;
709}

711static asciichat_error_t websocket_recv(acip_transport_t *transport, void **buffer, size_t *out_len,
712 void **out_allocated_buffer) {
713 websocket_transport_data_t *ws_data = (websocket_transport_data_t *)transport->impl_data;
714
715 // Wait for connection to be established (service thread may still be connecting)
716 // The service thread creates the transport immediately and returns, allowing the main
717 // thread to respond to input, but the connection may not be established yet.
718 // We need to wait for it with a timeout to avoid hanging forever.
719 const uint64_t CONNECT_WAIT_TIMEOUT_NS = 100 * 1000000ULL; // 100ms per wait iteration
720 uint64_t wait_start_ns = time_get_ns();
721
722 mutex_lock(&ws_data->state_mutex);
723 while (!ws_data->is_connected && !ws_data->connection_failed) {
724 uint64_t elapsed_ns = time_get_ns() - wait_start_ns;
725 if (elapsed_ns > 30 * 1000000000ULL) { // 30 second total timeout
726 log_error("🔴 WEBSOCKET_RECV: Connection timeout after 30 seconds, connection_failed=%d",
727 ws_data->connection_failed);
728 mutex_unlock(&ws_data->state_mutex);
729 return SET_ERRNO(ERROR_NETWORK, "WebSocket connection timeout");
730 }
731 // Wait for connection with timeout
732 cond_timedwait(&ws_data->state_cond, &ws_data->state_mutex, CONNECT_WAIT_TIMEOUT_NS);
733 }
734 bool connected = ws_data->is_connected;
735 bool connection_failed = ws_data->connection_failed;
736 mutex_unlock(&ws_data->state_mutex);
737
738 if (connection_failed && !connected) {
739 log_error("🔴 WEBSOCKET_RECV: Connection failed during establishment");
740 return SET_ERRNO(ERROR_NETWORK, "WebSocket connection failed");
741 }
742
743 mutex_lock(&ws_data->recv_mutex);
744
745 // Even if connection is closed, we should try to deliver any buffered data
746 // Check if there's data in the queue
747 bool has_queued_data = !ringbuffer_is_empty(ws_data->recv_queue);
748
749 if (!connected && !has_queued_data && ws_data->partial_size == 0) {
750 // Only fail if connection is closed AND no buffered data AND no leftover from previous call
751 uint64_t now_ns = time_get_ns();
752 log_fatal("🔴 WEBSOCKET_RECV: Connection closed! connected=%d, has_queued_data=%d, partial_size=%zu, wsi=%p, "
753 "timestamp=%llu",
754 connected, has_queued_data, ws_data->partial_size, (void *)ws_data->wsi, (unsigned long long)now_ns);
755 mutex_unlock(&ws_data->recv_mutex);
756 return SET_ERRNO(ERROR_NETWORK, "Connection closed");
757 }
758
759 // Reassemble fragmented WebSocket messages with SHORT timeout
760 // We queue each fragment from the LWS callback with first/final flags.
761 // Key insight: if we wait too long for final fragment, connection times out.
762 // Return partial messages quickly (500ms) to avoid blocking handler thread.
763 //
764 // CRITICAL FIX: Use persistent partial_buffer to handle packet boundaries
765 // that don't align with WebSocket frame boundaries. This prevents data loss
766 // when a single WebSocket message contains multiple ACIP packets.
767
768 uint8_t *assembled_buffer = NULL;
769 size_t assembled_size = 0;
770 size_t assembled_capacity = 0;
771 int fragment_count = 0;
772
773 // Start with leftover data from previous recv() call if available
774 if (ws_data->partial_size > 0) {
775 log_info("[WS_REASSEMBLE] Starting with leftover data from previous packet: %zu bytes", ws_data->partial_size);
776 assembled_buffer = ws_data->partial_buffer;
777 assembled_size = ws_data->partial_size;
778 assembled_capacity = ws_data->partial_capacity;
779
780 // Clear the partial buffer state so next call starts fresh if no more leftovers
781 ws_data->partial_buffer = NULL;
782 ws_data->partial_size = 0;
783 ws_data->partial_capacity = 0;
784 }
785
786 while (true) {
787 // Wait for fragment if queue is empty
788 while (ringbuffer_is_empty(ws_data->recv_queue)) {
789 // Don't timeout waiting for fragments. If the connection is dead, libwebsockets will
790 // set is_connected=false and we'll detect it below. Arbitrary timeouts just cause
791 // unnecessary disconnections while data is actually being transmitted.
792
793 // Check connection state - but don't fail immediately if connection closes
794 // while reassembling. Instead, return what we have so the handler can process it.
795 // This prevents losing buffered data due to connection timeouts.
796 mutex_lock(&ws_data->state_mutex);
797 bool still_connected = ws_data->is_connected;
798 mutex_unlock(&ws_data->state_mutex);
799
800 if (!still_connected && assembled_size > 0) {
801 // Connection closed but we have partial data - return it
802 log_info("[WS_REASSEMBLE] Connection closed mid-reassembly, returning %zu bytes received so far",
803 assembled_size);
804 *buffer = assembled_buffer;
805 *out_len = assembled_size;
806 *out_allocated_buffer = assembled_buffer;
807 mutex_unlock(&ws_data->recv_mutex);
808 return ASCIICHAT_OK;
809 }
810
811 if (!still_connected && assembled_size == 0) {
812 // Connection closed and no data yet
813 if (assembled_buffer) {
814 buffer_pool_free(NULL, assembled_buffer, assembled_capacity);
815 }
816 mutex_unlock(&ws_data->recv_mutex);
817 return SET_ERRNO(ERROR_NETWORK, "Connection closed");
818 }
819
820 // Wait for next fragment with 1ms timeout to allow signals to interrupt
821 cond_timedwait(&ws_data->recv_cond, &ws_data->recv_mutex, 1 * 1000000ULL);
822 }
823
824 // Read next fragment from queue
825 websocket_recv_msg_t frag;
826 bool success = ringbuffer_read(ws_data->recv_queue, &frag);
827 if (!success) {
828 if (assembled_buffer) {
829 buffer_pool_free(NULL, assembled_buffer, assembled_capacity);
830 }
831 mutex_unlock(&ws_data->recv_mutex);
832 return SET_ERRNO(ERROR_NETWORK, "Failed to read fragment from queue");
833 }
834
835 fragment_count++;
836 if (frag.len > 100 || fragment_count == 1) {
837 log_info("[WS_REASSEMBLE] Fragment #%d: %zu bytes, first=%d, final=%d, assembled_so_far=%zu", fragment_count,
838 frag.len, frag.first, frag.final, assembled_size);
839 } else {
840 log_debug("[WS_REASSEMBLE] Fragment #%d: %zu bytes, first=%d, final=%d", fragment_count, frag.len, frag.first,
841 frag.final);
842 }
843
844 // Sanity check: first fragment must have first=1, continuations must have first=0
845 if (assembled_size == 0 && !frag.first) {
846 log_error("[WS_REASSEMBLE] ERROR: Expected first=1 for first fragment, got first=%d", frag.first);
847 buffer_pool_free(NULL, frag.data, frag.len);
848 if (assembled_buffer) {
849 buffer_pool_free(NULL, assembled_buffer, assembled_capacity);
850 }
851 mutex_unlock(&ws_data->recv_mutex);
852 return SET_ERRNO(ERROR_NETWORK, "Protocol error: continuation fragment without first fragment");
853 }
854
855 // Grow assembled buffer if needed
856 size_t required_size = assembled_size + frag.len;
857 if (required_size > assembled_capacity) {
858 // Start with 8KB, grow by 1.5x, but cap at 4MB to prevent unbounded allocation
859 const size_t MAX_REASSEMBLY_SIZE = (4 * 1024 * 1024); // 4MB limit
860
861 size_t new_capacity = (assembled_capacity == 0) ? 8192 : (assembled_capacity * 3 / 2);
862 if (new_capacity < required_size) {
863 new_capacity = required_size;
864 }
865
866 // Enforce maximum reassembly buffer size
867 if (new_capacity > MAX_REASSEMBLY_SIZE) {
868 log_error("[WS_REASSEMBLE] Frame too large: need %zu bytes, max allowed is %zu", new_capacity,
869 MAX_REASSEMBLY_SIZE);
870 buffer_pool_free(NULL, frag.data, frag.len);
871 if (assembled_buffer) {
872 buffer_pool_free(NULL, assembled_buffer, assembled_capacity);
873 }
874 mutex_unlock(&ws_data->recv_mutex);
875 return SET_ERRNO(ERROR_NETWORK, "WebSocket frame exceeds maximum size (4MB)");
876 }
877
878 uint8_t *new_buffer = buffer_pool_alloc(NULL, new_capacity);
879 if (!new_buffer) {
880 log_error("[WS_REASSEMBLE] Failed to allocate reassembly buffer (%zu bytes)", new_capacity);
881 buffer_pool_free(NULL, frag.data, frag.len);
882 if (assembled_buffer) {
883 buffer_pool_free(NULL, assembled_buffer, assembled_capacity);
884 }
885 mutex_unlock(&ws_data->recv_mutex);
886 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate fragment reassembly buffer");
887 }
888
889 // Copy existing data to new buffer
890 if (assembled_size > 0) {
891 memcpy(new_buffer, assembled_buffer, assembled_size);
892 }
893
894 // Free old buffer
895 if (assembled_buffer) {
896 buffer_pool_free(NULL, assembled_buffer, assembled_capacity);
897 }
898
899 assembled_buffer = new_buffer;
900 assembled_capacity = new_capacity;
901 }
902
903 // Append fragment data with bounds checking
904 if (frag.len > 0 && frag.data) {
905 // Safety check: ensure we don't overflow the buffer
906 if (assembled_size + frag.len > assembled_capacity) {
907 log_error("[WS_REASSEMBLE] CRITICAL: Buffer overflow detected! assembled_size=%zu, frag.len=%zu, capacity=%zu",
908 assembled_size, frag.len, assembled_capacity);
909 buffer_pool_free(NULL, frag.data, frag.len);
910 if (assembled_buffer) {
911 buffer_pool_free(NULL, assembled_buffer, assembled_capacity);
912 }
913 mutex_unlock(&ws_data->recv_mutex);
914 return SET_ERRNO(ERROR_MEMORY, "Fragment reassembly buffer overflow");
915 }
916
917 memcpy(assembled_buffer + assembled_size, frag.data, frag.len);
918 assembled_size += frag.len;
919 }
920
921 // Free fragment data after copying (allocated in LWS callback with buffer_pool_alloc)
922 if (frag.data) {
923 // Use the fragment length that was stored when we allocated it
924 // This must match the size passed to buffer_pool_alloc in the LWS callback
925 buffer_pool_free(NULL, frag.data, frag.len);
926 frag.data = NULL; // Prevent accidental double-free
927 }
928
929 // Try to detect packet boundary using protocol structure
930 // ACIP packet header: magic(8) + type(2) + length(4) + crc(4) + client_id(4) = 22 bytes
931 // The length field (bytes 10-13) tells us the payload size
932 if (assembled_size >= 14) { // Need at least 14 bytes to read length field at offset 10
933 const uint8_t *data = assembled_buffer;
934
935 // Parse length field at offset 10 (4 bytes, big-endian)
936 uint32_t msg_payload_len = (data[10] << 24) | (data[11] << 16) | (data[12] << 8) | data[13];
937
938 // Sanity check: payload length should be reasonable (< 5MB)
939 if (msg_payload_len > 0 && msg_payload_len <= 5 * 1024 * 1024) {
940 // Total packet size = header(14 bytes up to and including length field) + payload
941 // Actually full header is 22 bytes but we can work with just the length field
942 const size_t HEADER_SIZE = 22;
943 size_t expected_size = HEADER_SIZE + msg_payload_len;
944
945 if (assembled_size >= expected_size) {
946 // Complete packet assembled based on header length field
947 log_info("[WS_REASSEMBLE] Complete message by length field: %zu bytes in %d fragments (payload=%u)",
948 expected_size, fragment_count, msg_payload_len);
949
950 // Check for leftover data after this packet
951 size_t leftover_size = assembled_size - expected_size;
952 if (leftover_size > 0) {
953 // Save leftover data for next recv() call
954 log_info("[WS_REASSEMBLE] Saving %zu bytes leftover for next recv() call", leftover_size);
955
956 // Create buffer for leftover (or reuse if we can)
957 uint8_t *leftover_buffer = buffer_pool_alloc(NULL, leftover_size);
958 if (leftover_buffer) {
959 memcpy(leftover_buffer, assembled_buffer + expected_size, leftover_size);
960 ws_data->partial_buffer = leftover_buffer;
961 ws_data->partial_size = leftover_size;
962 ws_data->partial_capacity = leftover_size;
963 } else {
964 log_error("[WS_REASSEMBLE] Failed to allocate leftover buffer (%zu bytes)", leftover_size);
965 // Fall through - return the packet anyway, we'll lose the leftover
966 }
967 }
968
969 // Create a new buffer with just the complete packet (not the leftover)
970 uint8_t *packet_buffer = buffer_pool_alloc(NULL, expected_size);
971 if (!packet_buffer) {
972 log_error("[WS_REASSEMBLE] Failed to allocate packet buffer (%zu bytes)", expected_size);
973 buffer_pool_free(NULL, assembled_buffer, assembled_capacity);
974 mutex_unlock(&ws_data->recv_mutex);
975 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate packet buffer");
976 }
977
978 memcpy(packet_buffer, assembled_buffer, expected_size);
979 buffer_pool_free(NULL, assembled_buffer, assembled_capacity);
980
981 *buffer = packet_buffer;
982 *out_len = expected_size;
983 *out_allocated_buffer = packet_buffer;
984 mutex_unlock(&ws_data->recv_mutex);
985 return ASCIICHAT_OK;
986 }
987 // Need more fragments to complete this message
988 }
989 }
990
991 // Fallback: check if we have the final WebSocket fragment
992 // BUT: Only return if we have a COMPLETE ACIP packet!
993 // ACIP packets can span multiple WebSocket frames, so final=true doesn't mean
994 // we have the complete ACIP packet - we must verify using the ACIP header's length field
995 if (frag.final) {
996 // Check if this is actually a complete ACIP packet by validating the length field
997 if (assembled_size >= 14) { // Minimum to read ACIP length field
998 const uint8_t *data = (const uint8_t *)assembled_buffer;
999 uint32_t msg_payload_len = (data[10] << 24) | (data[11] << 16) | (data[12] << 8) | data[13];
1000 const size_t HEADER_SIZE = 22;
1001 size_t expected_size = HEADER_SIZE + msg_payload_len;
1002
1003 if (assembled_size >= expected_size) {
1004 // We have a complete ACIP packet
1005 log_info("[WS_REASSEMBLE] Complete ACIP packet by WebSocket final fragment: %zu bytes in %d fragments",
1006 expected_size, fragment_count);
1007
1008 // Check for leftover data after this packet
1009 size_t leftover_size = assembled_size - expected_size;
1010 if (leftover_size > 0) {
1011 // Save leftover data for next recv() call
1012 log_info("[WS_REASSEMBLE] Saving %zu bytes leftover for next recv() call", leftover_size);
1013
1014 // Create buffer for leftover
1015 uint8_t *leftover_buffer = buffer_pool_alloc(NULL, leftover_size);
1016 if (leftover_buffer) {
1017 memcpy(leftover_buffer, assembled_buffer + expected_size, leftover_size);
1018 ws_data->partial_buffer = leftover_buffer;
1019 ws_data->partial_size = leftover_size;
1020 ws_data->partial_capacity = leftover_size;
1021 } else {
1022 log_error("[WS_REASSEMBLE] Failed to allocate leftover buffer (%zu bytes)", leftover_size);
1023 }
1024 }
1025
1026 // Create a new buffer with just the complete packet
1027 uint8_t *packet_buffer = buffer_pool_alloc(NULL, expected_size);
1028 if (!packet_buffer) {
1029 log_error("[WS_REASSEMBLE] Failed to allocate packet buffer (%zu bytes)", expected_size);
1030 buffer_pool_free(NULL, assembled_buffer, assembled_capacity);
1031 mutex_unlock(&ws_data->recv_mutex);
1032 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate packet buffer");
1033 }
1034
1035 memcpy(packet_buffer, assembled_buffer, expected_size);
1036 buffer_pool_free(NULL, assembled_buffer, assembled_capacity);
1037
1038 *buffer = packet_buffer;
1039 *out_len = expected_size;
1040 *out_allocated_buffer = packet_buffer;
1041 mutex_unlock(&ws_data->recv_mutex);
1042 return ASCIICHAT_OK;
1043 } else {
1044 // ACIP packet is incomplete even though WebSocket frame is final
1045 // This shouldn't happen - WebSocket is delivering corrupt data
1046 log_error("[WS_REASSEMBLE] ERROR: WebSocket final fragment but incomplete ACIP packet (have %zu, need %zu)",
1047 assembled_size, expected_size);
1048 buffer_pool_free(NULL, assembled_buffer, assembled_capacity);
1049 mutex_unlock(&ws_data->recv_mutex);
1050 return SET_ERRNO(ERROR_NETWORK, "WebSocket final fragment but incomplete ACIP packet");
1051 }
1052 }
1053 }
1054
1055 // More fragments coming, continue reassembling
1056 }
1057
1058 // Unreachable unless while loop breaks without returning
1059 mutex_unlock(&ws_data->recv_mutex);
1060 return SET_ERRNO(ERROR_NETWORK, "Unexpected exit from fragment reassembly loop");
1061}

1063static asciichat_error_t websocket_close(acip_transport_t *transport) {
1064 websocket_transport_data_t *ws_data = (websocket_transport_data_t *)transport->impl_data;
1065 uint64_t now_ns = time_get_ns();
1066
1067 // CRITICAL: Stop service thread BEFORE calling lws_close_reason()
1068 // Mark as destroying FIRST to signal service thread to stop immediately
1069 // This prevents service thread from trying to call lws_service on a destroying/destroyed context
1070 atomic_store_bool(&ws_data->is_destroying, true);
1071
1072 if (ws_data->service_running) {
1073 log_debug("[websocket_close] Stopping service thread to prevent deadlock during lws_close_reason()");
1074 ws_data->service_running = false;
1075 // Give service thread time to notice is_destroying flag
1076 platform_sleep_us(10000); // 10ms
1077 asciichat_thread_join(&ws_data->service_thread, NULL);
1078 log_debug("[websocket_close] Service thread stopped");
1079 }
1080
1081 mutex_lock(&ws_data->state_mutex);
1082
1083 if (!ws_data->is_connected) {
1084 mutex_unlock(&ws_data->state_mutex);
1085 log_info("websocket_close: Already closed (is_connected=false), wsi=%p, timestamp=%llu", (void *)ws_data->wsi,
1086 (unsigned long long)now_ns);
1087 return ASCIICHAT_OK; // Already closed
1088 }
1089
1090 log_fatal("🔴 websocket_close called! Setting is_connected=false, wsi=%p, timestamp=%llu", (void *)ws_data->wsi,
1091 (unsigned long long)now_ns);
1092 ws_data->is_connected = false;
1093 mutex_unlock(&ws_data->state_mutex);
1094
1095 // Close WebSocket connection
1096 // Safe to call lws_close_reason() now - service thread is stopped, so no callback contention
1097 if (ws_data->wsi) {
1098 log_fatal(" [websocket_close] Calling lws_close_reason for wsi=%p", (void *)ws_data->wsi);
1099 lws_close_reason(ws_data->wsi, LWS_CLOSE_STATUS_NORMAL, NULL, 0);
1100 log_fatal(" [websocket_close] lws_close_reason returned");
1101 }
1102
1103 // Wake any blocking recv() calls and send() waits
1104 // CRITICAL: Signal both recv_cond and state_cond to unblock all waiting threads
1105 // before the transport is destroyed. This prevents use-after-free when threads
1106 // wake from cond_timedwait() after the structure has been freed.
1107 cond_broadcast(&ws_data->recv_cond);
1108 cond_broadcast(&ws_data->state_cond);
1109
1110 log_info("WebSocket transport closed, wsi=%p", (void *)ws_data->wsi);
1111 return ASCIICHAT_OK;
1112}

1145static void websocket_destroy_impl(acip_transport_t *transport) {
1146 if (!transport || !transport->impl_data) {
1147 return;
1148 }
1149
1150 websocket_transport_data_t *ws_data = (websocket_transport_data_t *)transport->impl_data;
1151
1152 // Mark transport as destroying and broadcast condition variables
1153 // This signals all waiting threads to wake and exit naturally.
1154 log_debug("[WEBSOCKET_DESTROY] Marking transport as destroying");
1155 atomic_store_bool(&ws_data->is_destroying, true);
1156
1157 // Broadcast all condition variables to wake waiting threads so they can check
1158 // the is_destroying flag and exit gracefully
1159 cond_broadcast(&ws_data->state_cond);
1160 cond_broadcast(&ws_data->recv_cond);
1161
1162 // Cancel any pending libwebsockets service calls to interrupt blocking lws_service()
1163 // This prevents the service thread from getting stuck in lws_service()
1164 if (ws_data->context) {
1165 log_debug("[WEBSOCKET_DESTROY] Cancelling libwebsockets service");
1166 lws_cancel_service(ws_data->context);
1167 }
1168
1169 // Give threads a brief moment to detect the flag and exit
1170 // Threads should check is_destroying before acquiring mutexes
1171 platform_sleep_us(50 * US_PER_MS_INT); // 50ms for threads to detect flag
1172
1173 // Stop service thread (client-side only)
1174 if (ws_data->service_running) {
1175 log_debug("Stopping WebSocket service thread");
1176 ws_data->service_running = false;
1177 asciichat_thread_join(&ws_data->service_thread, NULL);
1178 log_debug("WebSocket service thread stopped");
1179 }
1180
1181 // Close WebSocket connection gracefully before destroying context
1182 // Skip lws_wsi_close() to avoid segfault in libwebsockets during cleanup
1183 // The context_destroy below will handle cleanup properly
1184 ws_data->wsi = NULL;
1185
1186 // Destroy WebSocket context (only if we own it - client transports only)
1187 // NOTE: Skip lws_context_destroy() to work around mbedTLS/OpenSSL 3.4.0 incompatibility
1188 // when libwebsockets is cached with wrong TLS backend. This causes memory leak but prevents crash.
1189 if (ws_data->context && ws_data->owns_context) {
1190 log_debug("Destroying WebSocket context (skipped due to mbedTLS/OpenSSL incompatibility)");
1191 // lws_context_destroy(ws_data->context); // DISABLED: causes crash on mbedTLS-built libwebsockets
1192 ws_data->context = NULL;
1193 }
1194
1195 // Clear receive queue and free buffered messages
1196 if (ws_data->recv_queue) {
1197 mutex_lock(&ws_data->recv_mutex);
1198
1199 websocket_recv_msg_t msg;
1200 while (ringbuffer_read(ws_data->recv_queue, &msg)) {
1201 if (msg.data) {
1202 buffer_pool_free(NULL, msg.data, msg.len);
1203 }
1204 }
1205
1206 mutex_unlock(&ws_data->recv_mutex);
1207 ringbuffer_destroy(ws_data->recv_queue);
1208 ws_data->recv_queue = NULL;
1209
1210 if (ws_data->send_queue) {
1211 // Drain send queue before destroying to free allocated message data
1212 // Use buffer_pool_free to match buffer_pool_alloc used for send messages
1213 websocket_recv_msg_t msg;
1214 while (ringbuffer_read(ws_data->send_queue, &msg)) {
1215 if (msg.data) {
1216 buffer_pool_free(NULL, msg.data, LWS_PRE + msg.len);
1217 }
1218 }
1219 ringbuffer_destroy(ws_data->send_queue);
1220 ws_data->send_queue = NULL;
1221 }
1222 }
1223
1224 // Free send buffer
1225 if (ws_data->send_buffer) {
1226 SAFE_FREE(ws_data->send_buffer);
1227 ws_data->send_buffer = NULL;
1228 }
1229
1230 // Free partial buffer (leftover data from previous recv())
1231 if (ws_data->partial_buffer) {
1232 buffer_pool_free(NULL, ws_data->partial_buffer, ws_data->partial_capacity);
1233 ws_data->partial_buffer = NULL;
1234 ws_data->partial_size = 0;
1235 ws_data->partial_capacity = 0;
1236 }
1237
1238 // Destroy synchronization primitives
1239 cond_destroy(&ws_data->state_cond);
1240 mutex_destroy(&ws_data->state_mutex);
1241 cond_destroy(&ws_data->recv_cond);
1242 mutex_destroy(&ws_data->recv_mutex);
1243 mutex_destroy(&ws_data->send_mutex);
1244 mutex_destroy(&ws_data->pending_free_mutex);
1245
1246 // Destroy pending-free queue
1247 if (ws_data->pending_free_queue) {
1248 ringbuffer_destroy(ws_data->pending_free_queue);
1249 ws_data->pending_free_queue = NULL;
1250 }
1251
1252 // Deregister websocket implementation data
1253 NAMED_UNREGISTER(ws_data);
1254
1255 // Clear impl_data pointer BEFORE freeing to prevent use-after-free in callbacks
1256 transport->impl_data = NULL;
1257
1258 // Free the websocket transport data structure
1259 SAFE_FREE(ws_data);
1260
1261 log_debug("Destroyed WebSocket transport resources");
1262}

1318 char path[256] = "/";

1333 SET_ERRNO(ERROR_INVALID_PARAM, "Invalid port number");
1334 return NULL;
1335 }
1336 } else {
1337 // No port specified, use default
1338 size_t host_len = path_start ? (size_t)(path_start - host_start) : strlen(host_start);
1339 if (host_len >= sizeof(host)) {
1340 SET_ERRNO(ERROR_INVALID_PARAM, "Host name too long");
1341 return NULL;
1342 }
1343 memcpy(host, host_start, host_len);
1344 host[host_len] = '\0';
1345 }

1347 // Extract path
1348 if (path_start) {
1349 SAFE_STRNCPY(path, path_start, sizeof(path) - 1);
1350 path[sizeof(path) - 1] = '\0';
1351 }

1358 SET_ERRNO(ERROR_MEMORY, "Failed to allocate WebSocket transport");

1366 SAFE_FREE(transport);
1367 SET_ERRNO(ERROR_MEMORY, "Failed to allocate WebSocket transport data");
1368 return NULL;

1376 if (!ws_data->recv_queue) {
1377 SAFE_FREE(ws_data);
1378 SAFE_FREE(transport);
1379 SET_ERRNO(ERROR_MEMORY, "Failed to create receive queue");
1380 return NULL;
1381 }
1382
1383 // Create send queue for client transport to buffer outgoing messages
1384 ws_data->send_queue = ringbuffer_create(sizeof(websocket_msg_t), WEBSOCKET_SEND_QUEUE_SIZE);
1385 if (!ws_data->send_queue) {
1386 ringbuffer_destroy(ws_data->recv_queue);
1387 SAFE_FREE(ws_data);
1388 SAFE_FREE(transport);
1389 SET_ERRNO(ERROR_MEMORY, "Failed to create send queue");
1390 return NULL;
1391 }
1392
1393 // Initialize synchronization primitives with transport-aware names
1394 char recv_name[64], state_name[64];
1395 snprintf(recv_name, sizeof(recv_name), "recv_%s", name);
1396 snprintf(state_name, sizeof(state_name), "state_%s", name);
1397
1398 if (mutex_init(&ws_data->recv_mutex, recv_name) != 0) {
1399 ringbuffer_destroy(ws_data->send_queue);
1400 ringbuffer_destroy(ws_data->recv_queue);
1401 SAFE_FREE(ws_data);
1402 SAFE_FREE(transport);
1403 SET_ERRNO(ERROR_INTERNAL, "Failed to initialize recv mutex");
1404 return NULL;
1405 }
1406
1407 if (cond_init(&ws_data->recv_cond, recv_name) != 0) {
1408 mutex_destroy(&ws_data->recv_mutex);
1409 ringbuffer_destroy(ws_data->send_queue);
1410 ringbuffer_destroy(ws_data->recv_queue);
1411 SAFE_FREE(ws_data);
1412 SAFE_FREE(transport);
1413 SET_ERRNO(ERROR_INTERNAL, "Failed to initialize recv condition variable");
1414 return NULL;
1415 }
1416
1417 if (mutex_init(&ws_data->state_mutex, state_name) != 0) {
1418 cond_destroy(&ws_data->recv_cond);
1419 mutex_destroy(&ws_data->recv_mutex);
1420 ringbuffer_destroy(ws_data->send_queue);
1421 ringbuffer_destroy(ws_data->recv_queue);
1422 SAFE_FREE(ws_data);
1423 SAFE_FREE(transport);
1424 SET_ERRNO(ERROR_INTERNAL, "Failed to initialize state mutex");
1425 return NULL;
1426 }

1428 if (cond_init(&ws_data->state_cond, state_name) != 0) {
1429 mutex_destroy(&ws_data->state_mutex);
1430 cond_destroy(&ws_data->recv_cond);
1431 mutex_destroy(&ws_data->recv_mutex);
1432 ringbuffer_destroy(ws_data->send_queue);
1433 ringbuffer_destroy(ws_data->recv_queue);
1434 SAFE_FREE(ws_data);
1435 SAFE_FREE(transport);
1436 SET_ERRNO(ERROR_INTERNAL, "Failed to initialize state condition variable");
1437 return NULL;
1438 }

1441 ws_data->send_buffer_capacity = LWS_PRE + 524288; // Match tx_packet_size (512KB)
1442 ws_data->send_buffer = SAFE_MALLOC(ws_data->send_buffer_capacity, uint8_t *);
1443 if (!ws_data->send_buffer) {
1444 mutex_destroy(&ws_data->state_mutex);
1445 cond_destroy(&ws_data->recv_cond);
1446 mutex_destroy(&ws_data->recv_mutex);
1447 ringbuffer_destroy(ws_data->send_queue);
1448 ringbuffer_destroy(ws_data->recv_queue);
1449 SAFE_FREE(ws_data);
1450 SAFE_FREE(transport);
1451 SET_ERRNO(ERROR_MEMORY, "Failed to allocate send buffer");
1452 return NULL;

1456 // Protocol array must persist for lifetime of context - use static
1457 // IMPORTANT: "http" protocol MUST be first for WebSocket upgrade handshake to work

1460 "http", // Required first for WebSocket upgrade
1461 websocket_callback,

1472 524288, // RX buffer size

1474 NULL, // User pointer (will be set from connect_info.userdata)
1475 524288 // TX packet size

1480 // Disable client compression for now - causes assertion in lws_set_extension_option()
1481 // This is a known issue with libwebsockets permessage-deflate negotiation
1482 // Server-side compression is still enabled, data will be compressed from server to client
1483 // but client->server traffic remains uncompressed (acceptable since client sends less data)
1484 // static const struct lws_extension client_extensions[] = {
1485 // {"permessage-deflate", lws_extension_callback_pm_deflate, "permessage-deflate; client_max_window_bits=15"},
1486 // {NULL, NULL, NULL}};

1505 .secs_since_valid_ping = 30, // Send PING after 30s idle
1506 .secs_since_valid_hangup = 35, // Hangup if still idle after 35s

1518 mutex_destroy(&ws_data->state_mutex);

1520 mutex_destroy(&ws_data->recv_mutex);

1535 connect_info.port = port;
1536 connect_info.path = path;

1546 log_debug("lws_client_connect_via_info returned: %p", (void *)ws_data->wsi);
1547 if (!ws_data->wsi) {
1548 lws_context_destroy(ws_data->context);
1549 SAFE_FREE(ws_data->send_buffer);
1550 cond_destroy(&ws_data->state_cond);
1551 mutex_destroy(&ws_data->state_mutex);
1552 cond_destroy(&ws_data->recv_cond);
1553 mutex_destroy(&ws_data->recv_mutex);

1555 ringbuffer_destroy(ws_data->recv_queue);

1567 transport->methods = &websocket_methods;
1568 transport->crypto_ctx = crypto_ctx;
1569 transport->impl_data = ws_data;

1572 // libwebsockets needs to process the initial connection handshake callbacks before lws_service()
1573 // is called repeatedly. Concurrent access to context during connection setup can trigger assertions.
1574 // Sleep 50ms to let lws_client_connect_via_info() callbacks complete initialization.

1576 platform_sleep_us(50000); // 50ms delay

1579 // Only the service thread should call lws_service() on this context
1580 log_debug("Starting WebSocket service thread...");
1581 ws_data->service_running = true;

1583 log_error("Failed to create WebSocket service thread");

1585 lws_context_destroy(ws_data->context);
1586 SAFE_FREE(ws_data->send_buffer);
1587 cond_destroy(&ws_data->state_cond);
1588 mutex_destroy(&ws_data->state_mutex);
1589 cond_destroy(&ws_data->recv_cond);
1590 mutex_destroy(&ws_data->recv_mutex);
1591 ringbuffer_destroy(ws_data->send_queue);
1592 ringbuffer_destroy(ws_data->recv_queue);
1593 SAFE_FREE(ws_data);
1594 SAFE_FREE(transport);
1595 SET_ERRNO(ERROR_INTERNAL, "Failed to create service thread");
1596 return NULL;
1597 }
1598 log_debug("WebSocket service thread started");
1599
1600 // CRITICAL FIX: Do NOT block the main thread waiting for connection!
1601 // This prevents stdin/keyboard input from being processed and causes the client to hang.
1602 // The service thread will establish the connection asynchronously.
1603 // The protocol layer (recv) will detect if connection fails and handle it there.
1604 // Return immediately so the main thread can respond to keyboard input.
1605 log_debug("WebSocket transport created, service thread will establish connection asynchronously");
1606
1607 // Register transport
1608 NAMED_REGISTER_TRANSPORT(transport, name, NULL);
1609
1610 // Register websocket implementation data with transport as parent
1611 NAMED_REGISTER_WEBSOCKET_IMPL(ws_data, "impl", (uintptr_t)(const void *)(transport));
1612
1613 // Return transport immediately - connection will be established by service thread
1614 // If connection fails, recv() will detect is_connected=false and return error
1615 return transport;
1616}
