240static void shutdown_client() {
241 // Guard against double cleanup (can be called explicitly + via atexit)
242 static bool shutdown_done = false;
243 if (shutdown_done) {
244 return;
245 }
246 shutdown_done = true;
247 log_debug("[SHUTDOWN] 1. Starting shutdown");

249 // Set global shutdown flag to stop all threads
250 signal_exit();
251
252 // Stop splash animation thread before any resource cleanup
253 splash_intro_done();
254
255 // IMPORTANT: Stop all protocol threads BEFORE cleaning up resources
256 // protocol_stop_connection() shuts down the socket to interrupt blocking recv(),
257 // then waits for the data reception thread and capture thread to exit.
258 // This prevents race conditions where threads access freed resources.
259 log_debug("[SHUTDOWN] 4. About to call protocol_stop_connection()");

275 // Legacy cleanup - will be removed after full migration to tcp_client

464static asciichat_error_t client_run(session_capture_ctx_t *capture, session_display_ctx_t *display, void *user_data) {
465 (void)capture; // Provided by session_client_like, used by protocol threads
466 (void)user_data; // Unused
467
468 // Make the framework-created display context available to protocol threads
469 display_set_context(display);
470
471 // Get the render loop's should_exit callback for monitoring
472 bool (*render_should_exit)(void *) = session_client_like_get_render_should_exit();
473 if (!render_should_exit) {
474 return SET_ERRNO(ERROR_INVALID_STATE, "Render should_exit callback not initialized");
475 }
476
477 // Attempt connection with fallback stages (TCP, WebRTC+STUN, WebRTC+TURN)
478 // Get pre-created TCP client from framework if available
479 tcp_client_t *framework_tcp_client = session_client_like_get_tcp_client();
480 asciichat_error_t connection_result = connection_attempt_tcp(
481 &g_client_session.connection_ctx,
482 g_client_session.discovered_address != NULL ? g_client_session.discovered_address : "",
483 (uint16_t)(g_client_session.discovered_port > 0 ? g_client_session.discovered_port : 27224),
484 framework_tcp_client);
485
486 // Check if shutdown was requested during connection attempt
487 if (should_exit()) {
488 log_info("Shutdown requested during connection attempt");
489 return ERROR_NETWORK;
490 }
491
492 if (connection_result != ASCIICHAT_OK) {
493 // Connection failed - need to stop audio threads that were initialized early
494 // (even though protocol_start_connection was never called)
495 audio_stop_thread();
496 // Framework will handle retry based on config
497 log_error("Connection attempt failed");
498 return connection_result;
499 }
500
501 // Connection successful - integrate transport into server layer
502 if (g_client_session.connection_ctx.active_transport) {
503 server_connection_set_transport(g_client_session.connection_ctx.active_transport);
504 g_client_session.connection_ctx.active_transport = NULL;
505 } else {
506 log_error("Connection succeeded but no active transport");
507 return ERROR_NETWORK;
508 }
509
510 // Log connection status
511 if (!g_client_session.has_ever_connected) {
512 log_info("Connected");
513 g_client_session.has_ever_connected = true;
514 } else {
515 log_info("Reconnected");
516 }
517
518 // Start protocol worker threads for this connection
519 if (protocol_start_connection() != 0) {
520 log_error("Failed to start connection protocols");
521 protocol_stop_connection();
522 server_connection_close();
523 return ERROR_NETWORK;
524 }
525
526 // Monitor connection until it breaks or shutdown is requested
527 static bool snapshot_check_logged_once = false;
528 while (!should_exit() && server_connection_is_active()) {
529 if (protocol_connection_lost()) {
530 log_debug("Connection lost detected");
531 break;
532 }
533
534 // Check if snapshot_delay timer has fired (in case frames stopped arriving)
535 // This is a safety check for when frame reception is sparse or stops entirely
536 extern uint64_t g_snapshot_first_frame_time_ns;
537 bool snapshot_mode_enabled = GET_OPTION(snapshot_mode);
538
539 if (!snapshot_check_logged_once) {
540 log_info("CLIENT_MAIN_LOOP: snapshot_mode=%d first_frame_time=%llu snapshot_delay=%.1f",
541 snapshot_mode_enabled, (unsigned long long)g_snapshot_first_frame_time_ns,
542 GET_OPTION(snapshot_delay));
543 snapshot_check_logged_once = true;
544 }
545
546 if (snapshot_mode_enabled && g_snapshot_first_frame_time_ns > 0) {
547 uint64_t current_time_ns = time_get_ns();
548 double elapsed = time_ns_to_s(time_elapsed_ns(g_snapshot_first_frame_time_ns, current_time_ns));
549 double snapshot_delay = GET_OPTION(snapshot_delay);
550
551 // Log progress every 500ms
552 static uint64_t last_progress_log_ns = 0;
553 if (current_time_ns - last_progress_log_ns > 500 * 1000 * 1000) {
554 log_debug("[MAIN_LOOP_SNAPSHOT_CHECK] elapsed=%.3f target=%.3f first_frame_ns=%llu current_ns=%llu",
555 elapsed, snapshot_delay, (unsigned long long)g_snapshot_first_frame_time_ns, (unsigned long long)current_time_ns);
556 last_progress_log_ns = current_time_ns;
557 }
558
559 if (elapsed >= snapshot_delay) {
560 char duration_str[32];
561 time_pretty((uint64_t)(elapsed * 1e9), -1, duration_str, sizeof(duration_str));
562 log_info("🎬 SNAPSHOT TIMEOUT FIRED (IN_MAIN_LOOP): Snapshot captured after %s! (target=%.3f)",
563 duration_str, snapshot_delay);
564 signal_exit();
565 break; // Exit loop, proper shutdown will follow
566 }
567 }
568
569 platform_sleep_us(100 * US_PER_MS_INT);
570 }
571
572 if (should_exit()) {
573 log_debug("Shutdown requested, cleaning up connection");
574 } else {
575 log_info("[ELSE_BRANCH] Connection lost path (should_exit=0, connection_active=0)");
576 // Connection lost - check if snapshot_delay timer has fired before exiting
577 extern uint64_t g_snapshot_first_frame_time_ns;
578 if (GET_OPTION(snapshot_mode) && g_snapshot_first_frame_time_ns > 0) {
579 uint64_t current_time_ns = time_get_ns();
580 double elapsed = time_ns_to_s(time_elapsed_ns(g_snapshot_first_frame_time_ns, current_time_ns));
581 double snapshot_delay = GET_OPTION(snapshot_delay);
582
583 log_info("[CONNECTION_LOSS_SNAPSHOT_CHECK] elapsed=%.3f target=%.3f (checking if timer fired)", elapsed, snapshot_delay);
584
585 if (elapsed >= snapshot_delay) {
586 char duration_str[32];
587 time_pretty((uint64_t)(elapsed * 1e9), -1, duration_str, sizeof(duration_str));
588 log_info("🎬 SNAPSHOT TIMEOUT FIRED (ON_CONNECTION_LOSS): Snapshot captured after %s! (target=%.3f)",
589 duration_str, snapshot_delay);
590 signal_exit();
591 // Don't return yet, allow normal cleanup to proceed below
592 }
593 }
594
595 if (!should_exit()) {
596 log_debug("Connection lost, preparing for reconnection attempt");
597 }
598 }
599
600 // Clean up this connection for potential reconnection
601 protocol_stop_connection();
602 server_connection_close();
603
604 // Clear transport reference from connection context since server_connection_close() already destroyed it
605 // This prevents double-free when connection_context_cleanup() is called below
606 g_client_session.connection_ctx.active_transport = NULL;
607
608 // Recreate thread pool for clean reconnection
609 if (g_client_worker_pool) {
610 thread_pool_destroy(g_client_worker_pool);
611 g_client_worker_pool = NULL;
612 }
613 g_client_worker_pool = thread_pool_create("client_reconnect");
614 if (!g_client_worker_pool) {
615 log_error("Failed to recreate worker thread pool");
616 return ERROR_THREAD;
617 }
618
619 // Reset connection context for next attempt
620 connection_context_cleanup(&g_client_session.connection_ctx);
621 memset(&g_client_session.connection_ctx, 0, sizeof(g_client_session.connection_ctx));
622 if (connection_context_init(&g_client_session.connection_ctx) != ASCIICHAT_OK) {
623 log_error("Failed to re-initialize connection context");
624 return ERROR_NETWORK;
625 }
626
627 // Return error to signal reconnection needed (framework handles the retry)
628 return ERROR_NETWORK;
629}

778 // Client mode supports:

780 // 2. LAN discovery (--scan) - handled above

802 discovered_port);
803 } else {
804 log_debug("Client: discovered_address=%s", discovered_address ? discovered_address : "NULL");
805 }
806
807 // Store discovered address/port in session state for client_run() callback
808 static char address_storage[BUFFER_SIZE_SMALL];
809 if (discovered_address) {
810 SAFE_STRNCPY(address_storage, discovered_address, sizeof(address_storage));
811 g_client_session.discovered_address = address_storage;
812 g_client_session.discovered_port = discovered_port;
813 } else {
814 const options_t *opts_fallback = options_get();
815 if (opts_fallback && opts_fallback->address[0] != '\0') {
816 SAFE_STRNCPY(address_storage, opts_fallback->address, sizeof(address_storage));
817 g_client_session.discovered_address = address_storage;
818 g_client_session.discovered_port = opts_fallback->port;
819 } else {
820 g_client_session.discovered_address = "localhost";
821 g_client_session.discovered_port = 27224;
822 }
823 }
824
825 // Initialize connection context for first attempt
826 asciichat_error_t ctx_init_result = connection_context_init(&g_client_session.connection_ctx);
827 if (ctx_init_result != ASCIICHAT_OK) {
828 log_error("Failed to initialize connection context");
829 return 1;
830 }

832 /* ========================================================================
833 * Configure and Run Shared Client-Like Session Framework
834 * ========================================================================
835 * session_client_like_run() handles all shared initialization:
836 * - Terminal output management (force stderr if piped)
837 * - Keepawake system (platform sleep prevention)
838 * - Splash screen lifecycle
839 * - Media source selection (webcam, file, URL, test pattern)
840 * - FPS probing for media files
841 * - Audio initialization and lifecycle
842 * - Display context creation
843 * - Proper cleanup ordering (critical for PortAudio)
844 *
845 * Client mode provides:
846 * - client_run() callback: connection loop, protocol startup, monitoring
847 * - client_should_reconnect() callback: reconnection policy
848 * - Reconnection configuration: attempts, delay, callbacks
849 */
850
851 // Get reconnect attempts setting (-1 = unlimited, 0 = no retry, >0 = retry N times)
852 int reconnect_attempts = GET_OPTION(reconnect_attempts);
853
854 // Create TCP client for network mode (will be used by session_client_like_run)
855 tcp_client_t *client_tcp = tcp_client_create();
856 if (!client_tcp) {
857 log_error("Failed to create TCP client for connection attempts");
858 return 1; // Return error to exit client
859 }
860
861 // Configure session_client_like with client-specific settings
862 session_client_like_config_t config = {
863 .run_fn = client_run,
864 .run_user_data = NULL,
865 .tcp_client = client_tcp,
866 .websocket_client = NULL,
867 .discovery = NULL,
868 .custom_should_exit = NULL,
869 .exit_user_data = NULL,
870 .keyboard_handler = NULL, // Client mode: server drives display
871 .max_reconnect_attempts = reconnect_attempts,
872 .should_reconnect_callback = client_should_reconnect,
873 .reconnect_user_data = NULL,
874 .reconnect_delay_ms = 1000, // 1 second delay between reconnection attempts
875 .print_newline_on_tty_exit = false, // Server/client manages cursor
876 };
877
878 log_debug("[CLIENT_MAIN] About to call session_client_like_run() with %d attempts", reconnect_attempts);
879 asciichat_error_t session_result = session_client_like_run(&config);
880 log_debug("[CLIENT_MAIN] session_client_like_run() returned %d", session_result);
881
882 // Note: TCP client lifecycle is managed by session_client_like_run() and connection attempts
883 // Do not destroy it here as it may be reused or already cleaned up
884
885 // Cleanup connection context
886 log_debug("[CLIENT_MAIN] About to call connection_context_cleanup()");
887 connection_context_cleanup(&g_client_session.connection_ctx);
888 log_debug("[CLIENT_MAIN] connection_context_cleanup() returned");
889
890 // Cleanup session log buffer (used by splash screen in session_client_like)
891 // NOTE: session_log_buffer_destroy() is already called in session_client_like_run()
892 // during its cleanup phase, so we don't call it here to avoid double-destroy of mutex
893
894 log_debug("[CLIENT_MAIN] ascii-chat client shutting down");
895
896 // IMPORTANT: Stop worker threads and join them BEFORE memory report
897 // atexit(shutdown_client) won't run if interrupted by SIGTERM, so call explicitly
898 log_debug("[CLIENT_MAIN] About to call shutdown_client()");
899 fflush(stderr);
900 shutdown_client();
901 log_debug("[CLIENT_MAIN] shutdown_client() returned");
902 fflush(stderr);
903
904 // Cleanup remaining shared subsystems (buffer pool, platform, etc.)
905 log_debug("[CLIENT_MAIN] About to call asciichat_shared_destroy()");
906 asciichat_shared_destroy();
907
908 // Log final session statistics before exit
909 // Note: Frame count is logged in protocol_stop_connection() if connection was established
910 // This is a fallback in case the connection never started
911 return (session_result == ASCIICHAT_OK) ? 0 : 1;
