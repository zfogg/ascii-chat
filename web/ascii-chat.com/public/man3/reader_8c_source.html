28asciichat_error_t terminal_fd_reader_create(int fd, int frame_height, terminal_fd_reader_t **out) {
29 if (fd < 0 || frame_height <= 0) {
30 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid fd or frame height: fd=%d, height=%d", fd, frame_height);
31 }
32
33 FILE *fp = fdopen(fd, "r");
34 if (!fp) {
35 return SET_ERRNO_SYS(ERROR_INVALID_PARAM, "fdopen failed for FD %d", fd);
36 }
37
38 terminal_fd_reader_t *reader = SAFE_CALLOC(1, sizeof(*reader), terminal_fd_reader_t *);
39 reader->fp = fp;
40 reader->frame_height = frame_height;

44 reader->eof_reached = false;

52asciichat_error_t terminal_fd_reader_next(terminal_fd_reader_t *reader, char **out_frame) {
53 if (!reader || !out_frame) {
54 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters to terminal_fd_reader_next");
55 }
56
57 if (reader->eof_reached) {
58 *out_frame = NULL;
59 return ASCIICHAT_OK; // EOF, but no error
60 }
61
62 // Allocate buffer for complete frame (frame_height lines)
63 // Estimate: 16KB per line should be enough for ANSI + ASCII
64 size_t frame_capacity = reader->frame_height * LINE_BUFFER_SIZE;
65 char *frame_data = SAFE_MALLOC(frame_capacity, char *);
66 size_t frame_pos = 0;
67
68 // Read exactly frame_height lines from FD
69 for (int line_num = 0; line_num < reader->frame_height; line_num++) {
70 // Read one line from FD
71 char *line = fgets(reader->line_buffer, (int)reader->line_buffer_size, reader->fp);
72
73 if (!line) {
74 // EOF reached
75 reader->eof_reached = true;
76
77 // If we read some lines but not a complete frame, that's ok - return partial
78 if (frame_pos > 0) {
79 log_debug("terminal_fd_reader: EOF reached after %d/%d lines, returning partial frame", line_num,
80 reader->frame_height);
81 // Remove trailing newline from last line if present
82 if (frame_pos > 0 && frame_data[frame_pos - 1] == '\n') {
83 frame_pos--;
84 }
85 *out_frame = frame_data;
86 return ASCIICHAT_OK;
87 }
88
89 SAFE_FREE(frame_data);
90 *out_frame = NULL;
91 return ASCIICHAT_OK; // EOF
92 }
93
94 // Append line to frame
95 size_t line_len = strlen(line);
96 if (frame_pos + line_len >= frame_capacity) {
97 log_warn("terminal_fd_reader: frame buffer overflow, line %d too large", line_num);
98 // Realloc and continue
99 frame_capacity *= 2;
100 char *new_frame = SAFE_MALLOC(frame_capacity, char *);
101 memcpy(new_frame, frame_data, frame_pos);
102 SAFE_FREE(frame_data);
103 frame_data = new_frame;
104 }
105
106 memcpy(frame_data + frame_pos, line, line_len);
107 frame_pos += line_len;
108 }
109
110 // Remove trailing newline from last line (we'll add it back in display)
111 if (frame_pos > 0 && frame_data[frame_pos - 1] == '\n') {
112 frame_data[frame_pos - 1] = '\0';
113 frame_pos--;
114 }
115
116 // Detect frame width from first frame (count visible chars in first line)
117 if (!reader->first_frame_processed && frame_pos > 0) {
118 reader->first_frame_processed = true;
119
120 // Find first line by looking for newline
121 char *first_newline = strchr(frame_data, '\n');
122 if (first_newline) {
123 // Null-terminate first line temporarily for width calculation
124 char saved_char = *first_newline;
125 *first_newline = '\0';
126 reader->frame_width = display_width(frame_data);
127 *first_newline = saved_char; // Restore
128 } else {
129 // Only one line in frame, use it for width
130 reader->frame_width = display_width(frame_data);
131 }
132
133 log_info("terminal_fd_reader: detected frame width %d from first frame", reader->frame_width);
134 }
135
136 *out_frame = frame_data;
137 log_debug_every(NS_PER_SEC_INT, "terminal_fd_reader: read frame (%zu bytes, %dx%d)", frame_pos, reader->frame_width,
138 reader->frame_height);
139 return ASCIICHAT_OK;
140}
