57void fps_counter_tick(fps_counter_t *counter) {
58 if (!counter) {
59 return;
60 }
61
62 // Record current time at head position
63 counter->frame_times[counter->head] = time_get_ns();
64
65 // Advance head pointer with wraparound
66 counter->head = (counter->head + 1) % FPS_WINDOW_SIZE;
67
68 // Track how many valid entries we have (up to FPS_WINDOW_SIZE)
69 if (counter->count < FPS_WINDOW_SIZE) {
70 counter->count++;
71 }
72}

74float fps_counter_get(fps_counter_t *counter) {
75 if (!counter || counter->count < 2) {
76 return 0.0f;
77 }
78
79 // Find indices of oldest and newest timestamps in the buffer
80 int oldest_idx = (counter->head - counter->count + FPS_WINDOW_SIZE) % FPS_WINDOW_SIZE;
81 int newest_idx = (counter->head - 1 + FPS_WINDOW_SIZE) % FPS_WINDOW_SIZE;
82
83 // Get the elapsed time between oldest and newest frames
84 uint64_t oldest_time = counter->frame_times[oldest_idx];
85 uint64_t newest_time = counter->frame_times[newest_idx];
86 uint64_t elapsed_ns = newest_time - oldest_time;
87
88 // Avoid division by zero
89 if (elapsed_ns == 0) {
90 return 0.0f;
91 }
92
93 // FPS = (number of frames) / elapsed_time
94 // We have (count - 1) frames across the elapsed time
95 // (count - 1) frames means count timestamps, so count-1 intervals
96 return (float)(counter->count - 1) * 1e9f / (float)elapsed_ns;
97}
