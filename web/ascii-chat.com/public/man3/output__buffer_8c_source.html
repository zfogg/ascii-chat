12// char* output buffer helpers
13//
14// ob_* - build strings with an output buffer
15// ob_reserve - reserve space in the output buffer
16// ob_putc - append a single character to the output buffer
17// ob_write - append a string to the output buffer
18// ob_term - terminate the output buffer
19// ob_u8 - append a decimal number to the output buffer
20// ob_u32 - append a decimal number to the output buffer

22void ob_reserve(outbuf_t *ob, size_t need) {
23 if (!ob)
24 return;
25 if (ob->cap == 0) {
26 // Always allocate at least default capacity on first call
27 size_t ncap = 4096;
28 while (ncap < ob->len + need)
29 ncap = (ncap * 3) / 2;
30 // SAFE_REALLOC handles NULL ptr as malloc - this is safe
31 ob->buf = SAFE_REALLOC(ob->buf, ncap, char *);
32 ob->cap = ncap;
33 } else if (ob->len + need > ob->cap) {
34 // Expand existing buffer
35 size_t ncap = ob->cap;
36 while (ncap < ob->len + need)
37 ncap = (ncap * 3) / 2;
38 ob->buf = SAFE_REALLOC(ob->buf, ncap, char *);
39 ob->cap = ncap;
40 }
41}

43void ob_putc(outbuf_t *ob, char c) {
44 if (!ob)
45 return;
46 ob_reserve(ob, 1);
47 ob->buf[ob->len++] = c;
48}

50void ob_write(outbuf_t *ob, const char *s, size_t n) {
51 if (!ob || n == 0)
52 return;
53 ob_reserve(ob, n);
54 SAFE_MEMCPY(ob->buf + ob->len, ob->cap - ob->len, s, n);
55 ob->len += n;
56}

58void ob_term(outbuf_t *ob) {
59 if (!ob)
60 return;
61 // Reserve space for null terminator if needed
62 if (ob->len >= ob->cap) {
63 ob_reserve(ob, 1);
64 }
65 // Append null terminator WITHOUT incrementing length
66 // (null terminator is not part of the string length)
67 ob->buf[ob->len] = '\0';
68}

70// Fast decimal for uint8_t (0..255)
71void ob_u8(outbuf_t *ob, uint8_t v) {
72 if (!ob)
73 return;
74 if (v >= 100) {
75 uint8_t d0 = v / 100;
76 uint8_t r = v % 100;
77 uint8_t d1 = r / 10;
78 uint8_t d2 = r % 10;
79 ob_putc(ob, '0' + d0);
80 ob_putc(ob, '0' + d1);
81 ob_putc(ob, '0' + d2);
82 } else if (v >= 10) {
83 uint8_t d1 = v / 10;
84 uint8_t d2 = v % 10;
85 ob_putc(ob, '0' + d1);
86 ob_putc(ob, '0' + d2);
87 } else {
88 ob_putc(ob, '0' + v);
89 }
90}

92void ob_u32(outbuf_t *ob, uint32_t v) {
93 if (!ob)
94 return;
95 char tmp[10];
96 int i = 0;
97 do {
98 tmp[i++] = '0' + (v % 10u);
99 v /= 10u;
100 } while (v);
101 ob_reserve(ob, (size_t)i);
102 while (i--)
103 ob->buf[ob->len++] = tmp[i];
104}

147// REP profitability calculation
148bool rep_is_profitable(uint32_t runlen) {
149 if (runlen <= 2)
150 return false;
151 uint32_t k = runlen - 1; // Extra repetitions beyond the first character
152 uint32_t rep_cost = (uint32_t)(digits_u32(k) + 3); // ESC [ digits b
153 return k > rep_cost; // Manual repetition cost vs REP cost
154}
