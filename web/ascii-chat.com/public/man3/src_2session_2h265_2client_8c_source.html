27h265_client_context_t *h265_client_context_create(uint16_t width, uint16_t height) {
28 if (width == 0 || height == 0) {
29 SET_ERRNO(ERROR_MEDIA_INIT, "Frame dimensions must be non-zero");
30 return NULL;
31 }
32
33 h265_client_context_t *ctx = SAFE_CALLOC(1, sizeof(h265_client_context_t), h265_client_context_t *);
34 if (!ctx) {
35 SET_ERRNO(ERROR_MEMORY, "Failed to allocate client context");
36 return NULL;
37 }
38
39 ctx->encoder = h265_encoder_create(width, height);
40 if (!ctx->encoder) {
41 SET_ERRNO(ERROR_MEDIA_INIT, "Failed to create H.265 encoder");
42 SAFE_FREE(ctx);

44 }
45
46 ctx->current_width = width;
47 ctx->current_height = height;
48 ctx->frame_buf_size = width * height;

50 if (!ctx->frame_buf) {
51 SET_ERRNO(ERROR_MEMORY, "Failed to allocate frame buffer");
52 h265_encoder_destroy(ctx->encoder);
53 SAFE_FREE(ctx);
54 return NULL;
55 }
56
57 log_info("H.265 client context created: %ux%u", width, height);
