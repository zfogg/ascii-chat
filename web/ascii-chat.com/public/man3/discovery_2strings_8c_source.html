195asciichat_error_t acds_string_generate(char *output, size_t output_size) {
196 if (!output || output_size < SESSION_STRING_BUFFER_SIZE) {
197 return SET_ERRNO(ERROR_INVALID_PARAM, "output buffer must be at least %zu bytes",
198 (size_t)SESSION_STRING_BUFFER_SIZE);
199 }
200
201 // libsodium is guaranteed to be initialized by acds_string_init() before this is called
202 // No need to call sodium_init() again (redundant initialization removed)
203
204 // Pick random adjective
205 uint32_t adj_idx = randombytes_uniform((uint32_t)adjectives_count);
206 const char *adj = adjectives[adj_idx];
207
208 // Pick two random nouns
209 uint32_t noun1_idx = randombytes_uniform((uint32_t)nouns_count);
210 uint32_t noun2_idx = randombytes_uniform((uint32_t)nouns_count);
211 const char *noun1 = nouns[noun1_idx];
212 const char *noun2 = nouns[noun2_idx];
213
214 // Format: adjective-noun-noun
215 int written = safe_snprintf(output, output_size, "%s-%s-%s", adj, noun1, noun2);
216 if (written < 0 || (size_t)written >= output_size) {
217 return SET_ERRNO(ERROR_BUFFER_OVERFLOW, "Session string too long for buffer");
218 }
219
220 log_dev("Generated session string: %s", output);
221 return ASCIICHAT_OK;
222}

224bool is_session_string(const char *str) {
225 if (!str || str[0] == '\0') {
226 SET_ERRNO(ERROR_INVALID_PARAM, "Session string is NULL or empty");
227 return false;
228 }
229
230 size_t len = strlen(str);
231
232 // Check length bounds (min 4, max defined by SESSION_STRING_MAX_LEN)
233 if (len < 4 || len > SESSION_STRING_MAX_LEN) {
234 SET_ERRNO(ERROR_INVALID_PARAM, "Session string length %zu outside valid range 4-%d", len, SESSION_STRING_MAX_LEN);
235 return false;
236 }
237
238 // Session strings must be ASCII-only (homograph attack prevention)
239 if (!utf8_is_ascii_only(str)) {
240 SET_ERRNO(ERROR_INVALID_PARAM, "Session string contains non-ASCII characters");
241 return false;
242 }
243
244 // Get compiled regex (lazy initialization)
245 pcre2_code *regex = session_format_regex_get();
246 if (!regex) {
247 SET_ERRNO(ERROR_INTERNAL, "Session string PCRE2 regex failed to initialize");
248 return false;
249 }
250
251 // Validate format using PCRE2 regex
252 pcre2_match_data *match_data = pcre2_match_data_create_from_pattern(regex, NULL);
253 if (!match_data) {
254 log_error("Failed to allocate match data for session string regex");
255 SET_ERRNO(ERROR_MEMORY, "Failed to allocate match data");
256 return false;
257 }
258
259 int match_result = pcre2_jit_match(regex, (PCRE2_SPTR8)str, len, 0, 0, match_data, NULL);
260
261 pcre2_match_data_free(match_data);
262
263 if (match_result < 0) {
264 // Format validation failed
265 SET_ERRNO(ERROR_INVALID_PARAM, "Session string format does not match pattern");
266 return false;
267 }
268
269 // Extract the three words by finding hyphens
270 // Format is guaranteed to be: word-word-word where each word is 2-12 lowercase letters
271 const char *hyphen1 = strchr(str, '-');
272 if (!hyphen1) {
273 SET_ERRNO(ERROR_INVALID_PARAM, "Session string missing first hyphen");
274 return false;
275 }
276
277 const char *hyphen2 = strchr(hyphen1 + 1, '-');
278 if (!hyphen2) {
279 SET_ERRNO(ERROR_INVALID_PARAM, "Session string missing second hyphen");
280 return false;
281 }
282
283 // Extract three words
284 size_t adj_len = hyphen1 - str;
285 size_t noun1_len = hyphen2 - hyphen1 - 1;
286 size_t noun2_len = len - (hyphen2 - str) - 1;
287
288 if (adj_len >= 32 || noun1_len >= 32 || noun2_len >= 32) {
289 SET_ERRNO(ERROR_INVALID_PARAM, "Session string word length out of bounds");
290 return false;
291 }
292
293 char adj[32], noun1[32], noun2[32];
294 memcpy(adj, str, adj_len);
295 adj[adj_len] = '\0';
296 memcpy(noun1, hyphen1 + 1, noun1_len);
297 noun1[noun1_len] = '\0';
298 memcpy(noun2, hyphen2 + 1, noun2_len);
299 noun2[noun2_len] = '\0';
300
301 // Lazy initialization: build validation caches on first use
302 // call_once ensures thread-safe exactly-once execution
303 build_validation_caches();
304 // If initialization fails silently (logs warning), hashtables will be empty
305 // and validation will fail, which is safe behavior
306
307 // Validate first word is an adjective
308 word_cache_entry_t *adj_entry = NULL;
309 HASH_FIND_STR(g_adjectives_cache, adj, adj_entry);
310 if (!adj_entry) {
311 SET_ERRNO(ERROR_INVALID_PARAM, "Session string first word '%s' is not a valid adjective", adj);
312 return false;
313 }
314
315 // Validate second and third words are nouns
316 word_cache_entry_t *noun_entry1 = NULL;
317 HASH_FIND_STR(g_nouns_cache, noun1, noun_entry1);
318 if (!noun_entry1) {
319 SET_ERRNO(ERROR_INVALID_PARAM, "Session string second word '%s' is not a valid noun", noun1);
320 return false;
321 }
322
323 word_cache_entry_t *noun_entry2 = NULL;
324 HASH_FIND_STR(g_nouns_cache, noun2, noun_entry2);
325 if (!noun_entry2) {
326 SET_ERRNO(ERROR_INVALID_PARAM, "Session string third word '%s' is not a valid noun", noun2);
327 return false;
328 }
329
330 log_dev("Valid session string: %s", str);
331 return true;
332}
