123 // Write to debug file directly to bypass logging system
124 FILE *dbg = fopen("/tmp/render-debug.txt", "a");
125 if (dbg) {
126 fprintf(dbg, "[RENDER_WRITE_FRAME] Called with ctx=%p, frame_len=%zu, captured_ns=%llu\n", (void *)ctx,
127 ansi_frame ? strlen(ansi_frame) : 0, (unsigned long long)captured_ns);
128 fclose(dbg);
129 }
130 log_info("render_file_write_frame: CALLED - ctx=%p, captured_ns=%llu", (void *)ctx, (unsigned long long)captured_ns);
131
132 if (!ctx) {
133 log_warn("render_file_write_frame: ctx is NULL");
134 return ASCIICHAT_OK;
135 }
136
137 if (!ansi_frame) {
138 log_warn("render_file_write_frame: ansi_frame is NULL");
139 return ASCIICHAT_OK;
140 }
141
142 size_t frame_len = strlen(ansi_frame);
143 log_info("render_file_write_frame: processing frame (len=%zu)", frame_len);
144 if (frame_len > 0) {
145 log_info(" first 100 chars: %.100s", ansi_frame);
146 }
147
148 asciichat_error_t err = term_renderer_feed(ctx->renderer, ansi_frame, frame_len);
149 if (err != ASCIICHAT_OK) {
150 log_warn("render_file_write_frame: term_renderer_feed failed: %s", asciichat_error_string(err));
151 return err;
152 }

154 const uint8_t *pixels = term_renderer_pixels(ctx->renderer);
155 int pitch = term_renderer_pitch(ctx->renderer);
156 int width_px = term_renderer_width_px(ctx->renderer);
157 int height_px = term_renderer_height_px(ctx->renderer);

159 log_info("render_file_write_frame: pixels=%p pitch=%d dims=%dx%d", (void *)pixels, pitch, width_px, height_px);

161 // Check pixels in multiple locations to verify content
162 if (pixels) {
163 // Check pixel at (0,0) - top left corner
164 uint8_t sample_r = pixels[0], sample_g = pixels[1], sample_b = pixels[2];
165
166 // Check pixel at (500, 100) where we injected a test red stripe
167 size_t test_offset = 100 * pitch + 500 * 3;
168 uint8_t test_r = pixels[test_offset], test_g = pixels[test_offset + 1], test_b = pixels[test_offset + 2];
169
170 // Check pixel at middle of screen
171 uint8_t sample_r_mid = pixels[(height_px / 2) * pitch], sample_g_mid = pixels[(height_px / 2) * pitch + 1],
172 sample_b_mid = pixels[(height_px / 2) * pitch + 2];
173
174 log_info(" pixel[0,0]: RGB(%u,%u,%u), test_stripe[500,100]: RGB(%u,%u,%u), pixel[0,%d]: RGB(%u,%u,%u)",
175 sample_r, sample_g, sample_b, test_r, test_g, test_b, height_px/2, sample_r_mid, sample_g_mid, sample_b_mid);
176 }
177
178 // CRITICAL: Copy pixel buffer before encoding
179 // term_renderer_pixels() returns a pointer to the renderer's internal buffer which gets
180 // overwritten on the next frame. We must copy the data before passing to the encoder.
181 uint8_t *pixels_copy = NULL;
182 if (pixels) {
183 size_t pixel_buffer_size = (size_t)pitch * (size_t)height_px;
184 pixels_copy = SAFE_MALLOC(pixel_buffer_size, uint8_t *);
185 memcpy(pixels_copy, pixels, pixel_buffer_size);
186 log_info("render_file_write_frame: copied %zu bytes of pixel data (was at %p, now at %p)",
187 pixel_buffer_size, (void *)pixels, (void *)pixels_copy);
188
189 // DEBUG: Inject test red stripe to verify encoder is working
190 static int frame_num = 0;
191 if (frame_num++ < 1) {
192 log_info("DEBUG: Injecting test RED vertical stripe into pixel buffer for verification");
193 // Fill a 100-pixel-wide vertical stripe with bright red at x=500
194 for (int y = 0; y < height_px; y++) {
195 for (int x = 500; x < 600 && x < width_px; x++) {
196 uint8_t *pixel = pixels_copy + y * pitch + x * 3;
197 pixel[0] = 255; // R
198 pixel[1] = 0; // G
199 pixel[2] = 0; // B
200 }
201 }
202 }
203 }
204
205 err = ffmpeg_encoder_write_frame(ctx->encoder, pixels_copy, pitch, captured_ns);
206 if (err != ASCIICHAT_OK) {
207 log_warn("render_file_write_frame: ffmpeg_encoder_write_frame failed: %s", asciichat_error_string(err));
208 }
209
210 // Free the pixel copy (ffmpeg_encoder_write_frame reads and converts it immediately)
211 SAFE_FREE(pixels_copy);
212
213 // Write synchronized audio for this frame if available
214 // Calculate how many samples correspond to one video frame
215 // Note: Audio samples per frame = 48000 Hz / FPS (e.g., 1600 samples @ 30fps)
216 if (ctx && ctx->audio_read_buf && ctx->encoder) {
217 // Assume default 60 FPS if we can't determine it (typically matches capture rate)
218 int fps = 60;
219 int samples_per_frame = ctx->audio_sample_rate / fps;
220 if (samples_per_frame > ctx->audio_buf_size) {
221 samples_per_frame = ctx->audio_buf_size;
222 }
223
224 int samples_read = 0;
225 static int audio_debug_once = 0;
226
227 // Read from whichever audio source is available (prefer media source, fallback to capture)
228 if (ctx->audio_media_source) {
229 // Read from media source (file/URL audio)
230 samples_read = media_source_read_audio(ctx->audio_media_source, ctx->audio_read_buf, samples_per_frame);
231 if (samples_read < 0) {
232 samples_read = 0; // On error, write silence
233 }
234 if (!audio_debug_once++) {
235 log_debug("render_file_write_frame: reading audio from media_source (samples_per_frame=%d, read=%d)",
236 samples_per_frame, samples_read);
237 }
238 } else if (ctx->audio_capture_rb) {
239 // Read from ring buffer (live mic capture)
240 samples_read =
241 audio_ring_buffer_read(ctx->audio_capture_rb, ctx->audio_read_buf, samples_per_frame);
242 if (samples_read < 0) {
243 samples_read = 0; // On error, write silence
244 }
245 if (!audio_debug_once++) {
246 log_debug("render_file_write_frame: reading audio from capture_rb (samples_per_frame=%d, read=%d)",
247 samples_per_frame, samples_read);
248 }
249 } else {
250 if (!audio_debug_once++) {
251 log_debug("render_file_write_frame: no audio source available (media=%p, capture=%p)",
252 ctx->audio_media_source, ctx->audio_capture_rb);
253 }
254 }
255
256 // Write audio samples if any were read
257 if (samples_read > 0) {
258 ffmpeg_encoder_write_audio(ctx->encoder, ctx->audio_read_buf, samples_read);
259 } else if (samples_read == 0 && (ctx->audio_media_source || ctx->audio_capture_rb)) {
260 // Source exists but no samples available - still need to write something for sync
261 // Zero-fill the buffer (silence) to maintain timing
262 memset(ctx->audio_read_buf, 0, samples_per_frame * sizeof(float));
263 ffmpeg_encoder_write_audio(ctx->encoder, ctx->audio_read_buf, samples_per_frame);
264 }
265 }
266
267 return err;
268}
