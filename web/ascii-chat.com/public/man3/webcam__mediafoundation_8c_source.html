36static HRESULT enumerate_devices_and_print(void) {
37 IMFAttributes *attr = NULL;
38 IMFActivate **devices = NULL;
39 UINT32 count = 0;
40 HRESULT hr;
41
42 // Create attribute store for device enumeration
43 hr = MFCreateAttributes(&attr, 1);
44 if (FAILED(hr)) {
45 return SET_ERRNO_SYS(ERROR_WEBCAM, "Failed to create MF attributes: 0x%08x", hr);
46 }
47
48 // Set the device type to video capture
49 hr =
50 IMFAttributes_SetGUID(attr, &MF_DEVSOURCE_ATTRIBUTE_SOURCE_TYPE, &MF_DEVSOURCE_ATTRIBUTE_SOURCE_TYPE_VIDCAP_GUID);
51 if (FAILED(hr)) {
52 SET_ERRNO_SYS(ERROR_WEBCAM, "Failed to set MF device type: 0x%08x", hr);
53 goto cleanup;
54 }
55
56 // Enumerate video capture devices
57 hr = MFEnumDeviceSources(attr, &devices, &count);
58 if (FAILED(hr)) {
59 SET_ERRNO_SYS(ERROR_WEBCAM, "Failed to enumerate MF devices: 0x%08x", hr);
60 goto cleanup;
61 }
62
63 log_info("Found %d video capture device(s):", count);
64 for (UINT32 i = 0; i < count; i++) {
65 LPWSTR friendlyName = NULL;
66 UINT32 nameLength = 0;
67
68 hr = IMFActivate_GetAllocatedString(devices[i], &MF_DEVSOURCE_ATTRIBUTE_FRIENDLY_NAME, &friendlyName, &nameLength);
69 if (SUCCEEDED(hr) && friendlyName) {
70 // Convert wide string to multibyte for logging
71 int len = WideCharToMultiByte(CP_UTF8, 0, friendlyName, -1, NULL, 0, NULL);
72 if (len > 0) {
73 char *mbName = SAFE_MALLOC((size_t)len, void *);
74 if (mbName && WideCharToMultiByte(CP_UTF8, 0, friendlyName, -1, mbName, len, NULL, NULL)) {
75 log_info(" Device %d: %s", i, mbName);
76 }
77 SAFE_FREE(mbName);
78 }
79 CoTaskMemFree(friendlyName);
80 } else {
81 log_info(" Device %d: <Unknown Name>", i);
82 }
83 }
84
85cleanup:
86 if (devices) {
87 for (UINT32 i = 0; i < count; i++) {
88 if (devices[i]) {
89 IMFActivate_Release(devices[i]);
90 }
91 }
92 CoTaskMemFree((void *)devices);
93 }
94 if (attr) {
95 IMFAttributes_Release(attr);
96 }
97
98 return hr;
99}

101asciichat_error_t webcam_init_context(webcam_context_t **ctx, unsigned short int device_index) {
102 log_info("Opening Windows webcam with Media Foundation, device index %d", device_index);
103
104 webcam_context_t *cam;
105 cam = SAFE_MALLOC(sizeof(webcam_context_t), webcam_context_t *);
106
107 // Initialize all fields
108 cam->device = NULL;
109 cam->reader = NULL;
110 cam->width = 0;
111 cam->height = 0;
112 cam->mf_initialized = FALSE;
113 cam->com_initialized = FALSE;
114 cam->cached_frame = NULL;
115
116 HRESULT hr;
117 IMFAttributes *attr = NULL;
118 IMFActivate **devices = NULL;
119 UINT32 count = 0;
120 int result = -1;
121
122 // Capture Windows Media Foundation's initialization output
123 LOG_IO("mf", {
124 // Initialize COM
125 hr = CoInitializeEx(NULL, COINIT_MULTITHREADED);
126 if (SUCCEEDED(hr)) {
127 cam->com_initialized = TRUE;
128 } else if (hr != RPC_E_CHANGED_MODE) {
129 SET_ERRNO_SYS(ERROR_WEBCAM, "Failed to initialize COM: 0x%08x", hr);
130 // Note: We can't use goto from within LOG_IO block, so we'll set hr to FAILED
131 hr = E_FAIL;
132 }
133
134 // Initialize Media Foundation
135 if (SUCCEEDED(hr)) {
136 hr = MFStartup(MF_VERSION, MFSTARTUP_NOSOCKET);
137 if (FAILED(hr)) {
138 SET_ERRNO_SYS(ERROR_WEBCAM, "Failed to startup Media Foundation: 0x%08x", hr);
139 } else {
140 cam->mf_initialized = TRUE;
141 }
142 }
143 });
144
145 if (FAILED(hr)) {
146 goto error;
147 }
148
149 // Enumerate and print all devices
150 enumerate_devices_and_print();
151
152 // Create attribute store for device enumeration
153 hr = MFCreateAttributes(&attr, 1);
154 if (FAILED(hr)) {
155 SET_ERRNO_SYS(ERROR_WEBCAM, "Failed to create MF attributes: 0x%08x", hr);
156 goto error;
157 }
158
159 hr =
160 IMFAttributes_SetGUID(attr, &MF_DEVSOURCE_ATTRIBUTE_SOURCE_TYPE, &MF_DEVSOURCE_ATTRIBUTE_SOURCE_TYPE_VIDCAP_GUID);
161 if (FAILED(hr)) {
162 SET_ERRNO_SYS(ERROR_WEBCAM, "Failed to set MF device type: 0x%08x", hr);
163 goto error;
164 }
165
166 // Enumerate devices
167 hr = MFEnumDeviceSources(attr, &devices, &count);
168 if (FAILED(hr)) {
169 SET_ERRNO_SYS(ERROR_WEBCAM, "Failed to enumerate MF devices: 0x%08x", hr);
170 goto error;
171 }
172
173 if (count == 0) {
174 SET_ERRNO(ERROR_WEBCAM, "No video capture devices found");
175 hr = E_FAIL;
176 goto error;
177 }
178
179 if (device_index >= count) {
180 SET_ERRNO(ERROR_WEBCAM, "Device index %d out of range (0-%d)", device_index, count - 1);
181 hr = E_FAIL;
182 goto error;
183 }
184
185 // Create media source for the specified device
186 hr = IMFActivate_ActivateObject(devices[device_index], &IID_IMFMediaSource, (void **)&cam->device);
187 log_info("IMFActivate_ActivateObject returned: 0x%08x", hr);
188
189 if (FAILED(hr)) {
190 log_error("CRITICAL: Failed to activate MF device: 0x%08x", hr);
191 log_error(" 0x80070005 = E_ACCESSDENIED (device in use)");
192 log_error(" 0xc00d3704 = Device already in use");
193 log_error(" 0xc00d3e85 = MF_E_VIDEO_RECORDING_DEVICE_INVALIDATED");
194 result = SET_ERRNO(ERROR_WEBCAM_IN_USE, "Media Foundation device %d is in use (HRESULT: 0x%08x)", device_index, hr);
195 goto error;
196 }
197
198 // Create source reader with GPU-accelerated video processing (Windows 8+)
199 // This enables automatic YUV->RGB conversion with hardware acceleration
200 IMFAttributes *readerAttrs = NULL;
201 hr = MFCreateAttributes(&readerAttrs, 1);
202 if (FAILED(hr)) {
203 log_error("Failed to create reader attributes: 0x%08x", hr);
204 result =
205 SET_ERRNO(ERROR_WEBCAM, "Failed to create reader attributes for device %d (HRESULT: 0x%08x)", device_index, hr);
206 goto error;
207 }
208
209 // Enable advanced video processing for GPU-accelerated YUV->RGB conversion (Windows 8+)
210 // This is the recommended attribute for webcam capture with format conversion
211 hr = IMFAttributes_SetUINT32(readerAttrs, &MF_SOURCE_READER_ENABLE_ADVANCED_VIDEO_PROCESSING, TRUE);
212 if (FAILED(hr)) {
213 log_warn("Failed to set advanced video processing attribute: 0x%08x", hr);
214 }
215
216 hr = MFCreateSourceReaderFromMediaSource(cam->device, readerAttrs, &cam->reader);
217 log_info("MFCreateSourceReaderFromMediaSource returned: 0x%08x, readerAttrs=%p", hr, (void *)readerAttrs);
218
219 if (readerAttrs) {
220 IMFAttributes_Release(readerAttrs);
221 }
222
223 if (FAILED(hr)) {
224 log_error("CRITICAL: Failed to create MF source reader: 0x%08x", hr);
225 result =
226 SET_ERRNO(ERROR_WEBCAM_IN_USE,
227 "Failed to create Media Foundation source reader for device %d (HRESULT: 0x%08x)", device_index, hr);
228 goto error;
229 }
230
231 // IMPORTANT: Select the video stream first before configuring
232 hr = IMFSourceReader_SetStreamSelection(cam->reader, (DWORD)MF_SOURCE_READER_ALL_STREAMS, FALSE);
233 log_info("SetStreamSelection (deselect all) returned: 0x%08x", hr);
234 if (FAILED(hr)) {
235 log_warn("Failed to deselect all streams: 0x%08x", hr);
236 }
237
238 hr = IMFSourceReader_SetStreamSelection(cam->reader, (DWORD)MF_SOURCE_READER_FIRST_VIDEO_STREAM, TRUE);
239 log_info("SetStreamSelection (select video) returned: 0x%08x", hr);
240 if (FAILED(hr)) {
241 log_error("Failed to select video stream: 0x%08x", hr);
242 result = SET_ERRNO(ERROR_WEBCAM, "Failed to select video stream for device %d (HRESULT: 0x%08x)", device_index, hr);
243 goto error;
244 }
245
246 // Request RGB32 output format (BGRA) at 640x480 resolution
247 // Combined with MF_SOURCE_READER_ENABLE_ADVANCED_VIDEO_PROCESSING, this enables
248 // GPU-accelerated YUV->RGB conversion
249 IMFMediaType *rgbType = NULL;
250 hr = MFCreateMediaType(&rgbType);
251 if (SUCCEEDED(hr)) {
252 IMFMediaType_SetGUID(rgbType, &MF_MT_MAJOR_TYPE, &MFMediaType_Video);
253 IMFMediaType_SetGUID(rgbType, &MF_MT_SUBTYPE, &MFVideoFormat_RGB32);
254
255 // Request 640x480 resolution (we only need 480x270 max for ascii-chat)
256 // This dramatically reduces pixel copy overhead (307,200 vs 8,294,400 pixels)
257 UINT64 frameSize = ((UINT64)640 << 32) | (UINT64)480;
258 hr = IMFMediaType_SetUINT64(rgbType, &MF_MT_FRAME_SIZE, frameSize);
259 if (FAILED(hr)) {
260 log_warn("Could not set frame size to 640x480: 0x%08x", hr);
261 }
262
263 // Use partial type - let MF fill in other details
264 hr = IMFSourceReader_SetCurrentMediaType(cam->reader, (DWORD)MF_SOURCE_READER_FIRST_VIDEO_STREAM, NULL, rgbType);
265 IMFMediaType_Release(rgbType);
266
267 if (SUCCEEDED(hr)) {
268 log_info("Successfully requested RGB32 output format at 640x480");
269 } else {
270 log_warn("Could not set RGB32 format: 0x%08x, will use native format", hr);
271 // Don't fail - just use whatever format the camera provides
272 }
273 }
274
275 // Get actual media type and dimensions
276 IMFMediaType *currentType = NULL;
277 hr = IMFSourceReader_GetCurrentMediaType(cam->reader, (DWORD)MF_SOURCE_READER_FIRST_VIDEO_STREAM, &currentType);
278 if (SUCCEEDED(hr)) {
279 UINT64 frameSize = 0;
280 hr = IMFMediaType_GetUINT64(currentType, &MF_MT_FRAME_SIZE, &frameSize);
281 if (SUCCEEDED(hr)) {
282 cam->width = (int)(frameSize >> 32);
283 cam->height = (int)(frameSize & 0xFFFFFFFF);
284 log_info("Media Foundation webcam opened: %dx%d", cam->width, cam->height);
285 } else {
286 // Default resolution if we can't get the actual size
287 cam->width = 640;
288 cam->height = 480;
289 log_warn("Could not get frame size, using default 640x480");
290 }
291 IMFMediaType_Release(currentType);
292 } else {
293 cam->width = 640;
294 cam->height = 480;
295 log_warn("Could not get media type, using default 640x480");
296 }
297
298 DWORD streamIndex, flags;
299 LONGLONG timestamp;
300 IMFSample *sample = NULL;
301
302 hr = IMFSourceReader_ReadSample(cam->reader, (DWORD)MF_SOURCE_READER_FIRST_VIDEO_STREAM,
303 0, // Regular synchronous read
304 &streamIndex, &flags, &timestamp, &sample);
305
306 if (FAILED(hr)) {
307 log_error("CRITICAL: Failed to read test frame during initialization: 0x%08x", hr);
308 result =
309 SET_ERRNO(ERROR_WEBCAM_IN_USE, "Failed to read test frame from device %d (HRESULT: 0x%08x)", device_index, hr);
310 goto error;
311 }
312
313 // Clean up the test sample if we got one
314 if (sample) {
315 IMFSample_Release(sample);
316 }
317
318 // Cleanup enumeration resources on success
319 if (devices) {
320 for (UINT32 i = 0; i < count; i++) {
321 if (devices[i]) {
322 IMFActivate_Release(devices[i]);
323 }
324 }
325 CoTaskMemFree((void *)devices);
326 devices = NULL;
327 }
328 if (attr) {
329 IMFAttributes_Release(attr);
330 attr = NULL;
331 }
332
333 *ctx = cam;
334
335 /* Register webcam context with named registry */
336 char device_name[32];
337 snprintf(device_name, sizeof(device_name), "device:%u", device_index);
338 NAMED_REGISTER(cam, device_name, "webcam", "0x%tx", NULL);
339
340 return ASCIICHAT_OK;
341
342error:
343 // Only cleanup devices and attr if they haven't been cleaned up yet
344 if (devices) {
345 for (UINT32 i = 0; i < count; i++) {
346 if (devices[i]) {
347 IMFActivate_Release(devices[i]);
348 }
349 }
350 CoTaskMemFree((void *)devices);
351 }
352 if (attr) {
353 IMFAttributes_Release(attr);
354 }
355 if (cam->reader) {
356 IMFSourceReader_Release(cam->reader);
357 }
358 if (cam->device) {
359 IMFMediaSource_Release(cam->device);
360 }
361 if (cam->mf_initialized) {
362 MFShutdown();
363 }
364 if (cam->com_initialized) {
365 CoUninitialize();
366 }
367 if (cam) {
368 SAFE_FREE(cam);
369 }
370 return result;
371}

386void webcam_cleanup_context(webcam_context_t *ctx) {
387 if (ctx) {
388 NAMED_UNREGISTER(ctx);
389
390 if (ctx->reader) {
391 // Flush before release to ensure no pending operations
392 webcam_flush_context(ctx);
393 IMFSourceReader_Release(ctx->reader);

397 IMFMediaSource_Release(ctx->device);
398 ctx->device = NULL;

402 ctx->mf_initialized = FALSE;
403 }
404 if (ctx->com_initialized) {
405 CoUninitialize();
406 ctx->com_initialized = FALSE;
407 }
408 // Free cached frame if it was allocated
409 if (ctx->cached_frame) {
410 image_destroy(ctx->cached_frame);
411 ctx->cached_frame = NULL;
412 }
413 SAFE_FREE(ctx);
414 log_debug("Windows Media Foundation webcam closed");
415 }

418image_t *webcam_read_context(webcam_context_t *ctx) {
419 if (!ctx || !ctx->reader) {
420 return NULL;
421 }
422
423 HRESULT hr;
424 IMFSample *sample = NULL;
425 DWORD streamIndex, flags;
426 LONGLONG timestamp;
427
428 // Timing diagnostic: measure ReadSample duration
429 LARGE_INTEGER freq, start, end;
430 QueryPerformanceFrequency(&freq);
431 QueryPerformanceCounter(&start);
432
433 // Read a sample from the source reader
434 // Use 0 for synchronous blocking read (DRAIN flag was wrong - that's for EOF)
435 hr = IMFSourceReader_ReadSample(ctx->reader, (DWORD)MF_SOURCE_READER_FIRST_VIDEO_STREAM,
436 0, // Regular synchronous read
437 &streamIndex, &flags, &timestamp, &sample);
438
439 QueryPerformanceCounter(&end);
440 double elapsed_ms = ((double)(end.QuadPart - start.QuadPart)) * 1000.0 / (double)freq.QuadPart;
441
442 char duration_str[32];
443 time_pretty((uint64_t)(elapsed_ms * 1e6), -1, duration_str, sizeof(duration_str));
444 log_info("ReadSample took %s (hr=0x%08x, flags=0x%08x, sample=%p)", duration_str, hr, flags, sample);
445
446 // Check for stream tick or other non-data flags
447 if (SUCCEEDED(hr) && (flags & MF_SOURCE_READERF_STREAMTICK)) {
448 log_info("Received stream tick, no sample yet");
449 return NULL;
450 }
451
452 if (SUCCEEDED(hr) && (flags & MF_SOURCE_READERF_ENDOFSTREAM)) {
453 log_warn("End of stream reached");
454 return NULL;
455 }
456
457 if (FAILED(hr)) {
458 log_error("CRITICAL: Failed to read MF sample on FIRST attempt: 0x%08x", hr);
459 log_error(" 0x80070005 = E_ACCESSDENIED (device in use)");
460 log_error(" 0xc00d3704 = Device already in use");
461 log_error(" 0xc00d3e85 = MF_E_VIDEO_RECORDING_DEVICE_INVALIDATED");
462 log_error(" 0x80004005 = E_FAIL (generic failure)");
463 log_error(" 0xc00d36b2 = MF_E_INVALIDREQUEST");
464 log_error(" 0xc00d36c4 = MF_E_HW_MFT_FAILED_START_STREAMING");
465
466 // Return NULL - device is likely in use
467 return NULL;
468 }
469
470 if (!sample) {
471 static int null_count = 0;
472 null_count++;
473 if (null_count <= 10) {
474 // This is normal when using DRAIN flag - it means no sample ready yet
475 if (null_count == 1) {
476 log_info("No sample available yet (this is normal during startup)");
477 }
478 } else if (null_count > 50) {
479 // Too many consecutive NULL samples - likely exclusive access issue
480 log_error("Too many consecutive NULL samples (%d) - device likely in use", null_count);
481 }
482 return NULL;
483 }
484
485 // Get the media buffer from the sample
486 IMFMediaBuffer *buffer = NULL;
487 hr = IMFSample_ConvertToContiguousBuffer(sample, &buffer);
488 if (FAILED(hr)) {
489 static int buffer_fail_count = 0;
490 buffer_fail_count++;
491 log_debug("Failed to get contiguous buffer: 0x%08x (failure #%d)", hr, buffer_fail_count);
492
493 if (buffer_fail_count > 20) {
494 log_error("CRITICAL: Failed to get media buffer %d times - webcam likely in use", buffer_fail_count);
495 }
496
497 IMFSample_Release(sample);
498 return NULL;
499 }
500
501 // Lock the buffer and get the data
502 BYTE *bufferData = NULL;
503 DWORD bufferLength = 0;
504 hr = IMFMediaBuffer_Lock(buffer, &bufferData, NULL, &bufferLength);
505 if (FAILED(hr)) {
506 static int lock_fail_count = 0;
507 lock_fail_count++;
508 log_debug("Failed to lock MF buffer: 0x%08x (failure #%d)", hr, lock_fail_count);
509
510 if (lock_fail_count > 20) {
511 log_error("CRITICAL: Failed to lock media buffer %d times - webcam likely in use", lock_fail_count);
512 }
513
514 IMFMediaBuffer_Release(buffer);
515 IMFSample_Release(sample);
516 return NULL;
517 }
518
519 // Use cached frame dimensions from context (no need to query media type every frame)
520 UINT32 width = (UINT32)ctx->width;
521 UINT32 height = (UINT32)ctx->height;
522
523 // Check for integer overflow before multiplication (prevents heap corruption).
524 // Check pixel count overflow
525 size_t pixel_count_check = 0;
526 if (checked_size_mul((size_t)width, (size_t)height, &pixel_count_check) != ASCIICHAT_OK) {
527 log_error("webcam_read_context: dimensions overflow: %u x %u", width, height);
528 IMFMediaBuffer_Unlock(buffer);
529 IMFMediaBuffer_Release(buffer);
530 IMFSample_Release(sample);
531 return NULL;
532 }
533
534 // Check pixel buffer size overflow
535 size_t pixel_buffer_size = 0;
536 if (checked_size_mul(pixel_count_check, sizeof(rgb_pixel_t), &pixel_buffer_size) != ASCIICHAT_OK) {
537 log_error("webcam_read_context: pixel buffer overflow: %zu pixels", pixel_count_check);
538 IMFMediaBuffer_Unlock(buffer);
539 IMFMediaBuffer_Release(buffer);
540 IMFSample_Release(sample);
541 return NULL;
542 }
543
544 // Allocate or reuse cached frame (allocated once, reused for each call)
545 // This matches the documented API contract: "Subsequent calls reuse the same buffer"
546 if (!ctx->cached_frame || ctx->cached_frame->w != (int)width || ctx->cached_frame->h != (int)height) {
547 // Free old cached frame if it exists with different dimensions
548 if (ctx->cached_frame) {
549 image_destroy(ctx->cached_frame);
550 }
551
552 // Create new image_t structure
553 image_t *img = SAFE_MALLOC(sizeof(image_t), image_t *);
554 img->w = (int)(unsigned int)width;
555 img->h = (int)(unsigned int)height;
556 // Use SIMD-aligned allocation for optimal NEON/AVX performance with vld3q_u8
557 img->pixels = SAFE_MALLOC_SIMD(pixel_buffer_size, rgb_pixel_t *);
558 ctx->cached_frame = img;
559 }
560
561 image_t *img = ctx->cached_frame;
562
563 // Copy RGB32 data (BGRA order in Media Foundation)
564 // Media Foundation converts YUV->RGB32 via GPU-accelerated pipeline
565 // RGB32 format is 4 bytes per pixel: B, G, R, A (we ignore alpha)
566
567 LARGE_INTEGER copy_start, copy_end;
568 QueryPerformanceCounter(&copy_start);
569
570 UINT32 pixel_count = width * height;
571 BYTE *src = bufferData;
572 rgb_pixel_t *dst = img->pixels;
573
574 // Optimized pixel copy - process 4 pixels at a time
575 UINT32 i;
576 for (i = 0; i + 4 <= pixel_count; i += 4) {
577 // Pixel 0
578 dst[0].b = src[0];
579 dst[0].g = src[1];
580 dst[0].r = src[2];
581 // Pixel 1
582 dst[1].b = src[4];
583 dst[1].g = src[5];
584 dst[1].r = src[6];
585 // Pixel 2
586 dst[2].b = src[8];
587 dst[2].g = src[9];
588 dst[2].r = src[10];
589 // Pixel 3
590 dst[3].b = src[12];
591 dst[3].g = src[13];
592 dst[3].r = src[14];
593
594 src += 16; // 4 pixels * 4 bytes
595 dst += 4;
596 }
597
598 // Handle remaining pixels
599 for (; i < pixel_count; i++) {
600 dst->b = src[0];
601 dst->g = src[1];
602 dst->r = src[2];
603 src += 4;
604 dst++;
605 }
606
607 QueryPerformanceCounter(&copy_end);
608 double copy_ms = ((double)(copy_end.QuadPart - copy_start.QuadPart)) * 1000.0 / (double)freq.QuadPart;
609
610 char copy_duration_str[32];
611 time_pretty((uint64_t)(copy_ms * 1e6), -1, copy_duration_str, sizeof(copy_duration_str));
612 log_info("Pixel copy took %s (%u pixels)", copy_duration_str, pixel_count);
613
614 // Unlock and cleanup
615 IMFMediaBuffer_Unlock(buffer);
616 IMFMediaBuffer_Release(buffer);
617 IMFSample_Release(sample);
618
619 return img;
620}

632asciichat_error_t webcam_list_devices(webcam_device_info_t **out_devices, unsigned int *out_count) {
633 if (!out_devices || !out_count) {
634 return SET_ERRNO(ERROR_INVALID_PARAM, "webcam_list_devices: invalid parameters");
635 }
636
637 *out_devices = NULL;
638 *out_count = 0;
639
640 IMFAttributes *attr = NULL;
641 IMFActivate **mf_devices = NULL;
642 UINT32 mf_count = 0;
643 HRESULT hr;
644 asciichat_error_t result = ASCIICHAT_OK;
645 BOOL com_initialized = FALSE;
646 BOOL mf_initialized = FALSE;
647
648 // Capture Windows Media Foundation's initialization output
649 LOG_IO("mf", {

651 hr = CoInitializeEx(NULL, COINIT_MULTITHREADED);
652 if (SUCCEEDED(hr)) {
653 com_initialized = TRUE;
654 } else if (hr != RPC_E_CHANGED_MODE) {
655 result = SET_ERRNO_SYS(ERROR_WEBCAM, "Failed to initialize COM: 0x%08x", hr);
656 hr = E_FAIL;
657 }
658

662 if (FAILED(hr)) {
663 result = SET_ERRNO_SYS(ERROR_WEBCAM, "Failed to startup Media Foundation: 0x%08x", hr);
664 } else {
665 mf_initialized = TRUE;
666 }
667 }
668 });

670 if (FAILED(hr)) {

672 }

674 // Create attribute store for device enumeration
675 hr = MFCreateAttributes(&attr, 1);
676 if (FAILED(hr)) {

678 goto cleanup;
679 }

682 hr =

684 if (FAILED(hr)) {

686 goto cleanup;
687 }

689 // Enumerate video capture devices
690 hr = MFEnumDeviceSources(attr, &mf_devices, &mf_count);
691 if (FAILED(hr)) {

693 goto cleanup;
694 }
695
696 if (mf_count == 0) {
697 // No devices found - not an error, just return empty list
698 goto cleanup;
699 }
700
701 // Allocate output array
702 webcam_device_info_t *devices = SAFE_CALLOC(mf_count, sizeof(webcam_device_info_t), webcam_device_info_t *);
703 if (!devices) {
704 result = SET_ERRNO(ERROR_MEMORY, "Failed to allocate device info array");
705 goto cleanup;
706 }
707
708 // Populate device info
709 for (UINT32 i = 0; i < mf_count; i++) {
710 devices[i].index = i;
711
712 LPWSTR friendlyName = NULL;
713 UINT32 nameLength = 0;
714
715 hr = IMFActivate_GetAllocatedString(mf_devices[i], &MF_DEVSOURCE_ATTRIBUTE_FRIENDLY_NAME, &friendlyName,
716 &nameLength);
717 if (SUCCEEDED(hr) && friendlyName) {
718 // Convert wide string to UTF-8
719 int len = WideCharToMultiByte(CP_UTF8, 0, friendlyName, -1, NULL, 0, NULL);
720 if (len > 0 && len < WEBCAM_DEVICE_NAME_MAX) {
721 WideCharToMultiByte(CP_UTF8, 0, friendlyName, -1, devices[i].name, WEBCAM_DEVICE_NAME_MAX, NULL);
722 } else {
723 SAFE_STRNCPY(devices[i].name, "<Unknown>", WEBCAM_DEVICE_NAME_MAX);
724 }
725 CoTaskMemFree(friendlyName);
726 } else {
727 SAFE_STRNCPY(devices[i].name, "<Unknown>", WEBCAM_DEVICE_NAME_MAX);
728 }
729 }
730
731 *out_devices = devices;
732 *out_count = mf_count;
733
734cleanup:
735 if (mf_devices) {
736 for (UINT32 i = 0; i < mf_count; i++) {
737 if (mf_devices[i]) {
738 IMFActivate_Release(mf_devices[i]);
739 }
740 }
741 CoTaskMemFree((void *)mf_devices);
742 }
743 if (attr) {
744 IMFAttributes_Release(attr);
745 }
746 if (mf_initialized) {
747 MFShutdown();
748 }
749 if (com_initialized) {
750 CoUninitialize();
751 }
752
753 return result;
754}
