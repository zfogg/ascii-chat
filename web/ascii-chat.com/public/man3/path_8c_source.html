87/* Normalize a path by resolving .. and . components
88 * Handles both Windows (\‍) and Unix (/) separators
89 * Returns a pointer to a static buffer (not thread-safe, but sufficient for __FILE__ normalization)
90 */
91static const char *normalize_path(const char *path) {
92 if (!path) {
93 SET_ERRNO(ERROR_INVALID_PARAM, "path is null");
94 return "unknown";
95 }
96
97 /* Use thread-local storage to avoid buffer corruption when multiple threads
98 * or rapid calls reuse the same static buffer. Each thread gets its own buffer. */
99 static _Thread_local char normalized[PLATFORM_MAX_PATH_LENGTH];
100 static _Thread_local char components[PLATFORM_MAX_PATH_LENGTH][256];
101 int component_count = 0;
102 size_t path_len = strlen(path);
103
104 if (path_len >= PLATFORM_MAX_PATH_LENGTH) {
105 return path; /* Can't normalize, return as-is */
106 }
107
108 const char *pos = path;
109 bool absolute = path_is_absolute(path);
110
111 /* Skip past the absolute path prefix if present */
112 if (absolute) {
113#ifdef _WIN32
114 if (path_len >= 3 && isalpha((unsigned char)path[0]) && path[1] == ':' && path[2] == PATH_DELIM) {
115 pos += 3; /* Skip the drive letter and colon and separator (e.g., "C:\") */
116 }
117#else
118 if (path_len >= 1 && path[0] == PATH_DELIM) {
119 pos += 1; /* Skip the root separator */
120 }
121#endif
122 }
123
124 /* Use PCRE2 to split by separators and detect dot components.
125 Allocate fresh match data for this operation and free immediately when done. */
126 pcre2_code *separator_regex = path_separator_regex_get();
127 pcre2_code *dot_regex = path_dot_component_regex_get();
128
129 if (separator_regex && dot_regex) {
130 pcre2_match_data *sep_match = path_match_data_create(separator_regex);
131 pcre2_match_data *dot_match = path_match_data_create(dot_regex);
132
133 if (sep_match && dot_match) {
134 size_t offset = 0;
135 size_t remaining_len = strlen(pos);
136 size_t last_component_end = 0;
137
138 while (offset <= remaining_len) {
139 /* Find next separator */
140 int rc = pcre2_jit_match(separator_regex, (PCRE2_SPTR8)pos, remaining_len, offset, 0, sep_match, NULL);
141 PCRE2_SIZE *ovector = pcre2_get_ovector_pointer(sep_match);
142 PCRE2_SIZE sep_start, sep_end;
143
144 if (rc >= 0) {
145 sep_start = ovector[0];
146 sep_end = ovector[1];
147 } else {
148 /* No more separators - handle last component if any */
149 sep_start = remaining_len;
150 sep_end = remaining_len;
151 }
152
153 /* Extract component between last_component_end and sep_start */
154 size_t component_len = sep_start - last_component_end;
155 if (component_len > 0) {
156 if (component_len >= sizeof(components[0])) {
157 component_len = sizeof(components[0]) - 1;
158 }
159
160 char temp_component[256];
161 memcpy(temp_component, pos + last_component_end, component_len);
162 temp_component[component_len] = '\0';
163
164 /* Check if component matches dot pattern (^\.\.*$) */
165 int is_dot = pcre2_jit_match(dot_regex, (PCRE2_SPTR8)temp_component, component_len, 0, 0, dot_match, NULL);
166
167 if (is_dot >= 0) {
168 /* It's ".", "..", or "..." */
169 if (component_len == 1) {
170 /* Skip "." */
171 } else if (component_len == 2) {
172 /* Handle ".." */
173 if (component_count > 0) {
174 component_count--;
175 } else if (!absolute) {
176 memcpy(components[component_count], temp_component, component_len);
177 components[component_count][component_len] = '\0';
178 component_count++;
179 }
180 } else {
181 /* "..." or more - treat as normal component */
182 memcpy(components[component_count], temp_component, component_len);
183 components[component_count][component_len] = '\0';
184 component_count++;
185 }
186 } else {
187 /* Normal component */
188 memcpy(components[component_count], temp_component, component_len);
189 components[component_count][component_len] = '\0';
190 component_count++;
191 }
192 }
193
194 if (rc < 0) {
195 break; /* No more separators */
196 }
197
198 offset = sep_end;
199 last_component_end = sep_end;
200 }
201
202 /* Free match data immediately - it was allocated for this operation only */
203 path_match_data_free(sep_match);
204 path_match_data_free(dot_match);
205 } else {
206 /* Fallback to manual parsing if match data allocation failed */
207 const char *parse_pos = pos;
208 while (*parse_pos) {
209 while (*parse_pos == '/' || *parse_pos == '\\') {
210 parse_pos++;
211 }
212 if (!*parse_pos)
213 break;
214 const char *component_start = parse_pos;
215 while (*parse_pos && *parse_pos != '/' && *parse_pos != '\\') {
216 parse_pos++;
217 }
218 size_t component_len = (size_t)(parse_pos - component_start);
219 if (component_len == 0)
220 continue;
221 if (component_len >= sizeof(components[0])) {
222 component_len = sizeof(components[0]) - 1;
223 }
224 if (component_len == 1 && component_start[0] == PATH_COMPONENT_DOT) {
225 continue;
226 }
227 if (component_len == 2 && component_start[0] == PATH_COMPONENT_DOT &&
228 component_start[1] == PATH_COMPONENT_DOT) {
229 if (component_count > 0) {
230 component_count--;
231 continue;
232 }
233 if (!absolute) {
234 memcpy(components[component_count], component_start, component_len);
235 components[component_count][component_len] = '\0';
236 component_count++;
237 }
238 continue;
239 }
240 memcpy(components[component_count], component_start, component_len);
241 components[component_count][component_len] = '\0';
242 component_count++;
243 }
244 }
245 } else {
246 /* Fallback to manual parsing if regex compilation failed */
247 const char *parse_pos = pos;
248 while (*parse_pos) {
249 while (*parse_pos == '/' || *parse_pos == '\\') {
250 parse_pos++;
251 }
252 if (!*parse_pos)
253 break;
254 const char *component_start = parse_pos;
255 while (*parse_pos && *parse_pos != '/' && *parse_pos != '\\') {
256 parse_pos++;
257 }
258 size_t component_len = (size_t)(parse_pos - component_start);
259 if (component_len == 0)
260 continue;
261 if (component_len >= sizeof(components[0])) {
262 component_len = sizeof(components[0]) - 1;
263 }
264 if (component_len == 1 && component_start[0] == PATH_COMPONENT_DOT) {
265 continue;
266 }
267 if (component_len == 2 && component_start[0] == PATH_COMPONENT_DOT && component_start[1] == PATH_COMPONENT_DOT) {
268 if (component_count > 0) {
269 component_count--;
270 continue;
271 }
272 if (!absolute) {
273 memcpy(components[component_count], component_start, component_len);
274 components[component_count][component_len] = '\0';
275 component_count++;
276 }
277 continue;
278 }
279 memcpy(components[component_count], component_start, component_len);
280 components[component_count][component_len] = '\0';
281 component_count++;
282 }
283 }
284
285 /* Build normalized path */
286 size_t out_pos = 0;
287#ifdef _WIN32
288 if (absolute && path_len >= 3 && isalpha((unsigned char)path[0]) && path[1] == ':') {
289 normalized[out_pos++] = path[0];
290 normalized[out_pos++] = ':';
291 normalized[out_pos++] = PATH_DELIM;
292 }
293#else
294 if (absolute) {
295 normalized[out_pos++] = PATH_DELIM;
296 }
297#endif
298
299 for (int i = 0; i < component_count; i++) {
300 if (i > 0) {
301 normalized[out_pos++] = PATH_DELIM;
302 }
303 size_t comp_len = strlen(components[i]);
304 if (out_pos + comp_len >= PLATFORM_MAX_PATH_LENGTH) {
305 break;
306 }
307 memcpy(normalized + out_pos, components[i], comp_len);
308 out_pos += comp_len;
309 }
310
311 normalized[out_pos] = '\0';
312 return normalized;
313}

330static char *find_project_root(void) {
331 // Cache the result so we don't repeatedly call stat() which can hang during shutdown
332 static char cached_root[PLATFORM_MAX_PATH_LENGTH] = {0};
333 static bool search_done = false;
334
335 if (search_done) {
336 if (cached_root[0] != '\0') {
337 // Return a new allocation using malloc (not SAFE_MALLOC) to avoid debug_malloc
338 // recursion when called during memory report while holding g_mem.mutex
339 char *copy = malloc(strlen(cached_root) + 1);
340 if (copy) {
341 strcpy(copy, cached_root);
342 }
343 return copy;
344 }
345 return NULL; // Not found
346 }
347
348 char cwd_buf[PLATFORM_MAX_PATH_LENGTH];
349 if (!platform_get_cwd(cwd_buf, sizeof(cwd_buf))) {
350 search_done = true;
351 return NULL;
352 }
353
354 char search_path[PLATFORM_MAX_PATH_LENGTH];
355 SAFE_STRNCPY(search_path, cwd_buf, sizeof(search_path));
356
357 // Search up to 128 levels deep for repository root
358 for (int i = 0; i < 128; i++) {
359 // Check for .git directory
360 char git_check[PLATFORM_MAX_PATH_LENGTH];
361 safe_snprintf(git_check, sizeof(git_check), "%s%s.git", search_path, PATH_SEPARATOR_STR);
362 struct stat git_stat;
363 bool git_found = stat(git_check, &git_stat) == 0 && S_ISDIR(git_stat.st_mode);
364 if (git_found) {
365 // Found .git directory - this is the repo root
366 safe_snprintf(cached_root, sizeof(cached_root), "%s", search_path);
367 search_done = true;
368 // Return a new allocation using malloc (not SAFE_MALLOC) to avoid debug_malloc
369 // recursion when called during memory report while holding g_mem.mutex
370 char *result = malloc(strlen(search_path) + 1);
371 if (result) {
372 strcpy(result, search_path);
373 }
374 return result;
375 }
376
377 // We might be built from a tarball and not a git repo. But I still want this to work.
378 // So check for LICENSE.txt and CMakeLists.txt (ascii-chat's license is in the repo root with the main
379 // CMakeLists.txt).
380 char license_check[PLATFORM_MAX_PATH_LENGTH];
381 safe_snprintf(license_check, sizeof(license_check), "%s%sLICENSE.txt", search_path, PATH_SEPARATOR_STR);
382 char cmake_check[PLATFORM_MAX_PATH_LENGTH];
383 safe_snprintf(cmake_check, sizeof(cmake_check), "%s%sCMakeLists.txt", search_path, PATH_SEPARATOR_STR);
384 struct stat license_stat;
385 bool license_found = stat(license_check, &license_stat) == 0 && S_ISREG(license_stat.st_mode);
386 struct stat cmake_stat;
387 bool cmake_found = stat(cmake_check, &cmake_stat) == 0 && S_ISREG(cmake_stat.st_mode);
388 if (license_found && cmake_found) {
389 // Found LICENSE.txt
390 safe_snprintf(cached_root, sizeof(cached_root), "%s", search_path);
391 search_done = true;
392 // Return a new allocation using malloc (not SAFE_MALLOC) to avoid debug_malloc
393 // recursion when called during memory report while holding g_mem.mutex
394 char *result = malloc(strlen(search_path) + 1);
395 if (result) {
396 strcpy(result, search_path);
397 }
398 return result;
399 }
400
401 // Go up one directory
402 size_t len = strlen(search_path);
403 if (len == 0 || (len == 1 && search_path[0] == '/')) {
404 // Hit root directory without finding .git or CMakeLists.txt
405 break;
406 }
407
408 // Remove trailing separator if present
409 if (search_path[len - 1] == '/' || search_path[len - 1] == '\\') {
410 search_path[len - 1] = '\0';
411 len--;
412 }
413
414 // Find last separator
415 int last_sep = -1;
416 for (int j = (int)len - 1; j >= 0; j--) {
417 if (search_path[j] == '/' || search_path[j] == '\\') {
418 last_sep = j;
419 break;
420 }
421 }
422
423 if (last_sep <= 0) {
424 // Can't go up further
425 break;
426 }
427
428 search_path[last_sep] = '\0';
429 }
430
431 search_done = true;
432 return NULL;
433}

435const char *extract_project_relative_path(const char *file) {
436 if (!file) {
437 SET_ERRNO(ERROR_INVALID_PARAM, "file is null");
438 return "unknown";
439 }
440
441#ifdef __EMSCRIPTEN__
442 // WASM builds: Return raw path to avoid recursion in logging system
443 // extract_project_relative_path is called from format_log_header, which is part of logging.
444 // If we call normalize_path or other functions that might use SET_ERRNO or log functions,
445 // we create infinite recursion: log_msg -> format_log_header -> extract_project_relative_path -> SET_ERRNO ->
446 // log_error -> log_msg For WASM, just return the filename without path processing.
447 const char *sep = strrchr(file, '/');
448 if (!sep) {
449 sep = strrchr(file, '\\');
450 }
451 return sep ? sep + 1 : file;
452#endif
453
454 /* First normalize the path to resolve .. and . components */
455 const char *normalized = normalize_path(file);
456
457 /* Try to find and strip the project root from the absolute path */
458 char *project_root = find_project_root();
459 if (project_root) {
460 size_t root_len = strlen(project_root);
461 size_t norm_len = strlen(normalized);
462
463 /* Check if normalized path starts with project root */
464 if (norm_len > root_len && strncmp(normalized, project_root, root_len) == 0) {
465 const char *remainder = normalized + root_len;
466
467 /* Skip the separator if present */
468 if (*remainder == PATH_DELIM || *remainder == '/' || *remainder == '\\') {
469 remainder++;
470 }
471
472 free(project_root);
473 return remainder;
474 }
475 free(project_root);
476 }
477
478 /* Fallback: Project root detection failed, but we should still try to return
479 * a meaningful relative path, not just the filename. Return the normalized path
480 * which should contain the full directory structure. */
481 return normalized;
482}

689bool path_normalize_copy(const char *path, char *out, size_t out_len) {
690 if (!path || !out || out_len == 0) {
691 SET_ERRNO(ERROR_INVALID_PARAM, "null path or out or out_len is 0");
692 return false;
693 }
694
695 const char *normalized = normalize_path(path);
696 if (!normalized) {
697 return false;
698 }
699
700 size_t len = strlen(normalized);
701 if (len + 1 > out_len) {
702 return false;
703 }
704
705 memcpy(out, normalized, len + 1);

737 char normalized_base[PLATFORM_MAX_PATH_LENGTH];

741 }

845static bool is_sensitive_system_path(const char *path) {
846 if (!path) {
847 return false;
848 }
849
850#ifdef _WIN32
851 // Windows system directories
852 const char *sensitive_paths[] = {"C:\\Windows", // System directory
853 "C:\\Program Files", // Program files
854 "C:\\Program Files (x86)", // 32-bit programs
855 "C:\\ProgramData", // All users data
856 "C:\\System Volume Information", // System recovery
857 "C:\\PerfLogs", // Performance logs
858 NULL};
859#else
860 // Unix/Linux/macOS system directories
861 const char *sensitive_paths[] = {"/etc", // System configuration
862 "/bin", // Essential binaries
863 "/sbin", // System binaries
864 "/usr/bin", // User binaries
865 "/usr/sbin", // User system binaries
866 "/usr/lib", // System libraries
867 "/lib", // Libraries
868 "/lib64", // 64-bit libraries
869 "/boot", // Boot files
870 "/sys", // System interface
871 "/proc", // Process interface
872 "/dev", // Devices
873 "/root", // Root home (should not write to)
874 "/var/lib", // Variable library data
875 "/var/cache", // Cache data
876 "/var/spool", // Spool data
877 NULL};
878
879#ifdef __APPLE__
880 // macOS-specific system paths
881 const char *macos_paths[] = {"/System", // Core system
882 "/Library", // System library
883 "/Applications", // Bundled apps
884 "/Developer", // Developer tools
885 "/Volumes", // Mounted volumes
886 NULL};
887#endif
888#endif
889
890 // Check each sensitive path
891 for (int i = 0; sensitive_paths[i] != NULL; i++) {
892 const char *base = sensitive_paths[i];
893 size_t base_len = strlen(base);
894
895 // Match if path equals base or starts with base + separator
896 if (strcmp(path, base) == 0) {
897 return true; // Exact match is sensitive
898 }
899 if (strncmp(path, base, base_len) == 0) {
900 // Make sure it's followed by a path separator, not a partial match
901 if (path[base_len] == PATH_DELIM || path[base_len] == '/' || path[base_len] == '\\') {
902 return true;
903 }
904 }
905 }
906
907#ifdef __APPLE__
908 // Check macOS paths
909 for (int i = 0; macos_paths[i] != NULL; i++) {
910 const char *base = macos_paths[i];
911 size_t base_len = strlen(base);
912
913 if (strcmp(path, base) == 0) {
914 return true;
915 }
916 if (strncmp(path, base, base_len) == 0) {
917 if (path[base_len] == PATH_DELIM || path[base_len] == '/' || path[base_len] == '\\') {
918 return true;
919 }
920 }
921 }
922#endif
923
924 return false;
925}

954static bool is_existing_ascii_chat_log(const char *path) {
955 if (!path) {
956 return false;
957 }
958
959 // Check if path is a regular file (not a directory) using platform abstraction
960 if (!platform_is_regular_file(path)) {
961 return false; // Not a regular file (could be directory, doesn't exist, symlink, etc.)
962 }
963
964 // Try to open and read the first line
965 FILE *f = platform_fopen("file_stream", path, "r");
966 if (!f) {
967 return false; // Can't read file
968 }
969
970 char buffer[256];
971 bool is_ascii_chat_log = false;
972
973 // Read first line and check for ascii-chat log signature
974 if (fgets(buffer, sizeof(buffer), f) != NULL) {
975 // ascii-chat logs start with timestamps like: [HH:MM:SS.microseconds] [LEVEL]
976 // Pattern: [digit][digit]:[digit][digit]:[digit][digit].[digits]
977 if (buffer[0] == '[' && isdigit((unsigned char)buffer[1]) && isdigit((unsigned char)buffer[2]) &&
978 buffer[3] == ':') {
979 is_ascii_chat_log = true;
980 }
981 }

983 fclose(f);

987asciichat_error_t path_validate_user_path(const char *input, path_role_t role, char **normalized_out) {
988 if (!normalized_out) {
989 return SET_ERRNO(map_role_to_error(role), "path_validate_user_path requires output pointer");
990 }
991 *normalized_out = NULL;
992
993 if (!input || *input == '\0') {
994 return SET_ERRNO(map_role_to_error(role), "Path is empty for role %d", role);
995 }
996
997 // SECURITY: For log files, if input is a simple filename (no separators or ..), constrain it to a safe directory
998 if (role == PATH_ROLE_LOG_FILE) {
999 // Check if input contains path separators or parent directory references
1000 bool is_simple_filename = true;
1001 for (const char *p = input; *p; p++) {
1002 if (*p == PATH_DELIM || *p == '/' || *p == '\\') {
1003 is_simple_filename = false;
1004 break;
1005 }
1006 }
1007 // Also reject ".." components (even without separators like "..something")
1008 if (strstr(input, "..") != NULL) {
1009 is_simple_filename = false;
1010 }
1011
1012 // If it's a simple filename, resolve it to a safe base directory
1013 if (is_simple_filename) {
1014 // Always prefer current working directory for simple log filenames
1015 // This ensures logs go to where the user is running the command from
1016 char safe_base[PLATFORM_MAX_PATH_LENGTH];
1017
1018 if (!platform_get_cwd(safe_base, sizeof(safe_base))) {
1019 // If cwd fails, try config dir as fallback
1020 char *config_dir = get_config_dir();
1021 if (config_dir) {
1022 SAFE_STRNCPY(safe_base, config_dir, sizeof(safe_base));
1023 SAFE_FREE(config_dir);
1024 } else {
1025 return SET_ERRNO(ERROR_LOGGING_INIT, "Failed to determine safe directory for log file");
1026 }
1027 }
1028
1029 // Build the full path: safe_base + separator + input
1030 size_t base_len = strlen(safe_base);
1031 size_t input_len = strlen(input);
1032 bool needs_sep = base_len > 0 && safe_base[base_len - 1] != PATH_DELIM;
1033 size_t total_len = base_len + (needs_sep ? 1 : 0) + input_len + 1;
1034
1035 if (total_len > PLATFORM_MAX_PATH_LENGTH) {
1036 return SET_ERRNO(ERROR_LOGGING_INIT, "Log file path too long: %s/%s", safe_base, input);
1037 }
1038
1039 char resolved_buf[PLATFORM_MAX_PATH_LENGTH];
1040 safe_snprintf(resolved_buf, sizeof(resolved_buf), "%s%s%s", safe_base, needs_sep ? PATH_SEPARATOR_STR : "",
1041 input);
1042
1043 // Normalize the resolved path
1044 char normalized_buf[PLATFORM_MAX_PATH_LENGTH];
1045 if (!path_normalize_copy(resolved_buf, normalized_buf, sizeof(normalized_buf))) {
1046 return SET_ERRNO(ERROR_LOGGING_INIT, "Failed to normalize log file path: %s", resolved_buf);
1047 }
1048
1049 // Allocate and return the result
1050 char *result = SAFE_MALLOC(strlen(normalized_buf) + 1, char *);
1051 if (!result) {
1052 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate normalized path");
1053 }
1054 safe_snprintf(result, strlen(normalized_buf) + 1, "%s", normalized_buf);
1055 *normalized_out = result;
1056 return ASCIICHAT_OK;
1057 }
1058 // If not a simple filename (contains separators), continue with normal validation below
1059 }
1060
1061 // For non-log-files or log files with path separators, validate as usual
1062 if (role != PATH_ROLE_LOG_FILE && !path_looks_like_path(input)) {
1063 return SET_ERRNO(map_role_to_error(role), "Value does not look like a filesystem path: %s", input);
1064 }
1065
1066 char *expanded = expand_path(input);
1067 if (!expanded) {
1068 return SET_ERRNO(map_role_to_error(role), "Failed to expand path: %s", input);
1069 }
1070
1071 char candidate_buf[PLATFORM_MAX_PATH_LENGTH];
1072 const char *candidate_path = expanded;
1073
1074 if (!path_is_absolute(candidate_path)) {
1075 char cwd_buf[PLATFORM_MAX_PATH_LENGTH];
1076 if (!platform_get_cwd(cwd_buf, sizeof(cwd_buf))) {
1077 SAFE_FREE(expanded);
1078 return SET_ERRNO(map_role_to_error(role), "Failed to determine current working directory");
1079 }
1080
1081 size_t total_len = strlen(cwd_buf) + 1 + strlen(candidate_path) + 1;
1082 if (total_len >= sizeof(candidate_buf)) {
1083 SAFE_FREE(expanded);
1084 return SET_ERRNO(map_role_to_error(role), "Resolved path is too long: %s/%s", cwd_buf, candidate_path);
1085 }
1086 if (strlen(candidate_path) > 0 && candidate_path[0] == PATH_DELIM) {
1087 safe_snprintf(candidate_buf, sizeof(candidate_buf), "%s%s", cwd_buf, candidate_path);
1088 } else {
1089 safe_snprintf(candidate_buf, sizeof(candidate_buf), "%s%c%s", cwd_buf, PATH_DELIM, candidate_path);
1090 }
1091 candidate_path = candidate_buf;
1092 }
1093
1094 char normalized_buf[PLATFORM_MAX_PATH_LENGTH];
1095 if (!path_normalize_copy(candidate_path, normalized_buf, sizeof(normalized_buf))) {
1096 SAFE_FREE(expanded);
1097 return SET_ERRNO(map_role_to_error(role), "Failed to normalize path: %s", candidate_path);
1098 }
1099
1100 if (!path_is_absolute(normalized_buf)) {
1101 SAFE_FREE(expanded);
1102 return SET_ERRNO(map_role_to_error(role), "Normalized path is not absolute: %s", normalized_buf);
1103 }
1104
1105 const char *bases[MAX_PATH_BASES] = {0};
1106 size_t base_count = 0;
1107
1108 // Always add current working directory as an allowed base
1109 // This is critical for log files and other paths relative to where the user runs the command
1110 char cwd_base[PLATFORM_MAX_PATH_LENGTH];
1111 if (platform_get_cwd(cwd_base, sizeof(cwd_base))) {
1112 append_base_if_valid(cwd_base, bases, &base_count);
1113 }
1114
1115 char temp_base[PLATFORM_MAX_PATH_LENGTH];
1116 if (platform_get_temp_dir(temp_base, sizeof(temp_base))) {
1117 append_base_if_valid(temp_base, bases, &base_count);
1118 }
1119
1120 char *config_dir = get_config_dir();
1121 if (config_dir) {
1122 append_base_if_valid(config_dir, bases, &base_count);
1123 }
1124
1125 const char *home_env = platform_get_home_dir();
1126 if (home_env) {
1127 append_base_if_valid(home_env, bases, &base_count);
1128 }
1129
1130 char ascii_chat_home[PLATFORM_MAX_PATH_LENGTH];
1131 if (home_env) {
1132 build_ascii_chat_path(home_env, ".ascii-chat", ascii_chat_home, sizeof(ascii_chat_home));
1133 append_base_if_valid(ascii_chat_home, bases, &base_count);
1134 }
1135
1136#ifndef _WIN32
1137 char ascii_chat_home_tmp[PLATFORM_MAX_PATH_LENGTH];
1138 build_ascii_chat_path("/tmp", ".ascii-chat", ascii_chat_home_tmp, sizeof(ascii_chat_home_tmp));
1139 append_base_if_valid(ascii_chat_home_tmp, bases, &base_count);
1140#endif
1141
1142 char ssh_home[PLATFORM_MAX_PATH_LENGTH];
1143 if (home_env) {
1144 build_ascii_chat_path(home_env, ".ssh", ssh_home, sizeof(ssh_home));
1145 append_base_if_valid(ssh_home, bases, &base_count);
1146 }
1147
1148#ifdef _WIN32
1149 char program_data_logs[PLATFORM_MAX_PATH_LENGTH];
1150 const char *program_data = platform_getenv("PROGRAMDATA");
1151 if (program_data) {
1152 build_ascii_chat_path(program_data, "ascii-chat", program_data_logs, sizeof(program_data_logs));
1153 append_base_if_valid(program_data_logs, bases, &base_count);
1154 }
1155#else
1156 // System-wide config directories (for server deployments)
1157 append_base_if_valid("/etc/ascii-chat", bases, &base_count);
1158 append_base_if_valid("/usr/local/etc/ascii-chat", bases, &base_count);
1159 append_base_if_valid("/var/log", bases, &base_count);
1160 append_base_if_valid("/var/tmp", bases, &base_count);
1161 append_base_if_valid("/tmp", bases, &base_count);
1162#ifdef __APPLE__
1163 // On macOS, /tmp is a symlink to /private/tmp
1164 append_base_if_valid("/private/tmp", bases, &base_count);
1165 // On macOS, all user home directories are under /Users
1166 append_base_if_valid("/Users", bases, &base_count);
1167#endif
1168#endif
1169
1170 // Security check: Reject paths that point to sensitive system files
1171 // This applies to all path roles, not just logs
1172 if (is_sensitive_system_path(normalized_buf)) {
1173 SAFE_FREE(expanded);
1174 if (config_dir) {
1175 SAFE_FREE(config_dir);
1176 }
1177 return SET_ERRNO(map_role_to_error(role), "Cannot write to protected system path: %s", normalized_buf);
1178 }
1179
1180 // For log files, apply special validation rules
1181 if (role == PATH_ROLE_LOG_FILE) {
1182 // Check if path is a regular file (not a directory, not non-existent)
1183 // On macOS, fopen() can succeed on directories, so we must use platform_is_regular_file()
1184 bool is_regular_file = platform_is_regular_file(normalized_buf);
1185
1186 // If a regular file exists, it MUST be an ascii-chat log or empty file to be overwritten
1187 if (is_regular_file && !is_existing_ascii_chat_log(normalized_buf) && !is_file_empty(normalized_buf)) {
1188 SAFE_FREE(expanded);
1189 if (config_dir) {
1190 SAFE_FREE(config_dir);
1191 }
1192 return SET_ERRNO(ERROR_LOGGING_INIT,
1193 "Cannot overwrite existing non-ascii-chat file: %s\n"
1194 "For safety, ascii-chat will only overwrite its own log files or empty files",
1195 normalized_buf);
1196 }
1197
1198 // If file doesn't exist (or is a directory), check that path is in safe locations
1199 if (!is_regular_file) {
1200 bool allowed = base_count == 0 ? true : path_is_within_any_base(normalized_buf, bases, base_count);
1201 if (!allowed) {
1202 SAFE_FREE(expanded);
1203 if (config_dir) {
1204 SAFE_FREE(config_dir);
1205 }
1206 return SET_ERRNO(ERROR_LOGGING_INIT,
1207 "Log path %s is outside allowed directories (use -L /tmp/file.log, ~/file.log, or "
1208 "relative/absolute paths in safe locations)",
1209 normalized_buf);
1210 }
1211 }
1212 } else {
1213 // For non-log-file paths, apply standard whitelist validation
1214 bool allowed = base_count == 0 ? true : path_is_within_any_base(normalized_buf, bases, base_count);
1215 if (!allowed) {
1216 SAFE_FREE(expanded);
1217 if (config_dir) {
1218 SAFE_FREE(config_dir);
1219 }
1220 return SET_ERRNO(map_role_to_error(role), "Path %s is outside allowed directories", normalized_buf);
1221 }
1222 }
1223
1224 char *result = SAFE_MALLOC(strlen(normalized_buf) + 1, char *);
1225 if (!result) {
1226 SAFE_FREE(expanded);
1227 if (config_dir) {
1228 SAFE_FREE(config_dir);
1229 }
1230 return SET_ERRNO(ERROR_MEMORY, "Failed to allocate normalized path");
1231 }
1232 safe_snprintf(result, strlen(normalized_buf) + 1, "%s", normalized_buf);
1233 *normalized_out = result;
1234
1235 SAFE_FREE(expanded);
1236 if (config_dir) {
1237 SAFE_FREE(config_dir);
1238 }
1239 return ASCIICHAT_OK;
1240}
