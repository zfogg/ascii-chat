60asciichat_error_t update_check_load_cache(update_check_result_t *result) {
61 if (!result) {
62 return SET_ERRNO(ERROR_INVALID_PARAM, "NULL result pointer");
63 }
64
65 memset(result, 0, sizeof(*result));
66
67 char *cache_path = get_cache_file_path();
68 if (!cache_path) {
69 return SET_ERRNO(ERROR_FILE_OPERATION, "Could not determine cache file path");
70 }
71
72 FILE *f = platform_fopen("file_stream", cache_path, "r");
73 if (!f) {
74 const char *error_msg = file_read_error_message(cache_path);
75 SAFE_FREE(cache_path);
76 return SET_ERRNO(ERROR_FILE_OPERATION, "%s", error_msg);
77 }
78
79 // Read line 1: timestamp
80 char line[512];
81 if (!fgets(line, sizeof(line), f)) {
82 fclose(f);
83 SAFE_FREE(cache_path);
84 return SET_ERRNO(ERROR_FILE_OPERATION, "Failed to read timestamp from cache");
85 }
86 result->last_check_time = (time_t)atoll(line);
87
88 // Read line 2: latest version (may be empty if check failed)
89 if (fgets(line, sizeof(line), f)) {
90 // Remove newline
91 size_t len = strlen(line);
92 if (len > 0 && line[len - 1] == '\n') {
93 line[len - 1] = '\0';
94 }
95 SAFE_STRNCPY(result->latest_version, line, sizeof(result->latest_version));
96 }
97
98 // Read line 3: latest SHA (may be empty if check failed)
99 if (fgets(line, sizeof(line), f)) {
100 // Remove newline
101 size_t len = strlen(line);
102 if (len > 0 && line[len - 1] == '\n') {
103 line[len - 1] = '\0';
104 }
105 SAFE_STRNCPY(result->latest_sha, line, sizeof(result->latest_sha));
106 }
107
108 fclose(f);
109
110 // Fill in current version/SHA
111 SAFE_STRNCPY(result->current_version, ASCII_CHAT_VERSION_STRING, sizeof(result->current_version));
112 SAFE_STRNCPY(result->current_sha, ASCII_CHAT_GIT_COMMIT_HASH, sizeof(result->current_sha));
113
114 // Determine if update is available using version comparison (if we have cached data)
115 if (result->latest_version[0] != '\0') {
116 semantic_version_t current_ver = version_parse(result->current_version);
117 semantic_version_t latest_ver = version_parse(result->latest_version);
118
119 if (current_ver.valid && latest_ver.valid) {
120 int cmp = version_compare(latest_ver, current_ver);
121 result->update_available = (cmp > 0); // Update available if latest > current
122 result->check_succeeded = true;
123 } else {
124 // Cache contains invalid version data - delete it
125 log_warn("Update cache contains invalid version data (current:%s, latest:%s) - deleting corrupted cache",
126 result->current_version, result->latest_version);
127 if (remove(cache_path) != 0) {
128 log_warn("Failed to delete corrupted cache file: %s", cache_path);
129 }
130 SAFE_FREE(cache_path);
131 return SET_ERRNO(ERROR_FORMAT, "Corrupted cache file deleted");
132 }
133 }
134
135 SAFE_FREE(cache_path);
136 return ASCIICHAT_OK;
137}

190static bool parse_json_string_field(const char *json, const char *field, char *output, size_t output_size) {
191 if (!json || !field || !output || output_size == 0) {
192 return false;
193 }
194
195 // Build search pattern: "field":"value"
196 char pattern[128];
197 snprintf(pattern, sizeof(pattern), "\"%s\"", field);
198
199 const char *field_start = strstr(json, pattern);
200 if (!field_start) {
201 return false;
202 }
203
204 // Skip past field name and find opening quote
205 const char *value_start = strchr(field_start + strlen(pattern), '"');
206 if (!value_start) {
207 return false;
208 }
209 value_start++; // Skip opening quote
210
211 // Find closing quote
212 const char *value_end = strchr(value_start, '"');
213 if (!value_end) {
214 return false;
215 }
216
217 // Copy value
218 size_t value_len = value_end - value_start;
219 if (value_len >= output_size) {
220 value_len = output_size - 1;
221 }
222
223 memcpy(output, value_start, value_len);
224 output[value_len] = '\0';
225
226 return true;
227}

267asciichat_error_t update_check_perform(update_check_result_t *result) {
268 if (!result) {
269 return SET_ERRNO(ERROR_INVALID_PARAM, "NULL result pointer");
270 }
271
272 memset(result, 0, sizeof(*result));
273
274 // Fill in current version and SHA
275 SAFE_STRNCPY(result->current_version, ASCII_CHAT_VERSION_STRING, sizeof(result->current_version));
276 SAFE_STRNCPY(result->current_sha, ASCII_CHAT_GIT_COMMIT_HASH, sizeof(result->current_sha));
277 result->last_check_time = time(NULL);
278
279 // Test DNS connectivity first
280 if (!dns_test_connectivity(GITHUB_API_HOSTNAME)) {
281 log_warn("No internet connectivity detected, skipping update check");
282 // Don't update cache - we'll retry when online
283 return SET_ERRNO(ERROR_NETWORK, "DNS connectivity test failed");
284 }
285
286 // Fetch latest release from GitHub API
287 log_info("Checking for updates from GitHub releases...");
288 char *response = https_get(GITHUB_API_HOSTNAME, GITHUB_RELEASES_PATH);
289 if (!response) {
290 log_warn("Failed to fetch GitHub releases API (timeout or network error)");
291 // Mark as checked even though it failed (prevents repeated offline attempts)
292 result->check_succeeded = false;
293 update_check_save_cache(result);
294 return SET_ERRNO(ERROR_NETWORK, "Failed to fetch GitHub releases");
295 }
296
297 // Parse JSON response
298 char latest_tag[64] = {0};
299 char latest_sha[41] = {0};
300 char release_url[512] = {0};
301
302 if (!parse_github_release_json(response, latest_tag, sizeof(latest_tag), latest_sha, sizeof(latest_sha), release_url,
303 sizeof(release_url))) {
304 log_error("Failed to parse GitHub API response");
305 SAFE_FREE(response);
306 result->check_succeeded = false;
307 update_check_save_cache(result);
308 return SET_ERRNO(ERROR_FORMAT, "Failed to parse GitHub API JSON");
309 }
310
311 SAFE_FREE(response);
312
313 // Fill in result
314 SAFE_STRNCPY(result->latest_version, latest_tag, sizeof(result->latest_version));
315 SAFE_STRNCPY(result->latest_sha, latest_sha, sizeof(result->latest_sha));
316 SAFE_STRNCPY(result->release_url, release_url, sizeof(result->release_url));
317 result->check_succeeded = true;
318
319 // Compare versions semantically
320 semantic_version_t current_ver = version_parse(result->current_version);
321 semantic_version_t latest_ver = version_parse(result->latest_version);
322
323 if (!current_ver.valid || !latest_ver.valid) {
324 log_warn("Failed to parse version strings for comparison (current: %s, latest: %s)", result->current_version,
325 result->latest_version);
326 result->update_available = false;
327 } else {
328 int cmp = version_compare(latest_ver, current_ver);
329 result->update_available = (cmp > 0); // Update available if latest > current
330 }
331
332 if (result->update_available) {
333 log_info("Update available: %s (%.*s) → %s (%.*s)", result->current_version, 8, result->current_sha,
334 result->latest_version, 8, result->latest_sha);
335 } else {
336 log_info("Already on latest version: %s (%.*s)", result->current_version, 8, result->current_sha);
337 }
338
339 // Save to cache
340 update_check_save_cache(result);
341
342 return ASCIICHAT_OK;
343}

399void update_check_get_upgrade_suggestion(install_method_t method, const char *latest_version, char *buffer,
400 size_t buffer_size) {
401 if (!buffer || buffer_size == 0) {
402 return;
403 }
404
405 switch (method) {
406 case INSTALL_METHOD_HOMEBREW:
407 snprintf(buffer, buffer_size, "brew upgrade ascii-chat");
408 break;
409
410 case INSTALL_METHOD_ARCH_AUR:
411 // Check if paru is available, otherwise suggest yay
412 if (system("command -v paru >/dev/null 2>&1") == 0) {
413 snprintf(buffer, buffer_size, "paru -S ascii-chat");
414 } else {
415 snprintf(buffer, buffer_size, "yay -S ascii-chat");
416 }
417 break;
418
419 case INSTALL_METHOD_GITHUB:
420 case INSTALL_METHOD_UNKNOWN:
421 default:
422 snprintf(buffer, buffer_size, "https://github.com/zfogg/ascii-chat/releases/tag/%s",
423 latest_version ? latest_version : "latest");
424 break;
425 }
426}

444asciichat_error_t update_check_startup(update_check_result_t *result) {
445 update_check_result_t local_result;
446 update_check_result_t *target = result ? result : &local_result;
447
448 // Try to load from cache first
449 asciichat_error_t cache_err = update_check_load_cache(target);
450 if (cache_err == ASCIICHAT_OK && update_check_is_cache_fresh(target)) {
451 // Cache is fresh, use it
452 log_debug("Using cached update check result (age: %.1f days)",
453 (time(NULL) - target->last_check_time) / (double)SEC_PER_DAY);
454 return ASCIICHAT_OK;
455 }
456
457 // Cache is stale or missing, perform fresh check
458 log_debug("Performing automatic update check (cache %s)", cache_err == ASCIICHAT_OK ? "stale" : "missing");
459 asciichat_error_t check_err = update_check_perform(target);
460 if (check_err != ASCIICHAT_OK) {
461 // Check failed, but don't fail startup
462 log_debug("Automatic update check failed (continuing startup)");
463 return check_err;
464 }
465
466 return ASCIICHAT_OK;
467}
