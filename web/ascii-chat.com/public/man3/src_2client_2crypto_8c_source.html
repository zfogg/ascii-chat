273 // Initialize crypto context (generates ephemeral X25519 keys)
274 result = crypto_handshake_init("crypto_client_local", &g_crypto_ctx, false); // false = client

303 g_crypto_ctx.client_gpg_key_id[0] = '\0';

325 // Password provided - use password-based initialization
326 log_debug("CLIENT_CRYPTO_INIT: Using password authentication");
327 result = crypto_handshake_init_with_password("crypto_client_local", &g_crypto_ctx, false,
328 GET_OPTION(password)); // false = client
329 if (result != ASCIICHAT_OK) {
330 FATAL(result, "Failed to initialize crypto handshake with password");
331 }
332 } else {
333 // No password or SSH key - use standard initialization with random keys
334 log_debug("CLIENT_CRYPTO_INIT: Using standard initialization");
335 result = crypto_handshake_init("crypto_client_local", &g_crypto_ctx, false); // false = client
336 if (result != ASCIICHAT_OK) {
337 FATAL(result, "Failed to initialize crypto handshake");
338 }
339 }

341 log_debug("CLIENT_CRYPTO_INIT: crypto_handshake_init succeeded");

343 // Set up server connection info for known_hosts
344 SAFE_STRNCPY(g_crypto_ctx.server_hostname, address, sizeof(g_crypto_ctx.server_hostname) - 1);
345 const char *server_ip = server_connection_get_ip();

349 // Use port from server_connection_set_port() if available (e.g., from WebSocket URL),

353 log_debug("CLIENT_CRYPTO_INIT: Set server_ip='%s', server_port=%u", g_crypto_ctx.server_ip, g_crypto_ctx.server_port);
354
355 // Configure server key verification if specified
356 if (strlen(server_key) > 0) {
357 g_crypto_ctx.verify_server_key = true;
358 SAFE_STRNCPY(g_crypto_ctx.expected_server_key, server_key, sizeof(g_crypto_ctx.expected_server_key) - 1);
359 log_debug("Server key verification enabled: %s", server_key);
360 }
361
362 // If --require-client-verify is set, perform ACDS session lookup for server identity
363 if (GET_OPTION(require_client_verify) && strlen(GET_OPTION(session_string)) > 0) {
364 log_debug("--require-client-verify enabled: performing ACDS session lookup for '%s'", GET_OPTION(session_string));
365
366 // Connect to ACDS server (configurable via --acds-server and --acds-port options)
367 acds_client_config_t acds_config;
368 acds_client_config_init_defaults(&acds_config);
369 SAFE_STRNCPY(acds_config.server_address, GET_OPTION(discovery_server), sizeof(acds_config.server_address));
370 acds_config.server_port = GET_OPTION(discovery_port);

373 // ACDS key verification (optional in debug builds, only if --discovery-service-key is provided)
374 if (strlen(GET_OPTION(discovery_service_key)) > 0) {
375 log_debug("Verifying ACDS server key for %s...", acds_config.server_address);
376 uint8_t acds_pubkey[32];
377 asciichat_error_t verify_result =
378 discovery_keys_verify(acds_config.server_address, GET_OPTION(discovery_service_key), acds_pubkey);
379 if (verify_result != ASCIICHAT_OK) {
380 log_error("ACDS key verification failed for %s", acds_config.server_address);
381 return -1;
382 }
383 log_debug("ACDS server key verified successfully");
384 }
385#ifndef NDEBUG
386 else {
387 log_debug("Skipping ACDS key verification (debug build, no --discovery-service-key provided)");
388 }
389#endif
390
391 acds_client_t acds_client;
392 asciichat_error_t acds_result = acds_client_connect(&acds_client, &acds_config);
393 if (acds_result != ASCIICHAT_OK) {
394 log_error("Failed to connect to ACDS server at %s:%d", acds_config.server_address, acds_config.server_port);
395 return -1;
396 }
397
398 // Perform SESSION_LOOKUP to get server's identity
399 acds_session_lookup_result_t lookup_result;
400 acds_result = acds_session_lookup(&acds_client, GET_OPTION(session_string), &lookup_result);
401 acds_client_disconnect(&acds_client);
402
403 if (acds_result != ASCIICHAT_OK || !lookup_result.found) {
404 log_error("ACDS session lookup failed for '%s': %s", GET_OPTION(session_string),
405 lookup_result.found ? "session not found" : "lookup error");
406 return -1;
407 }
408
409 // Convert server's Ed25519 public key (32 bytes) to hex string
410 char server_key_hex[65]; // 32 bytes * 2 + null terminator
411 for (size_t i = 0; i < 32; i++) {
412 SAFE_SNPRINTF(&server_key_hex[i * 2], 3, "%02x", lookup_result.host_pubkey[i]);
413 }
414 server_key_hex[64] = '\0';
415
416 // Set expected server key for verification during handshake
417 g_crypto_ctx.verify_server_key = true;
418 SAFE_STRNCPY(g_crypto_ctx.expected_server_key, server_key_hex, sizeof(g_crypto_ctx.expected_server_key) - 1);
419 log_debug("ACDS session lookup succeeded - server identity will be verified");
420 log_debug("Expected server key (from ACDS): %s", server_key_hex);
421 }
422
423 // Mark initialization as complete
424 lifecycle_init(&g_crypto_lc, "client_crypto");
425
426 log_debug("Client crypto handshake initialized");
427 log_debug("CLIENT_CRYPTO_INIT: Initialization complete");
428 return 0;

439int client_crypto_handshake(socket_t socket) {
440 // If client has --no-encrypt, skip handshake entirely
441 if (GET_OPTION(no_encrypt)) {
442 log_debug("Client has --no-encrypt, skipping crypto handshake");
443 return 0;
444 }
445
446 // If we reach here, crypto must be initialized for encryption
447 if (!lifecycle_is_initialized(&g_crypto_lc)) {
448 log_error("Crypto not initialized but server requires encryption");
449 log_error("Server requires encrypted connection but client has no encryption configured");
450 log_error("Use --key to specify a client key or --password for password authentication");
451 return CONNECTION_ERROR_AUTH_FAILED; // No retry - configuration error
452 }
453
454 log_debug("Starting crypto handshake with server...");
455
456 START_TIMER("client_crypto_handshake");
457
458 // Step 0a: Send protocol version to server
459 protocol_version_packet_t client_version = {0};
460 client_version.protocol_version = HOST_TO_NET_U16(1); // Protocol version 1
461 client_version.protocol_revision = HOST_TO_NET_U16(0); // Revision 0
462 client_version.supports_encryption = g_crypto_mode; // Send crypto mode bitmask
463 client_version.compression_algorithms = 0; // No compression for now
464 client_version.compression_threshold = 0;
465 client_version.feature_flags = 0;
466
467 int result = send_protocol_version_packet(socket, &client_version);
468 if (result != 0) {
469 log_error("Failed to send protocol version to server");
470 STOP_TIMER("client_crypto_handshake");
471 return -1;
472 }
473 log_debug("CLIENT_CRYPTO_HANDSHAKE: Protocol version sent successfully");
474
475 // Step 0b: Receive server's protocol version
476 packet_type_t packet_type;
477 void *payload = NULL;
478 size_t payload_len = 0;
479
480 result = receive_packet(socket, &packet_type, &payload, &payload_len);
481 if (result != ASCIICHAT_OK || packet_type != PACKET_TYPE_PROTOCOL_VERSION) {
482 log_error("Failed to receive server protocol version (got type %u)", packet_type);
483 log_error("Packet type 0x%x (decimal %u) - Expected 0x%x (decimal %d)", packet_type, packet_type,
484 PACKET_TYPE_PROTOCOL_VERSION, PACKET_TYPE_PROTOCOL_VERSION);
485 log_error("This suggests a protocol mismatch or packet corruption");
486 log_error("Raw packet type bytes: %02x %02x %02x %02x", (packet_type >> 0) & 0xFF, (packet_type >> 8) & 0xFF,
487 (packet_type >> 16) & 0xFF, (packet_type >> 24) & 0xFF);
488 if (payload) {
489 buffer_pool_free(NULL, payload, payload_len);
490 }
491 STOP_TIMER("client_crypto_handshake");
492 return -1;
493 }
494
495 if (payload_len != sizeof(protocol_version_packet_t)) {
496 log_error("Invalid protocol version packet size: %zu, expected %zu", payload_len,
497 sizeof(protocol_version_packet_t));
498 buffer_pool_free(NULL, payload, payload_len);
499 STOP_TIMER("client_crypto_handshake");
500 return -1;
501 }
502
503 protocol_version_packet_t server_version;
504 memcpy(&server_version, payload, sizeof(protocol_version_packet_t));
505 buffer_pool_free(NULL, payload, payload_len);
506
507 // Convert from network byte order
508 uint16_t server_proto_version = NET_TO_HOST_U16(server_version.protocol_version);
509 uint16_t server_proto_revision = NET_TO_HOST_U16(server_version.protocol_revision);
510 uint8_t server_mode = server_version.supports_encryption;
511
512 log_info("Server protocol version: %u.%u (crypto mode: 0x%02x)", server_proto_version, server_proto_revision,
513 server_mode);
514
515 // Verify server echoed our mode
516 if (server_mode != g_crypto_mode) {
517 log_error("Server mode mismatch: got 0x%02x, expected 0x%02x", server_mode, g_crypto_mode);
518 STOP_TIMER("client_crypto_handshake");
519 return CONNECTION_ERROR_AUTH_FAILED;
520 }
521
522 // Step 0c: Send crypto capabilities to server
523 log_debug("CLIENT_CRYPTO_HANDSHAKE: Sending crypto capabilities");
524 crypto_capabilities_packet_t client_caps = {0};
525 client_caps.supported_kex_algorithms = HOST_TO_NET_U16(KEX_ALGO_X25519);
526 client_caps.supported_auth_algorithms = HOST_TO_NET_U16(AUTH_ALGO_ED25519 | AUTH_ALGO_NONE);
527 client_caps.supported_cipher_algorithms = HOST_TO_NET_U16(CIPHER_ALGO_XSALSA20_POLY1305 | CIPHER_ALGO_NONE);
528 client_caps.requires_verification = 0; // Client doesn't require server verification (uses known_hosts)
529 client_caps.preferred_kex = KEX_ALGO_X25519;
530 client_caps.preferred_auth = ACIP_CRYPTO_HAS_AUTH(g_crypto_mode) ? AUTH_ALGO_ED25519 : AUTH_ALGO_NONE;
531 client_caps.preferred_cipher =
532 ACIP_CRYPTO_HAS_ENCRYPT(g_crypto_mode) ? CIPHER_ALGO_XSALSA20_POLY1305 : CIPHER_ALGO_NONE;
533
534 result = send_crypto_capabilities_packet(socket, &client_caps);
535 if (result != 0) {
536 log_error("Failed to send crypto capabilities to server");
537 STOP_TIMER("client_crypto_handshake");
538 return -1;
539 }
540 log_debug("CLIENT_CRYPTO_HANDSHAKE: Crypto capabilities sent successfully");
541
542 // Step 0d: Receive server's crypto parameters
543 log_debug("CLIENT_CRYPTO_HANDSHAKE: Receiving server crypto parameters");
544 payload = NULL;
545 payload_len = 0;
546
547 result = receive_packet(socket, &packet_type, &payload, &payload_len);
548 if (result != ASCIICHAT_OK || packet_type != PACKET_TYPE_CRYPTO_PARAMETERS) {
549 log_error("Failed to receive server crypto parameters (got type %u)", packet_type);
550 if (payload) {
551 buffer_pool_free(NULL, payload, payload_len);
552 }
553 STOP_TIMER("client_crypto_handshake");
554 return -1;
555 }
556

558 log_error("Invalid crypto parameters packet size: %zu, expected %zu", payload_len,
559 sizeof(crypto_parameters_packet_t));

584 FATAL(result, "Failed to set crypto parameters");

587 // Store verification flag - server will verify client identity (whitelist check)
588 // This is independent of whether server provides its own identity
589 if (server_params.verification_enabled) {

591 g_crypto_ctx.require_client_auth = true;
592 log_info("Server will verify client identity (whitelist enabled)");

596 if (server_params.selected_kex != KEX_ALGO_X25519) {

598 STOP_TIMER("client_crypto_handshake");
599 return CONNECTION_ERROR_AUTH_FAILED;
600 }
601
602 // Validate cipher selection based on negotiated mode
603 bool expect_cipher = ACIP_CRYPTO_HAS_ENCRYPT(g_crypto_mode);
604 if (expect_cipher && server_params.selected_cipher != CIPHER_ALGO_XSALSA20_POLY1305) {
605 log_error("Server selected unsupported cipher algorithm: %u", server_params.selected_cipher);
606 STOP_TIMER("client_crypto_handshake");
607 return CONNECTION_ERROR_AUTH_FAILED;
608 }
609 if (!expect_cipher && server_params.selected_cipher != CIPHER_ALGO_NONE) {
610 log_error("Server chose cipher %u but client requested no encryption", server_params.selected_cipher);
611 STOP_TIMER("client_crypto_handshake");
612 return CONNECTION_ERROR_AUTH_FAILED;
613 }
614
615 log_debug("CLIENT_CRYPTO_HANDSHAKE: Protocol negotiation completed successfully");

617 // Step 0.5: Send CRYPTO_CLIENT_HELLO with expected server key (multi-key support)
618 // This allows the server to select the correct identity key from g_server_identity_keys[]
619 if (g_crypto_ctx.verify_server_key && strlen(g_crypto_ctx.expected_server_key) > 0) {
620 log_debug("CLIENT_CRYPTO_HANDSHAKE: Sending expected server key to support multi-key selection");
621
622 // Parse expected server key to get the Ed25519 public key
623 public_key_t expected_keys[MAX_CLIENTS];
624 size_t num_expected_keys = 0;
625
626 if (parse_public_keys(g_crypto_ctx.expected_server_key, expected_keys, &num_expected_keys, MAX_CLIENTS) == 0 &&
627 num_expected_keys > 0) {
628 // Send the first expected key (client should only have one expected server key)
629 // Format: 32-byte Ed25519 public key
630 log_debug("Sending CRYPTO_CLIENT_HELLO with expected server key for multi-key selection");
631 result = send_packet(socket, PACKET_TYPE_CRYPTO_CLIENT_HELLO, expected_keys[0].key, ED25519_PUBLIC_KEY_SIZE);
632
633 if (result != ASCIICHAT_OK) {
634 FATAL(result, "Failed to send CRYPTO_CLIENT_HELLO packet");
635 }
636
637 // Log the key fingerprint for debugging
638 char hex_key[65];
639 pubkey_to_hex(expected_keys[0].key, hex_key);
640 log_debug("Sent expected server key: %s", hex_key);
641 } else {
642 log_warn("Failed to parse expected server key '%s' - server will use default key",
643 g_crypto_ctx.expected_server_key);
644 }
645 }

648 log_debug("CLIENT_CRYPTO_HANDSHAKE: Starting key exchange");
649 result = crypto_handshake_client_key_exchange_socket(&g_crypto_ctx, socket);
650 if (result != ASCIICHAT_OK) {
651#ifdef _WIN32

653 // Media Foundation creates background COM threads that can block exit() if not properly shut down

655#endif
656 FATAL(result, "Crypto key exchange failed");
