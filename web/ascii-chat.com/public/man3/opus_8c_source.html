19opus_codec_t *opus_codec_create_encoder(opus_application_t application, int sample_rate, int bitrate) {
20 if (sample_rate <= 0 || bitrate <= 0) {
21 SET_ERRNO(ERROR_INVALID_PARAM, "Invalid codec parameters: sample_rate=%d, bitrate=%d", sample_rate, bitrate);
22 return NULL;
23 }
24
25 opus_codec_t *codec = SAFE_MALLOC(sizeof(opus_codec_t), opus_codec_t *);
26 if (!codec) {
27 SET_ERRNO(ERROR_MEMORY, "Failed to allocate opus codec context");
28 return NULL;
29 }
30
31 // Create encoder
32 int error = OPUS_OK;
33 codec->encoder = opus_encoder_create(sample_rate, 1, (int)application, &error);
34 if (error != OPUS_OK || !codec->encoder) {
35 SET_ERRNO(ERROR_AUDIO, "Failed to create Opus encoder: %s", opus_strerror(error));
36 SAFE_FREE(codec);
37 return NULL;
38 }
39
40 // Set bitrate
41 error = opus_encoder_ctl(codec->encoder, OPUS_SET_BITRATE(bitrate));
42 if (error != OPUS_OK) {
43 SET_ERRNO(ERROR_AUDIO, "Failed to set Opus bitrate: %s", opus_strerror(error));
44 opus_encoder_destroy(codec->encoder);
45 SAFE_FREE(codec);
46 return NULL;
47 }
48
49 // Enable DTX (Discontinuous Transmission) for efficient silence frame encoding
50 // DTX allows the encoder to produce very small packets (3-4 bytes) for silence,
51 // reducing bandwidth usage while maintaining audio quality for voice/music
52 error = opus_encoder_ctl(codec->encoder, OPUS_SET_DTX(1));
53 if (error != OPUS_OK) {
54 log_warn("Failed to enable Opus DTX: %s (continuing without DTX)", opus_strerror(error));
55 }
56
57 codec->decoder = NULL;
58 codec->sample_rate = sample_rate;
59 codec->bitrate = bitrate;
60 codec->tmp_buffer = NULL;
61
62 log_debug("Opus encoder created: sample_rate=%d, bitrate=%d bps, DTX enabled", sample_rate, bitrate);
63
64 // Register Opus encoder for debugging
65 NAMED_REGISTER_CONTEXT(codec, "opus_encoder", "opus_codec", NULL);
66
67 return codec;
68}

112size_t opus_codec_encode(opus_codec_t *codec, const float *samples, int num_samples, uint8_t *out_data,
113 size_t out_size) {
114 if (!codec || !codec->encoder || !samples || num_samples <= 0 || !out_data || out_size == 0) {
115 SET_ERRNO(ERROR_INVALID_PARAM,
116 "Invalid encode parameters: codec=%p, encoder=%p, samples=%p, num_samples=%d, out_data=%p, "
117 "out_size=%zu",
118 (void *)codec, (void *)(codec ? codec->encoder : NULL), (const void *)samples, num_samples,
119 (void *)out_data, out_size);
120 return 0;
121 }
122
123 // Encode frame
124 opus_int32 encoded_bytes = opus_encode_float(codec->encoder, samples, num_samples, out_data, (opus_int32)out_size);
125
126 if (encoded_bytes < 0) {
127 SET_ERRNO(ERROR_AUDIO, "Opus encoding failed: %s", opus_strerror((int)encoded_bytes));
128 return 0;
129 }
130
131 if (encoded_bytes == 0) {
132 // DTX frame (silence) - encoder produced zero bytes
133 log_debug_every(LOG_RATE_VERY_FAST, "Opus DTX frame (silence detected)");
134 }
135
136 return (size_t)encoded_bytes;
137}

143int opus_codec_decode(opus_codec_t *codec, const uint8_t *data, size_t data_len, float *out_samples,
144 int out_num_samples) {
145 if (!codec || !codec->decoder || !out_samples || out_num_samples <= 0) {
146 SET_ERRNO(ERROR_INVALID_PARAM,
147 "Invalid decode parameters: codec=%p, decoder=%p, out_samples=%p, out_num_samples=%d", (void *)codec,
148 (void *)(codec ? codec->decoder : NULL), (void *)out_samples, out_num_samples);
149 return -1;
150 }
151
152 // If data is NULL, use PLC (Packet Loss Concealment)
153 if (!data || data_len == 0) {
154 log_debug_every(LOG_RATE_VERY_FAST, "Opus PLC (Packet Loss Concealment)");
155 int samples = opus_decode_float(codec->decoder, NULL, 0, out_samples, out_num_samples, 0);
156 if (samples < 0) {
157 SET_ERRNO(ERROR_AUDIO, "Opus PLC failed: %s", opus_strerror(samples));
158 return -1;
159 }
160 return samples;
161 }
162
163 // Decode frame
164 int samples = opus_decode_float(codec->decoder, data, (opus_int32)data_len, out_samples, out_num_samples, 0);
165
166 if (samples < 0) {
167 SET_ERRNO(ERROR_AUDIO, "Opus decoding failed: %s", opus_strerror(samples));
168 return -1;
169 }
170
171 return samples;
172}
