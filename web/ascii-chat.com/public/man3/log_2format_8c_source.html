44static log_template_t *parse_format_string(const char *format_str, bool console_only) {
45 if (!format_str) {
46 SET_ERRNO(ERROR_INVALID_PARAM, "Invalid format string: NULL");

50 bool is_valid = utf8_is_valid(format_str);

57 // Use regular malloc/calloc instead of SAFE_* to avoid debug tracking overhead at startup
58 // This avoids mutex contention and memory tracking overhead during log initialization
59 log_template_t *result = (log_template_t *)calloc(1, sizeof(log_template_t));
60 if (!result) {

255 * without needing custom parsing for each one. strftime will handle validation. */
256 const char *fmt_start = p;
257 size_t fmt_len = 0;
258
259 /* Collect the format code (usually 1-2 chars, but allow flexibility) */

262 fmt_len++;
263 }

265 if (fmt_len == 0) {

268 }

271 result->specs[spec_idx].literal = (char *)malloc(fmt_len + 1);
272 if (!result->specs[spec_idx].literal) {
273 goto cleanup;
274 }
275 memcpy(result->specs[spec_idx].literal, fmt_start, fmt_len);
276 result->specs[spec_idx].literal[fmt_len] = '\0';
277 result->specs[spec_idx].literal_len = fmt_len;

279 spec_idx++;
280 /* p already advanced in the while loop above */
281 }
282 }
283 } else {
284 /* Literal text until next escape or specifier */
285 const char *text_start = p;
286 while (*p && *p != '\\' && *p != '%') {
287 p++;
288 }
289
290 result->specs[spec_idx].type = LOG_FORMAT_LITERAL;

292 result->specs[spec_idx].literal = (char *)malloc(text_len + 1);
293 if (!result->specs[spec_idx].literal) {

300 result->specs[spec_idx].literal_len = text_len;
301 spec_idx++;
302 }
303 }

305 result->spec_count = spec_idx;

484int log_template_apply(const log_template_t *format, char *buf, size_t buf_size, log_level_t level,
485 const char *timestamp, const char *file, int line, const char *func, uint64_t tid,
486 const char *message, bool use_colors, uint64_t time_nanoseconds) {
487 /* Check all critical pointers - format could be freed by another thread (TOCTOU race) */
488 if (!format || !buf || buf_size == 0) {
489 return -1;
490 }
491
492 int total_written = 0;
493 char *p = buf;
494 size_t remaining = buf_size - 1; /* Reserve space for null terminator */
495
496 (void)timestamp; /* May be used in future; for now, LOG_FORMAT_TIME uses custom formatting */
497
498 /* Safety check: verify format->specs is valid before dereferencing
499 * This protects against use-after-free if format is freed by another thread (TOCTOU race).
500 * Between our initial NULL check and here, the main thread may have freed this memory. */
501 if (!format || !format->specs || format->spec_count == 0) {
502 return -1;
503 }
504
505 for (size_t i = 0; i < format->spec_count; i++) {
506 /* Re-check specs pointer in loop to catch use-after-free during iteration */
507 if (!format->specs || i >= format->spec_count) {
508 break; /* Format was freed, stop processing */
509 }
510 const log_format_spec_t *spec = &format->specs[i];
511
512 /* Additional safety check: if spec is null or invalid, stop processing */
513 if (!spec) {
514 break;
515 }
516
517 int written = 0;
518
519 switch (spec->type) {
520 case LOG_FORMAT_LITERAL:
521 written = safe_snprintf(p, remaining + 1, "%s", spec->literal ? spec->literal : "");
522 break;
523
524 case LOG_FORMAT_TIME:
525 /* Format time using custom time formatter */
526 if (spec->literal) {
527 written = time_format_now(spec->literal, p, remaining + 1);
528 if (written <= 0) {
529 log_debug("time_format_now failed for format: %s", spec->literal);
530 written = 0;
531 }
532 }
533 break;
534
535 case LOG_FORMAT_LEVEL:
536 /* Log level as string (DEV, DEBUG, INFO, WARN, ERROR, FATAL) */
537 written = safe_snprintf(p, remaining + 1, "%s", get_level_string(level));
538 break;
539
540 case LOG_FORMAT_LEVEL_ALIGNED:
541 /* Log level padded to exactly 5 characters (prevents truncation to [DEBU]/[ERRO]) */
542 written = safe_snprintf(p, remaining + 1, "%s", get_level_string_padded(level));
543 break;
544
545 case LOG_FORMAT_FILE:
546 if (file) {
547 written = safe_snprintf(p, remaining + 1, "%s", file);
548 }
549 break;
550
551 case LOG_FORMAT_FILE_RELATIVE:
552 if (file) {
553 const char *rel_file = extract_project_relative_path(file);
554 written = safe_snprintf(p, remaining + 1, "%s", rel_file);
555 }
556 break;
557
558 case LOG_FORMAT_LINE:
559 if (line > 0) {
560 written = safe_snprintf(p, remaining + 1, "%d", line);
561 }
562 break;
563
564 case LOG_FORMAT_FUNC:
565 if (func) {
566 written = safe_snprintf(p, remaining + 1, "%s", func);
567 }
568 break;
569
570 case LOG_FORMAT_TID:
571 written = safe_snprintf(p, remaining + 1, "%llu", (unsigned long long)tid);
572 break;
573
574 case LOG_FORMAT_TNAME: {
575 asciichat_thread_t thread = (asciichat_thread_t)tid;
576 uintptr_t key = asciichat_thread_to_key(thread);
577 const char *tname = named_get(key);
578 if (tname) {
579 written = safe_snprintf(p, remaining + 1, "%s", tname);
580 } else {
581 written = safe_snprintf(p, remaining + 1, "%llu", (unsigned long long)tid);
582 }
583 break;
584 }
585
586 case LOG_FORMAT_MICROSECONDS: {
587 /* Extract microseconds from nanoseconds (ns_value % 1_000_000_000 / 1000) */
588 long nanoseconds = (long)(time_nanoseconds % NS_PER_SEC_INT);
589 long microseconds = nanoseconds / 1000;
590 if (microseconds < 0)
591 microseconds = 0;
592 if (microseconds > 999999)
593 microseconds = 999999;
594 written = safe_snprintf(p, remaining + 1, "%06ld", microseconds);
595 break;
596 }
597
598 case LOG_FORMAT_NANOSECONDS: {
599 /* Extract nanoseconds component (ns_value % 1_000_000_000) */
600 long nanoseconds = (long)(time_nanoseconds % NS_PER_SEC_INT);
601 if (nanoseconds < 0)
602 nanoseconds = 0;
603 if (nanoseconds > 999999999)
604 nanoseconds = 999999999;
605 written = safe_snprintf(p, remaining + 1, "%09ld", nanoseconds);
606 break;
607 }
608
609 case LOG_FORMAT_STRFTIME_CODE: {
610 /* strftime format code (like %H, %M, %S, %A, %B, etc.)
611 * Let strftime handle all the parsing and validation */
612 if (spec->literal && spec->literal_len < 64) {
613 /* Construct format string with % prefix on stack */
614 char format_str[66]; /* 1 (%) + 64 (literal) + 1 (null) */
615 format_str[0] = '%';
616 memcpy(format_str + 1, spec->literal, spec->literal_len);
617 format_str[spec->literal_len + 1] = '\0';
618 written = time_format_now(format_str, p, remaining + 1);
619 if (written <= 0) {
620 log_debug("time_format_now failed for format code: %s", format_str);
621 written = 0;
622 }
623 }
624 break;
625 }
626
627 case LOG_FORMAT_MESSAGE:
628 if (message) {
629 /* Format hex addresses as named object descriptions (type/name format) */
630 const char *formatted_msg = log_named_format_or_original(message);
631 written = safe_snprintf(p, remaining + 1, "%s", formatted_msg);
632 }
633 break;
634
635 case LOG_FORMAT_COLORED_MESSAGE: {
636 /* Apply colorize_log_message() for number/unit/hex highlighting, then format named objects */
637 if (message) {
638 const char *formatted_msg = log_named_format_or_original(message);
639 const char *colorized_msg = colorize_log_message(formatted_msg);
640 written = safe_snprintf(p, remaining + 1, "%s", colorized_msg);
641 }
642 break;
643 }
644
645 case LOG_FORMAT_COLORLOG_LEVEL:
646 /* Color code for the log level (placeholder for future color support) */
647 (void)use_colors; /* Suppress unused parameter warning */
648 written = 0;
649 break;
650
651 case LOG_FORMAT_COLOR: {
652 /* Parse and apply %color(LEVEL, content)
653 * literal stores "LEVEL,content" where LEVEL is the color level name (or "*" for current level)
654 * and content is a format string to render and colorize */
655 if (!spec->literal || spec->literal_len == 0) {
656 written = 0;
657 break;
658 }
659
660 /* Find the comma separating LEVEL and content */
661 const char *comma_pos = strchr(spec->literal, ',');
662 if (!comma_pos) {
663 /* Invalid format - no comma found */
664 log_debug("log_template_apply: %%color format missing comma in: %s", spec->literal);
665 written = 0;
666 break;
667 }
668
669 /* Extract level name (before comma) */
670 size_t level_len = comma_pos - spec->literal;
671 char level_name[32];
672 if (level_len >= sizeof(level_name)) {
673 /* Level name too long */
674 written = 0;
675 break;
676 }
677 memcpy(level_name, spec->literal, level_len);
678 level_name[level_len] = '\0';
679
680 /* Parse level name to log_color_t (pass current level for "*" support) */
681 log_color_t color = parse_color_level(level_name, level);
682
683 /* Extract content (after comma), skip leading whitespace */
684 const char *content_start = comma_pos + 1;
685 while (*content_start == ' ' || *content_start == '\t') {
686 content_start++;
687 }
688
689 /* Render content recursively */
690 char content_buf[512];
691 int content_len = render_format_content(content_start, content_buf, sizeof(content_buf), level, timestamp, file,
692 line, func, tid, message, use_colors, time_nanoseconds);
693
694 if (content_len < 0 || content_len >= (int)sizeof(content_buf)) {
695 written = 0;
696 break;
697 }
698
699 /* Apply color to rendered content using colored_string */
700 const char *colored_content = colored_string(color, content_buf);
701
702 /* Copy colored content to output buffer */
703 written = safe_snprintf(p, remaining + 1, "%s", colored_content);
704 break;
705 }
706
707 case LOG_FORMAT_NEWLINE:
708 /* Platform-aware newline */
709#ifdef _WIN32
710 written = safe_snprintf(p, remaining + 1, "\r\n");
711#else
712 written = safe_snprintf(p, remaining + 1, "\n");
713#endif
714 break;
715
716 default:
717 break;
718 }
719
720 if (written < 0) {
721 /* snprintf error */
722 return -1;
723 }
724
725 if ((size_t)written > remaining) {
726 /* Buffer overflow prevention */
727 log_debug("log_template_apply: buffer overflow prevented");
728 return -1;
729 }
730
731 p += written;
732 remaining -= written;
733 total_written += written;
734 }
735
736 *p = '\0';
737 return total_written;
738}
