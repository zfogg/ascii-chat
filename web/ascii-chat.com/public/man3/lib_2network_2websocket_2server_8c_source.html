119 // Optimize TCP for high-throughput large message transfer.
120 // TCP delayed ACK (default ~40ms) causes the sender to stall waiting for ACKs,
121 // creating ~30ms gaps between 128KB chunk deliveries for large messages.
122 // TCP_QUICKACK forces immediate ACKs so the sender can push data continuously (Linux only).
123 // TCP_NODELAY disables Nagle's algorithm for the send path.
124#ifdef __linux__
125 {
126 int fd = lws_get_socket_fd(wsi);
127 if (fd >= 0) {
128 int quickack = 1;
129 if (setsockopt(fd, IPPROTO_TCP, TCP_QUICKACK, &quickack, sizeof(quickack)) < 0) {
130 log_warn("Failed to set TCP_QUICKACK: %s", SAFE_STRERROR(errno));
131 }
132 int nodelay = 1;
133 if (setsockopt(fd, IPPROTO_TCP, TCP_NODELAY, &nodelay, sizeof(nodelay)) < 0) {
134 log_warn("Failed to set TCP_NODELAY: %s", SAFE_STRERROR(errno));
135 }
136 // Do NOT set SO_RCVBUF/SO_SNDBUF manually.
137 // Setting SO_RCVBUF disables TCP autotuning on Linux, which locks the receive buffer
138 // at the rmem_default (212KB). Without manual override, the kernel can autotune up to
139 // tcp_rmem max (typically 6MB), allowing the entire 921KB video frame to fit in one
140 // TCP window without flow control stalls.
141 int actual_rcv = 0, actual_snd = 0;
142 socklen_t optlen = sizeof(int);
143 getsockopt(fd, SOL_SOCKET, SO_RCVBUF, &actual_rcv, &optlen);
144 getsockopt(fd, SOL_SOCKET, SO_SNDBUF, &actual_snd, &optlen);
145 log_info("[WS_SOCKET] TCP_QUICKACK=1, TCP_NODELAY=1, SO_RCVBUF=%d (autotuned), SO_SNDBUF=%d", actual_rcv,
146 actual_snd);
147 }
148 }
149#endif

556 // Record timing for this callback
557 websocket_callback_timing_record(&g_ws_callback_timing.receive, callback_enter_ns,
558 websocket_callback_timing_start());
559 break;
560 }
561
562 case LWS_CALLBACK_FILTER_HTTP_CONNECTION: {
563 // WebSocket upgrade handshake - allow all connections
564 log_info("[FILTER_HTTP_CONNECTION] WebSocket upgrade request (allow protocol upgrade)");
565 return 0; // Allow the connection
566 }
567
568 case LWS_CALLBACK_PROTOCOL_INIT: {
569 // Protocol initialization with timing instrumentation
570 uint64_t callback_start_ns = websocket_callback_timing_start();
571 log_info("[PROTOCOL_INIT] Protocol initialized, proto=%s", proto_name);
572 uint64_t callback_end_ns = websocket_callback_timing_start();
573 websocket_callback_timing_record(&g_ws_callback_timing.protocol_init, callback_start_ns, callback_end_ns);
574 break;
575 }
576
577 case LWS_CALLBACK_PROTOCOL_DESTROY: {
578 // Protocol destruction with timing instrumentation
579 uint64_t callback_start_ns = websocket_callback_timing_start();
580 log_info("[PROTOCOL_DESTROY] Protocol destroyed, proto=%s", proto_name);
581 uint64_t callback_end_ns = websocket_callback_timing_start();
582 websocket_callback_timing_record(&g_ws_callback_timing.protocol_destroy, callback_start_ns, callback_end_ns);
583 break;
584 }
585
586 case LWS_CALLBACK_EVENT_WAIT_CANCELLED: {
587 // Fired on the service thread when lws_cancel_service() is called from another thread.
588 // This is how we safely convert cross-thread send requests into writable callbacks.
589 // lws_callback_on_writable() is only safe from the service thread context.
590 //
591 // CRITICAL FIX: lws_cancel_service() wakes the event loop, which fires this callback.
592 // The `wsi` parameter may be the listening socket or any random connection, so we
593 // CANNOT rely on lws_get_protocol(wsi) to get the correct protocol.
594 // Instead, we MUST trigger WRITEABLE on ALL protocols that handle WebSocket frames:
595 // - "http" protocol (browser connects here for initial HTTP upgrade to WebSocket)
596 // - "acip" protocol (for ACIP over WebSocket connections)
597 // Both protocols use the same callback and handle frame transmission.
598 log_debug_every(
599 1 * NS_PER_SEC_INT,
600 ">>> LWS_CALLBACK_EVENT_WAIT_CANCELLED triggered - requesting writable callbacks for all protocols");
601
602 struct lws_context *ctx = lws_get_context(wsi);
603 if (!ctx) {
604 log_error("EVENT_WAIT_CANCELLED: Could not get context from wsi");
605 break;
606 }
607
608 // Trigger WRITEABLE on both protocols
609 // Browser clients connect with "http" protocol, so they MUST have WRITEABLE triggered
610 for (int i = 0; i < 2; i++) {
611 log_dev_every(1 * NS_PER_SEC_INT,
612 ">>> EVENT_WAIT_CANCELLED: Calling lws_callback_on_writable_all_protocol for protocol '%s'",
613 websocket_protocols[i].name);
614 lws_callback_on_writable_all_protocol(ctx, &websocket_protocols[i]);
615 }
616 break;
617 }
618
619 default:
620 break;
621 }
622
623 return 0;
624}

680asciichat_error_t websocket_server_init(websocket_server_t *server, const websocket_server_config_t *config) {
681 if (!server || !config || !config->client_handler) {
682 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid parameters");
683 }
684
685 memset(server, 0, sizeof(*server));
686 server->handler = config->client_handler;
687 server->user_data = config->user_data;
688 server->port = config->port;
689 atomic_store_bool(&server->running, true);
690
691 // Store server pointer in protocol user data so callbacks can access it
692 // Both the HTTP and ACIP protocols need access to the server
693 websocket_protocols[0].user = server; // http protocol
694 websocket_protocols[1].user = server; // acip protocol
695
696 // Enable libwebsockets logging through centralized logging system
697 lws_log_init_server();
698
699 // Configure libwebsockets context
700 struct lws_context_creation_info info = {0};
701 info.port = config->port;
702 info.protocols = websocket_protocols;
703 info.gid = (gid_t)-1; // Cast to avoid undefined behavior with unsigned type
704 info.uid = (uid_t)-1; // Cast to avoid undefined behavior with unsigned type
705 info.options = LWS_SERVER_OPTION_DO_SSL_GLOBAL_INIT; // Initialize SSL/TLS support (required for server binding)
706 info.extensions = NULL; // Disable permessage-deflate - causing connection issues
707 info.retry_and_idle_policy = &keep_alive_policy; // Configure keep-alive to prevent idle disconnects during handshake
708
709 // Configure TLS/WSS support if certificates are provided (check for non-empty strings)
710 if (config->tls_cert_path && config->tls_cert_path[0] != '\0' && config->tls_key_path &&
711 config->tls_key_path[0] != '\0') {
712#ifdef LWS_WITH_TLS
713 info.ssl_cert_filepath = config->tls_cert_path;
714 info.ssl_private_key_filepath = config->tls_key_path;
715 log_info("WebSocket server configured for WSS (TLS): cert=%s, key=%s", config->tls_cert_path, config->tls_key_path);
716#else
717 log_warn("WebSocket server: TLS support not compiled in libwebsockets; WSS unavailable");
718#endif
719 } else if ((config->tls_cert_path && config->tls_cert_path[0] != '\0') ||
720 (config->tls_key_path && config->tls_key_path[0] != '\0')) {
721 log_warn("WebSocket server: Both TLS certificate and key must be provided for WSS; using plain WS");
722 }
723
724 // Increase per-thread service buffer to prevent fragmentation of large messages
725 // Default is 4KB, causing large frames to fragment into many chunks
726 // Video frames are 921KB, so increase to 2MB to handle them without fragmentation
727 info.pt_serv_buf_size = 2 * 1024 * 1024; // 2MB per-thread service buffer
728
729 // Disable ALL default timeouts and keep-alive mechanisms
730 // Use only the explicit retry_and_idle_policy we set above (30/35 seconds)
731 info.ka_time = 0; // Disable TCP keep-alive probes (use WebSocket pings instead)
732 info.ka_probes = 0; // Disable TCP probes
733 info.ka_interval = 0; // Disable TCP probe intervals
734 info.keepalive_timeout = 0; // Disable HTTP keep-alive timeout entirely
735
736 // Explicitly set a very large idle timeout to prevent early disconnection
737 // Without this, libwebsockets defaults to 5 seconds for HTTP connections
738 // Set to effectively infinite (100 hours) since we use retry_and_idle_policy instead
739 info.timeout_secs = 360000; // 100 hours - effectively disabled
740
741 // Create libwebsockets context
742 server->context = lws_create_context(&info);
743 if (!server->context) {
744 return SET_ERRNO(ERROR_NETWORK_BIND, "Failed to create libwebsockets context");
745 }
746
747 // Create thread pool for handling client connections
748 // Use 4 worker threads to handle concurrent client handlers without blocking LWS event loop
749 server->handler_pool = thread_pool_create_with_workers("websocket_handlers", 4);
750 if (!server->handler_pool) {
751 lws_context_destroy(server->context);
752 server->context = NULL;
753 return SET_ERRNO(ERROR_THREAD, "Failed to create WebSocket handler thread pool");
754 }
755
756 /* Register WebSocket server with named registry */
757 char ws_port_name[32];
758 snprintf(ws_port_name, sizeof(ws_port_name), "ws:%d", server->port);
759 NAMED_REGISTER(server, ws_port_name, "websocket_server", "0x%tx", NULL);
760
761 /* Register server's sync primitives with hierarchical naming */
762 NAMED_REGISTER_ATOMIC(&server->running, "is_running", (uintptr_t)(const void *)(server));
763
764 /* Register WebSocket context as child of server */
765 NAMED_REGISTER_WEBSOCKET(server->context, "context", (uintptr_t)(const void *)(server));
766
767 log_info("WebSocket server initialized on port %d with static file serving", config->port);
768 return ASCIICHAT_OK;
769}

771asciichat_error_t websocket_server_run(websocket_server_t *server) {
772 if (!server || !server->context) {
773 return SET_ERRNO(ERROR_INVALID_PARAM, "Invalid server");
774 }
775
776 log_info("WebSocket server starting event loop on port %d", server->port);
777
778 // Run libwebsockets event loop
779 uint64_t last_service_ns = 0;
780 int service_call_count = 0;
781 while (atomic_load_bool(&server->running)) {
782 // Service libwebsockets with 16ms timeout to match 60 FPS frame rate.
783 // This provides frequent event processing (~60 callback invocations per second) so that
784 // WebSocket frames queued by the render loop are sent promptly, matching TCP performance.
785 // All client connections share this single server context, so a single lws_service() call
786 // services fragments and events for all clients simultaneously.
787 uint64_t service_start_ns = time_get_ns();
788 if (last_service_ns && service_start_ns - last_service_ns > 30 * US_PER_MS_INT) {
789 // > 30ms gap between service calls
790 double gap_ms = (double)(service_start_ns - last_service_ns) / 1e6;
791 log_info_every(1 * US_PER_MS_INT, "[LWS_SERVICE_GAP] %.1fms gap between lws_service calls", gap_ms);
792 }
793 service_call_count++;
794 log_debug_every(500 * US_PER_MS_INT, "[LWS_SERVICE] Call #%d, context=%s", service_call_count,
795 NAMED_DESCRIBE(server->context, "websocket_server"));
796 int result = lws_service(server->context, 1);
797 if (result < 0) {
798 log_error("libwebsockets service error: %d", result);
799 break;
800 }
801 }
802
803 log_info(
804 "WebSocket server event loop exited (context will be destroyed by main thread after handler threads complete)");
805 return ASCIICHAT_OK;
806}

814void websocket_server_destroy(websocket_server_t *server) {
815 if (!server) {
816 return;
817 }
818
819 atomic_store_bool(&server->running, false);
820
821 // Cancel any pending libwebsockets service calls to interrupt blocking lws_service()
822 // This prevents the event loop from getting stuck in lws_service()
823 if (server->context) {
824 log_debug("[WEBSOCKET_SERVER_DESTROY] Cancelling libwebsockets service");
825 lws_cancel_service(server->context);
826 }
827
828 // Unregister atomic fields
829 NAMED_UNREGISTER(&server->running);
830
831 // Destroy handler thread pool (waits for pending work to complete)
832 if (server->handler_pool) {
833 thread_pool_destroy(server->handler_pool);
834 server->handler_pool = NULL;
835 }
836
837 // Context is normally destroyed by websocket_server_run (from the event loop
838 // thread) for fast shutdown. This handles the case where run() wasn't called
839 // or didn't complete normally.
840 if (server->context) {
841 NAMED_UNREGISTER(server->context);
842 log_debug("WebSocket context still alive in destroy, cleaning up");
843 lws_context_destroy(server->context);
844 server->context = NULL;
845 }
846
847 log_debug("WebSocket server destroyed");
848}
