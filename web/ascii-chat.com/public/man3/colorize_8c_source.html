125static bool is_numeric_pattern(const char *str, size_t pos, size_t *end_pos) {
126 if (!isdigit(str[pos]) && str[pos] != '.') {
127 return false;
128 }
129
130 // Don't match numbers in the middle of words - check if preceded by alphanumeric, hyphen, or underscore
131 if (pos > 0 && (isalnum(str[pos - 1]) || str[pos - 1] == '-' || str[pos - 1] == '_')) {

169 if (isdigit(str[i + 1])) {
170 i++;
171 while (isdigit(str[i])) {
172 i++;
173 }
174 }
175 }
176
177 // Check for fraction format (1/2, 3/4, etc.)
178 if (str[i] == '/' && isdigit(str[i + 1])) {
179 i++;
180 while (isdigit(str[i])) {
181 i++;
182 }
183 }

185 // Check for known units with optional spaces (e.g., "25 MB", "1024 GiB", "69.9%")
186 // Only include units if they're in the known units list to avoid colorizing random words
187 size_t j = i;
188 bool has_space = false;
189
190 // First try: units immediately following (no space) like "69.9%"
191 if ((isalpha(str[j]) || str[j] == '%') && (j == i || !isalnum(str[j - 1]))) {

204 j = i;
205 }
206 }
207

209 if (i == j) {
210 while (str[j] == ' ' || str[j] == '\t') {
211 has_space = true;
212 j++;
213 }
214

216 // Collect potential unit (up to 32 characters)
217 size_t unit_start = j;

219 j++;
220 }

222
223 // Check if this is a known unit
224 if (is_known_unit(str + unit_start, unit_len)) {
225 // It's a known unit, include the space and unit
226 i = j;
227 }
228 // Otherwise, don't include - just the number (i stays as is)
229 }
